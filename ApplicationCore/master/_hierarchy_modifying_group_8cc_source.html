58  EntityOwner* HierarchyModifyingGroup::makeOwnerTree(EntityOwner* originalOwner, const std::string& qualifiedName) {
59  std::vector<std::string> splittedPath;
60  boost::split(splittedPath, qualifiedName, boost::is_any_of("/"));
61  bool moveToRoot = false;
62  size_t index = 0;
63  for(auto& pathElement : splittedPath) {
64  ++index;
65  if(pathElement.size() == 0) {
66  // If first element is an empty string, the qualifiedName starts with a slash, which means the entire tree
67  // shall be moved to the root.
68  if(_ownerTree.size() == 0) {
69  // the next path element needs to be moved to root
70  moveToRoot = true;
71  }
72  // Any empty strings should not create a VariableGroup. Multiple consecutive slashes are treated as single.
73  continue;
74  }
75 
76  // Do not create a group for the last element - this one is being represented directly by the
77  // HierarchyModifyingGroup
78  if(index == splittedPath.size()) {
79  break;
80  }
81 
82  // Create variable group
83  EntityOwner* theOwner;
84  if(_ownerTree.size() == 0) {
85  // The first group must be owned by the original owner passed to the contructor of the
86  // HierarchyModifyingGroup
87  theOwner = originalOwner;
88  }
89  else {
90  // Following groups are owned by the previous level
91  theOwner = &_ownerTree.back();
92  }
93  auto modifier = moveToRoot ? HierarchyModifier::moveToRoot : HierarchyModifier::none;
94  if(pathElement == "..") {
95  if(moveToRoot) {
96  throw ChimeraTK::logic_error("QualifiedName of HierarchyModifyingGroup must not start with '/../'!");
97  }
98  modifier = HierarchyModifier::oneUpAndHide;
99  }
100  _ownerTree.emplace_back(theOwner, pathElement, "", modifier);
101  moveToRoot = false;
102  }
103 
104  // If the last element is "..", the hierarchy modifier of the group itself needs to be changed to oneUpAndHide
105  if(splittedPath.back() == "..") {
106  assert(_hierarchyModifier == HierarchyModifier::none);
107  if(moveToRoot) {
108  throw ChimeraTK::logic_error("QualifiedName of HierarchyModifyingGroup must not be '/..'!");
109  }
110  _hierarchyModifier = HierarchyModifier::oneUpAndHide;
111  }
112 
113  // Special case: the qualifiedName is a single hierarchy element only. No additional groups have been created,
114  // and the resulting HierarchyModifyingGroup is identical to a plain VariableGroup.
115  if(_ownerTree.size() == 0) {
116  if(moveToRoot) {
117  // Only one path element which needs to be moved to root? Move the HierachyModifyingGroup itelf to root.
118  assert(_hierarchyModifier == HierarchyModifier::none);
119  _hierarchyModifier = HierarchyModifier::moveToRoot;
120  }
121  return originalOwner;
122  }
123 
124  // Otherwise return the last created group as owner for the HierarchyModifyingGroup itself.
125  return &_ownerTree.back();
126  }
