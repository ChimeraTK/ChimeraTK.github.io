21  TransferGroup::ExceptionHandlingResult TransferGroup::runPostReads(
22  std::set<boost::shared_ptr<TransferElement>>& elements) {
23  ExceptionHandlingResult result;
24 
25  for(auto& elem : elements) {
26  bool updateUserBuffer{true}; // local variable if this element had an exception, to pass on to postRead();
27  // check for exceptions on any of the element's low level elements
28  for(auto& lowLevelElem : elem->getHardwareAccessingElements()) {
29  if(lowLevelElem->_activeException) {
30  updateUserBuffer = false;
31  }
32  }
33 
34  result += handlePostExceptions([&] { elem->postRead(TransferType::read, updateUserBuffer); });
35  }
36 
37  return result;
38  }

41  TransferGroup::ExceptionHandlingResult TransferGroup::handlePostExceptions(Callable function) {
42  try {
43  function();
44  }
45  catch(ChimeraTK::runtime_error& ex) {
46  return ExceptionHandlingResult(true, ex.what());
47  }
48  catch(ChimeraTK::logic_error& ex) {
49  return ExceptionHandlingResult(true, ex.what());
50  }
51  catch(boost::numeric::bad_numeric_cast& ex) {
52  return ExceptionHandlingResult(true, ex.what());
53  }
54  catch(boost::thread_interrupted&) {
55  return ExceptionHandlingResult(true, {}, true); // report that we have seen a thread_interrupted exception
56  }
57  catch(...) {
58  std::cout << "BUG: Wrong exception type thrown in doPostRead() or doPostWrite()!" << std::endl;
59  std::terminate();
60  }
61  return ExceptionHandlingResult(); // result without exceptions
62  }

64  void TransferGroup::read() {
65  // reset exception flags
66  for(auto& it : lowLevelElementsAndExceptionFlags) {
67  it.second = false;
68  }
69 
70  for(auto& elem : highLevelElements) {
71  elem->preReadAndHandleExceptions(TransferType::read);
72  if(elem->_activeException) {
73  for(auto& lowLevelElem : elem->getHardwareAccessingElements()) {
74  lowLevelElementsAndExceptionFlags[lowLevelElem] = true;
75  }
76  }
77  }
78 
79  for(auto& elem : copyDecorators) {
80  elem->preReadAndHandleExceptions(TransferType::read);
81  if(elem->_activeException) {
82  for(auto& lowLevelElem : elem->getHardwareAccessingElements()) {
83  lowLevelElementsAndExceptionFlags[lowLevelElem] = true;
84  }
85  }
86  }
87 
88  for(auto& it : lowLevelElementsAndExceptionFlags) {
89  auto& elem = it.first;
90  auto& hasSeenException = it.second;
91  if(!hasSeenException) {
92  elem->handleTransferException([&] { elem->readTransfer(); });
93  }
94  }
95 
96  auto exceptionHandlingResult = runPostReads(copyDecorators);
97  exceptionHandlingResult += runPostReads(highLevelElements);
98 
99  exceptionHandlingResult.reThrow();
100  }

104  void TransferGroup::write(VersionNumber versionNumber) {
105  if(isReadOnly()) {
106  throw ChimeraTK::logic_error("TransferGroup::write() called, but the TransferGroup is read-only.");
107  }
108  for(auto& it : lowLevelElementsAndExceptionFlags) {
109  it.second = false;
110  }
111 
112  for(auto& elem : highLevelElements) {
113  elem->preWriteAndHandleExceptions(TransferType::write, versionNumber);
114  if(elem->_activeException) {
115  for(auto& lowLevelElem : elem->getHardwareAccessingElements()) {
116  lowLevelElementsAndExceptionFlags[lowLevelElem] = true;
117  }
118  }
119  }
120 
121  for(auto& it : lowLevelElementsAndExceptionFlags) {
122  auto& elem = it.first;
123  auto& hasSeenException = it.second;
124  if(!hasSeenException) {
125  elem->handleTransferException([&] { elem->writeTransfer(versionNumber); });
126  }
127  }
128 
129  ExceptionHandlingResult exceptionHandlingResult;
130  for(auto& elem : highLevelElements) {
131  exceptionHandlingResult = handlePostExceptions([&] { elem->postWrite(TransferType::write, versionNumber); });
132  }
133 
134  exceptionHandlingResult.reThrow();
135  }

143  void TransferGroup::addAccessor(TransferElementAbstractor& accessor) {
144  // check if accessor is already in a transfer group
145  if(accessor.getHighLevelImplElement()->_isInTransferGroup) {
146  throw ChimeraTK::logic_error("The given accessor is already in a TransferGroup and cannot be added "
147  "to another.");
148  }
149 
150  // Only accessors without wait_for_new_data can be used in a transfer group.
151  if(accessor.getAccessModeFlags().has(AccessMode::wait_for_new_data)) {
152  throw ChimeraTK::logic_error(
153  "A TransferGroup can only be used with transfer elements that don't have aAccessMode::wait_for_new_data.");
154  }
155 
156  // set flag on the accessors that it is now in a transfer group
157  accessor.getHighLevelImplElement()->_isInTransferGroup = true;
158 
159  auto highLevelElementsWithNewAccessor = highLevelElements;
160  highLevelElementsWithNewAccessor.insert(accessor.getHighLevelImplElement());
161 
162  // try replacing all internal elements in all high-level elements
163  for(auto& hlElem1 : highLevelElementsWithNewAccessor) {
164  auto list = hlElem1->getInternalElements();
165  list.push_front(hlElem1);
166 
167  for(auto& replacement : list) {
168  // try on the abstractor first, to make sure we replace at the highest
169  // level if possible
170  accessor.replaceTransferElement(replacement);
171  // try on all high-level elements already stored in the list
172  for(auto& hlElem : highLevelElementsWithNewAccessor) {
173  hlElem->replaceTransferElement(replacement); // note: this does nothing, if the replacement cannot
174  // be used by the hlElem!
175  }
176  }
177  }
178 
179  // store the accessor in the list of high-level elements
180  // this must be done only now, since it may have been replaced during the
181  // replacement process
182  highLevelElements.insert(accessor.getHighLevelImplElement());
183 
184  // update the list of hardware-accessing elements, since we might just have
185  // made some of them redundant since we are using a set to store the elements,
186  // duplicates are intrinsically avoided.
187  lowLevelElementsAndExceptionFlags.clear();
188  for(auto& hlElem : highLevelElements) {
189  for(auto& hwElem : hlElem->getHardwareAccessingElements())
190  lowLevelElementsAndExceptionFlags.insert({hwElem, false});
191  }
192 
193  // update the list of CopyRegisterDecorators
194  copyDecorators.clear();
195  for(auto& hlElem : highLevelElements) {
196  if(boost::dynamic_pointer_cast<ChimeraTK::CopyRegisterDecoratorTrait>(hlElem) != nullptr) {
197  copyDecorators.insert(hlElem);
198  }
199  for(auto& hwElem : hlElem->getInternalElements()) {
200  if(boost::dynamic_pointer_cast<ChimeraTK::CopyRegisterDecoratorTrait>(hwElem) != nullptr) {
201  copyDecorators.insert(hwElem);
202  }
203  }
204  }
205 
206  // Update read-only flag
207  if(accessor.isReadOnly()) readOnly = true;
208  }
