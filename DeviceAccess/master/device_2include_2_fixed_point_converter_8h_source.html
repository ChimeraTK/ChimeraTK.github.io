203  template<typename UserType>
204  UserType FixedPointConverter::toCooked(uint32_t rawValue) const {
205  UserType cooked;
206 
207  // extract sign and leave positive raw number
208  bool isNegative = (rawValue & _signBitMask);
209  if(isNegative) rawValue = ~rawValue; // bit inversion flips the sign, but will be off by 1!
210 
211  // leading, out of range bits are ignored. Just crop them.
212  rawValue &= _usedBitsMask;
213 
214  // Handle integer and floating-point types differently.
215  // This if-statement is likely to be optimised out at compile-time.
216  if(std::numeric_limits<UserType>::is_integer && _fractionalBits == 0) {
217  // cast into target type while checking against over/underflow
218  // as we flipped the sign above, we won't get a negative_overflow, thus
219  // catch the exception and re-throw the right one
220  try {
221  cooked = boost::numeric_cast<UserType>(rawValue);
222  }
223  catch(boost::numeric::positive_overflow&) {
224  if(!isNegative) {
225  throw FPC_OverflowException<boost::numeric::positive_overflow>(_variableName);
226  }
227  else {
228  throw FPC_OverflowException<boost::numeric::negative_overflow>(_variableName);
229  }
230  }
231 
232  // handle the sign
233  if(isNegative) {
234  // This if-statement also is likely to be optimised out at compile-time.
235  if(std::numeric_limits<UserType>::is_signed) {
236  cooked = -cooked - 1;
237  }
238  else {
239  cooked = 0;
240  throw FPC_OverflowException<boost::numeric::negative_overflow>(_variableName);
241  }
242  }
243  }
244  else {
245  // convert into double and scale to handle fractional bits
246  double d_cooked = _fractionalBitsCoefficient * static_cast<double>(rawValue);
247 
248  // Handle the sign. Since we inverted the rawValue to get a positive value,
249  // it was off by 1 before scaling
250  if(isNegative) d_cooked = -d_cooked - _fractionalBitsCoefficient;
251 
252  // Convert into target type. This conversion takes care of proper rounding
253  // when needed (and only then), and will throw
254  // boost::numeric::positive_overflow resp. boost::numeric::negative_overflow
255  // if the target range is exceeded.
256  typedef boost::numeric::converter<UserType, double, boost::numeric::conversion_traits<UserType, double>,
257  boost::numeric::def_overflow_handler, Round<double>>
258  converter;
259  try {
260  cooked = converter::convert(d_cooked);
261  }
262  catch(boost::numeric::positive_overflow&) {
263  throw FPC_OverflowException<boost::numeric::positive_overflow>(_variableName);
264  }
265  catch(boost::numeric::negative_overflow&) {
266  throw FPC_OverflowException<boost::numeric::negative_overflow>(_variableName);
267  }
268  }
269 
270  return cooked;
271  }

274  template<typename UserType>
275  uint32_t FixedPointConverter::toRaw(UserType cookedValue) const {
276  // Do a range check first. The later overflow check in the conversion is not
277  // sufficient, since we can have non-standard word sizes like 12 bits.
278  if(cookedValue < boost::fusion::at_key<UserType>(_minCookedValues)) {
279  return _minRawValue;
280  }
281  if(cookedValue > boost::fusion::at_key<UserType>(_maxCookedValues)) {
282  return _maxRawValue;
283  }
284 
285  // handle integer and floating-point types differently
286  if(std::numeric_limits<UserType>::is_integer && _fractionalBits == 0) {
287  // extract the sign and leave the positive number
288  bool isNegative = isNegativeUserType(cookedValue);
289  if(isNegative && !_isSigned) return _minRawValue;
290  if(isNegative)
291  cookedValue = -(cookedValue + 1); // bit inversion, ~ operator cannot be
292  // used as UserType might be double
293 
294  // cast into raw type
295  uint32_t rawValue = static_cast<uint32_t>(cookedValue);
296 
297  // handle sign
298  if(_isSigned && isNegative) {
299  rawValue = ~rawValue;
300  }
301 
302  // return with bit mask applied
303  return rawValue & _usedBitsMask;
304  }
305  else {
306  // convert into double and scale by fractional bit coefficient
307  double d_cooked = _inverseFractionalBitsCoefficient * static_cast<double>(cookedValue);
308 
309  // Convert into either signed or uinsigned int32_t, depending on _isSigned,
310  // so the conversion handls the sign correctly. Store always in a uint32_t,
311  // since this is our raw type. The conversion will properly round, when
312  // needed. Negative overflow exceptions need to be cought for some corner
313  // cases (e.g. number of fractional bits >= number of bits in total).
314  // Positive overflow cannot happen due to the rangecheck above (the negative
315  // branch has one more possible value).
316  uint32_t raw;
317  try {
318  if(_isSigned) {
319  typedef boost::numeric::converter<int32_t, double, boost::numeric::conversion_traits<int32_t, double>,
320  boost::numeric::def_overflow_handler, Round<double>>
321  converter_signed;
322  raw = converter_signed::convert(d_cooked);
323  }
324  else {
325  typedef boost::numeric::converter<uint32_t, double, boost::numeric::conversion_traits<uint32_t, double>,
326  boost::numeric::def_overflow_handler, Round<double>>
327  converter_unsigned;
328  raw = converter_unsigned::convert(d_cooked);
329  }
330  }
331  catch(boost::numeric::negative_overflow& e) {
332  raw = _minRawValue;
333  }
334 
335  // apply bit mask
336  return raw & _usedBitsMask;
337  }
338  }
