89  void StepperMotorWithReference::StateMachine::calibrationThreadFunction(){
90  _motor._calibrationFailed.exchange(false);
91  _stopAction.exchange(false);
92  _moveInterrupted.exchange(false);
93 
94  // Local variables for calibrated position
95  // TODO Move to else scope to remove cppcheck warning
96  int calibPositiveEndSwitchInSteps = 0;
97  int calibNegativeEndSwitchInSteps = 0;
98 
99  // At least one end switch disabled -> calibration not possible
100  if(!(_motor._positiveEndSwitchEnabled.load() && _motor._negativeEndSwitchEnabled.load())){
101 
102  _motor._motorControler->setCalibrationTime(0);
103 
104  _motor._calibrationFailed.exchange(true);
105  _motor._calibrationMode.exchange(StepperMotorCalibrationMode::NONE);
106  }
107  else{
108  findEndSwitch(Sign::POSITIVE);
109  calibPositiveEndSwitchInSteps = _motor.getCurrentPositionInSteps();
110  findEndSwitch(Sign::NEGATIVE);
111  calibNegativeEndSwitchInSteps = _motor.getCurrentPositionInSteps();
112 
113  if (_moveInterrupted.load() || _stopAction.load()){
114 
115  _motor._motorControler->setCalibrationTime(0);
116 
117  _motor._calibrationFailed.exchange(true);
118  _motor._errorMode.exchange(StepperMotorError::CALIBRATION_ERROR);
119  _motor._calibrationMode.exchange(StepperMotorCalibrationMode::NONE);
120  }else{
121  // Define positive axis with negative end switch as zero
122  calibPositiveEndSwitchInSteps = calibPositiveEndSwitchInSteps - calibNegativeEndSwitchInSteps;
123  calibNegativeEndSwitchInSteps = 0;
124 
125  _motor._motorControler->setCalibrationTime(time(nullptr));
126  _motor._motorControler->setPositiveReferenceSwitchCalibration(calibPositiveEndSwitchInSteps);
127  _motor._motorControler->setNegativeReferenceSwitchCalibration(0);
128  _motor._calibPositiveEndSwitchInSteps.exchange(calibPositiveEndSwitchInSteps);
129  _motor._calibNegativeEndSwitchInSteps.exchange(0);
130 
131  _motor._calibrationMode.exchange(StepperMotorCalibrationMode::FULL);
132  _motor.resetMotorControllerPositions(0);
133  }
134  }
135 
136  _asyncActionActive.exchange(false);
137  return;
138  }

140  void StepperMotorWithReference::StateMachine::findEndSwitch(Sign sign){
141  while (!_motor.isEndSwitchActive(sign)){
142  if (_stopAction.load() || _moveInterrupted.load()){
143  return;
144  }else if (_motor.getTargetPositionInSteps() != _motor.getCurrentPositionInSteps() &&
145  !_motor.isPositiveReferenceActive() && !_motor.isNegativeReferenceActive()){
146  _moveInterrupted.exchange(true);

148  }
149  moveToEndSwitch(sign);
150  }

209  double StepperMotorWithReference::StateMachine::getToleranceEndSwitch(Sign sign){
210  double meanMeasurement = 0;
211  double stdMeasurement = 0;
212  const int N_TOLERANCE_CALC_SAMPLES = 10;
213  double measurements[N_TOLERANCE_CALC_SAMPLES];
214 
215  int endSwitchPosition = getPositionEndSwitch(sign);
216 
217  // Get 10 samples for tolerance calculation
218  for (unsigned i=0; i<N_TOLERANCE_CALC_SAMPLES; i++){
219  if (_stopAction.load() || _moveInterrupted.load()){
220  break;
221  }
222 
223  //Move close to end switch
224  {
225  boost::lock_guard<boost::mutex> lck(_motor._mutex);
226  _motor._motorControler->setTargetPosition(endSwitchPosition - static_cast<int>(sign)*1000);
227  }
228  while(_motor._motorControler->isMotorMoving()){
229  std::this_thread::sleep_for(std::chrono::milliseconds(wakeupPeriodInMilliseconds));
230  }
231 
232  //Check if in expected position
233  if (!_motor.verifyMoveAction()){
234  _moveInterrupted.exchange(true);
235  break;
236  }
237 
238  // Try to move beyond end switch
239  {
240  boost::lock_guard<boost::mutex> lck(_motor._mutex);
241  _motor._motorControler->setTargetPosition(endSwitchPosition + static_cast<int>(sign)*1000);
242  }
243  while(_motor._motorControler->isMotorMoving()){
244  std::this_thread::sleep_for(std::chrono::milliseconds(wakeupPeriodInMilliseconds));
245  }
246  if (!_motor.isEndSwitchActive(sign)){
247  _moveInterrupted.exchange(true);
248  break;
249  }
250 
251  // Mean calculation
252  meanMeasurement += static_cast<double>(_motor.getCurrentPositionInSteps())
253  / N_TOLERANCE_CALC_SAMPLES;
254  measurements[i] = _motor.getCurrentPositionInSteps();
255  } /* for (i in [0,9]) */
256 
257  // Compute variance
258  if (!(_stopAction.load() || _moveInterrupted.load()) ){
259  for (unsigned int i=0; i<N_TOLERANCE_CALC_SAMPLES; i++){
260  stdMeasurement += ((measurements[i] - meanMeasurement) / (N_TOLERANCE_CALC_SAMPLES-1)) * (measurements[i] - meanMeasurement);
261  }
262  }
263 
264  return sqrt(stdMeasurement);
265  }
266 }
