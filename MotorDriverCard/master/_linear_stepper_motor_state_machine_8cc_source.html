86  void LinearStepperMotor::StateMachine::calibrationThreadFunction(){
87  _motor._calibrationFailed.exchange(false);
88  _stopAction.exchange(false);
89  _moveInterrupted.exchange(false);
90 
91  // Local variables for calibrated position
92  // TODO Move to else scope to remove cppcheck warning
93  int calibPositiveEndSwitchInSteps = 0;
94  int calibNegativeEndSwitchInSteps = 0;
95 
96  // At least one end switch disabled -> calibration not possible
97  if(!(_motor._positiveEndSwitchEnabled.load() && _motor._negativeEndSwitchEnabled.load())){
98 
99  _motor._motorControler->setCalibrationTime(0);
100 
101  _motor._calibrationFailed.exchange(true);
102  _motor._calibrationMode.exchange(CalibrationMode::NONE);
103  }
104  else{
105  findEndSwitch(Sign::POSITIVE);
106  calibPositiveEndSwitchInSteps = _motor.getCurrentPositionInSteps();
107  findEndSwitch(Sign::NEGATIVE);
108  calibNegativeEndSwitchInSteps = _motor.getCurrentPositionInSteps();
109 
110  if (_moveInterrupted.load() || _stopAction.load()){
111 
112  _motor._motorControler->setCalibrationTime(0);
113 
114  _motor._calibrationFailed.exchange(true);
115  _motor._errorMode.exchange(Error::CALIBRATION_ERROR);
116  _motor._calibrationMode.exchange(CalibrationMode::NONE);
117  }else{
118  // Define positive axis with negative end switch as zero
119  calibPositiveEndSwitchInSteps = calibPositiveEndSwitchInSteps - calibNegativeEndSwitchInSteps;
120  calibNegativeEndSwitchInSteps = 0;
121 
122  _motor._motorControler->setCalibrationTime(time(nullptr));
123  _motor._motorControler->setPositiveReferenceSwitchCalibration(calibPositiveEndSwitchInSteps);
124  _motor._motorControler->setNegativeReferenceSwitchCalibration(0);
125  _motor._calibPositiveEndSwitchInSteps.exchange(calibPositiveEndSwitchInSteps);
126  _motor._calibNegativeEndSwitchInSteps.exchange(0);
127 
128  _motor._calibrationMode.exchange(CalibrationMode::FULL);
129  _motor.resetMotorControllerPositions(0);
130  }
131  }
132 
133  _asyncActionActive.exchange(false);
134  return;
135  }

137  void LinearStepperMotor::StateMachine::findEndSwitch(Sign sign){

139  if (_stopAction.load() || _moveInterrupted.load()){

143  _moveInterrupted.exchange(true);

145  }

207  double LinearStepperMotor::StateMachine::getToleranceEndSwitch(Sign sign){
208  double meanMeasurement = 0;
209  double stdMeasurement = 0;
210  const int N_TOLERANCE_CALC_SAMPLES = 10;
211  double measurements[N_TOLERANCE_CALC_SAMPLES];
212 
213  int endSwitchPosition = getPositionEndSwitch(sign);
214 
215  // Get 10 samples for tolerance calculation
216  for (unsigned i=0; i<N_TOLERANCE_CALC_SAMPLES; i++){
217  if (_stopAction.load() || _moveInterrupted.load()){
218  break;
219  }
220 
221  //Move close to end switch
222  {
223  boost::lock_guard<boost::mutex> lck(_motor._mutex);
224  _motor._motorControler->setTargetPosition(endSwitchPosition - static_cast<int>(sign)*1000);
225  }
226  while(_motor._motorControler->isMotorMoving()){
227  std::this_thread::sleep_for(std::chrono::milliseconds(wakeupPeriodInMilliseconds));
228  }
229 
230  //Check if in expected position
231  if (!_motor.verifyMoveAction()){
232  _moveInterrupted.exchange(true);
233  break;
234  }
235 
236  // Try to move beyond end switch
237  {
238  boost::lock_guard<boost::mutex> lck(_motor._mutex);
239  _motor._motorControler->setTargetPosition(endSwitchPosition + static_cast<int>(sign)*1000);
240  }
241  while(_motor._motorControler->isMotorMoving()){
242  std::this_thread::sleep_for(std::chrono::milliseconds(wakeupPeriodInMilliseconds));
243  }
244  if (!_motor.isEndSwitchActive(sign)){
245  _moveInterrupted.exchange(true);
246  break;
247  }
248 
249  // Mean calculation
250  meanMeasurement += static_cast<double>(_motor.getCurrentPositionInSteps())
251  / N_TOLERANCE_CALC_SAMPLES;
252  measurements[i] = _motor.getCurrentPositionInSteps();
253  } /* for (i in [0,9]) */
254 
255  // Compute variance
256  if (!(_stopAction.load() || _moveInterrupted.load()) ){
257  for (unsigned int i=0; i<N_TOLERANCE_CALC_SAMPLES; i++){
258  stdMeasurement += ((measurements[i] - meanMeasurement) / (N_TOLERANCE_CALC_SAMPLES-1)) * (measurements[i] - meanMeasurement);
259  }
260  }
261 
262  return sqrt(stdMeasurement);
263  }
264 }
