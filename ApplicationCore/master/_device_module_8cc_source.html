114  VirtualModule& DeviceModule::virtualiseFromCatalog() const {
115  if(virtualisedModuleFromCatalog_isValid) return virtualisedModuleFromCatalog;
116 
117  virtualisedModuleFromCatalog = VirtualModule(deviceAliasOrURI, "Device module", ModuleType::Device);
118 
119  if (!deviceIsInitialized){
120  device = Device(deviceAliasOrURI);
121  deviceIsInitialized = true;
122  }
123  // obtain register catalogue
124  auto catalog = device.getRegisterCatalogue();
125  // iterate catalogue, create VariableNetworkNode for all registers starting
126  // with the registerNamePrefix
127  size_t prefixLength = registerNamePrefix.length();
128  for(auto& reg : catalog) {
129  if(std::string(reg.getRegisterName()).substr(0, prefixLength) != std::string(registerNamePrefix)) continue;
130 
131  // ignore 2D registers
132  if(reg.getNumberOfDimensions() > 1) continue;
133 
134  // guess direction and determine update mode
135  VariableDirection direction;
136  UpdateMode updateMode;
137  if(reg.isWriteable()) {
138  direction = {VariableDirection::consuming, false};
139  updateMode = UpdateMode::push;
140  }
141  else {
142  direction = {VariableDirection::feeding, false};
143  if(reg.getSupportedAccessModes().has(AccessMode::wait_for_new_data)) {
144  updateMode = UpdateMode::push;
145  }
146  else {
147  updateMode = UpdateMode::poll;
148  }
149  }
150 
151  // guess type
152  const std::type_info* valTyp{&typeid(AnyType)};
153  auto& dd = reg.getDataDescriptor(); // numeric, string, boolean, nodata, undefined
154  if(dd.fundamentalType() == RegisterInfo::FundamentalType::numeric) {
155  if(dd.isIntegral()) {
156  if(dd.isSigned()) {
157  if(dd.nDigits() > 11) {
158  valTyp = &typeid(int64_t);
159  }
160  else if(dd.nDigits() > 6) {
161  valTyp = &typeid(int32_t);
162  }
163  else if(dd.nDigits() > 4) {
164  valTyp = &typeid(int16_t);
165  }
166  else {
167  valTyp = &typeid(int8_t);
168  }
169  }
170  else {
171  if(dd.nDigits() > 10) {
172  valTyp = &typeid(uint64_t);
173  }
174  else if(dd.nDigits() > 5) {
175  valTyp = &typeid(uint32_t);
176  }
177  else if(dd.nDigits() > 3) {
178  valTyp = &typeid(uint16_t);
179  }
180  else {
181  valTyp = &typeid(uint8_t);
182  }
183  }
184  }
185  else { // fractional
186  valTyp = &typeid(double);
187  }
188  }
189  else if(dd.fundamentalType() == RegisterInfo::FundamentalType::boolean) {
190  valTyp = &typeid(int32_t);
191  }
192  else if(dd.fundamentalType() == RegisterInfo::FundamentalType::string) {
193  valTyp = &typeid(std::string);
194  }
195  else if(dd.fundamentalType() == RegisterInfo::FundamentalType::nodata) {
196  valTyp = &typeid(int32_t);
197  }
198 
199  auto name = std::string(reg.getRegisterName()).substr(prefixLength);
200  auto lastSlash = name.find_last_of("/");
201  auto dirname = name.substr(0, lastSlash);
202  auto basename = name.substr(lastSlash + 1);
203  VariableNetworkNode node(
204  basename, deviceAliasOrURI, reg.getRegisterName(), updateMode, direction, *valTyp, reg.getNumberOfElements());
205  virtualisedModuleFromCatalog.createAndGetSubmoduleRecursive(dirname).addAccessor(node);
206  }
207 
208  virtualisedModuleFromCatalog_isValid = true;
209  return virtualisedModuleFromCatalog;
210  }

214  void DeviceModule::reportException(std::string errMsg) {
215  // In testable mode, we need to increase the testableMode_counter before releasing the testableModeLock, but we have
216  // to release the testableModeLock lock before obtaining the errorMutex, which in turn has to be obtained before
217  // writing to the errorQueue. Otherwise deadlocks might occur (with race conditions).
218  if(owner->isTestableModeEnabled()) {
219  assert(owner->testableModeTestLock());
220  ++owner->testableMode_counter;
221  }
222  owner->testableModeUnlock("reportException");
223  std::unique_lock<std::mutex> lk(errorMutex);
224  retry:
225  bool success = errorQueue.push(errMsg);
226  if(!success) {
227  // retry error reporting until the message has been sent. It has to be successful since testableMode_counter
228  // has been increased already
229  goto retry;
230  }
231  errorCondVar.wait(lk);
232  // release errorMutex before obtaining testableModeLock to prevent deadlocks
233  lk.unlock();
234  owner->testableModeLock("reportException");
235  }

239  void DeviceModule::handleException() {
240  Application::registerThread("DM_" + getName());
241  std::string error;
242  owner->testableModeLock("Startup");
243  try {
244  while(!device.isOpened()) {
245  try {
246  boost::this_thread::interruption_point();
247  usleep(500000);
248  device.open();
249  if(deviceError.status != 0) {
250  deviceError.status = 0;
251  deviceError.message = "";
252  deviceError.setCurrentVersionNumber({});
253  deviceError.writeAll();
254  }
255  for(auto& te : writeAfterOpen) {
256  te->write();
257  }
258  }
259  catch(ChimeraTK::runtime_error& e) {
260  if(deviceError.status != 1) {
261  deviceError.status = 1;
262  deviceError.message = error;
263  deviceError.setCurrentVersionNumber({});
264  deviceError.writeAll();
265  }
266  }
267  }
268  while(true) {
269  owner->testableModeUnlock("Wait for exception");
270  errorQueue.pop_wait(error);
271  boost::this_thread::interruption_point();
272  owner->testableModeLock("Process exception");
273  if(owner->isTestableModeEnabled()) --owner->testableMode_counter;
274  std::lock_guard<std::mutex> lk(errorMutex);
275  // report exception to the control system
276  deviceError.status = 1;
277  deviceError.message = error;
278  deviceError.setCurrentVersionNumber({});
279  deviceError.writeAll();
280 
281  // wait some time until retrying
282  owner->testableModeUnlock("Wait for recovery");
283  usleep(500000);
284  owner->testableModeLock("Try recovery");
285 
286  // empty exception reporting queue
287  while(errorQueue.pop()) {
288  if(owner->isTestableModeEnabled()) --owner->testableMode_counter;
289  }
290 
291  // reset exception state and try again
292  deviceError.status = 0;
293  deviceError.message = "";
294  deviceError.setCurrentVersionNumber({});
295  deviceError.writeAll();
296  errorCondVar.notify_all();
297  }
298  }
299  catch(...) {
300  // before we leave this thread, we might need to notify other waiting
301  // threads. boost::this_thread::interruption_point() throws an exception
302  // when the thread should be interrupted, so we will end up here
303  errorCondVar.notify_all();
304  throw;
305  }
306  }
