543  template<typename LAMBDATYPE>
544  void callForType(const std::type_info& type, LAMBDATYPE lambda) {
545  bool done = false;
546 
547  // iterate through a userTypeMap() and test all types
548  for_each(userTypeMap(), [&type, &lambda, &done](auto pair) {
549  if(type != typeid(pair.second)) return;
550  // if the right type has been found, call the original lambda
551  lambda(pair.second);
552  done = true;
553  });
554 
555  // Check if done flag has been set. If not, an unknown type has been passed.
556  if(!done) {
557  class myBadCast : public std::bad_cast {
558  public:
559  myBadCast(const std::string& desc) : _desc(desc) {}
560  const char* what() const noexcept override { return _desc.c_str(); }
561 
562  private:

564  };
