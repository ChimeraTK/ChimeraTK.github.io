239  DummyMultiplexedRegisterAccessor(DummyBackend *dev, std::string module, std::string name)
240  : _dev(dev), pitch(0)
241  {
242  _dev->_registerMapping->getRegisterInfo(MULTIPLEXED_SEQUENCE_PREFIX+name, registerInfo, module);
243 
244  int i = 0;
245  while(true) {
246  // obtain register information for sequence
247  RegisterInfoMap::RegisterInfo elem;
248  std::stringstream sequenceNameStream;
249  sequenceNameStream << SEQUENCE_PREFIX << name << "_" << i++;
250  try{
251  _dev->_registerMapping->getRegisterInfo( sequenceNameStream.str(), elem, module );
252  }
253  catch(ChimeraTK::logic_error &) {
254  break;
255  }
256  // create fixed point converter for sequence
257  fpc.push_back( FixedPointConverter(module+"/"+name,elem.width, elem.nFractionalBits, elem.signedFlag) );
258  // store offsets and number of bytes per word
259  offsets.push_back(elem.address);
260  nbytes.push_back(elem.nBytes);
261  // determine pitch
262  pitch += elem.nBytes;
263  }
264 
265  if(fpc.empty()) {
266  throw ChimeraTK::logic_error("No sequenes found for name \""+name+"\".");
267  }
268 
269  // compute number of elements per sequence
270  nElements = registerInfo.nBytes/pitch;
271  }
272 
274  unsigned int getNumberOfElements()
275  {
276  return nElements;
277  }

283  }
