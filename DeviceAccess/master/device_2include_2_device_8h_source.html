680 template <typename UserType>
681 void Device::write(const RegisterPath &registerPathName,
682  const std::vector<UserType> &vector,
683  size_t wordOffsetInRegister, const AccessModeFlags &flags) {
684  auto acc = getOneDRegisterAccessor<UserType>(registerPathName, vector.size(),
685  wordOffsetInRegister, flags);
686  // the vector stays constant, but we have to work around so we can swap it and
687  // later swap it back...
688  std::vector<UserType> &mutable_vector =
689  const_cast<std::vector<UserType> &>(vector);
690  acc.swap(mutable_vector);
691  acc.write();
692  acc.swap(mutable_vector);
693 }
