77  mtca4u::TransferElementID readAny(
78  std::list<std::reference_wrapper<TransferElementType>> elementsToRead) {
79 
80  // build list of TransferFutures for all elements. Since readAsync() is a virtual call and we need to visit all
81  // elements at least twice (once in wait_for_any and a second time for sorting by version number), this is assumed
82  // to be less expensive than calling readAsync() on the fly in the TransferFutureIterator instead.
83  std::list<TransferFuture> futureList;
84  for(auto &elem : elementsToRead) {
85  futureList.push_back(elem.get().readAsync());
86  }
87  // wait until any future is ready
88  auto iter = boost::wait_for_any(detail::TransferFutureIterator(futureList.begin()),
89  detail::TransferFutureIterator(futureList.end()));
90 
91  // Find the variable which has the oldest version number (to guarantee the order of updates).
92  // Start with assuming that the future returned by boost::wait_for_any() has the oldes version number.
93  TransferFuture theUpdate = iter.getTransferFuture();
94  for(auto future : futureList) {
95  // skip if this is the future returned by boost::wait_for_any()
96  if(future == theUpdate) continue;
97  // also skip if the future is not yet ready
98  if(!future.hasNewData()) continue;
99  // compare version number with the version number of the stored future
100  if(future.getBoostFuture().get()->_versionNumber < theUpdate.getBoostFuture().get()->_versionNumber) {
101  theUpdate = future;
102  }
103  }
104 
105  // complete the transfer (i.e. run postRead())
106  theUpdate.wait();
107 
108  // return the transfer element as a shared pointer
109  return theUpdate.getTransferElementID();
110 
111  }
