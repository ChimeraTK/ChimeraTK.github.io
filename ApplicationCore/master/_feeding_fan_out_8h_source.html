72  void addSlave(boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> slave, VariableNetworkNode&) override {
73  // check if array shape is compatible, unless the receiver is a trigger
74  // node, so no data is expected
75  if(slave->getNumberOfSamples() != 0 &&
76  (slave->getNumberOfChannels() != 1 || slave->getNumberOfSamples() != this->getNumberOfSamples())) {
77  std::string what = "FeedingFanOut::addSlave(): Trying to add a slave '" + slave->getName();
78  what += "' with incompatible array shape! Name of fan out: '" + this->getName() + "'";
79  throw ChimeraTK::logic_error(what.c_str());
80  }
81 
82  // make sure slave is writeable
83  if(!slave->isWriteable()) {
84  throw ChimeraTK::logic_error("FeedingFanOut::addSlave() has been called "
85  "with a receiving implementation!");
86  }
87 
88  // handle return channels
89  if(_withReturn) {
90  if(slave->isReadable()) {
91  if(_hasReturnSlave) {
92  throw ChimeraTK::logic_error("FeedingFanOut: Cannot add multiple slaves with return channel!");
93  }
94  _hasReturnSlave = true;
95  _returnSlave = slave;
96 
97  // Determine the AccessModeFlags from the return slave
98  // As this becomes the implemention of the feeding output, the flags are determined by that slave accessor
99  // In other cases, the information is not relevant because the feeding node is on output which is never read
100  ChimeraTK::NDRegisterAccessor<UserType>::_accessModeFlags = _returnSlave->getAccessModeFlags();
101 
102  TransferElement::_readQueue = _returnSlave->getReadQueue();
103  }
104  }
105 
106  // add the slave
107  FanOut<UserType>::slaves.push_back(slave);
108  }

127  void doPostRead(TransferType type, bool hasNewData) override {
128  assert(_withReturn);
129  assert(_hasReturnSlave);
130 
131  auto _ = cppext::finally([&] {
132  if(!hasNewData) return;
133  _returnSlave->accessChannel(0).swap(ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0]);
134  // distribute return-channel update to the other slaves
135  for(auto& slave : FanOut<UserType>::slaves) { // send out copies to slaves
136  if(slave == _returnSlave) continue;
137  if(slave->getNumberOfSamples() != 0) { // do not send copy if no data is expected (e.g. trigger)
138  slave->accessChannel(0) = ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0];
139  }
140  slave->writeDestructively(_returnSlave->getVersionNumber());
141  }
142  });
143 
144  _returnSlave->postRead(type, hasNewData);
145  }

147  void doPreWrite(TransferType type, VersionNumber versionNumber) override {
148  for(auto& slave : FanOut<UserType>::slaves) { // send out copies to slaves
149  if(slave->getNumberOfSamples() != 0) { // do not send copy if no data is expected (e.g. trigger)
150  if(slave == FanOut<UserType>::slaves.front()) { // in case of first slave, swap instead of copy
151  slave->accessChannel(0).swap(ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0]);
152  }
153  else { // not the first slave: copy the data from the first slave
154  slave->accessChannel(0) = FanOut<UserType>::slaves.front()->accessChannel(0);
155  }
156  }
157  slave->setDataValidity(this->dataValidity());
158  }
159 
160  // pre write may only be called on the target accessors after we have filled
161  // them all, otherwise the first accessor might take us the data away...
162  RuntimeErrorCollector ec;
163  for(auto& slave : FanOut<UserType>::slaves) {
164  ec.wrap([&] { slave->preWrite(type, versionNumber); });
165  }
166 
167  ec.unwrap();
168  }

170  bool doWriteTransfer(ChimeraTK::VersionNumber versionNumber) override {
171  bool dataLost = false;
172  bool isFirst = true;
173  for(auto& slave : FanOut<UserType>::slaves) {
174  bool ret;
175  if(isFirst) {
176  isFirst = false;
177  ret = slave->writeTransfer(versionNumber);
178  }
179  else {
180  ret = slave->writeTransferDestructively(versionNumber);
181  }
182  if(ret) dataLost = true;
183  }
184  return dataLost;
185  }
