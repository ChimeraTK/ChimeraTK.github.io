88  bool write(ChimeraTK::VersionNumber versionNumber={}) override {
89  bool dataLost = false;
90  boost::shared_ptr<mtca4u::NDRegisterAccessor<UserType>> firstSlave; // will have the data for the other slaves after swapping
91  for(auto &slave : FanOut<UserType>::slaves) { // send out copies to slaves
92  if(slave->getNumberOfSamples() != 0) { // do not send copy if no data is expected (e.g. trigger)
93  if(!firstSlave) { // in case of first slave, swap instead of copy
94  firstSlave = slave;
95  firstSlave->accessChannel(0).swap(mtca4u::NDRegisterAccessor<UserType>::buffer_2D[0]);
96  }
97  else { // not the first slave: copy the data from the first slave
98  slave->accessChannel(0) = firstSlave->accessChannel(0);
99  }
100  }
101  bool ret = slave->write(versionNumber);
102  if(ret) dataLost = true;
103  }
104  // swap back the data from the first slave so we still have it available
105  if(firstSlave) {
106  firstSlave->accessChannel(0).swap(mtca4u::NDRegisterAccessor<UserType>::buffer_2D[0]);
107  }
108  return dataLost;
109  }
