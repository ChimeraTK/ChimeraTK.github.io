140 uint32_t RebotBackend::getServerProtocolVersion() {
141  // send a negotiation to the server:
142  // sendClientProtocolVersion
143  _lastSendTime=testable_rebot_sleep::now();
144  std::vector<uint32_t> clientHelloMessage = frameClientHello();
145  _tcpCommunicator->sendData(clientHelloMessage);
146 
147  // Kludge is needed to work around server bug.
148  // We have a bug with the old version were only one word is returned for multiple
149  // unrecognized command. Fetching one word for the 3 words send is a workaround.
150  auto serverHello = _tcpCommunicator->receiveData(1);
151 
152  if (serverHello.at(0) == rebot::UNKNOWN_INSTRUCTION) {
153  return 0; // initial protocol version 0.0
154  }
155 
156  auto remainingBytesOfAValidServerHello =
157  _tcpCommunicator->receiveData(rebot::LENGTH_OF_HELLO_TOKEN_MESSAGE - 1);
158 
159  serverHello.insert(serverHello.end(),
160  remainingBytesOfAValidServerHello.begin(),
161  remainingBytesOfAValidServerHello.end());
162  return parseRxServerHello(serverHello);
163 }

179  void RebotBackend::heartbeatLoop(boost::shared_ptr<ThreadInformerMutex> threadInformerMutex){
180  while (true){
181  try{
182  // only send a heartbeat if the connection was inactive for half of the timeout period
183  if ( (testable_rebot_sleep::now() - _lastSendTime) >
184  boost::chrono::milliseconds(_connectionTimeout/2) ){
185  // scope for the lock guard is in the if statement
186  std::lock_guard<std::mutex> lock(_threadInformerMutex->mutex);
187  //To handle the race condition that the thread woke up while the desructor was holding the lock
188  //and closes the socket: Check the flag and quit if set
189  if (threadInformerMutex->quitThread){
190  break;
191  }
192  // always update the last send time. Otherwise the sleep will be ineffective for a closed
193  // connection and go to 100 & CPU load
194  _lastSendTime=testable_rebot_sleep::now();
195  if (_protocolImplementor){
196  _protocolImplementor->sendHeartbeat();
197  }
198  }
199  // sleep without holding the lock. Sleep for half of the connection timeout (plus 1 ms)
200  testable_rebot_sleep::sleep_until( _lastSendTime + boost::chrono::milliseconds(_connectionTimeout/2 +1 ) );
201  }catch(RebotBackendException &e){
202  std::cerr << "RebotBackend: Sending heartbeat failed. Caught exception: " << e.what() <<std::endl;
203  std::cerr << "Closing connection." << std::endl;
204  close();
205  }
206  }//while true
207  }
