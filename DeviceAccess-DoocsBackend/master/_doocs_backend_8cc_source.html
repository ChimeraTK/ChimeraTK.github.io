166  boost::shared_ptr<DeviceBackend> DoocsBackend::createInstance(
167  std::string address, std::map<std::string, std::string> parameters) {
168  // if address is empty, build it from parameters (for compatibility with SDM)
169  if(address.empty()) {
170  RegisterPath serverAddress;
171  serverAddress /= parameters["facility"];
172  serverAddress /= parameters["device"];
173  serverAddress /= parameters["location"];
174  address = std::string(serverAddress).substr(1);
175  }
176  std::string cacheFile{};
177  try {
178  cacheFile = parameters.at("cacheFile");
179  }
180  catch(std::out_of_range&) {
181  // empty cacheFile string => no caching
182  }
183  // create and return the backend
184  return boost::shared_ptr<DeviceBackend>(new DoocsBackend(address, cacheFile));

209  template<typename UserType>
210  boost::shared_ptr<NDRegisterAccessor<UserType>> DoocsBackend::getRegisterAccessor_impl(
211  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
212  NDRegisterAccessor<UserType>* p;
213  std::string path = _serverAddress + registerPathName;
214 
215  // check for additional hierarchy level, which indicates an access to a field of a complex property data type
216  bool hasExtraLevel = false;
217  if(!boost::starts_with(path, "doocs://") && !boost::starts_with(path, "epics://")) {
218  size_t nSlashes = std::count(path.begin(), path.end(), '/');
219  if(nSlashes == 4) {
220  hasExtraLevel = true;
221  }
222  else if(nSlashes < 3 || nSlashes > 4) {
223  throw ChimeraTK::logic_error(std::string("DOOCS address has an illegal format: ") + path);
224  }
225  }
226  else if(boost::starts_with(path, "doocs://")) {
227  size_t nSlashes = std::count(path.begin(), path.end(), '/');
228  // we have 3 extra slashes compared to the standard syntax without "doocs:://"
229  if(nSlashes == 4 + 3) {
230  hasExtraLevel = true;
231  }
232  else if(nSlashes < 3 + 3 || nSlashes > 4 + 3) {
233  throw ChimeraTK::logic_error(std::string("DOOCS address has an illegal format: ") + path);
234  }
235  }
236 
237  // split the path into property name and field name
238  std::string field;
239  if(hasExtraLevel) {
240  field = path.substr(path.find_last_of('/') + 1);
241  path = path.substr(0, path.find_last_of('/'));
242  }
243 
244  // read property once
245  EqAdr ea;
246  EqCall eq;
247  EqData src, dst;
248  ea.adr(std::string(path).c_str());
249  int rc = eq.get(&ea, &src, &dst);
250  if(rc) {
251  throw ChimeraTK::runtime_error("Cannot open DOOCS property '" + path + "': " + dst.get_string());
252  }
253 
254  // check type and create matching accessor
255  bool extraLevelUsed = false;
256  if(dst.type() == DATA_INT || dst.type() == DATA_A_INT || dst.type() == DATA_BOOL || dst.type() == DATA_A_BOOL ||
257  dst.type() == DATA_A_SHORT) {
258  p = new DoocsBackendIntRegisterAccessor<UserType>(
259  this, path, registerPathName, numberOfWords, wordOffsetInRegister, flags);
260  }
261  else if(dst.type() == DATA_A_LONG) {
262  p = new DoocsBackendLongRegisterAccessor<UserType>(
263  this, path, registerPathName, numberOfWords, wordOffsetInRegister, flags);
264  }
265  else if(dst.type() == DATA_FLOAT || dst.type() == DATA_A_FLOAT || dst.type() == DATA_DOUBLE ||
266  dst.type() == DATA_A_DOUBLE || dst.type() == DATA_SPECTRUM) {
267  p = new DoocsBackendFloatRegisterAccessor<UserType>(
268  this, path, registerPathName, numberOfWords, wordOffsetInRegister, flags);
269  }
270  else if(dst.type() == DATA_TEXT || dst.type() == DATA_STRING || dst.type() == DATA_STRING16) {
271  p = new DoocsBackendStringRegisterAccessor<UserType>(
272  this, path, registerPathName, numberOfWords, wordOffsetInRegister, flags);
273  }
274  else if(dst.type() == DATA_IIII) {
275  p = new DoocsBackendIIIIRegisterAccessor<UserType>(
276  this, path, registerPathName, numberOfWords, wordOffsetInRegister, flags);
277  }
278  else if(dst.type() == DATA_IFFF) {
279  extraLevelUsed = true;
280  p = new DoocsBackendIFFFRegisterAccessor<UserType>(
281  this, path, field, registerPathName, numberOfWords, wordOffsetInRegister, flags);
282  }
283  else {
284  throw ChimeraTK::logic_error("Unsupported DOOCS data type " + std::string(dst.type_string()) + " of property '" +
285  _serverAddress + registerPathName + "'");
286  }
287 
288  // if the field name has been specified but the data type does not use it, throw an exception
289  if(hasExtraLevel && !extraLevelUsed) {
290  throw ChimeraTK::logic_error("Specifiaction of field name is not supported for the DOOCS data type " +
291  std::string(dst.type_string()) + ": " + _serverAddress + registerPathName);
292  }
293 
294  return boost::shared_ptr<NDRegisterAccessor<UserType>>(p);
295  }

325 void CatalogueFetcher::fillCatalogue(std::string fixedComponents, long level) const {
326  // obtain list of elements within the given partial address
327  EqAdr ea;
328  EqCall eq;
329  EqData src, propList;
330  ea.adr((fixedComponents + "/*").c_str());
331  int rc = eq.names(&ea, &propList);
332  if(rc) {
333  // if the enumeration failes, maybe the server is not available (but
334  // exists in ENS) -> just ignore this address
335  return;
336  }
337 
338  // iterate over list
339  for(int i = 0; i < propList.array_length() && (isCancelled() == false); ++i) {
340  // obtain the name of the element
341  char c[255];
342  propList.get_string_arg(i, c, 255);
343  std::string name = c;
344  name = name.substr(0, name.find_first_of(" ")); // ignore comment which is following the space
345 
346  // if we are not yet at the property-level, recursivly call the function
347  // again to resolve the next hierarchy level
348  if(level < 2) {
349  fillCatalogue(fixedComponents + "/" + name, level + 1);
350  }
351  else {
352  // this is a property: create RegisterInfo entry and set its name
353  bool skipRegister = false;
354  for(uint i = 0; i < ctk::SIZE_IGNORE_PATTERNS; i++) {
355  std::string pattern = ctk::IGNORE_PATTERNS[i];
356  if(ignorePattern(name, pattern)) {
357  skipRegister = true;
358  break;
359  }
360  }
361  if(skipRegister) {
362  continue;
363  }
364  boost::shared_ptr<DoocsBackendRegisterInfo> info(new DoocsBackendRegisterInfo());
365  std::string fqn = fixedComponents + "/" + name;
366  info->name = fqn.substr(std::string(serverAddress_).length());
367 
368  // read property once to determine its length and data type
370  EqData dst;
371  ea.adr(fqn.c_str()); // strip leading slash
372  rc = eq.get(&ea, &src, &dst);
373  if(rc) {
374  // if the property is not accessible, ignore it. This happens
375  // frequently e.g. for archiver-related properties
376  continue;
377  }
378 
379  if(checkZmqAvailability(fixedComponents, name)) info->accessModeFlags.add(ctk::AccessMode::wait_for_new_data);
380 
381  info->length = dst.array_length();
382  if(info->length == 0) info->length = 1; // DOOCS reports 0 if not an array
383  if(dst.type() == DATA_TEXT || dst.type() == DATA_STRING || // in case of strings, DOOCS reports
384  // the length of the string
385  dst.type() == DATA_STRING16 || dst.type() == DATA_USTR) {
386  info->length = 1;
387  info->dataDescriptor =
388  ChimeraTK::RegisterInfo::DataDescriptor(ChimeraTK::RegisterInfo::FundamentalType::string);
389  }
390  else if(dst.type() == DATA_INT || dst.type() == DATA_A_INT || dst.type() == DATA_A_SHORT ||
391  dst.type() == DATA_A_LONG || dst.type() == DATA_A_BYTE || dst.type() == DATA_IIII) { // integral data types
392  size_t digits;
393  if(dst.type() == DATA_A_SHORT) { // 16 bit signed
394  digits = 6;
395  }
396  else if(dst.type() == DATA_A_BYTE) { // 8 bit signed
397  digits = 4;
398  }
399  else if(dst.type() == DATA_A_LONG) { // 64 bit signed
400  digits = 20;
401  }
402  else { // 32 bit signed
403  digits = 11;
404  }
405  if(dst.type() == DATA_IIII) info->length = 4;
406 
407  info->dataDescriptor = ChimeraTK::RegisterInfo::DataDescriptor(
408  ChimeraTK::RegisterInfo::FundamentalType::numeric, true, true, digits);
409  }
410  else if(dst.type() == DATA_IFFF) {
411  info->name = fqn.substr(std::string(serverAddress_).length()) + "/I";
412  info->dataDescriptor = ChimeraTK::RegisterInfo::DataDescriptor(
413  ChimeraTK::RegisterInfo::FundamentalType::numeric, true, true, 11); // 32 bit integer
414 
415  boost::shared_ptr<DoocsBackendRegisterInfo> infoF1(new DoocsBackendRegisterInfo(*info));
416  infoF1->name = fqn.substr(std::string(serverAddress_).length()) + "/F1";
417  infoF1->dataDescriptor = ChimeraTK::RegisterInfo::DataDescriptor(
418  ChimeraTK::RegisterInfo::FundamentalType::numeric, false, true, 320, 300); // float
419 
420  boost::shared_ptr<DoocsBackendRegisterInfo> infoF2(new DoocsBackendRegisterInfo(*infoF1));
421  infoF2->name = fqn.substr(std::string(serverAddress_).length()) + "/F2";
422 
423  boost::shared_ptr<DoocsBackendRegisterInfo> infoF3(new DoocsBackendRegisterInfo(*infoF1));
424  infoF3->name = fqn.substr(std::string(serverAddress_).length()) + "/F3";
425 
426  catalogue_->addRegister(infoF1);
427  catalogue_->addRegister(infoF2);
428  catalogue_->addRegister(infoF3);
429  // the info for the integer is added below
430  }
431  else { // floating point data types: always treat like double
432  info->dataDescriptor = ChimeraTK::RegisterInfo::DataDescriptor(
433  ChimeraTK::RegisterInfo::FundamentalType::numeric, false, true, 320, 300);
434  }
435 
436  // add info to catalogue
437  catalogue_->addRegister(info);
438  }

444 bool CatalogueFetcher::checkZmqAvailability(
445  const std::string& fullLocationPath, const std::string& propertyName) const {
446  int rc;
447  float f1;
448  float f2;
449  char* sp;
450  time_t tm;
451  EqAdr ea;
452  EqData dat;
453  EqData dst;
454  EqCall eq;
455  int portp;
456 
457  ea.adr(fullLocationPath + "/SPN");
458 
459  // get channel port number
460  dat.set(1, 0.0f, 0.0f, time_t{0}, propertyName, 0);
461 
462  // call get () to see whether it is supported
463  rc = eq.get(&ea, &dat, &dst);
464  if(rc) {
465  return false;
466  }
467 
468  rc = dst.get_ustr(&portp, &f1, &f2, &tm, &sp, 0);
469  if(rc && !portp && !static_cast<int>(f1 + f2)) rc = 0; // not supported
470 
471  if(!rc) {
472  dst.get_ustr(&portp, &f1, &f2, &tm, &sp, 0);
473  // get () not supported, call set ()
474  rc = eq.set(&ea, &dat, &dst);
475  if(rc) {
476  return false;
477  }
478  }
479 
480  if(dst.type() == DATA_INT) {
481  portp = dst.get_int();
482  }
483  else {
484  dst.get_ustr(&portp, &f1, &f2, &tm, &sp, 0);
485  }
486  return portp != 0;
487 }

570  auto catalogue = std::make_unique<ctk::RegisterCatalogue>();
571  auto parser = createDomParser(xmlfile);

573

586  void saveCatalogue(ctk::RegisterCatalogue& c, const std::string& xmlfile) {
587  xmlpp::Document doc;
588 
589  auto rootNode = doc.create_root_node("catalogue");
590  rootNode->set_attribute("version", "1.0");
591
