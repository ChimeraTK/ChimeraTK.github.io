25  NumericAddressedBackendRegisterAccessor(boost::shared_ptr<DeviceBackend> dev,
26  const RegisterPath &registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
27  : SyncNDRegisterAccessor<UserType>(registerPathName),
28  _fixedPointConverter(registerPathName),
29  isRaw(false),
30  _registerPathName(registerPathName),
31  _numberOfWords(numberOfWords)
32  {
33  try {
34  // check for unknown flags
35  flags.checkForUnknownFlags({AccessMode::raw});
36 
37  // check device backend
38  _dev = boost::dynamic_pointer_cast<NumericAddressedBackend>(dev);
39  if(!_dev) {
40  throw DeviceException("NumericAddressedBackendRegisterAccessor is used with a backend which is not "
41  "a NumericAddressedBackend.", DeviceException::WRONG_PARAMETER);
42  }
43 
44  // obtain register information
45  boost::shared_ptr<RegisterInfo> info = _dev->getRegisterInfo(registerPathName);
46  _registerInfo = boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(info);
47  _bar = _registerInfo->bar;
48  _startAddress = _registerInfo->address + wordOffsetInRegister*sizeof(int32_t);
49 
50  // check number of words
51  if(_numberOfWords == 0) {
52  _numberOfWords = _registerInfo->getNumberOfElements() - wordOffsetInRegister;
53  }
54  if(_numberOfWords + wordOffsetInRegister > _registerInfo->getNumberOfElements()) {
55  throw DeviceException("Requested number of words exceeds the size of the register '"+_registerPathName+"'!",
56  DeviceException::WRONG_PARAMETER);
57  }
58  if(wordOffsetInRegister >= _registerInfo->getNumberOfElements()) {
59  throw DeviceException("Requested offset exceeds the size of the register'"+_registerPathName+"'!",
60  DeviceException::WRONG_PARAMETER);
61  }
62 
63  // create low-level transfer element handling the actual data transfer to the hardware with raw data
64  _rawAccessor.reset(new NumericAddressedLowLevelTransferElement(_dev,_bar,_startAddress,_numberOfWords));
65 
66  // allocated the buffers
67  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
68  NDRegisterAccessor<UserType>::buffer_2D[0].resize(_numberOfWords);
69 
70  // configure fixed point converter
71  if(!flags.has(AccessMode::raw)) {
72  _fixedPointConverter = FixedPointConverter(_registerPathName, _registerInfo->width, _registerInfo->nFractionalBits,
73  _registerInfo->signedFlag);
74  }
75  else {
76  if(typeid(UserType) != typeid(int32_t)) {
77  throw DeviceException("Given UserType when obtaining the NumericAddressedBackendRegisterAccessor in raw mode does not "
78  "match the expected type. Use an int32_t instead! (Register name: "+_registerPathName+"')", DeviceException::WRONG_PARAMETER);
79  }
80  _fixedPointConverter = FixedPointConverter(_registerPathName, 32, 0, true);
81  isRaw = true;
82  }
83  }
84  catch(...) {
85  this->shutdown();
86  throw;
87  }
88  }
