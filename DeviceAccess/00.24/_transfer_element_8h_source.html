341  inline boost::shared_ptr<TransferElement> TransferElement::readAny(std::list<std::reference_wrapper<TransferElement>> elementsToRead) {
342  std::list<TransferFuture*> futureList;
343  for(auto elem : elementsToRead) {
344  auto *future = &(elem.get().readAsync());
345  futureList.push_back(future);
346  }
347  // See class description of TransferFutureIterator. We need this trick to obtain the transfer element from the
348  // returned iterator.
349  auto iter = boost::wait_for_any(TransferFutureIterator(futureList.begin()), TransferFutureIterator(futureList.end()));
350  boost::this_thread::interruption_point();
351  iter.getTransferFuture().wait(); // complete the transfer (i.e. run postRead())
352  boost::this_thread::interruption_point();
353  return iter.getTransferFuture().getTransferElement().shared_from_this();
354  }
