56  template<typename T>
57  ChimeraTK::ScalarRegisterAccessor<T> getScalar(const ChimeraTK::RegisterPath& name) const {
58  // check for existing accessor in cache
59  if(boost::fusion::at_key<T>(scalarMap.table).count(name) > 0) {
60  return boost::fusion::at_key<T>(scalarMap.table)[name];
61  }
62 
63  // obtain accessor from ControlSystemPVManager
64  auto pv = pvManager->getProcessArray<T>(name);
65  if(pv == nullptr) {
66  throw ChimeraTK::logic_error("Process variable '" + name + "' does not exist.");
67  }
68 
69  // obtain variable id from pvIdMap and transfer it to idMap (required by the
70  // TestableModeAccessorDecorator)
71  size_t varId = Application::getInstance().pvIdMap[pv->getUniqueId()];
72 
73  // decorate with TestableModeAccessorDecorator if variable is sender and
74  // receiver is not poll-type, and store it in cache
75  if(pv->isWriteable() && !Application::getInstance().testableMode_isPollMode[varId]) {
76  auto deco = boost::make_shared<TestableModeAccessorDecorator<T>>(pv, false, true, varId, varId);
77  Application::getInstance().testableMode_names[varId] = "ControlSystem:" + name;
78  boost::fusion::at_key<T>(scalarMap.table)[name].replace(ChimeraTK::ScalarRegisterAccessor<T>(deco));
79  }
80  else {
81  boost::fusion::at_key<T>(scalarMap.table)[name].replace(ChimeraTK::ScalarRegisterAccessor<T>(pv));
82  }
83 
84  // return the accessor as stored in the cache
85  return boost::fusion::at_key<T>(scalarMap.table)[name];
86  }

161  // Cache (possible decorated) accessors to avoid the need to create accessors
162  // multiple times. This would not work if the accessor is decorated, since the
163  // buffer would be lost and thus the current value could no longer be
164  // obtained. This has to be done separately for scalar and array accessors and
165  // in dependence of the user type. Since this is a cache and does not change
166  // the logical behaviour of the class, the maps are defined mutable.
167  template<typename UserType>
168  using ScalarMap = std::map<std::string, ChimeraTK::ScalarRegisterAccessor<UserType>>;
169  mutable ChimeraTK::TemplateUserTypeMap<ScalarMap> scalarMap;
170 
171  template<typename UserType>
172  using ArrayMap = std::map<std::string, ChimeraTK::OneDRegisterAccessor<UserType>>;
173  mutable ChimeraTK::TemplateUserTypeMap<ArrayMap> arrayMap;
