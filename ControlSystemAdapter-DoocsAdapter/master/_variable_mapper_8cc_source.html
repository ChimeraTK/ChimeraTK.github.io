225  void VariableMapper::import(std::string importSource, std::string importLocationName,

228  if (importSource == "/"){

230  }
231

233  for (auto const & processVariable : _inputVariables){
234  if (_usedInputVariables.find(processVariable) != _usedInputVariables.end()){
235  continue;
236  }
237

239  // processVariable starts with wanted source

241  // we use the register path because it removes duplicate separators and allows to use
242  // . as separater to replace all / with .

245  if (importLocationName.empty()){
