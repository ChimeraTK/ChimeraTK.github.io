16 /*
17  * Select all mount points that mount a physical hard drive on the local machine.
18  * The boot partition is not selected.
19  * Only devices that include /dev/sd are considered.
20  */
21 std::set<std::pair<std::string, std::string> > findMountPoints(){
22  std::set<std::pair<std::string, std::string> > out;
23  std::ifstream in("/proc/mounts");
24  std::string strin[4];
25  int iin[2];
26  while(in.good()){
27  in >> strin[0] >> strin[1] >> strin[2] >> strin[3] >> iin[0] >> iin[1];
28  bfs::path p(strin[0]);
29  if(strin[0].substr(0,7).compare("/dev/sd") == 0){
30  if(strin[1].substr(0,6).compare("/boot/") != 0){
31  out.insert(std::make_pair(p.filename().string(),strin[1]));
32  }
33  }
34  }
35  in.close();
36  return out;
37 }

39 /*
40  * Select all non virtual network adapter.
41  */
42 std::set<std::string> findNetworkDevices(){
43  std::set<std::string> out;
44  bfs::path p("/sys/class/net");
45  try {
46  if (exists(p)){ // does p actually exist?
47  if (is_directory(p)){ // is p a directory?
48  for(auto i = bfs::directory_iterator(p); i != bfs::directory_iterator(); i++){
49  if(bfs::canonical(i->path()).string().find("virtual") == std::string::npos)
50  out.insert(i->path().filename().string());
51  }
52  }
53  }
54  } catch (const bfs::filesystem_error& ex) {
55  std::cout << ex.what() << '\n';
56  }
57  return out;
58 }
