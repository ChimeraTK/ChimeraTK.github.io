56  VirtualModule& DeviceModule::virtualiseFromCatalog() const {
57  if(virtualisedModuleFromCatalog_isValid) return virtualisedModuleFromCatalog;
58 
59  virtualisedModuleFromCatalog = VirtualModule(deviceAliasOrURI, "Device module", ModuleType::Device);
60 
61  // obtain register catalogue
62  Device d;
63  d.open(deviceAliasOrURI);
64  auto catalog = d.getRegisterCatalogue();
65 
66  // iterate catalogue, create VariableNetworkNode for all registers starting with the registerNamePrefix
67  size_t prefixLength = registerNamePrefix.length();
68  for(auto &reg : catalog) {
69  if(std::string(reg.getRegisterName()).substr(0,prefixLength) != std::string(registerNamePrefix)) continue;
70 
71  // ignore 2D registers
72  if(reg.getNumberOfDimensions() > 1) continue;
73 
74  // guess direction and determine update mode
75  VariableDirection direction;
76  UpdateMode updateMode;
77  if(reg.isWriteable()) {
78  direction = {VariableDirection::consuming, false};
79  updateMode = UpdateMode::push;
80  }
81  else {
82  direction = {VariableDirection::feeding, false};
83  if(reg.getSupportedAccessModes().has(AccessMode::wait_for_new_data)) {
84  updateMode = UpdateMode::push;
85  }
86  else {
87  updateMode = UpdateMode::poll;
88  }
89  }
90 
91  // guess type
92  const std::type_info *valTyp{&typeid(AnyType)};
93  auto &dd = reg.getDataDescriptor(); //numeric, string, boolean, nodata, undefined
94  if(dd.fundamentalType() == RegisterInfo::FundamentalType::numeric) {
95  if(dd.isIntegral()) {
96  if(dd.isSigned()) {
97  if(dd.nDigits() > 11) {
98  valTyp = &typeid(int64_t);
99  }
100  else if(dd.nDigits() > 6) {
101  valTyp = &typeid(int32_t);
102  }
103  else if(dd.nDigits() > 4) {
104  valTyp = &typeid(int16_t);
105  }
106  else {
107  valTyp = &typeid(int8_t);
108  }
109  }
110  else {
111  if(dd.nDigits() > 10) {
112  valTyp = &typeid(uint64_t);
113  }
114  else if(dd.nDigits() > 5) {
115  valTyp = &typeid(uint32_t);
116  }
117  else if(dd.nDigits() > 3) {
118  valTyp = &typeid(uint16_t);
119  }
120  else {
121  valTyp = &typeid(uint8_t);
122  }
123  }
124  }
125  else { // fractional
126  valTyp = &typeid(double);
127  }
128  }
129  else if(dd.fundamentalType() == RegisterInfo::FundamentalType::boolean) {
130  valTyp = &typeid(int32_t);
131  }
132  else if(dd.fundamentalType() == RegisterInfo::FundamentalType::string) {
133  valTyp = &typeid(std::string);
134  }
135 
136  auto name = std::string(reg.getRegisterName()).substr(prefixLength);
137  auto lastSlash = name.find_last_of("/");
138  auto dirname = name.substr(0,lastSlash);
139  auto basename = name.substr(lastSlash+1);
140  VariableNetworkNode node(basename, deviceAliasOrURI, reg.getRegisterName(), updateMode,
141  direction, *valTyp, reg.getNumberOfElements());
142  virtualisedModuleFromCatalog.createAndGetSubmoduleRecursive(dirname).addAccessor(node);
143 
144  }
145 
146  virtualisedModuleFromCatalog_isValid = true;
147  return virtualisedModuleFromCatalog;
148  }
