153  inline const STATE& getState(VirtualTime time, bool forceNoInterpolation=false) {
154 #ifndef ENABLE_EXPERIMENTAL_FEATURES
155  (void)forceNoInterpolation; // avoid warning
156 #endif
157  // check if time is the current time and return the latest element
158  if(time >= currentTime && time < currentTime + validityPeriod) {
159  return getLatestState();
160  }
161  // search in buffer: find the first element after the requested time
162  auto it = buffer.upper_bound(time);
163  if(it == buffer.begin()) {
164  std::stringstream s;
165  s << "Value request is too far into the past: ";
166  s << "requested time = " << time << ", oldest history = " << buffer.begin()->first;
167  s << ", current time = " << currentTime;
168  throw StateVariableSetException(s.str(),StateVariableSetException::REQUEST_FAR_PAST);
169  }
170  // if this is end(), a new value needs to be computed
171  if(it == buffer.end()) {
172 #ifdef ENABLE_EXPERIMENTAL_FEATURES
173  if(!forceNoInterpolation) {
174  return getValueInterpolated(time, buffer.rbegin()->first);
175  }
176  else {
177 #endif
178  return getValueFromCallback(time);
179 #ifdef ENABLE_EXPERIMENTAL_FEATURES
180  }
181 #endif
182  }
183  // decrement to get the most recent sample before the requested time
184  --it;
185  // if sample is too old, request one via callback
186  if(time >= it->first + validityPeriod) {
187 #ifdef ENABLE_EXPERIMENTAL_FEATURES
188  if(!forceNoInterpolation) {
189  return getValueInterpolated(time, it->first);
190  }
191  else {
192 #endif
193  return getValueFromCallback(time);
194 #ifdef ENABLE_EXPERIMENTAL_FEATURES
195  }
196 #endif
197  }
198  // return value from buffer
199  return it->second;
200  }

245  inline void truncateFuture(VirtualTime time) {
246  if(time < currentTime) {
247  auto firstToDelete = buffer.upper_bound(time);
248  // Note: getState() already ensures that the request does not go before the first map element, so we are
249  // never deleting the entire map here.

266  inline const STATE& getValueInterpolated(VirtualTime timeRequested, VirtualTime previousStep) {
267 #ifndef ENABLE_EXPERIMENTAL_FEATURES
268  class ExperimentalFeatureNotEnabled{};
269  throw ExperimentalFeatureNotEnabled();
270 #endif
271  // no interpolation or requested time is later than the interpolation period (maxGap)
272  if(!enableInterpolation || timeRequested >= previousStep + maxGap) {
273  return getValueFromCallback(timeRequested);
274  }
275 
276  // save interpolated state into the buffer and return it
277  auto state1 = getState(previousStep, true);
278  auto state2 = getState(previousStep + maxGap, true);
279  buffer[timeRequested] = interpolate(state1, state2, previousStep, previousStep+maxGap, timeRequested);
280  return buffer[timeRequested];
281  }

284  inline const STATE& getValueFromCallback(VirtualTime time) {
285 
286  // If request goes into the past (w.r.t. currentTime), erase newer states to force recomputing them.
287  truncateFuture(time);
288 
289  // Compute the number of steps necessary to reach the requested time, obaying hugeGap. The last step at the
290  // requested time itself is not computed here, since maxGap shall be obeyed in the last gap (see below).
291  // Note: this is effectively rounding-up the integer division (time-CurrentTime)/hugeGap
292  unsigned int nHugeSteps = (time-currentTime-1)/hugeGap + 1;
293 
294  // Loop over the time in nHugeSteps steps. The first step is potentially smaller, all consequtive steps are
295  // maxGap big. Omit the last step, so the requested time is not yet reached.
296  for(unsigned int i=0; i<nHugeSteps-1; i++) {
297  VirtualTime t = time-(nHugeSteps-1)*hugeGap + i*hugeGap;
298 
299  // obtain the new state
300  STATE state = compute(t);
301 
302  // save value into buffer and update the current time
303  buffer[t] = state;
304  currentTime = t;
305 
306  }
307 
308  // Compute the number of steps necessary to reach the requested time, obaying maxGap.
309  // Note: this is effectively rounding-up the integer division (time-CurrentTime)/maxGap
310  unsigned int nSteps = (time-currentTime-1)/maxGap + 1;
311 
312  // Loop over the time in nSteps steps. The first step is potentially smaller, all consequtive steps are
313  // maxGap big.
314  for(unsigned int i=0; i<nSteps; i++) {
315  VirtualTime t = time-(nSteps-1)*maxGap + i*maxGap;
316 
317  // obtain the new state
318  STATE state = compute(t);
319 
320  // save value into buffer and update the current time
321  buffer[t] = state;
322  currentTime = t;
323 
324  }
325 
326  // remove old history
327  truncatePast();
328 
329  // return the new state (do not use buffer[], as it will silently insert new objects if something went wrong)
330  return buffer.find(time)->second;
331  }
