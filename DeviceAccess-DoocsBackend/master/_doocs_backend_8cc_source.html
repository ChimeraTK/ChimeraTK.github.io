103  void DoocsBackend::fillCatalogue(std::string fixedComponents, long level) const {
104 
105  // obtain list of elements within the given partial address
106  EqAdr ea;
107  EqCall eq;
108  EqData src,dst;
109  ea.adr((fixedComponents+"/*").c_str());
110  int rc = eq.names(&ea, &dst);
111  if(rc) {
112  // if the enumeration failes, maybe the server is not available (but exists in ENS) -> just ignore this address
113  return;
114  }
115 
116  // iterate over list
117  for(int i=0; i<dst.array_length(); ++i) {
118 
119  // obtain the name of the element
120  char c[255];
121  dst.get_string_arg(i, c, 255);
122  std::string name = c;
123  name = name.substr(0, name.find_first_of(" ")); // ignore comment which is following the space
124 
125  // if we are not yet at the property-level, recursivly call the function again to resolve the next hierarchy level
126  if(level < 2) {
127  fillCatalogue(fixedComponents+"/"+name, level+1);
128  }
129  else {
130 
131  // this is a property: create RegisterInfo entry and set its name
132  boost::shared_ptr<DoocsBackendRegisterInfo> info(new DoocsBackendRegisterInfo());
133  std::string fqn = fixedComponents+"/"+name;
134  info->name = fqn.substr(std::string(_serverAddress).length());
135 
136  // read property once to determine its length and data type
138  EqAdr ea;
139  EqCall eq;
140  EqData src,dst;
141  ea.adr(fqn.c_str()); // strip leading slash
142  int rc = eq.get(&ea, &src, &dst);
143  if(rc) {
144  // if the property is not accessible, ignore it. This happens frequently e.g. for archiver-related properties
145  continue;
146  }
147  info->length = dst.array_length();
148  if(info->length == 0) info->length = 1; // DOOCS reports 0 if not an array
149  if(dst.type() == DATA_TEXT || dst.type() == DATA_STRING || // in case of strings, DOOCS reports the length of the string
150  dst.type() == DATA_STRING16 || dst.type() == DATA_USTR ) {
151  info->length = 1;
152  info->dataDescriptor = ChimeraTK::RegisterInfo::DataDescriptor( ChimeraTK::RegisterInfo::FundamentalType::string );
153  }
154  else if(dst.type() == DATA_INT || dst.type() == DATA_A_INT || dst.type() == DATA_A_SHORT ||
155  dst.type() == DATA_A_LONG || dst.type() == DATA_A_BYTE || dst.type() == DATA_IIII ) { // integral data types
156  size_t digits;
157  if(dst.type() == DATA_A_SHORT) { // 16 bit signed
158  digits = 6;
159  }
160  else if(dst.type() == DATA_A_BYTE) { // 8 bit signed
161  digits = 4;
162  }
163  else if(dst.type() == DATA_A_LONG) { // 64 bit signed
164  digits = 20;
165  }
166  else { // 32 bit signed
167  digits = 11;
168  }
169  if (dst.type() == DATA_IIII)
170  info->length = 4;
171 
172  info->dataDescriptor = ChimeraTK::RegisterInfo::DataDescriptor( ChimeraTK::RegisterInfo::FundamentalType::numeric,
173  true, true, digits );
174  }
175  else { // floating point data types: always treat like double
176  info->dataDescriptor = ChimeraTK::RegisterInfo::DataDescriptor( ChimeraTK::RegisterInfo::FundamentalType::numeric,
177  false, true, 320, 300 );
178  }
179 
180  // add info to catalogue
181  _catalogue_mutable.addRegister(info);
182  }
183 
184  }

197  void DoocsBackend::fillCatalogue() const {
198  if(!catalogueFilled && _opened) {
199  // Fill the catalogue:
200  // first, count number of elements in address part given in DMAP file to determine how many components we have to
201  // iterate over
202  size_t nSlashes;
203  if(!boost::starts_with(_serverAddress, "doocs://")) {
204  nSlashes = std::count(_serverAddress.begin(), _serverAddress.end(), '/');
205  }
206  else{
207  nSlashes = std::count(_serverAddress.begin(), _serverAddress.end(), '/') - 3;
208  }
209  // next, iteratively call the function to fill the catalogue
210  fillCatalogue(_serverAddress, nSlashes);
211  catalogueFilled = true;
212  }
213  }

231  template<typename UserType>
232  boost::shared_ptr< NDRegisterAccessor<UserType> > DoocsBackend::getRegisterAccessor_impl(
233  const RegisterPath &registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
234 
235  NDRegisterAccessor<UserType> *p;
236  std::string path = _serverAddress+registerPathName;
237 
238  // read property once
239  EqAdr ea;
240  EqCall eq;
241  EqData src,dst;
242  ea.adr(std::string(path).c_str());
243  int rc = eq.get(&ea, &src, &dst);
244  if(rc) {
245  throw ChimeraTK::runtime_error(std::string("Cannot open DOOCS property: ")+dst.get_string());
246  }
247 
248  // check type and create matching accessor
249  if( dst.type() == DATA_INT || dst.type() == DATA_A_INT ||
250  dst.type() == DATA_BOOL || dst.type() == DATA_A_BOOL ||
251  dst.type() == DATA_A_SHORT) {
252  p = new DoocsBackendIntRegisterAccessor<UserType>(path, numberOfWords, wordOffsetInRegister, flags);
253  }
254  else if( dst.type() == DATA_A_LONG ) {
255  p = new DoocsBackendLongRegisterAccessor<UserType>(path, numberOfWords, wordOffsetInRegister, flags);
256  }
257  else if( dst.type() == DATA_FLOAT || dst.type() == DATA_A_FLOAT ||
258  dst.type() == DATA_DOUBLE || dst.type() == DATA_A_DOUBLE || dst.type() == DATA_SPECTRUM ) {
259  p = new DoocsBackendFloatRegisterAccessor<UserType>(path, numberOfWords, wordOffsetInRegister, flags);
260  }
261  else if( dst.type() == DATA_TEXT || dst.type() == DATA_STRING || dst.type() == DATA_STRING16) {
262  p = new DoocsBackendStringRegisterAccessor<UserType>(path, numberOfWords, wordOffsetInRegister, flags);
263  }
264  else if( dst.type() == DATA_IIII) {
265  p = new DoocsBackendIIIIRegisterAccessor<UserType>(path, numberOfWords, wordOffsetInRegister, flags);
266  }
267  else {
268  throw ChimeraTK::logic_error("Unsupported DOOCS data type: "+std::string(dst.type_string()));
269  }
270 
271 
272  return boost::shared_ptr< NDRegisterAccessor<UserType> > ( p );
273  }
