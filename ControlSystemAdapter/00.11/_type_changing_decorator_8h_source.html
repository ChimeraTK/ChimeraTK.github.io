346  template<class UserType>
347  boost::shared_ptr<mtca4u::NDRegisterAccessor<UserType>> getDecorator(const boost::shared_ptr<mtca4u::TransferElement> &transferElement,
348  DecoratorType decoratorType) {
349 
350  // check if there already is a decorator for the transfer element
351  auto decoratorMapEntry = getGlobalDecoratorMap().find(transferElement);
352  if (decoratorMapEntry != getGlobalDecoratorMap().end()) {
353  // There already is a decorator for this transfer element
354  // The decorator has to have a matching type, otherwise we can only throw
355  auto castedType = boost::dynamic_pointer_cast<mtca4u::NDRegisterAccessor<UserType>>(decoratorMapEntry->second);
356  if (castedType) { // User type matches, but the decorator type also has to match
357  auto decoTypeHolder = boost::dynamic_pointer_cast<DecoratorTypeHolder>(decoratorMapEntry->second);
358  assert(decoTypeHolder);
359  if (decoTypeHolder->getDecoratorType() == decoratorType) {
360  // decorator matches, return it
361  return castedType;
362  }
363  else {
364  // sorry there is a decorator, but it's the wrong user type
365  throw std::logic_error("ChimeraTK::ControlSystemAdapter: Decorator for TransferElement " +
366  transferElement->getName() +" already exists as a different decorator type.");
367  }
368  }
369  else {
370  // sorry there is a decorator, but it's the wrong user type
371  throw std::logic_error("ChimeraTK::ControlSystemAdapter: Decorator for TransferElement " +
372  transferElement->getName() +" already exists with a different user type.");
373  }
374  }
375 
376  DecoratorFactory< UserType > factory(transferElement, decoratorType );
377  boost::fusion::for_each(mtca4u::userTypeMap(), factory);
378  getGlobalDecoratorMap()[transferElement] = factory.createdDecorator;
379  return factory.createdDecorator;
380  }
