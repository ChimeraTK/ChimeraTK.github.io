70  bool isMergeable(const boost::shared_ptr<TransferElement const>& other) const {
71  if(!_dev->canMergeRequests()) return false;
72 
73  // accessor type, device and bar must be the same
74  auto rhsCasted = boost::dynamic_pointer_cast<const NumericAddressedLowLevelTransferElement>(other);
75  if(!rhsCasted) return false;
76  if(_dev != rhsCasted->_dev) return false;
77  if(_bar != rhsCasted->_bar) return false;
78 
79  // only allow adjacent and overlapping address areas to be merged
80  if(_startAddress + _numberOfBytes < rhsCasted->_startAddress) return false;
81  if(_startAddress > rhsCasted->_startAddress + rhsCasted->_numberOfBytes) return false;
82  return true;
83  }
