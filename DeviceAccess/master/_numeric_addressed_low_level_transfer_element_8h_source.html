112  bool isMergeable(const boost::shared_ptr<TransferElement const> &other) const {
113  // accessor type, device and bar must be the same
114  auto rhsCasted = boost::dynamic_pointer_cast< const NumericAddressedLowLevelTransferElement >(other);
115  if(!rhsCasted) return false;
116  if(_dev != rhsCasted->_dev) return false;
117  if(_bar != rhsCasted->_bar) return false;
118 
119  // only allow adjacent and overlapping address areas to be merged
