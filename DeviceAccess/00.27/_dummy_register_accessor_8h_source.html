247  DummyMultiplexedRegisterAccessor(DummyBackend *dev, std::string module, std::string name)
248  : _dev(dev), pitch(0)
249  {
250  _dev->_registerMapping->getRegisterInfo(MULTIPLEXED_SEQUENCE_PREFIX+name, registerInfo, module);
251 
252  int i = 0;
253  while(true) {
254  // obtain register information for sequence
255  RegisterInfoMap::RegisterInfo elem;
256  std::stringstream sequenceNameStream;
257  sequenceNameStream << SEQUENCE_PREFIX << name << "_" << i++;
258  try{
259  _dev->_registerMapping->getRegisterInfo( sequenceNameStream.str(), elem, module );
260  }
261  catch(MapFileException &) {
262  break;
263  }
264  // create fixed point converter for sequence
265  fpc.push_back( FixedPointConverter(module+"/"+name,elem.width, elem.nFractionalBits, elem.signedFlag) );
266  // store offsets and number of bytes per word
267  offsets.push_back(elem.address);
268  nbytes.push_back(elem.nBytes);
269  // determine pitch
270  pitch += elem.nBytes;
271  }
272 
273  if(fpc.empty()) {
274  throw DummyRegisterException("No sequenes found for name \""+name+"\".",DummyRegisterException::EMPTY_AREA);
275  }
276 
277  // compute number of elements per sequence
278  nElements = registerInfo.nBytes/pitch;
279  }
280

283  {
284  return nElements;
285  }
