145  inline const STATE& getState(VirtualTime time, bool forceNoInterpolation=false) {
146 #ifndef ENABLE_EXPERIMENTAL_FEATURES
147  (void)forceNoInterpolation; // avoid warning
148 #endif
149  // check if time is the current time and return the latest element
150  if(time >= currentTime && time < currentTime + validityPeriod) {
151  return getLatestState();
152  }
153  // search in buffer: find the first element after the requested time
154  auto it = buffer.upper_bound(time);
155  if(it == buffer.begin()) {
156  std::stringstream s;
157  s << "Value request is too far into the past: ";
158  s << "requested time = " << time << ", oldest history = " << buffer.begin()->first;
159  s << ", current time = " << currentTime;
160  throw ChimeraTK::logic_error(s.str());
161  }
162  // if this is end(), a new value needs to be computed
163  if(it == buffer.end()) {
164 #ifdef ENABLE_EXPERIMENTAL_FEATURES
165  if(!forceNoInterpolation) {
166  return getValueInterpolated(time, buffer.rbegin()->first);
167  }
168  else {
169 #endif
170  return getValueFromCallback(time);
171 #ifdef ENABLE_EXPERIMENTAL_FEATURES
172  }
173 #endif
174  }
175  // decrement to get the most recent sample before the requested time
176  --it;
177  // if sample is too old, request one via callback
178  if(time >= it->first + validityPeriod) {
179 #ifdef ENABLE_EXPERIMENTAL_FEATURES
180  if(!forceNoInterpolation) {
181  return getValueInterpolated(time, it->first);
182  }
183  else {
184 #endif
185  return getValueFromCallback(time);
186 #ifdef ENABLE_EXPERIMENTAL_FEATURES
187  }
188 #endif
189  }
190  // return value from buffer
191  return it->second;
192  }

237  inline void truncateFuture(VirtualTime time) {
238  if(time < currentTime) {
239  auto firstToDelete = buffer.upper_bound(time);
240  // Note: getState() already ensures that the request does not go before the first map element, so we are

250  // clear old values from history

258  inline const STATE& getValueInterpolated(VirtualTime timeRequested, VirtualTime previousStep) {
259 #ifndef ENABLE_EXPERIMENTAL_FEATURES
260  class ExperimentalFeatureNotEnabled{};
261  throw ExperimentalFeatureNotEnabled();
262 #endif
263  // no interpolation or requested time is later than the interpolation period (maxGap)
264  if(!enableInterpolation || timeRequested >= previousStep + maxGap) {
265  return getValueFromCallback(timeRequested);
266  }
267 
268  // save interpolated state into the buffer and return it
269  auto state1 = getState(previousStep, true);
270  auto state2 = getState(previousStep + maxGap, true);
271  buffer[timeRequested] = interpolate(state1, state2, previousStep, previousStep+maxGap, timeRequested);
272  return buffer[timeRequested];
273  }

276  inline const STATE& getValueFromCallback(VirtualTime time) {
277 
278  // If request goes into the past (w.r.t. currentTime), erase newer states to force recomputing them.
279  truncateFuture(time);
280 
281  // Compute the number of steps necessary to reach the requested time, obaying hugeGap. The last step at the
282  // requested time itself is not computed here, since maxGap shall be obeyed in the last gap (see below).
283  // Note: this is effectively rounding-up the integer division (time-CurrentTime)/hugeGap
284  unsigned int nHugeSteps = (time-currentTime-1)/hugeGap + 1;
285 
286  // Loop over the time in nHugeSteps steps. The first step is potentially smaller, all consequtive steps are
287  // maxGap big. Omit the last step, so the requested time is not yet reached.
288  for(unsigned int i=0; i<nHugeSteps-1; i++) {
289  VirtualTime t = time-(nHugeSteps-1)*hugeGap + i*hugeGap;
290 
291  // obtain the new state
292  STATE state = compute(t);
293 
294  // save value into buffer and update the current time
295  buffer[t] = state;
296  currentTime = t;
297 
298  }
299 
300  // Compute the number of steps necessary to reach the requested time, obaying maxGap.
301  // Note: this is effectively rounding-up the integer division (time-CurrentTime)/maxGap
302  unsigned int nSteps = (time-currentTime-1)/maxGap + 1;
303 
304  // Loop over the time in nSteps steps. The first step is potentially smaller, all consequtive steps are
305  // maxGap big.
306  for(unsigned int i=0; i<nSteps; i++) {
307  VirtualTime t = time-(nSteps-1)*maxGap + i*maxGap;
308 
309  // obtain the new state
310  STATE state = compute(t);
311 
312  // save value into buffer and update the current time
313  buffer[t] = state;
314  currentTime = t;
315 
316  }
317 
318  // remove old history
319  truncatePast();
320 
321  // return the new state (do not use buffer[], as it will silently insert new objects if something went wrong)
322  return buffer.find(time)->second;
323  }
