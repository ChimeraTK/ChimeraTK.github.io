181  // FIXME: If the same function is registered more than one, it may be executed multiple times
182

204  bool DummyBackend::isWriteRangeOverlap( AddressRange firstRange, AddressRange secondRange){
205  if (firstRange.bar != secondRange.bar){
206  return false;
207  }
208 
209  uint32_t startAddress = std::max( firstRange.offset, secondRange.offset );
210  uint32_t endAddress = std::min( firstRange.offset + firstRange.sizeInBytes,
211  secondRange.offset + secondRange.sizeInBytes );
212 
213  // if at least one register is writable there is an overlap of writable registers
214  for ( uint32_t address = startAddress; address < endAddress; address += sizeof(int32_t) ){
215  if ( isReadOnly(firstRange.bar, address)==false ){
216  return true;
217  }
218  }
219 
220  // we looped all possible registers, none is writable
221  return false;
222  }
