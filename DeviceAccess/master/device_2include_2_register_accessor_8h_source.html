46  template<typename ConvertedDataType>
47  void read(ConvertedDataType* convertedData, size_t nWords = 1, uint32_t wordOffsetInRegister = 0) const {
48  if(nWords == 0) return;
49  // obtain the accessor through the handler
50  auto& accessorHandler = boost::fusion::at_key<ConvertedDataType>(_convertingAccessorHandlers.table);
51  accessorHandler.checkAndResize(nWords, wordOffsetInRegister, false /*not raw*/, _backend, _registerPathName);
52  // now we can be sure that the accessor is valid and has enough memory to perform the memcpy
53  accessorHandler.accessor->read();
54  // copy data to target buffer. The accessor might not start at index 0, so we have to calculate the correct
55  // offset inside the accessor
56  memcpy(convertedData,
57  accessorHandler.accessor->accessChannel(0).data() + wordOffsetInRegister - accessorHandler.beginIndex,
58  nWords * sizeof(ConvertedDataType));
59  }

200  template<class UserType>
201  class AccessorHandler {
202  public:
203  AccessorHandler() : beginIndex(0), endIndex(0) {}
204 
205  boost::shared_ptr<NDRegisterAccessor<UserType>> accessor;
206  size_t beginIndex; // the first index of the accessor;
207  size_t endIndex; // the end index (first index after the valid range, liken end iterator)
208  void checkAndResize(size_t nWords, size_t wordOffsetInRegister, bool isRaw,
209  boost::shared_ptr<DeviceBackend> const& backend, RegisterPath const& registerPathName) {
210  // no need for out of register size checking. getRegisterAccessor does the job for us and throws in case of
211  // error.
212  if((wordOffsetInRegister < beginIndex) || (wordOffsetInRegister + nWords > (endIndex) || !accessor)) {
213  // we have to (re)allocate
214  size_t newBeginIndex;
215  if(!accessor) {
216  newBeginIndex = wordOffsetInRegister;
217  }
218  else {
219  newBeginIndex = std::min(wordOffsetInRegister, beginIndex);
220  }
221  size_t newEndIndex = std::max(wordOffsetInRegister + nWords, endIndex);
222  accessor = backend->getRegisterAccessor<UserType>(
223  registerPathName, newEndIndex - newBeginIndex, newBeginIndex, isRaw); // 0 offset, raw
224  // we have to update the accessor with the current hardware information. It might be that a partial write is
225  // going to happen.
226  accessor->read();
227  // only if creating the new accessor succeeds we change the index bookkeeping variables
228  beginIndex = newBeginIndex;
229  endIndex = newEndIndex;
230  }
231  }
232  };
