75  void StateMachine::performTransition(Event event){
76  std::map< Event, State::TransitionData >::const_iterator it;
77 
78  const State::TransitionTable& transitionTable = _currentState->getTransitionTable();
79  it = transitionTable.find(event);
80  if(it != transitionTable.end()){
81 
82  _requestedState = ((it->second).targetState);
83  _requestedInternalCallback = (it->second).internalCallbackAction;
84 
85  // Apply new state right away, if no async action active
86  if(!_asyncActionActive.load()){
87  _currentState = _requestedState;
88  _requestedState = nullptr;
89  _internalEventCallback = _requestedInternalCallback;
90  }
91  (it->second).entryCallbackAction();
92  }
