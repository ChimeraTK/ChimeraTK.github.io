411  // the factory function. You don't have to care about the IMPL_Type when requesting a decorator. Implementation, declared at the top of this file
412  template<class UserType>
413  boost::shared_ptr< mtca4u::NDRegisterAccessor< UserType > > getDecorator( mtca4u::TransferElement & transferElement, DecoratorType decoratorType){
414 
415  // check if there already is a decorator for the transfer element
416  auto decoratorMapEntry = getGlobalDecoratorMap().find( transferElement.getHighLevelImplElement() );
417  if (decoratorMapEntry != getGlobalDecoratorMap().end()){
418  // There already is a decorator for this transfer element
419  // The decorator has to have a matching type, otherwise we can only throw
420  auto castedType = boost::dynamic_pointer_cast< DecoratorTypeAccessor< UserType > >(decoratorMapEntry->second);
421  if (castedType){// User type matches, but the decorator type also has to match
422  if (castedType->getDecoratorType() == decoratorType){
423  // decorator matches, return it
424  return castedType;
425  }else{
426  // sorry there is a decorator, but it's the wrong user type
427  throw std::logic_error("ChimeraTK::ControlSystemAdapter: Decorator for TransferElement " +
428  transferElement.getName() +" already exists as a different decorator type.");
429  }
430  }else{
431  // sorry there is a decorator, but it's the wrong user type
432  throw std::logic_error("ChimeraTK::ControlSystemAdapter: Decorator for TransferElement " +
433  transferElement.getName() +" already exists with a different user type.");
434  }
435  }
436 
437  DecoratorFactory< UserType > factory( transferElement.getHighLevelImplElement(), decoratorType );
438  boost::fusion::for_each(mtca4u::userTypeMap(), factory);
439  getGlobalDecoratorMap()[transferElement.getHighLevelImplElement()]=factory.createdDecorator;
440  return factory.createdDecorator;
441  }
