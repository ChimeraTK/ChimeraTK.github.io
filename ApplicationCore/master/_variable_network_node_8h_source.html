242  template<typename UserType>
243  VariableNetworkNode VariableNetworkNode::makeConstant(bool makeFeeder, UserType value, size_t length) {
244  VariableNetworkNode node;
245  node.pdata = boost::make_shared<VariableNetworkNode_data>();
246  node.pdata->constNode.reset(new ConstantAccessor<UserType>(value, length));
247  node.pdata->type = NodeType::Constant;
248  node.pdata->valueType = &typeid(UserType);
249  node.pdata->nElements = length;
250  node.pdata->name = "*UNNAMED CONSTANT*";
251  if(makeFeeder) {
252  node.pdata->direction = VariableDirection::feeding;
253  node.pdata->mode = UpdateMode::push;
254  }
255  else {
256  node.pdata->direction = VariableDirection::consuming;
257  node.pdata->mode = UpdateMode::poll;
258  }
259  return node;
260  }
