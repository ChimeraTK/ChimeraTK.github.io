36  void VariableNetwork::addNode(VariableNetworkNode &a) {
37  assert(!a.hasOwner());
38 
39  // change owner of the node: erase from Application's unconnectedNodeList and set this as owner
40  a.setOwner(this);
41 
42  // if node is feeding, save as feeder for this network
43  if(a.getDirection() == VariableDirection::feeding) {
44  // make sure we only have one feeding node per network
45  if(hasFeedingNode()) {
46  std::stringstream msg;
47  msg << "Trying to add a feeding accessor to a network already having a feeding accessor." << std::endl;
48  msg << "The network you were trying to add the new accessor to:" << std::endl;
49  dump("", msg);
50  msg << "The node you were trying to add:" << std::endl;
51  a.dump(msg);
52  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalVariableNetwork>(msg.str());
53  }
54  // force value type, engineering unit and description of the network if set in this feeding node
55  if(a.getValueType() != typeid(AnyType)) valueType = &(a.getValueType());
56  if(a.getUnit() != mtca4u::TransferElement::unitNotSet) engineeringUnit = a.getUnit();
57  if(a.getDescription() != "") description = a.getDescription();
58  }
59  else {
60  // update value type and engineering unit, if not yet set
61  if(valueType == &typeid(AnyType)) valueType = &(a.getValueType());
62  if(engineeringUnit == mtca4u::TransferElement::unitNotSet) engineeringUnit = a.getUnit();
63  if(description == "") description = a.getDescription();
64  }
65  // add node to node list
66  nodeList.push_back(a);
67  }

117  if(consumer.getDirection() != VariableDirection::consuming) continue;
118  stream << linePrefix << " # " << ++count << ":";

120  }
121  if(hasFeedingNode()) {

140  VariableNetwork::TriggerType VariableNetwork::getTriggerType(bool verboseExceptions) const {
141  if(!hasFeedingNode()) return TriggerType::none;
142  const auto &feeder = getFeedingNode();
143  // network has an external trigger
144  if(feeder.hasExternalTrigger()) {
145  if(feeder.getMode() == UpdateMode::push) {
146  std::stringstream msg;
147  msg << "Providing an external trigger to a variable network which is fed by a pushing variable is not allowed." << std::endl;
148  if(verboseExceptions) {
149  msg << "The illegal network:" << std::endl;
150  dump("", msg);
151  }
152  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalVariableNetwork>(msg.str());
153  }
154  return TriggerType::external;
155  }
156  // network is fed by a pushing node
157  if(feeder.getMode() == UpdateMode::push) {
158  return TriggerType::feeder;
159  }
160  // network is fed by a poll-type node: must have exactly one polling consumer
161  size_t nPollingConsumers = count_if( nodeList.begin(), nodeList.end(),
162  [](const VariableNetworkNode &n) {
163  return n.getDirection() == VariableDirection::consuming && n.getMode() == UpdateMode::poll;
164  } );
165  if(nPollingConsumers != 1) {
166  std::stringstream msg;
167  msg << "In a network with a poll-type feeder and no external trigger, there must be exactly one polling consumer. Maybe you forgot to add a trigger?" << std::endl;
168  if(verboseExceptions) {
169  msg << "The illegal network:" << std::endl;
170  dump("", msg);
171  }
172  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalVariableNetwork>(msg.str());
173  }
174  return TriggerType::pollingConsumer;
175  }

179  void VariableNetwork::check() {
180  // must have consuming nodes
181  if(countConsumingNodes() == 0) {
182  std::stringstream msg;
183  msg << "No consuming nodes connected to this network!" << std::endl;
184  msg << "The illegal network:" << std::endl;
185  dump("", msg);
186  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalVariableNetwork>(msg.str());
187  }
188 
189  // must have a feeding node
190  if(!hasFeedingNode()) {
191  std::stringstream msg;
192  msg << "No feeding node connected to this network!" << std::endl;
193  msg << "The illegal network:" << std::endl;
194  dump("", msg);
195  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalVariableNetwork>(msg.str());
196  }
197 
198  // the network's value type must be correctly set
199  if(*valueType == typeid(AnyType)) {
200  std::stringstream msg;
201  msg << "No data type specified for any of the nodes in this network!" << std::endl;
202  msg << "The illegal network:" << std::endl;
203  dump("", msg);
204  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalVariableNetwork>(msg.str());
205  }
206 
207  // the feeder node must have a non-zero length
208  size_t length = getFeedingNode().getNumberOfElements();
209  if(length == 0) {
210  std::stringstream msg;
211  msg << "The feeding node has zero (or undefined) length!" << std::endl;
212  msg << "The illegal network:" << std::endl;
213  dump("", msg);
214  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalVariableNetwork>(msg.str());
215  }
216 
217  // all consumers must have the same length as the feeder or a zero length for trigger receivers
218  for(auto &node : nodeList) {
219  if(node.getType() != NodeType::TriggerReceiver) {
220  if(node.getNumberOfElements() != length) {
221  std::stringstream msg;
222  msg << "The network contains a node with a different length than the feeding node!" << std::endl;
223  msg << "The illegal network:" << std::endl;
224  dump("", msg);
225  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalVariableNetwork>(msg.str());
226  }
227  }
228  else {
229  if(node.getNumberOfElements() != 0) {
230  std::stringstream msg;
231  msg << "The network contains a trigger receiver node with a non-zero length!" << std::endl;
232  msg << "The illegal network:" << std::endl;
233  dump("", msg);
234  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalVariableNetwork>(msg.str());
235  }
236  }
237  }
238 
239  // all nodes must have this network as the owner and a value type equal the network's value type
240  for(auto &node : nodeList) {
241  assert(&(node.getOwner()) == this);
242  if(node.getValueType() == typeid(AnyType)) node.setValueType(*valueType);
243  if(node.getValueType() != *valueType) {
244  std::stringstream msg;
245  msg << "The network contains variables of different value types, which is not supported!" << std::endl;
246  msg << "The illegal network:" << std::endl;
247  dump("", msg);
248  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalVariableNetwork>(msg.str());
249  }
250  }
251 
252  // if the feeder is an application node, it must be in push mode
253  if(getFeedingNode().getType() == NodeType::Application) {
254  assert(getFeedingNode().getMode() == UpdateMode::push);
255  }
256 
257  // check if trigger is correctly defined (the return type doesn't matter, only the checks done in the function are needed)
258  getTriggerType();
259  }
