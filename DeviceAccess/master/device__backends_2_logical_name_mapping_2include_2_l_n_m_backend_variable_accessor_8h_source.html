28  LNMBackendVariableAccessor(boost::shared_ptr<DeviceBackend> dev, const RegisterPath &registerPathName,
29  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
30  : SyncNDRegisterAccessor<UserType>(registerPathName),
31  _registerPathName(registerPathName),
32  _wordOffsetInRegister(wordOffsetInRegister),
33  _fixedPointConverter(registerPathName, 32, 0, 1)
34  {
35  try {
36  // cast device
37  _dev = boost::dynamic_pointer_cast<LogicalNameMappingBackend>(dev);
38  // check for unknown flags
39  flags.checkForUnknownFlags({AccessMode::raw});
40  // obtain the register info
41  _info = boost::static_pointer_cast<LNMBackendRegisterInfo>(
42  _dev->getRegisterCatalogue().getRegister(_registerPathName));
43  if(numberOfWords == 0) numberOfWords = _info->length;
44  // check for illegal parameter combinations
45  if(wordOffsetInRegister+numberOfWords > _info->length) {
46  throw ChimeraTK::logic_error("Requested number of words and/or offset exceeds length of register '"
47  +registerPathName+"'.");
48  }
49  if(flags.has(AccessMode::raw)) {
50  if(typeid(UserType) != typeid(int32_t)) {
51  throw ChimeraTK::logic_error("Given UserType when obtaining the LNMBackendBufferingVariableAccessor in"
52  " raw mode does not match the expected type. Use an int32_t instead!");
53  }
54  std::cout << "WARNING: You are using AccessMode::raw on a variable/constant-type register in a logical "
55  "name mapping device. This is DEPRECATED and support for it will be removed soon!" << std::endl;
56  }
57  // check for incorrect usage of this accessor
58  if( _info->targetType != LNMBackendRegisterInfo::TargetType::INT_CONSTANT &&
59  _info->targetType != LNMBackendRegisterInfo::TargetType::INT_VARIABLE ) {
60  throw ChimeraTK::logic_error("LNMBackendVariableAccessor used for wrong register type."); // LCOV_EXCL_LINE (impossible to test...)
61  }
62  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
63  NDRegisterAccessor<UserType>::buffer_2D[0].resize(numberOfWords);
64  doPostRead();
65  }
66  catch(...) {
67  this->shutdown();
68  throw;
69  }
70  }

85  bool doWriteTransfer(ChimeraTK::VersionNumber versionNumber={}) override {
86  if(isReadOnly()) {
87  throw ChimeraTK::logic_error("Writing to constant-type registers of logical name mapping devices is not possible.");
88  }
89  for(size_t i=0; i < NDRegisterAccessor<UserType>::buffer_2D[0].size(); ++i) {
90  auto &v = _info->value_int[i+_wordOffsetInRegister];
91  v = _fixedPointConverter.toRaw(NDRegisterAccessor<UserType>::buffer_2D[0][i]);
92  }
93  currentVersion = versionNumber;
94  return false;
95  }
