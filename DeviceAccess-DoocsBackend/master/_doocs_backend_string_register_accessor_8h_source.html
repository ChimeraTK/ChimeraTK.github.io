42  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
43  : DoocsBackendRegisterAccessor<UserType>(
44  backend, path, registerPathName, numberOfWords, wordOffsetInRegister, flags, false) {
45  try {
46  // set buffer size (nElements will be the number of characters, so the
47  // buffer allocation in the base class would be incorrect)
48  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
49  NDRegisterAccessor<UserType>::buffer_2D[0].resize(1);
50 
51  // check UserType
52  if(typeid(UserType) != typeid(std::string)) {
53  throw ChimeraTK::logic_error("Trying to access a string DOOCS property "
54  "with a non-string user data type.");
55  }
56 
57  // initialise fully only if backend is open
58  if(backend->isOpen()) {
59  this->initialise();
60  }
61  }
62  catch(...) {
63  this->shutdown();
64  throw;
65  }
66  }
