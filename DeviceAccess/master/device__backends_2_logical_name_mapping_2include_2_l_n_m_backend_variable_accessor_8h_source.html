24  template<typename UserType>
25  class LNMBackendVariableAccessor : public SyncNDRegisterAccessor<UserType> {
26  public:
27  LNMBackendVariableAccessor(boost::shared_ptr<DeviceBackend> dev, const RegisterPath& registerPathName,
28  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
29  : SyncNDRegisterAccessor<UserType>(registerPathName), _registerPathName(registerPathName),
30  _wordOffsetInRegister(wordOffsetInRegister), _fixedPointConverter(registerPathName, 32, 0, 1) {
31  try {
32  // cast device
33  _dev = boost::dynamic_pointer_cast<LogicalNameMappingBackend>(dev);
34  // check for unknown flags
35  flags.checkForUnknownFlags({AccessMode::raw});
36  // obtain the register info
37  _info = boost::static_pointer_cast<LNMBackendRegisterInfo>(
38  _dev->getRegisterCatalogue().getRegister(_registerPathName));
39  if(numberOfWords == 0) numberOfWords = _info->length;
40  // check for illegal parameter combinations
41  if(wordOffsetInRegister + numberOfWords > _info->length) {
42  throw ChimeraTK::logic_error("Requested number of words and/or offset "
43  "exceeds length of register '" +
44  registerPathName + "'.");
45  }
46  if(flags.has(AccessMode::raw)) {
47  if(typeid(UserType) != typeid(int32_t)) {
48  throw ChimeraTK::logic_error("Given UserType when obtaining the "
49  "LNMBackendBufferingVariableAccessor in"
50  " raw mode does not match the expected "
51  "type. Use an int32_t instead!");
52  }
53  std::cout << "WARNING: You are using AccessMode::raw on a "
54  "variable/constant-type register in a logical "
55  "name mapping device. This is DEPRECATED and support for "
56  "it will be removed soon!"
57  << std::endl;
58  }
59  // check for incorrect usage of this accessor
60  if(_info->targetType != LNMBackendRegisterInfo::TargetType::INT_CONSTANT &&
61  _info->targetType != LNMBackendRegisterInfo::TargetType::INT_VARIABLE) {
62  throw ChimeraTK::logic_error("LNMBackendVariableAccessor used for "
63  "wrong register type."); // LCOV_EXCL_LINE
64  // (impossible to
65  // test...)
66  }
67  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
68  NDRegisterAccessor<UserType>::buffer_2D[0].resize(numberOfWords);
69  doPostRead();
70  }
71  catch(...) {
72  this->shutdown();
73  throw;
74  }
75  }
76 
77  virtual ~LNMBackendVariableAccessor() override { this->shutdown(); }
78 
79  void doReadTransfer() override {}
80 
81  bool doReadTransferNonBlocking() override { return true; }
82 
83  bool doReadTransferLatest() override { return true; }
84 
85  bool doWriteTransfer(ChimeraTK::VersionNumber versionNumber = {}) override {
86  if(isReadOnly()) {
87  throw ChimeraTK::logic_error("Writing to constant-type registers of logical name mapping devices "
88  "is not possible.");
89  }
90  for(size_t i = 0; i < NDRegisterAccessor<UserType>::buffer_2D[0].size(); ++i) {
91  auto& v = _info->value_int[i + _wordOffsetInRegister];
92  v = _fixedPointConverter.toRaw(NDRegisterAccessor<UserType>::buffer_2D[0][i]);
93  }
94  currentVersion = versionNumber;
95  return false;
96  }
97 
98  bool mayReplaceOther(const boost::shared_ptr<TransferElement const>&) const override {
99  return false; // never replace, since it does not optimise anything
100  }
101 
102  bool isReadOnly() const override { return _info->targetType == LNMBackendRegisterInfo::TargetType::INT_CONSTANT; }
103 
104  bool isReadable() const override { return true; }
105 
106  bool isWriteable() const override { return _info->targetType != LNMBackendRegisterInfo::TargetType::INT_CONSTANT; }
107 
108  void doPostRead() override {
109  for(size_t i = 0; i < NDRegisterAccessor<UserType>::buffer_2D[0].size(); ++i) {
110  auto& v = _info->value_int[i + _wordOffsetInRegister];
111  NDRegisterAccessor<UserType>::buffer_2D[0][i] = _fixedPointConverter.toCooked<UserType>(v);
112  }
113  currentVersion = {};
114  }
115 
116  AccessModeFlags getAccessModeFlags() const override { return {}; }
117 
118  VersionNumber getVersionNumber() const override { return currentVersion; }
119 
120  protected:
122  RegisterPath _registerPathName;
123 
125  boost::shared_ptr<LogicalNameMappingBackend> _dev;
126 
130  boost::shared_ptr<LNMBackendRegisterInfo> _info;
131 
133  size_t _wordOffsetInRegister;
134

139 
140  std::vector<boost::shared_ptr<TransferElement>> getHardwareAccessingElements() override {
141  return {boost::enable_shared_from_this<TransferElement>::shared_from_this()};
142  }
143 
144  std::list<boost::shared_ptr<TransferElement>> getInternalElements() override { return {}; }
145 
146  void replaceTransferElement(boost::shared_ptr<TransferElement> /*newElement*/) override {} // LCOV_EXCL_LINE
147 
148  // Version number of the last transfer
149  VersionNumber currentVersion;
150  };
