427  inline TransferElement::ID TransferElement::readAny(std::list<std::reference_wrapper<TransferElement>> elementsToRead) {
428  std::list<TransferFuture*> futureList;
429  for(auto &elem : elementsToRead) {
430  auto *future = &(elem.get().readAsync());
431  futureList.push_back(future);
432  }
433  // wait until any future is ready
434  auto iter = boost::wait_for_any(TransferFutureIterator(futureList.begin()), TransferFutureIterator(futureList.end()));
435 
436  // Find the variable which has the oldest version number (to guarantee the order of updates).
437  // Start with assuming that the future returned by boost::wait_for_any() has the oldes version number.
438  TransferFuture *theUpdate = &(iter.getTransferFuture());
439  for(auto future : futureList) {
440  // skip if this is the future returned by boost::wait_for_any()
441  if(future == theUpdate) continue;
442  // also skip if the future is not yet ready
443  if(future->getBoostFuture().wait_for(boost::chrono::duration<int, boost::centi>(0)) == boost::future_status::timeout) {
444  continue;
445  }
446  // compare version number with the version number of the stored future
447  if(future->getBoostFuture().get()->_versionNumber < theUpdate->getBoostFuture().get()->_versionNumber) {
448  theUpdate = future;
449  }
450  }
451 
452  // complete the transfer (i.e. run postRead())
453  theUpdate->wait();
454 
455  // return the transfer element as a shared pointer
456  return theUpdate->getTransferElement().getId();
457  }
