194 bool comparHelper(UserType a, UserType b) {
195  bool isFloat = std::is_floating_point<decltype(a)>::value;

197  return std::abs(a - b) < (std::max(a, b) / 1e6);

199  else {
200  return a == b;

215 // Helper macro to compare the value on an accessor and the expected 2D value
216 // Note: we use a macro and not a function, so BOOST_ERROR prints us the line number of the actual test!
217 #define CHECK_EQUALITY(accessor, expectedValue) \
218  { \
219  std::string fail = ""; \
220  BOOST_REQUIRE(accessor.getNChannels() == expectedValue.size()); \
221  BOOST_REQUIRE(accessor.getNElementsPerChannel() == expectedValue[0].size()); \
222  for(size_t i = 0; i < expectedValue.size(); ++i) { \
223  for(size_t k = 0; k < expectedValue.size(); ++k) { \
224  if(!comparHelper(accessor[i][k], expectedValue[i][k])) { \
225  if(fail.size() == 0) { \
226  fail = "Accessor content differs from expected value. First difference at index [" + std::to_string(i) + \
227  "][" + std::to_string(k) + "]: " + std::to_string(accessor[i][k]) + \
228  " != " + std::to_string(expectedValue[i][k]); \
229  } \
230  } \
231  } \
232  } \
233  if(fail != "") { \
234  BOOST_ERROR(fail); \
235  } \
236  } \
