151  template<typename UserType>
152  DoocsBackendRegisterAccessor<UserType>::DoocsBackendRegisterAccessor(const std::string &path, size_t numberOfWords,
153  size_t wordOffsetInRegister, AccessModeFlags flags, bool allocateBuffers)
154  : NDRegisterAccessor<UserType>(path),
155  _path(path),
156  elementOffset(wordOffsetInRegister),
157  useZMQ(false)
158  {
159  try {
160  // check for unknown access mode flags
161  flags.checkForUnknownFlags({AccessMode::wait_for_new_data});
162 
163  // set address
164  ea.adr(std::string(path).c_str());
165 
166  // try to read data, to check connectivity and to obtain size of the register
167  doReadTransfer();
168 
169  // obtain number of elements
170  size_t actualLength = dst.array_length();
171  if(actualLength == 0 && dst.length() == 1) {
172  actualLength = 1;
173  isArray = false;
174  }
175  else {
176  if (actualLength == 0)
177  actualLength = dst.length();
178  isArray = true;
179  }
180  if(numberOfWords == 0) {
181  nElements = actualLength;
182  }
183  else {
184  nElements = numberOfWords;
185  }
186  if(nElements + elementOffset > actualLength) {
187  throw ChimeraTK::logic_error("Requested number of words exceeds the length of the DOOCS property!");
188  }
189  if(nElements == actualLength && elementOffset == 0) {
190  isPartial = false;
191  }
192  else {
193  isPartial = true;
194  }
195 
196  // allocate buffers
197  if(allocateBuffers) {
198  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
199  NDRegisterAccessor<UserType>::buffer_2D[0].resize(nElements);
200  }
201 
202  // set proper type information in the source EqData
203  src.set_type(dst.type());
204  if(allocateBuffers && dst.type() != DATA_IIII) {
205  src.length(actualLength);
206  }
207 
208  // use ZeroMQ with AccessMode::wait_for_new_data
209  if(flags.has(AccessMode::wait_for_new_data)) {
210  // set flag
211  useZMQ = true;
212 
213  // create notification queue and TransferFuture
214  notifications = cppext::future_queue<EqData>(3);
215  activeFuture = TransferFuture(notifications.then<void>([](EqData&){}, std::launch::deferred), this);
216  futureCreated = true;
217 
218  // subscribe to property
219  int err = dmsg_attach(&ea, &dst, (void*)this, &zmq_callback);
220  if(err) {
221  throw ChimeraTK::runtime_error(std::string("Cannot subscribe to DOOCS property via ZeroMQ: ")+dst.get_string());
222  }
223  // run dmsg_start() once
224  std::unique_lock<std::mutex> lck(DoocsBackend::dmsgStartCalled_mutex);
225  if(!DoocsBackend::dmsgStartCalled) {
226  dmsg_start();
227  DoocsBackend::dmsgStartCalled = true;
228  }
229  }
230  }
231  catch(...) {
232  this->shutdown();
233  throw;
234  }
235  }

246  template<typename UserType>
247  bool DoocsBackendRegisterAccessor<UserType>::doReadTransferNonBlocking() {
248  if(!useZMQ) {
249  this->doReadTransfer();
250  return true;
251  }
252  else {
253  // loop to allow retries in case of timeout errors
254  while(true) {
255  // wait until new data has been received
256  bool gotData = notifications.pop(dst);
257  if(!gotData) return false;
258  // check for an error
259  if(dst.error() != 0) {
260  // try obtaining data through RPC call instead to verify error
261  int rc = eq.get(&ea, &src, &dst);
262  // if again error received, throw exception
263  if(rc) {
264  throw ChimeraTK::runtime_error(std::string("Cannot read from DOOCS property: ")+dst.get_string());
265  }
266  // otherwise try again
267  continue;
268  }
269  // we have received data, so return
270  return true;
271  }
272  }
273  }

292  template<typename UserType>
293  void DoocsBackendRegisterAccessor<UserType>::doReadTransfer() {
294  boost::this_thread::interruption_point();
295  if(!useZMQ) {
296  // read data
297  int rc = eq.get(&ea, &src, &dst);
298  // check error
299  if(rc) {
300  throw ChimeraTK::runtime_error(std::string("Cannot read from DOOCS property: ")+dst.get_string());
301  }
302  }
303  else {
304  // loop to allow retries in case of timeout errors
305  while(true) {
306  // wait until new data has been received
307  notifications.pop_wait(dst);
308  // check for an error
309  if(dst.error() != 0) {
310  // try obtaining data through RPC call instead to verify error
311  int rc = eq.get(&ea, &src, &dst);
312  // if again error received, throw exception
313  if(rc) {
314  throw ChimeraTK::runtime_error(std::string("Cannot read from DOOCS property: ")+dst.get_string());
315  }
316  // otherwise try again
317  continue;
318  }
319  // terminate loop, since no retry needed
320  break;
321  }
322  }
323  // check for error code in dst
324  if(dst.error() != 0) {
325  throw ChimeraTK::runtime_error(std::string("Cannot read from DOOCS property: ")+dst.get_string());
326  }

356  template<typename UserType>
357  TransferFuture DoocsBackendRegisterAccessor<UserType>::doReadTransferAsync() {
358  // create future_queue if not already created and continue it to enusre postRead is called (in the user thread,
359  // so we use the deferred launch policy)
360  if(!futureCreated) {
361  notifications = cppext::future_queue<EqData>(2);
362  activeFuture = TransferFuture(notifications.then<void>([](EqData&){}), this);
363  futureCreated = true;
364  }
365 
366  if(!useZMQ) {
367  // launch doReadTransfer in separate thread
368  readAsyncThread = boost::thread(
369  [this] {
370  try {
371  this->doReadTransfer();
372  }
373  catch(...) {
374  this->notifications.push_exception(std::current_exception());
375  throw;
376  }
377  this->notifications.push({});
378  }
379  );
380  }
381 
382  // return the TransferFuture
383  return activeFuture;
