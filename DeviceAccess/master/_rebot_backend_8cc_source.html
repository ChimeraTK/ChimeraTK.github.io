102  uint32_t RebotBackend::getServerProtocolVersion() {
103  // send a negotiation to the server:
104  // sendClientProtocolVersion
105  _lastSendTime = testable_rebot_sleep::now();
106  std::vector<uint32_t> clientHelloMessage = frameClientHello();
107  _tcpCommunicator->sendData(clientHelloMessage);
108 
109  // Kludge is needed to work around server bug.
110  // We have a bug with the old version were only one word is returned for
111  // multiple unrecognized command. Fetching one word for the 3 words send is a
112  // workaround.
113  auto serverHello = _tcpCommunicator->receiveData(1);
114 
115  if(serverHello.at(0) == rebot::UNKNOWN_INSTRUCTION) {
116  return 0; // initial protocol version 0.0
117  }
118 
119  auto remainingBytesOfAValidServerHello = _tcpCommunicator->receiveData(rebot::LENGTH_OF_HELLO_TOKEN_MESSAGE - 1);
120 
121  serverHello.insert(
122  serverHello.end(), remainingBytesOfAValidServerHello.begin(), remainingBytesOfAValidServerHello.end());
123  return parseRxServerHello(serverHello);
124  }

139  void RebotBackend::heartbeatLoop(boost::shared_ptr<ThreadInformerMutex> threadInformerMutex) {
140  while(true) {
141  try {
142  // only send a heartbeat if the connection was inactive for half of the
143  // timeout period
144  if((testable_rebot_sleep::now() - _lastSendTime) > boost::chrono::milliseconds(_connectionTimeout / 2)) {
145  // scope for the lock guard is in the if statement
146  std::lock_guard<std::mutex> lock(_threadInformerMutex->mutex);
147  // To handle the race condition that the thread woke up while the
148  // desructor was holding the lock and closes the socket: Check the flag
149  // and quit if set
150  if(threadInformerMutex->quitThread) {
151  break;
152  }
153  // always update the last send time. Otherwise the sleep will be
154  // ineffective for a closed connection and go to 100 & CPU load
155  _lastSendTime = testable_rebot_sleep::now();
156  if(_protocolImplementor) {
157  _protocolImplementor->sendHeartbeat();
158  }
159  }
160  // sleep without holding the lock. Sleep for half of the connection
161  // timeout (plus 1 ms)
162  testable_rebot_sleep::sleep_until(_lastSendTime + boost::chrono::milliseconds(_connectionTimeout / 2 + 1));
163  }
164  catch(ChimeraTK::runtime_error& e) {
165  std::cerr << "RebotBackend: Sending heartbeat failed. Caught exception: " << e.what() << std::endl;
166  std::cerr << "Closing connection." << std::endl;
167  close();
168  }
169  } // while true
170  }
