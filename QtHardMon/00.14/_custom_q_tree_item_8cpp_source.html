94 std::vector<int> RegisterItem::fetchElementsFromCard(
95  const TableWidgetData& tabledata) {
96  RegisterInfo_t regInfo = this->getRegisterMapElement();
97 
98  int numberOfElements = regInfo.reg_elem_nr;
99  std::vector<int> buffer(numberOfElements);
100 
101  unsigned int registerBar = regInfo.reg_bar;
102  Device_t const& mtcadevice = tabledata.device;
103  int maxrows = tabledata.tableMaxRowCount;
104  size_t nBytesToRead =
105  std::min(numberOfElements, maxrows) * qthardmon::WORD_SIZE_IN_BYTES;
106  unsigned int registerAddress = regInfo.reg_address;
107 
108  if (registerBar == 0xD) {
109  mtcadevice->readDMA(registerAddress, &(buffer[0]), nBytesToRead,
110  registerBar);
111  } else {
112  mtcadevice->readArea(registerAddress, &(buffer[0]), nBytesToRead,
113  registerBar);
114  }
115  return buffer;
116 }
