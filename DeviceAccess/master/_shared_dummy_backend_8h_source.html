189  if(getInstanceMap().find(instanceId) == getInstanceMap().end()) {
190  boost::shared_ptr<DeviceBackend> ptr(new T(std::forward<Args>(arguments)...));
191  getInstanceMap().insert(std::make_pair(instanceId, ptr));
192  return ptr;
