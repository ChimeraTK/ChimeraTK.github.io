21  void TransferGroup::read() {
22  std::stringstream combinedMessage;
23  bool hadException{false};
24 
25  for(auto& elem : highLevelElements) {
26  elem->preRead(TransferType::read);
27  }
28 
29  for(auto& elem : copyDecorators) {
30  elem->preRead(TransferType::read);
31  }
32 
33  for(auto& elem : lowLevelElements) {
34  elem->preRead(TransferType::read);
35  }
36 
37  for(auto& elem : lowLevelElements) {
38  elem->readTransfer();
39  }
40 
41  // check for any "pending" exceptions
42  for(auto& elem : lowLevelElements) {
43  if(elem->hasSeenException) hadException = true;
44  }
45 
46  for(auto& elem : lowLevelElements) {
47  try {
48  elem->postRead(TransferType::read, !hadException);
49  }
50  catch(runtime_error& ex) {
51  assert(hadException); // must not receive unexpected exception
52  combinedMessage << ex.what() << "\n";
53  }
54  }
55 
56  for(auto& elem : copyDecorators) {
57  assert(!elem->hasSeenException); // no readTransfer(), no exception
58  try {
59  elem->postRead(TransferType::read, !hadException);
60  }
61  catch(runtime_error& ex) {
62  assert(hadException); // must not receive unexpected exception
63  combinedMessage << ex.what() << "\n";
64  }
65  }
66 
67  for(auto& elem : highLevelElements) {
68  assert(!elem->hasSeenException); // no readTransfer(), no exception
69  try {
70  elem->postRead(TransferType::read, !hadException);
71  }
72  catch(runtime_error& ex) {
73  assert(hadException); // must not receive unexpected exception
74  combinedMessage << ex.what() << "\n";
75  }
76  }
77 
78  if(hadException) throw runtime_error(combinedMessage.str());
79  }

104  void TransferGroup::addAccessor(TransferElementAbstractor& accessor) {
105  // check if accessor is already in a transfer group
106  if(accessor.getHighLevelImplElement()->isInTransferGroup) {
107  throw ChimeraTK::logic_error("The given accessor is already in a TransferGroup and cannot be added "
108  "to another.");
109  }
110 
111  // set flag on the accessors that it is now in a transfer group
112  accessor.getHighLevelImplElement()->isInTransferGroup = true;
113 
114  auto highLevelElementsWithNewAccessor = highLevelElements;
115  highLevelElementsWithNewAccessor.insert(accessor.getHighLevelImplElement());
116 
117  // try replacing all internal elements in all high-level elements
118  for(auto& hlElem1 : highLevelElementsWithNewAccessor) {
119  auto list = hlElem1->getInternalElements();
120  list.push_front(hlElem1);
121 
122  for(auto& replacement : list) {
123  // try on the abstractor first, to make sure we replace at the highest
124  // level if possible
125  accessor.replaceTransferElement(replacement);
126  // try on all high-level elements already stored in the list
127  for(auto& hlElem : highLevelElementsWithNewAccessor) {
128  hlElem->replaceTransferElement(replacement); // note: this does nothing, if the replacement cannot
129  // be used by the hlElem!
130  }
131  }
132  }
133 
134  // store the accessor in the list of high-level elements
135  // this must be done only now, since it may have been replaced during the
136  // replacement process
137  highLevelElements.insert(accessor.getHighLevelImplElement());
138 
139  // update the list of hardware-accessing elements, since we might just have
140  // made some of them redundant since we are using a set to store the elements,
141  // duplicates are intrinsically avoided.
142  lowLevelElements.clear();
143  for(auto& hlElem : highLevelElements) {
144  for(auto& hwElem : hlElem->getHardwareAccessingElements()) lowLevelElements.insert(hwElem);
145  }
146 
147  // update the list of CopyRegisterDecorators
148  copyDecorators.clear();
149  for(auto& hlElem : highLevelElements) {
150  if(boost::dynamic_pointer_cast<ChimeraTK::CopyRegisterDecoratorTrait>(hlElem) != nullptr) {
151  copyDecorators.insert(hlElem);
152  }
153  for(auto& hwElem : hlElem->getInternalElements()) {
154  if(boost::dynamic_pointer_cast<ChimeraTK::CopyRegisterDecoratorTrait>(hwElem) != nullptr) {
155  copyDecorators.insert(hwElem);
156  }
157  }
158  }
159 
160  // Update read-only flag
161  if(accessor.isReadOnly()) readOnly = true;
162  }
