27  LNMBackendChannelAccessor(boost::shared_ptr<DeviceBackend> dev, const RegisterPath &registerPathName,
28  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
29  : SyncNDRegisterAccessor<UserType>(registerPathName),
30  _registerPathName(registerPathName)
31  {
32  try {
33  // check for unknown flags
34  flags.checkForUnknownFlags({AccessMode::raw});
35  // check for illegal parameter combinations
36  if(flags.has(AccessMode::raw)) {
37  throw DeviceException("LNMBackendChannelAccessor: raw access not yet supported!", DeviceException::NOT_IMPLEMENTED);
38  }
39  _dev = boost::dynamic_pointer_cast<LogicalNameMappingBackend>(dev);
40  // copy the register info and create the internal accessors, if needed
41  _info = *( boost::static_pointer_cast<LNMBackendRegisterInfo>(
42  _dev->getRegisterCatalogue().getRegister(_registerPathName)) );
43  _info.createInternalAccessors(dev);
44  // check for incorrect usage of this accessor
45  if( _info.targetType != LNMBackendRegisterInfo::TargetType::CHANNEL ) {
46  throw DeviceException("LNMBackendChannelAccessor used for wrong register type.",
47  DeviceException::WRONG_PARAMETER); // LCOV_EXCL_LINE (impossible to test...)
48  }
49  // get target device and accessor
50  std::string devName = _info.deviceName;
51  boost::shared_ptr<DeviceBackend> targetDevice;
52  if(devName != "this") {
53  targetDevice = _dev->_devices[devName];
54  }
55  else {
56  targetDevice = dev;
57  }
58  _accessor = targetDevice->getRegisterAccessor<UserType>(RegisterPath(_info.registerName), numberOfWords,wordOffsetInRegister, false);
59  // allocate the buffer
60  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
61  NDRegisterAccessor<UserType>::buffer_2D[0].resize(_accessor->getNumberOfSamples());
62  }
63  catch(...) {
64  this->shutdown();
65  throw;
66  }
67  }
