335  int StepperMotor::truncateMotorPosition(int newPosition) {
336  int position = newPosition;

351  StepperMotorStatusAndError StepperMotor::determineMotorStatusAndError() {
352 
353  //InternalMutex intMutex(this);
354  boost::lock_guard<boost::mutex> guard(_mutex);
355 
356  _motorStatus = StepperMotorStatusTypes::M_OK;
357  _motorError = StepperMotorErrorTypes::M_NO_ERROR;
358 
359  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
360  // ERROR CONDITION CHECK - it should be done first.
361  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
362 
363 
364  if (_softwareLimitsEnabled && (_maxPositionLimitInSteps <= _minPositionLimitInSteps)) {
365  _motorStatus = StepperMotorStatusTypes::M_ERROR;
366  _motorError = StepperMotorErrorTypes::M_CONFIG_ERROR_MIN_POS_GRATER_EQUAL_TO_MAX;
367 
368  return StepperMotorStatusAndError(_motorStatus, _motorError);
369  }
370 
371  //first check if disabled
372  if (!_motorControler->isEnabled()) {
373  _motorStatus = StepperMotorStatusTypes::M_DISABLED;
374  return StepperMotorStatusAndError(_motorStatus, _motorError);
375  }
376 
377  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
378  // STATUS CHECK
379  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
380 
381  if (_motorControler->getStatus().getStandstillIndicator() == 0) {
382  _motorStatus = StepperMotorStatusTypes::M_IN_MOVE;
383  return StepperMotorStatusAndError(_motorStatus, _motorError);
384  }
385 
386 
387  if (_targetPositionInSteps != _motorControler->getActualPosition()) {
388  _motorStatus = StepperMotorStatusTypes::M_NOT_IN_POSITION;
389  return StepperMotorStatusAndError(_motorStatus, _motorError);
390  }
391 
392  if (_softwareLimitsEnabled) {
393  if (_targetPositionInSteps >= _maxPositionLimitInSteps) {
394  _motorStatus = StepperMotorStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON;
395  return StepperMotorStatusAndError(_motorStatus, _motorError);
396  }
397 
398  if (_targetPositionInSteps <= _minPositionLimitInSteps) {
399  _motorStatus = StepperMotorStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON;
400  return StepperMotorStatusAndError(_motorStatus, _motorError);
401  }
402  }
403 
404  return StepperMotorStatusAndError(_motorStatus, _motorError);
405  }
