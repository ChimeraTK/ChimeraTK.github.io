128  const RegisterPath &registerPathName, size_t numberOfElements, size_t elementsOffset,
129  boost::shared_ptr<DeviceBackend> _backend )
130  : SyncNDRegisterAccessor<UserType>(registerPathName),
131  _ioDevice(boost::dynamic_pointer_cast<NumericAddressedBackend>(_backend)),
132  _registerPathName(registerPathName),
133  _numberOfElements(numberOfElements),
134  _elementsOffset(elementsOffset)
135  {
136  try {
137  // re-split register and module after merging names by the last dot (to allow module.register in the register name)
138  _registerPathName.setAltSeparator(".");
139  auto moduleAndRegister = MapFileParser::splitStringAtLastDot(_registerPathName.getWithAltSeparator());
140  _moduleName = moduleAndRegister.first;
141  _registerName = moduleAndRegister.second;
142 
143  // build name of area as written in the map file
144  std::string areaName = MULTIPLEXED_SEQUENCE_PREFIX+_registerName;
145 
146  // Obtain information about the area
147  auto registerMapping = _ioDevice->getRegisterMap();
148  RegisterInfoMap::RegisterInfo areaInfo;
149  registerMapping->getRegisterInfo(areaName, areaInfo, _moduleName);
150 
151  // Obtain information for each sequence (= channel) in the area:
152  // Create a fixed point converter for each sequence and store the sequence information in a vector
153  size_t iSeq = 0;
154  while(true) {
155 
156  // build name of the next sequence as written in the map file
157  RegisterInfoMap::RegisterInfo sequenceInfo;
158  std::stringstream sequenceNameStream;
159  sequenceNameStream << SEQUENCE_PREFIX << _registerName << "_" << iSeq++;
160 
161  // try to find sequence
162  try {
163  registerMapping->getRegisterInfo(sequenceNameStream.str(), sequenceInfo, _moduleName);
164  }
165  catch(MapFileException &) {
166  // no sequence found: we are done
167  break;
168  }
169 
170  // check consistency
171  if(sequenceInfo.nElements != 1) {
172  throw MultiplexedDataAccessorException( "Sequence words must have exactly one element",
173  MultiplexedDataAccessorException::INVALID_N_ELEMENTS );
174  }
175 
176  // store sequence info and fixed point converter
177  _sequenceInfos.push_back(sequenceInfo);
178  _converters.push_back( FixedPointConverter(registerPathName, sequenceInfo.width, sequenceInfo.nFractionalBits, sequenceInfo.signedFlag) );
179  }
180 
181  // check if no sequences were found
182  if(_converters.empty()){
183  throw MultiplexedDataAccessorException( "No sequences found for name \""+_registerName+"\".",
184  MultiplexedDataAccessorException::EMPTY_AREA );
185  }
186 
187  // compute size of one block in bytes (one sample for all channels)
188  bytesPerBlock = 0;
189  for(unsigned int i=0; i<_converters.size(); i++) {
190  uint32_t nbt = _sequenceInfos[i].nBytes;
191  bytesPerBlock += nbt;
192  if(nbt != 1 && nbt != 2 && nbt != 4) {
193  throw MultiplexedDataAccessorException( "Sequence word size must correspond to a primitive type",
194  MultiplexedDataAccessorException::INVALID_WORD_SIZE );
195  }
196  }
197 
198  // compute number of blocks (number of samples for each channel)
199  _nBlocks = std::floor(areaInfo.nBytes / bytesPerBlock);
200 
201  // check number of words
202  if(_elementsOffset >= _nBlocks) {
203  throw DeviceException("Requested element offset exceeds the size of the register!",
204  DeviceException::WRONG_PARAMETER);
205  }
206  if(_numberOfElements == 0) {
207  _numberOfElements = _nBlocks - _elementsOffset;
208  }
209  if(_numberOfElements + _elementsOffset > _nBlocks) {
210  throw DeviceException("Requested number of elements exceeds the size of the register!",
211  DeviceException::WRONG_PARAMETER);
212  }
213 
214  // compute the address taking into account the selected area of interest
215  _bar = areaInfo.bar;
216  _address = areaInfo.address + bytesPerBlock*_elementsOffset;
217  _nBytes = bytesPerBlock * _numberOfElements;
218  if(_nBytes % sizeof(int32_t) > 0) _nBytes += 4 - _nBytes % sizeof(int32_t); // round up to the next multiple of 4
219  if(_nBytes > areaInfo.nBytes) {
220  throw DeviceException("Requested number of elements exceeds the size of the register (late, redundant safety check)!",
221  DeviceException::WRONG_PARAMETER);
222  }
223  _nBlocks = _numberOfElements;
224 
225  // allocate the buffer for the converted data
226  NDRegisterAccessor<UserType>::buffer_2D.resize(_converters.size());
227  for(size_t i=0; i<_converters.size(); ++i) {
228  NDRegisterAccessor<UserType>::buffer_2D[i].resize(_nBlocks);
229  }
230 
231  // allocate the raw io buffer
232  _ioBuffer.resize(_nBytes/sizeof(int32_t));
233  }
234  catch(...) {
235  this->shutdown();
236  throw;
237  }
