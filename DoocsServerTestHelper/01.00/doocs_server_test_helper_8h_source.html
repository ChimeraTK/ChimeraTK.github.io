222 template<typename TYPE>
223 std::vector<TYPE> DoocsServerTestHelper::doocsGetArray( const std::string & name ) {
224  EqAdr ad;
225  EqData ed, res;
226  // obtain location pointer
227  ad.adr(name.c_str());
228  EqFct *p = eq_get(&ad);
229  ASSERT(p != NULL, std::string("Could not get location for property ")+name);
230  // for D_BSpectrum: set IIII structure to obtain always the latest buffer (shouldn't hurt for others)
231  IIII iiii;
232  iiii.i1_data = -1;
233  iiii.i2_data = -1;
234  iiii.i3_data = -1;
235  iiii.i4_data = -1;
236  ed.set(&iiii);
237  // obtain values
238  p->lock();
239  p->get(&ad,&ed,&res);
240  p->unlock();
241  // check for errors
242  ASSERT(res.error() == 0, std::string("Error reading property ")+name);
243  // copy to vector and return it
244  std::vector<TYPE> val;
245  if(std::is_integral<TYPE>()) {
246  if(res.type() != DATA_A_LONG) {
247  for(int i=0; i<res.length(); i++) val.push_back(res.get_int(i));
248  }
249  else {
250  for(int i=0; i<res.length(); i++) val.push_back(res.get_long(i));
251  }
252  }
253  else if(std::is_floating_point<TYPE>()) {
254  for(int i=0; i<res.length(); i++) val.push_back(res.get_float(i));
255  }
256  else {
257  ASSERT(false, "Wrong type passed as tempalate argument.");
258  }
259  return val;
260 }
