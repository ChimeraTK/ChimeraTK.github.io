133  inline void ReadAnyGroup::add(TransferElementAbstractor element) {
134  if(isFinalised) {

150  inline void ReadAnyGroup::finalise() {
151  if(isFinalised) throw std::logic_error("ReadAnyGroup has already been finalised, calling finalise() is no longer allowed.");
152  std::vector<cppext::future_queue<void>> queueList;
153  bool groupEmpty = true;
154  for(auto &e : push_elements) {
155  auto tf = e.readAsync();
156  queueList.push_back( detail::getFutureQueueFromTransferFuture(tf) );
157  groupEmpty = false;
158  }
159  if(groupEmpty) {
160  throw std::logic_error("ReadAnyGroup has no element with AccessMode::wait_for_new_data.");
161  }
162  notification_queue = cppext::when_any(queueList.begin(), queueList.end());
163  isFinalised = true;
164  }

168  inline TransferElementID ReadAnyGroup::readAny() {
169  size_t idx;
170  push_elements[0].transferFutureWaitCallback();
171  notification_queue.pop_wait(idx);
172  push_elements[idx].readAsync().wait();
173  for(auto &e : poll_elements) {
174  if(!e.getAccessModeFlags().has(AccessMode::wait_for_new_data)) e.readLatest();
175  }
176  return push_elements[idx].getId();
177  }

196  inline void ReadAnyGroup::readUntilAll(const std::vector<TransferElementID> &ids) {
197  std::map<TransferElementID, bool> found;
198  for(auto &id : ids) found[id] = false; // initialise map so we can tell from the map if a variable is in the vector
199  size_t leftToFind = ids.size();
200  while(true) {
201  auto read = readAny();
202  if(found.count(read) == 0) continue; // variable is not in the vector ids
203  if(found[read] == true) continue; // variable has been read already
204  found[read] = true;
205  --leftToFind;
206  if(leftToFind == 0) return;
207  }
208  }
