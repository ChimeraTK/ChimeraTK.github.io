311  template<class T>
312  void BidirectionalProcessArray<T>::doReadTransfer() {
313  do {
314  _receiver->read();
315  // We only update the current value (stored in the sender) when the version
316  // number of the data that we received is greater than or equal the current
317  // version number. This ensures that old updates (that might arrive late due
318  // to the asynchronous nature of the transfer logic) do not overwrite newer
319  // values and also helps to ensure that we do not get a feedback loop where
320  // two (or more) bidirectional process variables "play ping-pong" (see issue
321  // #2 for the full discussion).
322  if(_receiver->getVersionNumber() >= _versionNumber) return;
323  if(valueRejectCallback) valueRejectCallback();
324  } while(true);
325  }

383  template<class T>
384  void BidirectionalProcessArray<T>::doPostRead() {
385  this->accessChannel(0).swap(_receiver->accessChannel(0));
386  // After receiving, our new time stamp and version number are the ones
387  // that we got from the receiver.
388  _versionNumber = _receiver->getVersionNumber();
389 
390  // Pass on data validity flag from sender to receiver and make it our
391  // our internal validity flag
392  _dataValidity = _receiver->dataValidity();
393  _sender->setDataValidity(_dataValidity);
394 
395  // If we have a persistent data-storage, we have to update it. We have to
396  // do this because a (new) value received from the other side should be
397  // treated like a value sent by this side.
398  if(_persistentDataStorage) {
399  _persistentDataStorage->updateValue(_persistentDataStorageID, this->accessChannel(0));
400  }
401  }

410  template<class T>
411  bool BidirectionalProcessArray<T>::doWriteTransfer(ChimeraTK::VersionNumber versionNumber) {
412  // We have to copy our current value to the sender. We cannot swap it
413  // because this would mean that we would lose the current value.
414  _sender->accessChannel(0) = this->accessChannel(0);
415  // We already copied the value, so the sender does not have to copy the
416  // value again.
417 
418  // Propagate validity flag
419  _sender->setDataValidity(_dataValidity);
420 
421  bool lostData = _sender->writeDestructively(versionNumber);
422  // After sending the new value, our current version number are the one from
423  // the sender.
424  _versionNumber = versionNumber;
425 
426  // If we have a persistent data-storage, we have to update it.
427  if(_persistentDataStorage) {
428  _persistentDataStorage->updateValue(_persistentDataStorageID, this->accessChannel(0));
429  }
430  if(_sendNotificationListener) {
431  typename ProcessArray<T>::SharedPtr partner = _partner.lock();
432  if(partner) {
433  _sendNotificationListener->notify(partner);
434  }
435  }
436  return lostData;
437  }
