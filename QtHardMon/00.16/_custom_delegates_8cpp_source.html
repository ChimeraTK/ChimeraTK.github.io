18 /*
19  * Method displays double values in the cell with decimal places. ((eg: 123.6500
20  * if precision required is 4)). The desired number of decimal places
21  * (precision) can be specified by setting
22  * _doubleSpinBoxPrecision
23  */
24 QString CustomDelegates::displayText(const QVariant& value,
25  const QLocale& locale) const {
26  if (value.type() == QVariant::Double) { // frame and return a formatted
27  // QString if handling a cell with
28  // double value
29  return locale.toString((value.toDouble()), 'f', _doubleSpinBoxPrecision);
30  } else if (value.type() == QVariant::UserType) {
31  // The hex type is the only defined type at this point so this check should
32  // be OK
33 
34  HexData hexValue = value.value<HexData>(); // get the datatype stored in this cell
35  // QString::number has a bug and gives negative values as 64 bit hex.
36  // So we have to do it manually.
37  int intValue = hexValue.value; // covert the value to 32 bit int
38  std::stringstream s;
39  s << "0x" << std::hex << intValue;
40  return QString(s.str().c_str());
41 
42  } else { // else let the default delegate take care of formatting if
43  // content is not double
44  return QStyledItemDelegate::displayText(value, locale);
45  }
46 }

49  * Use a customized QDoubleSpinbox to modify cells with double value. The
50  * spinbox supports a precision specified by _doubleSpinBoxPrecision.
51  */
52 QWidget* CustomDelegates::createEditor(QWidget* parent_,
53  const QStyleOptionViewItem& option,
54  const QModelIndex& index) const {
55 
56  if (index.data(Qt::EditRole).type() ==
57  QVariant::Double) { // create spinbox with custom precision
58  // for cells in the double column
59  QDoubleSpinBox* doubleSpinBox = new QDoubleSpinBox(parent_);
60  doubleSpinBox->setDecimals(_doubleSpinBoxPrecision);
61  doubleSpinBox->setRange(INT_MIN, INT_MAX);
62  return doubleSpinBox;
63  } else if (index.data(Qt::EditRole).type() == QVariant::Int) {
64  // Want to customize the decimal spin boxes to limit the max and minimum
65  // value they can display
66  QSpinBox* decimalSpinbox = new QSpinBox(parent_);
67  decimalSpinbox->setRange(INT_MIN, INT_MAX);
68  return decimalSpinbox;
69  } else if (index.data(Qt::EditRole).type() == QVariant::UserType) {
70  // TODO: make the check above specific to our custom HexDataType
71  HexSpinBox* hexSpinBox = new HexSpinBox(parent_);
72  hexSpinBox->setRange(INT_MIN, INT_MAX);
73  return hexSpinBox;
74  } else {
75  return QStyledItemDelegate::createEditor(parent_, option, index);
76  }
77 }
