69  bool isMergeable(const boost::shared_ptr<TransferElement const> &other) const {
70  // accessor type, device and bar must be the same
71  auto rhsCasted = boost::dynamic_pointer_cast< const NumericAddressedLowLevelTransferElement >(other);
72  if(!rhsCasted) return false;
