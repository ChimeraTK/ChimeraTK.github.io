165  void BackendFactory::loadPluginLibrary(std::string soFile){
166  // Create a copy of the original creator map. If we unload the library because the signature is not
167  // recognised we have to restore the original map because it could be an old backed that has a working
168  // registerer (or the programmer forgot the signature function). In this case we would leave the
169  // factory with a dangling function pointer which causes a segfault if the backend is opened.
170  auto originalCreatorMap = creatorMap;
171 
172  // first do an open which does not load the symbols yet
173  // FIXME: Only works for functions, not for variables = registerer :-(
174  void *hndl = dlopen(soFile.c_str() , RTLD_LAZY );
175  if(hndl == NULL){
176  throw DeviceException( dlerror(), DeviceException::WRONG_PARAMETER);
177  }
178 
179  // try to find the symbol for the version function.
180  const char * (*versionFunction)();
181  // We have to use an instance of the function pointer with the right signature
182  // and reinterpred cast it because dlsym is giving out a void pointer which in
183  // pedantic C++ cannot be casted directly to a function pointer.
184  *reinterpret_cast<void**>(&versionFunction) = dlsym(hndl, "deviceAccessVersionUsedToCompile");
185 
186  if (versionFunction == NULL){
187  creatorMap = originalCreatorMap;
188  dlclose(hndl);
189  std::stringstream errorMessage;
190  errorMessage << "Symbol 'deviceAccessVersionUsedToCompile' not found in " <<soFile;
191  throw DeviceException( errorMessage.str(), DeviceException::WRONG_PARAMETER);
192  }
193 
194  if (std::string(versionFunction()) != CHIMERATK_DEVICEACCESS_VERSION){
195  std::stringstream errorMessage;
196  errorMessage << soFile << " was compiled with the wrong DeviceAccess version " << versionFunction()
197  << ". Recompile with DeviceAccess version " << CHIMERATK_DEVICEACCESS_VERSION;
198  // Do not restore the original creator map. If the registerer worked there is a function which throws an
199  // exception that tells the user that the backend has been compiled against the wrong version.
200  // This is better than just printing a warning because the exception which is thrown here is caught
201  // if the library is loaded via dmap file (to keep dmap files working if broken backends are not used).
202  dlclose(hndl);
203  throw DeviceException( errorMessage.str(), DeviceException::WRONG_PARAMETER);
204  }
205 
206  // it is a correct plugin, load all the symbols now
207  dlopen(soFile.c_str() , RTLD_NOW);
208  }

210  void BackendFactory::loadAllPluginsFromDMapFile(){
211  if (_dMapFile.empty()){
212  return;
213  }
214 
215  auto dmap = DMapFileParser().parse(_dMapFile);
216 
217  for ( auto lib : dmap->getPluginLibraries() ){
218  try{
219  loadPluginLibrary(lib);
220  }catch(DeviceException &e){
221  //Ignore library loading errors when doing this automatically for all plugins in the dmap file
222  //to keep dmap files usable if the broken backends are not used.
223  //Print a warning that the loading failed.
224  std::cerr << "Error: Caught exception loading plugin '"<< lib <<"' specified in dmap file: "
225  << e.what() << std::endl;
226  std::cerr << "Some backends with not be available!" << std::endl;
227  }
228  }
