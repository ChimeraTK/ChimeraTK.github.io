69  bool RegisterInfoMap::check(ErrorList &err, RegisterInfoMap::ErrorList::ErrorElem::TYPE level) {
70  std::vector<addresses> v_addresses;
71  std::vector<addresses>::iterator v_iter;
72  addresses address;
73 
74  std::vector<RegisterInfo> map_file;
75  for(auto it = _catalogue.begin(); it != _catalogue.end(); ++it) {
76  auto infoCast = boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(it.get());
77  map_file.push_back(*infoCast);
78  }
79 
80  std::vector<RegisterInfo>::iterator iter_p, iter_n;
81  bool ret = true;
82 
83  err.clear();
84  if (map_file.size() < 2)
85  return true;
86  std::sort(map_file.begin(), map_file.end(), compareRegisterInfosByName_functor());
87  iter_p = map_file.begin();
88  iter_n = iter_p + 1;
89  while (1) {
90  if ( (iter_p->name == iter_n->name) && (iter_p->module == iter_n->module) ) {
91  err.insert(ErrorList::ErrorElem(ErrorList::ErrorElem::ERROR,
92  ErrorList::ErrorElem::NONUNIQUE_REGISTER_NAME,
93  *iter_p, *iter_n, _mapFileName));
94  ret = false;
95  }
96  iter_p = iter_n;
97  iter_n = ++iter_n;
98  if (iter_n == map_file.end()) {
99  break;
100  }
101  }
102 
103  for (iter_p = map_file.begin(); iter_p != map_file.end(); ++iter_p) {
104  address.start = iter_p->address;
105  address.end = iter_p->address + iter_p->nBytes;
106  address.iter = iter_p;
107  for (v_iter = v_addresses.begin(); v_iter != v_addresses.end(); ++v_iter) {
108  // only addresses in the same module are considered to overlap
109  if (iter_p->module != v_iter->iter->module){
110  continue;
111  }
112 
113  if (iter_p->bar == (v_iter->iter)->bar) {
114  if ( (v_iter->start >= iter_p->address) &&
115  (v_iter->start < (iter_p->address + iter_p->nBytes)) ) {
116  if (level >= ErrorList::ErrorElem::WARNING){
117  err.insert(ErrorList::ErrorElem(ErrorList::ErrorElem::WARNING,
118  ErrorList::ErrorElem::WRONG_REGISTER_ADDRESSES,
119  *iter_p, *(v_iter->iter), _mapFileName));
120  ret = false;
121  }
122  } else if ( (v_iter->start <= iter_p->address) && (iter_p->address < v_iter->end) ) {
123  if (level >= ErrorList::ErrorElem::WARNING){
124  err.insert(ErrorList::ErrorElem(ErrorList::ErrorElem::WARNING,
125  ErrorList::ErrorElem::WRONG_REGISTER_ADDRESSES,
126  *iter_p, *(v_iter->iter), _mapFileName));
127  ret = false;
128  }
129  }
130  }
131  }
132  v_addresses.push_back(address);
133  }
134  return ret;

235  std::list< RegisterInfoMap::RegisterInfo > RegisterInfoMap::getRegistersInModule( std::string const & moduleName){
236  // first sort all elements accordind the names (module first, then register in module)
237  // make a copy to keep the original order from the map file
238  std::vector<RegisterInfo> sortedRegisterInfos;
239  for(auto it = _catalogue.begin(); it != _catalogue.end(); ++it) {
240  auto infoCast = boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(it.get());
241  sortedRegisterInfos.push_back(*infoCast);
242  }
243  std::sort(sortedRegisterInfos.begin(), sortedRegisterInfos.end(), compareRegisterInfosByName_functor());
244 
245  // The vector is sorted, first module, than register name.
246  // Find the first iterator with the module name, starting at the beginning,
247  // then search for the first iterator with another name, starting from the previousy found first match.
248  std::vector<RegisterInfo>::iterator firstMatchingIterator =
249  std::find_if( sortedRegisterInfos.begin(), sortedRegisterInfos.end(), compareModuleName_pred( moduleName ) );
250  std::vector<RegisterInfo>::iterator firstNotMatchingIterator =
251  std::find_if( firstMatchingIterator, sortedRegisterInfos.end(),
252  std::not1( compareModuleName_pred(moduleName) ) );
253 
254  // fill the list
255  std::list<RegisterInfoMap::RegisterInfo> registerInfoList;
256  for (std::vector<RegisterInfo>::iterator it = firstMatchingIterator;
257  it != firstNotMatchingIterator; ++it){
258  registerInfoList.push_back( *it );
259  }
260 
261  return registerInfoList;
262  }
