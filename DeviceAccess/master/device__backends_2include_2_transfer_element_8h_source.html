110  void read() {
111  if(TransferElement::isInTransferGroup) {
112  throw ChimeraTK::logic_error("Calling read() or write() on an accessor "
113  "which is part of a TransferGroup "
114  "is not allowed.");
115  }
116  if(hasActiveFuture) {
117  activeFuture.wait();
118  return;
119  }
120  this->readTransactionInProgress = false;
121  preRead(TransferType::read);
122  readTransfer();
123 
124  postRead(TransferType::read, !hasSeenException);
125  }

138  bool readNonBlocking() {
139  if(hasActiveFuture) {
140  if(activeFuture.hasNewData()) {
141  activeFuture.wait();
142  return true;
143  }
144  else {
145  return false;
146  }
147  }
148  this->readTransactionInProgress = false;
149  preRead(TransferType::readNonBlocking);
150  bool hasNewData = readTransferNonBlocking();
151  postRead(TransferType::readNonBlocking, hasNewData);
152  return hasNewData;
153  }
154 
159  bool readLatest() {
160  bool ret = false;
161  if(hasActiveFuture) {
162  if(activeFuture.hasNewData()) {
163  activeFuture.wait();
164  ret = true;
165  }
166  else {
167  return false;
168  }
169  }
170  this->readTransactionInProgress = false;
171  preRead(TransferType::readLatest);
172  bool ret2 = readTransferLatest();
173  postRead(TransferType::readLatest, ret2);
174  return ret || ret2;
175  }

202  TransferFuture& readAsync() {
203  if(!hasActiveFuture) {
204  // call preRead
205  this->readTransactionInProgress = false;
206  this->preRead(TransferType::readAsync);
207 
208  // initiate asynchronous transfer and return future
209  activeFuture = readTransferAsync();
210  hasActiveFuture = true;
211  }
212  return activeFuture;
213  }
