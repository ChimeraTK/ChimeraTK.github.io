48  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
49  NDRegisterAccessor<UserType>::buffer_2D[0].resize(1);
50 
51  // check data type
52  if(DoocsBackendRegisterAccessor<UserType>::dst.type() != DATA_TEXT &&
53  DoocsBackendRegisterAccessor<UserType>::dst.type() != DATA_STRING &&
54  DoocsBackendRegisterAccessor<UserType>::dst.type() != DATA_STRING16) {
55  throw ChimeraTK::logic_error("DOOCS data type not supported by "
