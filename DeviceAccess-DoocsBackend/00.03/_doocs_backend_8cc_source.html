76  template<typename UserType>
77  boost::shared_ptr< NDRegisterAccessor<UserType> > DoocsBackend::getRegisterAccessor_impl(
78  const RegisterPath &registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
79 
80  NDRegisterAccessor<UserType> *p;
81  RegisterPath path = _serverAddress/registerPathName;
82 
83  // read property once
84  EqAdr ea;
85  EqCall eq;
86  EqData src,dst;
87  ea.adr(std::string(path).substr(1).c_str()); // strip leading slash
88  int rc = eq.get(&ea, &src, &dst);
89  if(rc) {
90  throw DeviceException(std::string("Cannot open DOOCS property: ")+dst.get_string(),
91  DeviceException::CANNOT_OPEN_DEVICEBACKEND);
92  }
93 
94  // check type and create matching accessor
95  if( dst.type() == DATA_INT || dst.type() == DATA_A_INT ||
96  dst.type() == DATA_BOOL || dst.type() == DATA_A_BOOL ||
97  dst.type() == DATA_A_SHORT ) {
98  p = new DoocsBackendIntRegisterAccessor<UserType>(path, numberOfWords, wordOffsetInRegister, flags);
99  }
100  else if( dst.type() == DATA_A_LONG ) {
101  p = new DoocsBackendLongRegisterAccessor<UserType>(path, numberOfWords, wordOffsetInRegister, flags);
102  }
103  else if( dst.type() == DATA_FLOAT || dst.type() == DATA_A_FLOAT ||
104  dst.type() == DATA_DOUBLE || dst.type() == DATA_A_DOUBLE ) {
105  p = new DoocsBackendFloatRegisterAccessor<UserType>(path, numberOfWords, wordOffsetInRegister, flags);
106  }
107  else if( dst.type() == DATA_TEXT || dst.type() == DATA_STRING || dst.type() == DATA_STRING16) {
108  p = new DoocsBackendStringRegisterAccessor<UserType>(path, numberOfWords, wordOffsetInRegister, flags);
109  }
110  else {
111  throw DeviceException("Unsupported DOOCS data type: "+std::string(dst.type_string()),
112  DeviceException::WRONG_PARAMETER);
113  }
114 
115 
116  return boost::shared_ptr< NDRegisterAccessor<UserType> > ( p );
117  }
