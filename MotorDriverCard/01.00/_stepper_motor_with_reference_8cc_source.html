115  StepperMotorError StepperMotorWithReference::getError(){
116  boost::lock_guard<boost::mutex> guard(_mutex);
117  if (isPositiveReferenceActive() && isNegativeReferenceActive()){
118  return BOTH_END_SWITCHES_ON;
119  }
120  if (stateMachineInIdleAndNoEvent()){
121  if (_toleranceCalcFailed || _calibrationFailed){
122  return ACTION_ERROR;
123  }else if ((_motorControler->getTargetPosition() != _motorControler->getActualPosition()) &&
124  !isPositiveReferenceActive() &&
125  !isNegativeReferenceActive()){
126  return ACTION_ERROR;
127  }
128  }
129  if (_toleranceCalculated){
130  if (isPositiveReferenceActive() && fabs(_motorControler->getActualPosition() - _calibPositiveEndSwitchInSteps) > 3 * _tolerancePositiveEndSwitch){
131  return CALIBRATION_LOST;
132  }else if(isNegativeReferenceActive() && fabs(_motorControler->getActualPosition() - _calibNegativeEndSwitchInSteps) > 3 * _toleranceNegativeEndSwitch){
133  return CALIBRATION_LOST;
134  }
135  }
136  return NO_ERROR;
137  }
