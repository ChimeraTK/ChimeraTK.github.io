209  template<typename UserType>
210  void DoocsBackendRegisterAccessor<UserType>::initialise() {
211  if(isInitialised) return;
212 
213  // try to read data, to check connectivity and to obtain size of the register
214  EqData tmp;
215  int rc = eq.get(&ea, &tmp, &dst);
216  if(rc) {
217  if(rc == eq_errors::ill_property || rc == eq_errors::ill_location ||
218  rc == eq_errors::ill_address) { // no property by that name
219  throw ChimeraTK::logic_error("Property does not exist: " + _path);
220  }
221  // runtime error will later be thrown by the transfer...
222  return;
223  }
224 
225  // obtain number of elements
226  size_t actualLength = dst.array_length();
227  if(actualLength == 0 && dst.length() == 1) {
228  actualLength = 1;
229  isArray = false;
230  }
231  else {
232  if(actualLength == 0) actualLength = dst.length();
233  isArray = true;
234  }
235  if(nElements == 0) {
236  nElements = actualLength;
237  }
238  if(nElements + elementOffset > actualLength) {
239  throw ChimeraTK::logic_error("Requested number of words exceeds the length of the DOOCS property!");
240  }
241  if(nElements == actualLength && elementOffset == 0) {
242  isPartial = false;
243  }
244  else {
245  isPartial = true;
246  }
247 
248  // allocate buffers
249  if(_allocateBuffers) {
250  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
251  NDRegisterAccessor<UserType>::buffer_2D[0].resize(nElements);
252  }
253 
254  // set proper type information in the source EqData
255  src.set_type(dst.type());
256  if(_allocateBuffers && dst.type() != DATA_IIII) {
257  src.length(actualLength);
258  }
259 
260  // use ZeroMQ with AccessMode::wait_for_new_data
261  if(useZMQ) {
262  // create TransferFuture
263  activeFuture = TransferFuture(notifications.then<void>([](EqData&) {}, std::launch::deferred), this);
264  futureCreated = true;
265 
266  // subscribe via subscription manager
267  DoocsBackendNamespace::ZMQSubscriptionManager::getInstance().subscribe(_path, this);
268  }
269 
270  initialiseImplementation();
271  isInitialised = true;
272  }

326  bool DoocsBackendRegisterAccessor<UserType>::doReadTransferNonBlocking() {
327  if(!isInitialised) {
328  // if initialise() fails to contact the server, it cannot yet throw a runtime_error, so we have to do this here.
329  throw ChimeraTK::runtime_error(std::string("Cannot read from DOOCS property: ") + dst.get_string());
330  }
331 
332  if(!useZMQ) {
333  this->doReadTransfer();
334  return true;
335  }
336  else {
337  // loop to allow retries in case of timeout errors
338  while(true) {
339  // wait until new data has been received
340  bool gotData = notifications.pop(dst);
341  if(!gotData) return false;
342  // check for an error
343  if(dst.error() != 0) {
344  // try obtaining data through RPC call instead to verify error
345  EqData tmp;
346  int rc = eq.get(&ea, &tmp, &dst);
347  // if again error received, throw exception
348  if(rc) {
349  throw ChimeraTK::runtime_error(std::string("Cannot read from DOOCS property: ") + dst.get_string());
350  }
351  // otherwise try again
352  continue;
353  }
354  // we have received data, so return
355  return true;
356  }
357  }

447  TransferFuture DoocsBackendRegisterAccessor<UserType>::doReadTransferAsync() {
448  if(!isInitialised) {
449  // if initialise() fails to contact the server, it cannot yet throw a runtime_error, so we have to do this here.
450  throw ChimeraTK::runtime_error(std::string("Cannot read from DOOCS property: ") + dst.get_string());

452

459  futureCreated = true;
