121 void DoocsServerTestHelper::doocsSet( const std::string &name, TYPE value ) {
122  EqAdr ad;
123  EqData ed, res;
124  // obtain location pointer
125  ad.adr(name.c_str());
126  EqFct *p = eq_get(&ad);
127  ASSERT(p != NULL, std::string("Could not get location for property ")+name);
128  // set value
129  ed.set(value);
130  p->lock();
131  p->set(&ad,&ed,&res);
132  p->unlock();
133  // check for error
134  ASSERT(res.error() == 0, std::string("Error writing property ")+name);

212 template<typename TYPE>
213 std::vector<TYPE> DoocsServerTestHelper::doocsGetArray( const char *name ) {
214  EqAdr ad;
215  EqData ed, res;
216  // obtain location pointer
217  ad.adr(name);
218  EqFct *p = eq_get(&ad);
219  ASSERT(p != NULL, std::string("Could not get location for property ")+name);
220  // for D_BSpectrum: set IIII structure to obtain always the latest buffer (shouldn't hurt for others)
221  IIII iiii;
222  iiii.i1_data = -1;
223  iiii.i2_data = -1;
224  iiii.i3_data = -1;
225  iiii.i4_data = -1;
226  ed.set(&iiii);
227  // obtain values
228  p->lock();
229  p->get(&ad,&ed,&res);
230  p->unlock();
231  // check for errors
232  ASSERT(res.error() == 0, std::string("Error reading property ")+name);
233  // copy to vector and return it
234  std::vector<TYPE> val;
235  if(std::is_integral<TYPE>()) {
236  if(res.type() != DATA_A_LONG) {
237  for(int i=0; i<res.length(); i++) val.push_back(res.get_int(i));
238  }
239  else {
240  for(int i=0; i<res.length(); i++) val.push_back(res.get_long(i));
241  }
242  }
243  else if(std::is_floating_point<TYPE>()) {
244  for(int i=0; i<res.length(); i++) val.push_back(res.get_float(i));
245  }
246  else {
247  ASSERT(false, "Wrong type passed as tempalate argument.");
248  }
249  return val;
250 }
