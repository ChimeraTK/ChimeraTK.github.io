70 bool RegisterTypeAbstractorImpl<USER_DATA_TYPE>::setData(unsigned int channelIndex, unsigned int elementIndex, const QVariant & data){
71 
72  if (data.type() == QVariant::UserType){ //in the user type the data is hex representation
73  if (! data.canConvert<HexData>() ){
74  return false;
75  }
76  // The variant has a templated function called value() which returns the template type,
77  // HexData in out case. The HexData class itself has a member called value, which holds the
78  // actual value. Phew...
79  _accessor[channelIndex][elementIndex] = data.value<HexData>().value;
80  }else{
81  if (! data.canConvert<USER_DATA_TYPE>() ){
82  return false;
83  }
84  _accessor[channelIndex][elementIndex] = data.value<USER_DATA_TYPE>();
85  }
86  return true;
