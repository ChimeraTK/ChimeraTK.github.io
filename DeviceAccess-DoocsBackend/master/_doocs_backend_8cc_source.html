50 static std::unique_ptr<ctk::RegisterCatalogue> fetchCatalogue(
51  std::string serverAddress, std::string cacheFile, std::future<void> cancelFlag) {
52  auto result = CatalogueFetcher(serverAddress, std::move(cancelFlag)).fetch();
53  auto catalogue = std::move(result.first);
54  auto isCatalogueComplete = result.second;
55  bool isCacheFileNameSpecified = not cacheFile.empty();
56 
57  if(isCatalogueComplete && isCacheFileNameSpecified) {
58  Cache::saveCatalogue(*catalogue, cacheFile);
59  }
60  return catalogue;
61 }

117  boost::shared_ptr<DeviceBackend> DoocsBackend::createInstance(
118  std::string address, std::map<std::string, std::string> parameters) {
119  // if address is empty, build it from parameters (for compatibility with SDM)
120  if(address.empty()) {
121  RegisterPath serverAddress;
122  serverAddress /= parameters["facility"];
123  serverAddress /= parameters["device"];
124  serverAddress /= parameters["location"];
125  address = std::string(serverAddress).substr(1);
126  }
127  std::string cacheFile{};
128  try {
129  cacheFile = parameters.at("cacheFile");
130  }
131  catch(std::out_of_range&) {
132  // empty cacheFile string => no caching
133  }
134  // create and return the backend
135  return boost::shared_ptr<DeviceBackend>(new DoocsBackend(address, cacheFile));
136  }

162  template<typename UserType>
163  boost::shared_ptr<NDRegisterAccessor<UserType>> DoocsBackend::getRegisterAccessor_impl(
164  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
165  boost::shared_ptr<NDRegisterAccessor<UserType>> p;
166  std::string path = _serverAddress + registerPathName;
167 
168  // check for additional hierarchy level, which indicates an access to a field of a complex property data type
169  bool hasExtraLevel = false;
170  if(!boost::starts_with(path, "doocs://") && !boost::starts_with(path, "epics://")) {
171  size_t nSlashes = std::count(path.begin(), path.end(), '/');
172  if(nSlashes == 4) {
173  hasExtraLevel = true;
174  }
175  else if(nSlashes < 3 || nSlashes > 4) {
176  throw ChimeraTK::logic_error(std::string("DOOCS address has an illegal format: ") + path);
177  }
178  }
179  else if(boost::starts_with(path, "doocs://")) {
180  size_t nSlashes = std::count(path.begin(), path.end(), '/');
181  // we have 3 extra slashes compared to the standard syntax without "doocs:://"
182  if(nSlashes == 4 + 3) {
183  hasExtraLevel = true;
184  }
185  else if(nSlashes < 3 + 3 || nSlashes > 4 + 3) {
186  throw ChimeraTK::logic_error(std::string("DOOCS address has an illegal format: ") + path);
187  }
188  }
189 
190  // split the path into property name and field name
191  std::string field;
192  if(hasExtraLevel) {
193  field = path.substr(path.find_last_of('/') + 1);
194  path = path.substr(0, path.find_last_of('/'));
195  }
196 
197  // if backend is open, read property once to obtain type. otherwise use the (potentially cached) catalogue
198  int doocsTypeId = DATA_NULL;
199  if(isOpen()) {
200  EqAdr ea;
201  EqCall eq;
202  EqData src, dst;
203  ea.adr(path);
204  int rc = eq.get(&ea, &src, &dst);
205  if(rc) {
206  throw ChimeraTK::runtime_error("Cannot open DOOCS property '" + path + "': " + dst.get_string());
207  }
208  doocsTypeId = dst.type();
209  }
210  else {
211  auto reg =
212  boost::dynamic_pointer_cast<DoocsBackendRegisterInfo>(getRegisterCatalogue().getRegister(registerPathName));
213  doocsTypeId = reg->doocsTypeId;
214  }
215 
216  // check type and create matching accessor
217  bool extraLevelUsed = false;
218  auto sharedThis = boost::static_pointer_cast<DoocsBackend>(shared_from_this());
219  if(doocsTypeId == DATA_INT || doocsTypeId == DATA_A_INT || doocsTypeId == DATA_BOOL || doocsTypeId == DATA_A_BOOL ||
220  doocsTypeId == DATA_A_SHORT) {
221  p.reset(new DoocsBackendIntRegisterAccessor<UserType>(
222  sharedThis, path, registerPathName, numberOfWords, wordOffsetInRegister, flags));
223  }
224  else if(doocsTypeId == DATA_A_LONG) {
225  p.reset(new DoocsBackendLongRegisterAccessor<UserType>(
226  sharedThis, path, registerPathName, numberOfWords, wordOffsetInRegister, flags));
227  }
228  else if(doocsTypeId == DATA_FLOAT || doocsTypeId == DATA_A_FLOAT || doocsTypeId == DATA_DOUBLE ||
229  doocsTypeId == DATA_A_DOUBLE || doocsTypeId == DATA_SPECTRUM) {
230  p.reset(new DoocsBackendFloatRegisterAccessor<UserType>(
231  sharedThis, path, registerPathName, numberOfWords, wordOffsetInRegister, flags));
232  }
233  else if(doocsTypeId == DATA_TEXT || doocsTypeId == DATA_STRING || doocsTypeId == DATA_STRING16) {
234  p.reset(new DoocsBackendStringRegisterAccessor<UserType>(
235  sharedThis, path, registerPathName, numberOfWords, wordOffsetInRegister, flags));
236  }
237  else if(doocsTypeId == DATA_IIII) {
238  p.reset(new DoocsBackendIIIIRegisterAccessor<UserType>(
239  sharedThis, path, registerPathName, numberOfWords, wordOffsetInRegister, flags));
240  }
241  else if(doocsTypeId == DATA_IFFF) {
242  extraLevelUsed = true;
243  p.reset(new DoocsBackendIFFFRegisterAccessor<UserType>(
244  sharedThis, path, field, registerPathName, numberOfWords, wordOffsetInRegister, flags));
245  }
246  else {
247  throw ChimeraTK::logic_error("Unsupported DOOCS data type " + std::string(EqData().type_string(doocsTypeId)) +
248  " of property '" + _serverAddress + registerPathName + "'");
249  }
250 
251  // if the field name has been specified but the data type does not use it, throw an exception
252  if(hasExtraLevel && !extraLevelUsed) {
253  throw ChimeraTK::logic_error("Specifiaction of field name is not supported for the DOOCS data type " +
254  std::string(EqData().type_string(doocsTypeId)) + ": " + _serverAddress + registerPathName);
255  }
256 
257  return p;
258  }
