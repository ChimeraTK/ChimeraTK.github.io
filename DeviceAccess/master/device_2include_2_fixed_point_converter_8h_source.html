211  template<typename UserType>
212  UserType FixedPointConverter::toCooked(uint32_t rawValue) const
213  {
214  UserType cooked;
215 
216  // extract sign and leave positive raw number
217  bool isNegative = (rawValue & _signBitMask);
218  if(isNegative) rawValue = ~rawValue; // bit inversion flips the sign, but will be off by 1!
219 
220  // leading, out of range bits are ignored. Just crop them.
221  rawValue &= _usedBitsMask;
222 
223  // Handle integer and floating-point types differently.
224  // This if-statement is likely to be optimised out at compile-time.
225  if(std::numeric_limits<UserType>::is_integer && _fractionalBits == 0) {
226 
227  // cast into target type while checking against over/underflow
228  // as we flipped the sign above, we won't get a negative_overflow, thus catch the exception and
229  // re-throw the right one
230  try {
231  cooked = boost::numeric_cast<UserType>(rawValue);
232  }
233  catch( boost::numeric::positive_overflow &) {
234  if(!isNegative) {
235  throw FPC_OverflowException<boost::numeric::positive_overflow>(_variableName);
236  }
237  else {
238  throw FPC_OverflowException<boost::numeric::negative_overflow>(_variableName);
239  }
240  }
241 
242  // handle the sign
243  if(isNegative) {
244  // This if-statement also is likely to be optimised out at compile-time.
245  if(std::numeric_limits<UserType>::is_signed) {
246  cooked = -cooked - 1;
247  }
248  else {
249  cooked = 0;
250  throw FPC_OverflowException<boost::numeric::negative_overflow>(_variableName);
251  }
252  }
253 
254  }
255  else {
256 
257  // convert into double and scale to handle fractional bits
258  double d_cooked = _fractionalBitsCoefficient * static_cast<double>(rawValue);
259 
260  // Handle the sign. Since we inverted the rawValue to get a positive value, it was off by 1 before scaling
261  if(isNegative) d_cooked = -d_cooked - _fractionalBitsCoefficient;
262 
263  // Convert into target type. This conversion takes care of proper rounding when needed (and only then), and will
264  // throw boost::numeric::positive_overflow resp. boost::numeric::negative_overflow if the target
265  // range is exceeded.
266  typedef boost::numeric::converter<UserType, double, boost::numeric::conversion_traits<UserType,double>,
267  boost::numeric::def_overflow_handler,
268  Round<double> > converter;
269  try{
270  cooked = converter::convert(d_cooked);
271  }catch( boost::numeric::positive_overflow & ) {
272  throw FPC_OverflowException<boost::numeric::positive_overflow>(_variableName);
273  }catch( boost::numeric::negative_overflow & ) {
274  throw FPC_OverflowException<boost::numeric::negative_overflow>(_variableName);
275  }
276  }
277 
278  return cooked;
279 
280  }

283  template<typename UserType>
284  uint32_t FixedPointConverter::toRaw(UserType cookedValue) const
285  {
286 
287  // Do a range check first. The later overflow check in the conversion is not sufficient, since we can have
288  // non-standard word sizes like 12 bits.
289  if(cookedValue < boost::fusion::at_key<UserType>(_minCookedValues)) {
290  return _minRawValue;
291  }
292  if(cookedValue > boost::fusion::at_key<UserType>(_maxCookedValues)) {
293  return _maxRawValue;
294  }
295 
296  // handle integer and floating-point types differently
297  if(std::numeric_limits<UserType>::is_integer && _fractionalBits == 0) {
298 
299  // extract the sign and leave the positive number
300  bool isNegative = isNegativeUserType(cookedValue);
301  if(isNegative && !_isSigned) return _minRawValue;
302  if(isNegative) cookedValue = -(cookedValue+1); // bit inversion, ~ operator cannot be used as UserType might be double
303 
304  // cast into raw type
305  uint32_t rawValue = static_cast<uint32_t>(cookedValue);
306 
307  // handle sign
308  if(_isSigned && isNegative) {
309  rawValue = ~rawValue;
310  }
311 
312  // return with bit mask applied
313  return rawValue & _usedBitsMask;
314  }
315  else {
316 
317  // convert into double and scale by fractional bit coefficient
318  double d_cooked = _inverseFractionalBitsCoefficient * static_cast<double>(cookedValue);
319 
320  // Convert into either signed or uinsigned int32_t, depending on _isSigned, so the conversion handls
321  // the sign correctly. Store always in a uint32_t, since this is our raw type.
322  // The conversion will properly round, when needed.
323  // Negative overflow exceptions need to be cought for some corner cases (e.g. number of fractional bits >= number
324  // of bits in total). Positive overflow cannot happen due to the rangecheck above (the negative branch has one
325  // more possible value).
326  uint32_t raw;
327  try {
328  if(_isSigned) {
329  typedef boost::numeric::converter<int32_t, double, boost::numeric::conversion_traits<int32_t,double>,
330  boost::numeric::def_overflow_handler,
331  Round<double> > converter_signed;
332  raw = converter_signed::convert(d_cooked);
333  }
334  else {
335  typedef boost::numeric::converter<uint32_t, double, boost::numeric::conversion_traits<uint32_t,double>,
336  boost::numeric::def_overflow_handler,
337  Round<double> > converter_unsigned;
338  raw = converter_unsigned::convert(d_cooked);
339  }
340  }
341  catch(boost::numeric::negative_overflow &e) {
342  raw = _minRawValue;
343  }
344 
345 
346  // apply bit mask
347  return raw & _usedBitsMask;
348  }
349  }
