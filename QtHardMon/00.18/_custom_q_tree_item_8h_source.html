280 template <typename T>
281 inline void CustomQTreeItem::putValuesIntoTable(
282  const TableWidgetData& tabledata, std::vector<T> const& buffer) {
283  int column = 0;
284  if (typeid(T) == typeid(int)) {
285  column = qthardmon::FIXED_POINT_DISPLAY_COLUMN;
286  } else if (typeid(T) == typeid(double)) {
287  column = qthardmon::FLOATING_POINT_DISPLAY_COLUMN;
288  }
289 
290  clearTable(tabledata);
291 
292  QTableWidget* table = tabledata.table;
293  unsigned int maxRow = tabledata.tableMaxRowCount;
294 
295  size_t rowCount = (buffer.size() > maxRow) ? maxRow : buffer.size();
296  // Prepare the appropriate number of rows in the table
297  table->setRowCount(rowCount);
298 
299  for (size_t row = 0; row < rowCount; row++) {
300  // Prepare a data item with a QVariant. The QVariant takes care that the
301  // type is recognised as
302  // int and a proper editor (spin box) is used when editing in the GUI.
303  QTableWidgetItem* dataItem = new QTableWidgetItem();
304  QTableWidgetItem* rowItem = new QTableWidgetItem();
305  std::stringstream rowAsText;
306  rowAsText << row;
307 
308  rowItem->setText(rowAsText.str().c_str());
309  table->setVerticalHeaderItem(row, rowItem);
310 
311  if (row == maxRow) { // The register is too large to display. Show that it
312  // is truncated and stop reading
313  dataItem->setText("truncated");
314  dataItem->setFlags(dataItem->flags() & ~Qt::ItemIsSelectable &
315  ~Qt::ItemIsEditable);
316  dataItem->setToolTip("List is truncated. You can change the number of "
317  "words displayed in the preferences.");
318  table->setItem(row, 0, dataItem);
319  break;
320  }
321 
322  T registerContent = buffer[row];
323  dataItem->setData(0, QVariant(registerContent)); // 0 is the default role
324  table->setItem(row, column, dataItem);
325  } // for row
326 }

329 template <typename T>
330 inline std::vector<T> CustomQTreeItem::copyValuesFromTable(
331  const TableWidgetData& tabledata, unsigned int count) {
332 
333  unsigned int maxRowCount = tabledata.tableMaxRowCount;
334  QTableWidget const* table = tabledata.table;
335  std::vector<T> buffer(count);
336 
337  for (unsigned int row = 0; (row < count) && (row < maxRowCount); row++) {
338  int registerContent =
339  table->item(row, qthardmon::FIXED_POINT_DISPLAY_COLUMN)
340  ->data(0 /*default role*/)
341  .toInt();
342  buffer[row] = registerContent;
343  }
344  return buffer;
345 }
