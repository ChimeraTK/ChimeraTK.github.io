151  template<typename UserType>
152  DoocsBackendRegisterAccessor<UserType>::DoocsBackendRegisterAccessor(const RegisterPath &path, size_t numberOfWords,
153  size_t wordOffsetInRegister, AccessModeFlags flags, bool allocateBuffers)
154  : NDRegisterAccessor<UserType>(path),
155  _path(path),
156  elementOffset(wordOffsetInRegister),
157  useZMQ(false)
158  {
159  try {
160  // check for unknown access mode flags
161  flags.checkForUnknownFlags({AccessMode::wait_for_new_data});
162 
163  // set address
164  ea.adr(std::string(path).substr(1).c_str()); // strip leading slash
165 
166  // try to read data, to check connectivity and to obtain size of the register
167  doReadTransfer();
168 
169  // obtain number of elements
170  size_t actualLength = dst.array_length();
171  if(actualLength == 0 && dst.length() == 1) {
172  actualLength = 1;
173  isArray = false;
174  }
175  else {
176  if (actualLength == 0)
177  actualLength = dst.length();
178  isArray = true;
179  }
180  if(numberOfWords == 0) {
181  nElements = actualLength;
182  }
183  else {
184  nElements = numberOfWords;
185  }
186  if(nElements + elementOffset > actualLength) {
187  throw DeviceException("Requested number of words exceeds the length of the DOOCS property!",
188  DeviceException::WRONG_PARAMETER);
189  }
190  if(nElements == actualLength && elementOffset == 0) {
191  isPartial = false;
192  }
193  else {
194  isPartial = true;
195  }
196 
197  // allocate buffers
198  if(allocateBuffers) {
199  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
200  NDRegisterAccessor<UserType>::buffer_2D[0].resize(nElements);
201  }
202 
203  // set proper type information in the source EqData
204  src.set_type(dst.type());
205  if(allocateBuffers && dst.type() != DATA_IIII) {
206  src.length(actualLength);
207  }
208 
209  // use ZeroMQ with AccessMode::wait_for_new_data
210  if(flags.has(AccessMode::wait_for_new_data)) {
211  // set flag
212  useZMQ = true;
213 
214  // create notification queue and TransferFuture
215  notifications = cppext::future_queue<EqData>(3);
216  activeFuture = TransferFuture(notifications.then<void>([](EqData&){}, std::launch::deferred), this);
217  futureCreated = true;
218 
219  // subscribe to property
220  int err = dmsg_attach(&ea, &dst, (void*)this, &zmq_callback);
221  if(err) {
222  throw DeviceException(std::string("Cannot subscribe to DOOCS property via ZeroMQ: ")+dst.get_string(),
223  DeviceException::CANNOT_OPEN_DEVICEBACKEND);
224  }
225  // run dmsg_start() once
226  std::unique_lock<std::mutex> lck(DoocsBackend::dmsgStartCalled_mutex);
227  if(!DoocsBackend::dmsgStartCalled) {
228  dmsg_start();
229  DoocsBackend::dmsgStartCalled = true;
230  }
231  }
232  }
233  catch(...) {
234  this->shutdown();
235  throw;
236  }
237  }

298  template<typename UserType>
299  void DoocsBackendRegisterAccessor<UserType>::doReadTransfer() {
300  boost::this_thread::interruption_point();
301  if(!useZMQ) {
302  // read data
303  int rc = eq.get(&ea, &src, &dst);
304  // check error
305  if(rc) {
306  throw DeviceException(std::string("Cannot read from DOOCS property: ")+dst.get_string(),
307  DeviceException::CANNOT_OPEN_DEVICEBACKEND);
308  }
309  }
310  else {
311  // loop to allow retries in case of timeout errors
312  while(true) {
313  // wait until new data has been received
314  notifications.pop_wait(dst);
315  // check for an error
316  if(dst.error() != 0) {
317  // try obtaining data through RPC call instead to verify error
318  int rc = eq.get(&ea, &src, &dst);
319  // if again error received, throw exception
320  if(rc) {
321  throw DeviceException(std::string("Cannot read from DOOCS property: ")+dst.get_string(),
322  DeviceException::CANNOT_OPEN_DEVICEBACKEND);
323  }
324  // otherwise try again
325  continue;
326  }
327  // terminate loop, since no retry needed
328  break;
329  }
330  }
331  // check for error code in dst
332  if(dst.error() != 0) {
333  throw DeviceException(std::string("Cannot read from DOOCS property: ")+dst.get_string(),
334  DeviceException::CANNOT_OPEN_DEVICEBACKEND);
335  }
336  }

366  template<typename UserType>
367  TransferFuture DoocsBackendRegisterAccessor<UserType>::doReadTransferAsync() {
368  // create future_queue if not already created and continue it to enusre postRead is called (in the user thread,
369  // so we use the deferred launch policy)
370  if(!futureCreated) {
371  notifications = cppext::future_queue<EqData>(2);
372  activeFuture = TransferFuture(notifications.then<void>([](EqData&){}), this);
373  futureCreated = true;
374  }
375 
376  if(!useZMQ) {
377  // launch doReadTransfer in separate thread
378  readAsyncThread = boost::thread(
379  [this] {
380  try {
381  this->doReadTransfer();
382  }
383  catch(...) {
384  this->notifications.push_exception(std::current_exception());
385  throw;
386  }
387  this->notifications.push({});
388  }
389  );
390  }
391 
392  // return the TransferFuture
393  return activeFuture;
394  }
