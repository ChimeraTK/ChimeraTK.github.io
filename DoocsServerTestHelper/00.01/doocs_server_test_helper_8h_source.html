214  template<typename TYPE>
215  std::vector<TYPE> doocsServerTestHelper::doocsGetArray( const char *name ) {
216  EqAdr ad;
217  EqData ed, res;
218  // obtain location pointer
219  ad.adr(name);
220  EqFct *p = eq_get(&ad);
221  ASSERT(p != NULL, std::string("Could not get location for property ")+name);
222  // for D_BSpectrum: set IIII structure to obtain always the latest buffer (shouldn't hurt for others)
223  IIII iiii;
224  iiii.i1_data = -1;
225  iiii.i2_data = -1;
226  iiii.i3_data = -1;
227  iiii.i4_data = -1;
228  ed.set(&iiii);
229  // obtain values
230  p->lock();
231  p->get(&ad,&ed,&res);
232  p->unlock();
233  // check for errors
234  ASSERT(res.error() == 0, std::string("Error reading property ")+name);
235  // copy to vector and return it
236  std::vector<TYPE> val;
237  if(std::is_integral<TYPE>()) {
238  if(res.type() != DATA_A_LONG) {
239  for(int i=0; i<res.length(); i++) val.push_back(res.get_int(i));
240  }
241  else {
242  for(int i=0; i<res.length(); i++) val.push_back(res.get_long(i));
243  }
244  }
245  else if(std::is_floating_point<TYPE>()) {
246  for(int i=0; i<res.length(); i++) val.push_back(res.get_float(i));
247  }
248  else {
249  ASSERT(false, "Wrong type passed as tempalate argument.");
250  }
251  return val;
252  }
