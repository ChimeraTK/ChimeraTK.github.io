317  void callForType(const std::type_info& type, LAMBDATYPE lambda) {
318  bool done = false;
319 
320  // iterate through a userTypeMap() and test all types
321  for_each(userTypeMap(), [&type, &lambda, &done](auto pair) {
322  if(type != typeid(pair.second)) return;
323  // if the right type has been found, call the original lambda
324  lambda(pair.second);
325  done = true;
326  });
327 
328  // Check if done flag has been set. If not, an unknown type has been passed.
