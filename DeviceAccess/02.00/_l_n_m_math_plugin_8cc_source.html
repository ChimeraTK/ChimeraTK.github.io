66  template<typename UserType>
67  MathPluginDecorator<UserType>::MathPluginDecorator(boost::shared_ptr<LogicalNameMappingBackend>& backend,
68  const boost::shared_ptr<ChimeraTK::NDRegisterAccessor<double>>& target,
69  const std::map<std::string, std::string>& parameters)
70  : ChimeraTK::NDRegisterAccessorDecorator<UserType, double>(target) {
71  // 2D arrays are not yet supported
72  if(_target->getNumberOfChannels() != 1) {
73  throw ChimeraTK::logic_error(
74  "The LogicalNameMapper MathPlugin supports only scalar or 1D array registers. Register name: " +
75  this->getName());
76  }
77 
78  // Although this is a decorator we do not use the exceptions backend from the target because this decorator is given out by a backend
79  target->setExceptionBackend(backend);
80  this->_exceptionBackend = backend;
81 
82  // create exprtk parser
83  exprtk::parser<double> parser;
84 
85  // add basic constants like pi
86  symbols.add_constants();
87 
88  // Create vector view for the value and add it to the symbol table. We need to use a vector view instead of adding
89  // the buffer directly as a vector, since our buffers might be swapped and hence the address of the data can
90  // change.
91  valueView = std::make_unique<exprtk::vector_view<double>>(
92  exprtk::make_vector_view(_target->accessChannel(0), _target->accessChannel(0).size()));
93  symbols.add_vector("x", *valueView);
94 
95  // iterate parameters, add all but 'formula' parameter as a variable
96  for(auto& parpair : parameters) {
97  if(parpair.first == "formula") continue;
98  auto acc = backend->getRegisterAccessor<double>(parpair.second, 0, 0, {});
99  acc->setExceptionBackend(backend);
100  if(acc->getNumberOfChannels() != 1) {
101  throw ChimeraTK::logic_error(
102  "The LogicalNameMapper MathPlugin supports only scalar or 1D array registers. Register name: '" +
103  this->getName() + "', parameter name: '" + parpair.first + "'");
104  }
105  auto view = std::make_unique<exprtk::vector_view<double>>(
106  exprtk::make_vector_view(acc->accessChannel(0), acc->getNumberOfSamples()));
107  symbols.add_vector(parpair.first, *view);
108  params[acc] = std::move(view);
109  }
110 
111  // compile the expression
112  expression.register_symbol_table(symbols);
113  bool success = parser.compile(parameters.at("formula"), expression);
114  if(!success) {
115  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + this->getName() +
116  "': failed to compile expression '" + parameters.at("formula") + "': " + parser.error());
117  }
118  }

122  template<typename UserType>
123  void MathPluginDecorator<UserType>::doPostRead(TransferType type, bool hasNewData) {
124  _target->postRead(type, hasNewData);
125  if(!hasNewData) return;
126 
127  // update data pointer
128  valueView->rebase(_target->accessChannel(0).data());
129 
130  // update parameters
131  for(auto& p : params) {
132  p.first->read();
133  p.second->rebase(p.first->accessChannel(0).data());
134  }
135 
136  // evaluate the expression, obtain the result in a way so it also works when using the return statement
137  double valueWhenNotUsingReturn = expression.value();
138  exprtk::results_context<double> results = expression.results();
139 
140  if(results.count() == 0) {
141  // if results.count() is 0, the return statement presumably has not been used
142  if(buffer_2D[0].size() != 1) {
143  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + this->getName() +
144  "': The expression returns a scalar but " + std::to_string(buffer_2D[0].size()) +
145  " expected for read operations.");
146  }
147  buffer_2D[0][0] = numericToUserType<UserType>(valueWhenNotUsingReturn);
148  this->_versionNumber = _target->getVersionNumber();
149  this->_dataValidity = _target->dataValidity();
150  }
151  else if(results.count() == 1) {
152  // return statement has been used to return exactly one value (note: this value might be an array)
153  exprtk::type_store<double> result = results[0];
154 
155  // make sure we got a numeric result
156  if(result.type != exprtk::type_store<double>::e_scalar && result.type != exprtk::type_store<double>::e_vector) {
157  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + this->getName() +
158  "': The expression did not return a numeric result.");
159  }
160 
161  // create vector view and check that its size matches
162  exprtk::type_store<double>::type_view<double> view(result);
163  if(view.size() != buffer_2D[0].size()) {
164  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + this->getName() +
165  "': The expression returns " + std::to_string(view.size()) + " elements but " +
166  std::to_string(buffer_2D[0].size()) + " expected for read operations.");
167  }
168 
169  // convert and copy data into user buffer
170  for(size_t k = 0; k < view.size(); ++k) {
171  buffer_2D[0][k] = numericToUserType<UserType>(view[k]);
172  }
173  this->_versionNumber = _target->getVersionNumber();
174  this->_dataValidity = _target->dataValidity();
175  }
176  else {
177  // multiple results in return statement are unexpected
178  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + this->getName() +
179  "': The expression returned " + std::to_string(results.count()) + " results, expect exactly one result.");
180  }
181  }

187  // convert from UserType to double - use the target accessor's buffer as a temporary buffer (this is a bit a hack,
188  // but it is safe to overwrite the buffer and we can avoid the need for an additional permanent buffer which might
189  // not even be used if the register is never written).
190  for(size_t k = 0; k < buffer_2D[0].size(); ++k) {
191  _target->accessData(0, k) = userTypeToNumeric<double>(buffer_2D[0][k]);
192  }
193 
194  // inform the value view of the new data pointer - the buffer might have been swapped
195  valueView->rebase(_target->accessChannel(0).data());
196 
197  // update parameters
198  for(auto& p : params) {
199  p.first->read();
200  p.second->rebase(p.first->accessChannel(0).data());
201  }
202

204  double valueWhenNotUsingReturn = expression.value();
205  exprtk::results_context<double> results = expression.results();
206  if(results.count() == 0) {
207  // if results.count() is 0, the return statement presumably has not been used
208  if(_target->accessChannel(0).size() != 1) {
209  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + this->getName() +
210  "': The expression returns a scalar but " + std::to_string(buffer_2D[0].size()) +
211  " expected for write operations.");
212  }
213  _target->accessData(0, 0) = valueWhenNotUsingReturn;
214  }
215  else if(results.count() == 1) {
216  // return statement has been used to return exactly one value (note: this value might be an array)
217  exprtk::type_store<double> result = results[0];
218

221  throw ChimeraTK::logic_error("LogicalNameMapping MathPlugin for register '" + this->getName() +
222  "': The expression did not return a numeric result.");
