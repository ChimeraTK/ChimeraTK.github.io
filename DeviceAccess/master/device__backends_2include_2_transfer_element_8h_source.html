132  bool readLatest() {

175  TransferFuture& readAsync() {
176  if(!hasActiveFuture) {
177  // call preRead
178  this->readTransactionInProgress = false;
179  this->preRead();
180 
181  // initiate asynchronous transfer and return future

202  preWrite();
