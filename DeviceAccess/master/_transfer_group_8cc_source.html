59  void TransferGroup::addAccessor(TransferElementAbstractor &accessor) {
60 
61  // check if accessor is already in a transfer group
62  if(accessor.getHighLevelImplElement()->isInTransferGroup) {
63  throw ChimeraTK::logic_error("The given accessor is already in a TransferGroup and cannot be added to another.");
64  }
65 
66  // set flag on the accessors that it is now in a transfer group
67  accessor.getHighLevelImplElement()->isInTransferGroup = true;
68 
69  auto highLevelElementsWithNewAccessor = highLevelElements;
70  highLevelElementsWithNewAccessor.insert(accessor.getHighLevelImplElement());
71 
72  // try replacing all internal elements in all high-level elements
73  for(auto &hlElem1 : highLevelElementsWithNewAccessor) {
74 
75  auto list = hlElem1->getInternalElements();
76  list.push_front(hlElem1);
77 
78  for(auto &replacement : list) {
79  // try on the abstractor first, to make sure we replace at the highest level if possible
80  accessor.replaceTransferElement(replacement);
81  // try on all high-level elements already stored in the list
82  for(auto &hlElem : highLevelElementsWithNewAccessor) {
83  hlElem->replaceTransferElement(replacement); // note: this does nothing, if the replacement cannot be used by the hlElem!
84  }
85  }
86  }
87 
88  // store the accessor in the list of high-level elements
89  // this must be done only now, since it may have been replaced during the replacement process
90  highLevelElements.insert(accessor.getHighLevelImplElement());
91 
92  // update the list of hardware-accessing elements, since we might just have made some of them redundant
93  // since we are using a set to store the elements, duplicates are intrinsically avoided.
94  lowLevelElements.clear();
95  for(auto &hlElem : highLevelElements) {
96  for(auto &hwElem : hlElem->getHardwareAccessingElements()) lowLevelElements.insert(hwElem);
97  }
98 
99  // update the list of CopyRegisterDecorators
100  copyDecorators.clear();
101  for(auto &hlElem : highLevelElements) {
102  if(boost::dynamic_pointer_cast<ChimeraTK::CopyRegisterDecoratorTrait>(hlElem) != nullptr) {
103  copyDecorators.insert(hlElem);
104  }
105  for(auto &hwElem : hlElem->getInternalElements()) {
106  if(boost::dynamic_pointer_cast<ChimeraTK::CopyRegisterDecoratorTrait>(hwElem) != nullptr) {
107  copyDecorators.insert(hwElem);
108  }
109  }
110  }
111 
112  // Update read-only flag
113  if(accessor.isReadOnly()) readOnly = true;
114  }
