27  LNMBackendChannelAccessor(boost::shared_ptr<DeviceBackend> dev, const RegisterPath &registerPathName,
28  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
29  : NDRegisterAccessor<UserType>(registerPathName),

31  {
32  // check for unknown flags
33  flags.checkForUnknownFlags({AccessMode::raw});
34  // check for illegal parameter combinations
35  if(flags.has(AccessMode::raw)) {
36  throw DeviceException("LNMBackendChannelAccessor: raw access not yet supported!", DeviceException::NOT_IMPLEMENTED);
37  }
38  _dev = boost::dynamic_pointer_cast<LogicalNameMappingBackend>(dev);
39  // copy the register info and create the internal accessors, if needed
40  _info = *( boost::static_pointer_cast<LNMBackendRegisterInfo>(
41  _dev->getRegisterCatalogue().getRegister(_registerPathName)) );
42  _info.createInternalAccessors(dev);
43  // check for incorrect usage of this accessor
44  if( _info.targetType != LNMBackendRegisterInfo::TargetType::CHANNEL ) {
45  throw DeviceException("LNMBackendChannelAccessor used for wrong register type.",
46  DeviceException::WRONG_PARAMETER); // LCOV_EXCL_LINE (impossible to test...)
47  }
48  // get target device and accessor
49  std::string devName = _info.deviceName;
50  if(devName != "this") {
51  _targetDevice = _dev->_devices[devName];
52  }
53  else {
54  _targetDevice = dev;
55  }
56  _accessor = _targetDevice->getRegisterAccessor<UserType>(RegisterPath(_info.registerName), numberOfWords,wordOffsetInRegister, false);
57  // allocate the buffer
58  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
59  NDRegisterAccessor<UserType>::buffer_2D[0].resize(_accessor->getNumberOfSamples());
60  }
