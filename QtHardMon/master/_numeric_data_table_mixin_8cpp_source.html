40  // We have two editable fields - The decimal field and double field.
41  // The values reflect each other and to avoid an infinite

43  // only if required
44  //

58  if (currentValueInDoubleColumn == fractionalVersionOfUserValue)
59  return; // same value in the corresponding double cell, so not updating
60  // this cell
61  }
62  // If here, This is a new value. Trigger update of the float cell
63 
64  writeCell<double>(row, qthardmon::FLOATING_POINT_DISPLAY_COLUMN,
65  fractionalVersionOfUserValue);
66 
67  } else if (column == qthardmon::FLOATING_POINT_DISPLAY_COLUMN) {
68  double userUpdatedValueInCell = readCell<double>(row, column);
69  int FixedPointVersionOfUserValue =
70  convertToFixedPoint(userUpdatedValueInCell);
71 
72  if (isValidCell(row, qthardmon::FIXED_POINT_DISPLAY_COLUMN)) {
73  int currentValueInFixedPointCell =
74  readCell<int>(row, qthardmon::FIXED_POINT_DISPLAY_COLUMN);
75  double convertedValueFrmFPCell =
76  convertToDouble(currentValueInFixedPointCell);
77  if (userUpdatedValueInCell == convertedValueFrmFPCell)
78  return;
79  }
80 
81  writeCell<int>(
82  row, qthardmon::FIXED_POINT_DISPLAY_COLUMN,
83  FixedPointVersionOfUserValue); // This will trigger an update to
84  // the fixed point display column,
85  // which will in turn correct the
86  // value in this double cell to a
87  // valid one (In case the user entered one is not supported by the floating
88  // point converter settings)
89  } else if (column == qthardmon::HEX_VALUE_DISPLAY_COLUMN) {
90  HexData hexInCell = readCell<HexData>(row, column);

94  int currentValueInFixedPointCell =
95  readCell<int>(row, qthardmon::FIXED_POINT_DISPLAY_COLUMN);
