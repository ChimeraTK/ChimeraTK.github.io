86  void doPreWrite() override {
87  for(auto &slave : FanOut<UserType>::slaves) { // send out copies to slaves
88  if(slave->getNumberOfSamples() != 0) { // do not send copy if no data is expected (e.g. trigger)
89  if(slave == FanOut<UserType>::slaves.front()) { // in case of first slave, swap instead of copy
90  slave->accessChannel(0).swap(mtca4u::NDRegisterAccessor<UserType>::buffer_2D[0]);
91  }
92  else { // not the first slave: copy the data from the first slave
93  slave->accessChannel(0) = FanOut<UserType>::slaves.front()->accessChannel(0);
94  }
95  }
96  }
97  // pre write may only be called on the target accessors after we have filled them all, otherwise the first
98  // accessor might take us the data away...
99  for(auto &slave : FanOut<UserType>::slaves) {
100  slave->preWrite();
101  }
102  }
