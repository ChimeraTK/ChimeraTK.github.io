27  LNMBackendBitAccessor(boost::shared_ptr<DeviceBackend> dev, const RegisterPath &registerPathName,
28  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
29  : SyncNDRegisterAccessor<UserType>(registerPathName),
30  _registerPathName(registerPathName),
31  _fixedPointConverter(registerPathName, 32, 0, 1)
32  {
33  try {
34  // check for unknown flags
35  flags.checkForUnknownFlags({AccessMode::raw});
36  // check for illegal parameter combinations
37  if(flags.has(AccessMode::raw)) {
38  throw ChimeraTK::logic_error("LNMBackendBitAccessor: raw access not supported!");
39  }
40  _dev = boost::dynamic_pointer_cast<LogicalNameMappingBackend>(dev);
41  // copy the register info and create the internal accessors, if needed
42  _info = *( boost::static_pointer_cast<LNMBackendRegisterInfo>(
43  _dev->getRegisterCatalogue().getRegister(_registerPathName)) );
44  // check for incorrect usage of this accessor
45  if( _info.targetType != LNMBackendRegisterInfo::TargetType::BIT ) {
46  throw ChimeraTK::logic_error("LNMBackendBitAccessor used for wrong register type."); // LCOV_EXCL_LINE (impossible to test...)
47  }
48  // get target device and accessor
49  std::string devName = _info.deviceName;
50  boost::shared_ptr<DeviceBackend> targetDevice;
51  if(devName != "this") {
52  targetDevice = _dev->_devices[devName];
53  }
54  else {
55  targetDevice = dev;
56  }
57  _accessor = targetDevice->getRegisterAccessor<uint64_t>(RegisterPath(_info.registerName), numberOfWords,wordOffsetInRegister, false);
58  // allocate and initialise the buffer
59  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
60  NDRegisterAccessor<UserType>::buffer_2D[0].resize(1);
61  NDRegisterAccessor<UserType>::buffer_2D[0][0] = _fixedPointConverter.toCooked<UserType>(false);
62  // set the bit mask
63  _bitMask = 1 << _info.bit;
64  }
65  catch(...) {
66  this->shutdown();
67  throw;
68  }
69  }
