73  VirtualModule& DeviceModule::virtualiseFromCatalog() const {
74  if(virtualisedModuleFromCatalog_isValid) return virtualisedModuleFromCatalog;
75 
76  virtualisedModuleFromCatalog = VirtualModule(deviceAliasOrURI, "Device module", ModuleType::Device);
77 
78  // obtain register catalogue
79  Device d;
80  d.open(deviceAliasOrURI);
81  auto catalog = d.getRegisterCatalogue();
83 
84  // iterate catalogue, create VariableNetworkNode for all registers starting
85  // with the registerNamePrefix
86  size_t prefixLength = registerNamePrefix.length();
87  for(auto& reg : catalog) {
88  if(std::string(reg.getRegisterName()).substr(0, prefixLength) != std::string(registerNamePrefix)) continue;
89 
90  // ignore 2D registers
91  if(reg.getNumberOfDimensions() > 1) continue;
92 
93  // guess direction and determine update mode
94  VariableDirection direction;
95  UpdateMode updateMode;
96  if(reg.isWriteable()) {
97  direction = {VariableDirection::consuming, false};
98  updateMode = UpdateMode::push;
99  }
100  else {
101  direction = {VariableDirection::feeding, false};
102  if(reg.getSupportedAccessModes().has(AccessMode::wait_for_new_data)) {
103  updateMode = UpdateMode::push;
104  }
105  else {
106  updateMode = UpdateMode::poll;
107  }
108  }
109 
110  // guess type
111  const std::type_info* valTyp{&typeid(AnyType)};
112  auto& dd = reg.getDataDescriptor(); // numeric, string, boolean, nodata, undefined
113  if(dd.fundamentalType() == RegisterInfo::FundamentalType::numeric) {
114  if(dd.isIntegral()) {
115  if(dd.isSigned()) {
116  if(dd.nDigits() > 11) {
117  valTyp = &typeid(int64_t);
118  }
119  else if(dd.nDigits() > 6) {
120  valTyp = &typeid(int32_t);
121  }
122  else if(dd.nDigits() > 4) {
123  valTyp = &typeid(int16_t);
124  }
125  else {
126  valTyp = &typeid(int8_t);
127  }
128  }
129  else {
130  if(dd.nDigits() > 10) {
131  valTyp = &typeid(uint64_t);
132  }
133  else if(dd.nDigits() > 5) {
134  valTyp = &typeid(uint32_t);
135  }
136  else if(dd.nDigits() > 3) {
137  valTyp = &typeid(uint16_t);
138  }
139  else {
140  valTyp = &typeid(uint8_t);
141  }
142  }
143  }
144  else { // fractional
145  valTyp = &typeid(double);
146  }
147  }
148  else if(dd.fundamentalType() == RegisterInfo::FundamentalType::boolean) {
149  valTyp = &typeid(int32_t);
150  }
151  else if(dd.fundamentalType() == RegisterInfo::FundamentalType::string) {
152  valTyp = &typeid(std::string);
153  }
154 
155  auto name = std::string(reg.getRegisterName()).substr(prefixLength);
156  auto lastSlash = name.find_last_of("/");
157  auto dirname = name.substr(0, lastSlash);
158  auto basename = name.substr(lastSlash + 1);
159  VariableNetworkNode node(
160  basename, deviceAliasOrURI, reg.getRegisterName(), updateMode, direction, *valTyp, reg.getNumberOfElements());
161  virtualisedModuleFromCatalog.createAndGetSubmoduleRecursive(dirname).addAccessor(node);
162  }
163 
164  virtualisedModuleFromCatalog_isValid = true;
165  return virtualisedModuleFromCatalog;
166  }

178  void DeviceModule::handleException() {
179  Application::registerThread("DM_" + getName());
180  Device d;
181  std::string error;
182 
183  try {
184  while(true) {
185  errorQueue.pop_wait(error);
186  boost::this_thread::interruption_point();
187  std::lock_guard<std::mutex> lk(errorMutex);
188  deviceError.status = 1;
189  deviceError.message = error;
190  deviceError.setCurrentVersionNumber({});
191  deviceError.writeAll();
192  while(true) {
193  boost::this_thread::interruption_point();
194  try {
195  d.open(deviceAliasOrURI);
196  if(d.isOpened()) {
197  break;
198  }
199  }
200  catch(std::exception& ex) {
201  deviceError.status = 1;
202  deviceError.message = ex.what();
203  deviceError.setCurrentVersionNumber({});
204  deviceError.writeAll();
205  }
206  usleep(500000);
207  }
208  deviceError.status = 0;
209  deviceError.message = "";
210  deviceError.setCurrentVersionNumber({});
211  deviceError.writeAll();
212  errorCondVar.notify_all();
213  }
214  }
215  catch(...) {
216  // before we leave this thread, we might need to notify other waiting
217  // threads. boost::this_thread::interruption_point() throws an exception
218  // when the thread should be interrupted, so we will end up here
219  errorCondVar.notify_all();
220  throw;
221  }
222  }
