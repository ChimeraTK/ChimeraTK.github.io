42 std::pair<DATA_TYPE, bool> qvariantToStandardDataType(const QVariant & data){
43  if (data.type() == HexDataType){
44  if (! data.canConvert<HexData>() ){
45  return {DATA_TYPE(), false};
46  }
47  // The variant has a templated function called value() which returns the template type,
48  // HexData in our case. The HexData class itself has a member called value, which holds the
49  // actual value. Phew...
50  return {data.value<HexData>().value, true};
51  }else{
52  if (! data.canConvert<DATA_TYPE>() ){
53  return {DATA_TYPE(), false};
54  }
55  return {data.value<DATA_TYPE>(), true};
56  }
