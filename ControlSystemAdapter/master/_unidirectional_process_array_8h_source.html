408  // It would be better to do the validation before initializing, but this

410  if(!this->isReadable()) {

585  template<class T>
586  bool UnidirectionalProcessArray<T>::writeInternal(VersionNumber newVersionNumber, bool shouldCopy) {
587  // thread safety check, if enabled (only active with debug flags enabled)
588  assert(checkThreadSafety());
589 
590  assert(this->isWriteable());
591 
592  // A version should never be send with a version number that is equal to or
593  // even less than the last version number used. Such an attempt indicates
594  // that there is a problem in the logic attempting the write operation.
595  if(newVersionNumber < _versionNumber) {
596  throw ChimeraTK::logic_error("The version number passed to write() is less than the last version number used.");
597  }
598 
599  // First update the persistent data storage, if any was associated. This
600  // cannot be done after sending, since the value might no longer be available
601  // within this instance.
602  if(_persistentDataStorage) {
603  _persistentDataStorage->updateValue(_persistentDataStorageID, ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0]);
604  }
605 
606  // Set time stamp and version number
607  _localBuffer._versionNumber = newVersionNumber;
608  _localBuffer._dataValidity = dataValidity();
609  _versionNumber = newVersionNumber;
610 
611  // set the data by copying or swapping
612  assert(_localBuffer._value.size() == ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0].size());
613  if(shouldCopy) {
614  _localBuffer._value = ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0];
615  }
616  else {
617  _localBuffer._value.swap(ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0]);
618  }
619 
620  // send the data to the queue
621  bool dataNotLost = _sharedState._queue.push_overwrite(std::move(_localBuffer));
622 
623  // notify send notification listener, if present
624  if(_sendNotificationListener) {
625  _sendNotificationListener->notify(_receiver);
626  }
627 
628  return !dataNotLost;
629  }
