63  void run() {
64  Application::registerThread("TrFO" + externalTrigger->getName());
65  Application::testableModeLock("start");
66  while(true) {
67  // wait for external trigger
68  boost::this_thread::interruption_point();
69  Profiler::stopMeasurement();
70  externalTrigger->read();
71  Profiler::startMeasurement();
72  boost::this_thread::interruption_point();
73  // receive data. We need to catch exceptions here, since the ExceptionHandlingDecorator cannot do this for us
74  // inside a TransferGroup, if the exception is thrown inside doReadTransfer() (as it is directly called on the
75  // lowest-level TransferElements inside the group).
76  retry:
77  try {
78  transferGroup.read();
79  }
80  catch(ChimeraTK::runtime_error& e) {
81  dm.reportException(e.what());
82  goto retry;
83  }
84  // send the data to the consumers
85  auto version = externalTrigger->getVersionNumber();
86  boost::fusion::for_each(fanOutMap.table, SendDataToConsumers(version));
87  }
88  }

105  fanOut->accessChannel(0).swap(feeder->accessChannel(0));
