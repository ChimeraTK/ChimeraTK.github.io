357  inline boost::shared_ptr<TransferElement> TransferElement::readAny(std::list<std::reference_wrapper<TransferElement>> elementsToRead) {
358  std::list<TransferFuture*> futureList;
359  for(auto elem : elementsToRead) {
360  auto *future = &(elem.get().readAsync());
361  futureList.push_back(future);
362  }
363  // wait until any future is ready
364  auto iter = boost::wait_for_any(TransferFutureIterator(futureList.begin()), TransferFutureIterator(futureList.end()));
365 
366  // Find the variable which has the oldest version number (to guarantee the order of updates).
367  // Start with assuming that the future returned by boost::wait_for_any() has the oldes version number.
368  TransferFuture *theUpdate = &(iter.getTransferFuture());
369  for(auto future : futureList) {
370  // skip if this is the future returned by boost::wait_for_any()
371  if(future == theUpdate) continue;
372  // also skip if the future is not yet ready
373  if(future->getBoostFuture().wait_for(boost::chrono::duration<int, boost::centi>(0)) == boost::future_status::timeout) {
374  continue;
375  }
376  // compare version number with the version number of the stored future
377  if(future->getBoostFuture().get()->_versionNumber < theUpdate->getBoostFuture().get()->_versionNumber) {
378  theUpdate = future;
379  }
380  }
381 
382  // complete the transfer (i.e. run postRead())
383  theUpdate->wait();
384 
385  // return the transfer element as a shared pointer
386  return theUpdate->getTransferElement().shared_from_this();
387  }
