83  void StepperMotorWithReferenceStateMachine::calibrationThreadFunction(){
84  _motor._calibrationFailed.exchange(false);
85  _stopAction.exchange(false);
86  _moveInterrupted.exchange(false);
87 
88  // Local variables for calibrated position
89  int calibPositiveEndSwitchInSteps = 0;
90  int calibNegativeEndSwitchInSteps = 0;
91 
92  // At least one end switch disabled -> calibration not possible
93  if(!(_motor._positiveEndSwitchEnabled.load() && _motor._negativeEndSwitchEnabled.load())){
94 
95  _motor._motorControler->setCalibrationTime(0);
96 
97  _motor._calibrationFailed.exchange(true);
98  _motor._calibrationMode.exchange(StepperMotorCalibrationMode::NONE);
99  }
100  else{
101  findEndSwitch(Sign::POSITIVE);
102  calibPositiveEndSwitchInSteps = _motor.getCurrentPositionInSteps();
103  findEndSwitch(Sign::NEGATIVE);
104  calibNegativeEndSwitchInSteps = _motor.getCurrentPositionInSteps();
105 
106  if (_moveInterrupted.load() || _stopAction.load()){
107 
108  _motor._motorControler->setCalibrationTime(0);
109 
110  _motor._calibrationFailed.exchange(true);
111  _motor._errorMode.exchange(StepperMotorError::CALIBRATION_ERROR);
112  _motor._calibrationMode.exchange(StepperMotorCalibrationMode::NONE);
113  }else{
114  // Define positive axis with negative end switch as zero
115  calibPositiveEndSwitchInSteps = calibPositiveEndSwitchInSteps - calibNegativeEndSwitchInSteps;
116  calibNegativeEndSwitchInSteps = 0;
117 
118  _motor._motorControler->setCalibrationTime(time(nullptr));
119  _motor._motorControler->setPositiveReferenceSwitchCalibration(calibPositiveEndSwitchInSteps);
120  _motor._motorControler->setNegativeReferenceSwitchCalibration(0);
121  _motor._calibPositiveEndSwitchInSteps.exchange(calibPositiveEndSwitchInSteps);
122  _motor._calibNegativeEndSwitchInSteps.exchange(0);
123 
124  _motor._calibrationMode.exchange(StepperMotorCalibrationMode::FULL);
125  _motor.resetMotorControllerPositions(0);
126  }
127  }
128 
129  _asyncActionActive.exchange(false);
130  return;
131  }

202  double StepperMotorWithReferenceStateMachine::getToleranceEndSwitch(Sign sign){
203  double meanMeasurement = 0;
204  double stdMeasurement = 0;
205  const int N_TOLERANCE_CALC_SAMPLES = 10;
206  double measurements[N_TOLERANCE_CALC_SAMPLES];
207 
208  int endSwitchPosition = getPositionEndSwitch(sign);
209 
210  // Get 10 samples for tolerance calculation
211  for (unsigned i=0; i<N_TOLERANCE_CALC_SAMPLES; i++){
212  if (_stopAction.load() || _moveInterrupted.load()){
213  break;
214  }
215 
216  //Move close to end switch
217  {
218  boost::lock_guard<boost::mutex> lck(_motor._mutex);
219  _motor._motorControler->setTargetPosition(endSwitchPosition - static_cast<int>(sign)*1000);
220  }
221  while(_motor._motorControler->isMotorMoving()){
222  std::this_thread::sleep_for(std::chrono::milliseconds(wakeupPeriodInMilliseconds));
223  }
224 
225  //Check if in expected position
226  if (!_motor.verifyMoveAction()){
227  _moveInterrupted.exchange(true);
228  break;
229  }
230 
231  // Try to move beyond end switch
232  {
233  boost::lock_guard<boost::mutex> lck(_motor._mutex);
234  _motor._motorControler->setTargetPosition(endSwitchPosition + static_cast<int>(sign)*1000);
235  }
236  while(_motor._motorControler->isMotorMoving()){
237  std::this_thread::sleep_for(std::chrono::milliseconds(wakeupPeriodInMilliseconds));
238  }
239  if (!_motor.isEndSwitchActive(sign)){
240  _moveInterrupted.exchange(true);
241  break;
242  }
243 
244  // Mean calculation
245  meanMeasurement += static_cast<double>(_motor.getCurrentPositionInSteps())
246  / N_TOLERANCE_CALC_SAMPLES;
247  measurements[i] = _motor.getCurrentPositionInSteps();
248  } /* for (i in [0,9]) */
249 
250  // Compute variance
251  if (!(_stopAction.load() || _moveInterrupted.load()) ){
252  for (unsigned int i=0; i<N_TOLERANCE_CALC_SAMPLES; i++){
253  stdMeasurement += ((measurements[i] - meanMeasurement) / (N_TOLERANCE_CALC_SAMPLES-1)) * (measurements[i] - meanMeasurement);
254  }
255  }
256 
257  return sqrt(stdMeasurement);
258  }
