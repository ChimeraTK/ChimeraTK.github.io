28  LNMBackendVariableAccessor(boost::shared_ptr<DeviceBackend> dev, const RegisterPath &registerPathName,
29  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
30  : SyncNDRegisterAccessor<UserType>(registerPathName),
31  _registerPathName(registerPathName),
32  _fixedPointConverter(registerPathName, 32, 0, 1)
33  {
34  try {
35  // check for unknown flags
36  flags.checkForUnknownFlags({AccessMode::raw});
37  // check for illegal parameter combinations
38  if(wordOffsetInRegister != 0 || numberOfWords > 1) {
39  throw DeviceException("LNMBackendBufferingVariableAccessor: offset and number of words not "
40  "supported!", DeviceException::NOT_IMPLEMENTED); // LCOV_EXCL_LINE (impossible to test...)
41  }
42  if(flags.has(AccessMode::raw)) {
43  if(typeid(UserType) != typeid(int32_t)) {
44  throw DeviceException("Given UserType when obtaining the LNMBackendBufferingVariableAccessor in raw mode"
45  " does not match the expected type. Use an int32_t instead!", DeviceException::WRONG_PARAMETER);
46  }
47  }
48  _dev = boost::dynamic_pointer_cast<LogicalNameMappingBackend>(dev);
49  // obtain the register info
50  _info = boost::static_pointer_cast<LNMBackendRegisterInfo>(
51  _dev->getRegisterCatalogue().getRegister(_registerPathName));
52  // check for incorrect usage of this accessor
53  if( _info->targetType != LNMBackendRegisterInfo::TargetType::INT_CONSTANT &&
54  _info->targetType != LNMBackendRegisterInfo::TargetType::INT_VARIABLE ) {
55  throw DeviceException("LNMBackendBufferingVariableAccessor used for wrong register type.",
56  DeviceException::WRONG_PARAMETER); // LCOV_EXCL_LINE (impossible to test...)
57  }
58  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
59  NDRegisterAccessor<UserType>::buffer_2D[0].resize(1);
60  NDRegisterAccessor<UserType>::buffer_2D[0][0] = _fixedPointConverter.toCooked<UserType>(_info->value);
61  }
62  catch(...) {
63  this->shutdown();
64  throw;
65  }
66  }
