452  }

537  template<class T>
538  bool UnidirectionalProcessArray<T>::doReadTransferLatest() {
539 
540  // As long as there is more than one valid element on the queue, discard it.
541  // Due to our implementation there is always one unfulfilled future in the queue, so
542  // we must pop until there are two elements left in order not to flush out the newest valid value.
543  auto theFuture = mtca4u::TransferElement::readAsync().getBoostFuture();
544  while( theFuture.wait_for(boost::chrono::duration<int, boost::centi>(0)) != boost::future_status::timeout
545  && _sharedState->_fullBufferQueue.read_available() > 2 ) {
546  // discard data by moving the buffer to the empty buffer queue
547  TransferFuture::Data *discardedBuffer = theFuture.get();
548  _sharedState->_fullBufferQueue.pop();
549  mtca4u::TransferElement::hasActiveFuture = false;
550  _sharedState->_emptyBufferQueue.push(static_cast<Buffer*>(discardedBuffer)); // static cast is ok, we never put something else into the queue
551  theFuture = mtca4u::TransferElement::readAsync().getBoostFuture();
552  }
553  // Check whether data is present in the atomic triple buffer. If it is newer, we also need to discard the last
554  // valid element in the queue. If it is older, we need to discard the data in the triple buffer.
555  // First update the triple buffer if it is not valid on our side
556  if(!(_tripleBufferIndex->_isValid)) {
557  _tripleBufferIndex = _sharedState->_tripleBufferIndex.exchange(_tripleBufferIndex);
558  }
559  // if it now contains a valid element, check if it is newer
560  if(_tripleBufferIndex->_isValid) {
561  uint64_t tripleBufferVersion = _tripleBufferIndex->_internalVersionNumber;
562  uint64_t queueVersion = static_cast<Buffer*>(theFuture.get())->_internalVersionNumber;
563  if(tripleBufferVersion > queueVersion) {
564  TransferFuture::Data *discardedBuffer = theFuture.get();
565  _sharedState->_fullBufferQueue.pop();
566  _sharedState->_emptyBufferQueue.push(static_cast<Buffer*>(discardedBuffer));
567  }
568  else {
569  _tripleBufferIndex->_isValid = false;
570  }
571  }
572  return this->doReadTransferNonBlocking();
573  }

577  template<class T>
578  mtca4u::TransferFuture UnidirectionalProcessArray<T>::doReadTransferAsync() {
579 
580  if (!this->isReadable()) {
581  throw std::logic_error("Receive operation is only allowed for a receiver process variable.");
582  }
583 
584  // Obtain future from queue but do not pop it yet, since we need to determine first whether to use it nor not.
585  // The future will only be popped from the queue in postRead(). This makes sure that subsequent calls to this
586  // function even if the future has not yet been used still result in the correct behaviour.
587  assert(_sharedState->_fullBufferQueue.read_available() > 0); // guaranteed by our logic
588  TransferFuture::PlainFutureType future = _sharedState->_fullBufferQueue.front();
589 
590  // if the future blocks, check for data in the tripple buffer
591  auto status = future.wait_for(boost::chrono::duration<int, boost::centi>(0));
592  if(status == boost::future_status::timeout) {
593  // if local buffer of triple buffer is invalid, swap with shared buffer first
594  if(!(_tripleBufferIndex->_isValid)) {
595  _tripleBufferIndex = _sharedState->_tripleBufferIndex.exchange(_tripleBufferIndex);
596  }
597  // if now the local buffer contains valid data, use that data first
598  if(_tripleBufferIndex->_isValid) {
599  future = boost::make_ready_future(static_cast<TransferFuture::Data*>(_tripleBufferIndex)).share();
600  }
601  }
602 
603  // return the future
604  return TransferFuture(future, this);

610  template<class T>
611  void UnidirectionalProcessArray<T>::doPostRead() {
612 
613  // We have to check that the vector that we currently own still has the
614  // right size. Otherwise, the code using the sender might get into
615  // trouble when it suddenly experiences a vector of the wrong size.
616  if(_currentIndex->value.size() != _vectorSize) {
617  throw std::runtime_error("Cannot run read operation because the size of the vector belonging to the current"
618  " buffer has been modified.");
619  }
620 
621  // If local buffer of triple buffer is invalid, swap with shared buffer first
622  if(!(_tripleBufferIndex->_isValid)) {
623  _tripleBufferIndex = _sharedState->_tripleBufferIndex.exchange(_tripleBufferIndex);
624  }
625 
626  // Determine version number of data in the triple buffer
627  bool tripleBufferHasData = _tripleBufferIndex->_isValid;
628 
629  // Determine version number of data in the queue
630  assert(_sharedState->_fullBufferQueue.read_available() > 0); // guaranteed by our logic
631  TransferFuture::PlainFutureType future = _sharedState->_fullBufferQueue.front();
632  auto status = future.wait_for(boost::chrono::duration<int, boost::centi>(0));
633  bool queueHasData = (status != boost::future_status::timeout);
634 
635  // If postRead() has been called, there should be new data somewhere
636  assert(queueHasData || tripleBufferHasData);
637 
638  // Determine whether to use the data from the queue or from the triple buffer. If in both places data is present,
639  // use the older data (by the version number).
640  bool useTripleBuffer;
641  if(queueHasData && !tripleBufferHasData) {
642  // only data in queue present
643  useTripleBuffer = false;
644  }
645  else if(!queueHasData && tripleBufferHasData) {
646  // only data in atomic triple buffer present
647  useTripleBuffer = true;
648  }
649  else { // queueHasData && tripleBufferHasData -> see the assert above
650  uint64_t tripleBufferVersion = _tripleBufferIndex->_internalVersionNumber;
651  uint64_t queueVersion = static_cast<Buffer*>(future.get())->_internalVersionNumber;
652  if(tripleBufferVersion < queueVersion) {
653  useTripleBuffer = true;
654  }
655  else {
656  useTripleBuffer = false;
657  }
658  }
659 
660  // perform the determined action
661  if(!useTripleBuffer) {
662  // remove future from queue, since we only obtained it through front() so far
663  _sharedState->_fullBufferQueue.pop();
664  // get buffer from future
665  TransferFuture::Data *nextBuffer = future.get();
666  // put old buffer back on empty buffer queue and update current index
667  _sharedState->_emptyBufferQueue.push(_currentIndex);
668  _currentIndex = static_cast<Buffer*>(nextBuffer); // static_cast should be ok, since we control the queue
669  }
670  else {
671  Buffer *nextBuffer = _tripleBufferIndex;
672  // put old buffer back into triple buffer (after invalidating int) and update current index
673  _currentIndex->_isValid = false;
674  _tripleBufferIndex = _currentIndex;
675  _currentIndex = nextBuffer;
676  }
677 
678  // swap data out of the queue buffer
679  mtca4u::NDRegisterAccessor<T>::buffer_2D[0].swap( _currentIndex->value );
680 
681  }

742  template<class T>
743  bool UnidirectionalProcessArray<T>::writeInternal(
744  TimeStamp newTimeStamp, VersionNumber newVersionNumber, bool shouldCopy) {
745  if(!this->isWriteable()) {
746  throw std::logic_error(
747  "Send operation is only allowed for a sender process variable. Variable name: "+this->getName());
748  }
749  // We have to check that the vector that we currently own still has the
750  // right size. Otherwise, the code using the receiver might get into
751  // trouble when it suddenly experiences a vector of the wrong size.
752  if(mtca4u::NDRegisterAccessor<T>::buffer_2D[0].size() != _vectorSize) {
753  throw std::runtime_error(
754  "Cannot run receive operation because the size of the vector belonging to the current buffer has been "
755  "modified. Variable name: "+this->getName());
756  }
757  // A version should never be send with a version number that is equal to or
758  // even less than the last version number used. Such an attempt indicates
759  // that there is a problem in the logic attempting the write operation.
760  if (newVersionNumber <= getVersionNumber()) {
761  throw std::runtime_error(
762  "The version number passed to write is less than or equal to the last version number used.");
763  }
764 
765  // First update the persistent data storage, if any was associated. This cannot be done after sending, since the
766  // value might no longer be available within this instance.
767  if(_persistentDataStorage) {
768  _persistentDataStorage->updateValue(_persistentDataStorageID, mtca4u::NDRegisterAccessor<T>::buffer_2D[0]);
769  }
770 
771  // Before sending the value, we have to set the associated time-stamp.
772  _currentIndex->timeStamp = newTimeStamp;
773 
774  // set the version numbers
775  _currentIndex->_versionNumber = newVersionNumber;
776  _currentIndex->_internalVersionNumber = _lastInternalVersionNumber;
777  ++_lastInternalVersionNumber;
778 
779  // flag current data as valid
780  _currentIndex->_isValid = true;
781 
782  // set the data by copying or swapping
783  if(shouldCopy) {
784  _currentIndex->value = mtca4u::NDRegisterAccessor<T>::buffer_2D[0];
785  }
786  else {
787  _currentIndex->value.swap( mtca4u::NDRegisterAccessor<T>::buffer_2D[0] );
788  }
789  // send the data to the queue by first obtaining a new empty buffer from the empty buffer queue
790  Buffer *nextIndex;
791  bool lostData = false;
792  if(_sharedState->_emptyBufferQueue.pop(nextIndex)) {
793  // Create a new promise first and push its future to the full buffer queue before fulfilling the current promise.
794  // This makes sure that always at least one unfulfilled future is available for the receier to wait on.
795  TransferFuture::PromiseType nextPromise;
796  _sharedState->_fullBufferQueue.push(nextPromise.get_future());
797  _sharedState->_notificationPromise.set_value(_currentIndex);
798  _sharedState->_notificationPromise = std::move(nextPromise);
799  }
800  else {
801  // No empty buffer available: write to special atomic triple buffer
802  nextIndex = _tripleBufferIndex;
803  _tripleBufferIndex = _currentIndex;
804  _tripleBufferIndex = _sharedState->_tripleBufferIndex.exchange(_tripleBufferIndex);
805  // if the buffer swapped out of the shared state contained valid data, we have now lost that data
806  if(_tripleBufferIndex->_isValid) {
807  lostData = true;
808  // mark as invalid so we can always just swap it with the shared state
809  _tripleBufferIndex->_isValid = false;
810  }
811  }
812 
813  // change current index to the new empty buffer
814  if(shouldCopy) {
815  nextIndex->timeStamp = newTimeStamp;
816  nextIndex->_versionNumber = newVersionNumber;
817  }
818  _currentIndex = nextIndex;
819 
820  // notify send notification listener, if present
821  if(_sendNotificationListener) {
822  _sendNotificationListener->notify(_receiver);
823  }
824 
825  return lostData;
826  }
