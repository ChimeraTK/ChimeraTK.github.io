128  template<typename UserType>
129  DoocsBackendRegisterAccessor<UserType>::DoocsBackendRegisterAccessor(const RegisterPath &path, size_t numberOfWords,
130  size_t wordOffsetInRegister, AccessModeFlags flags, bool allocateBuffers)
131  : SyncNDRegisterAccessor<UserType>(path),
132  _path(path),
133  elementOffset(wordOffsetInRegister),
134  useZMQ(false)
135  {
136  try {
137  // check for unknown access mode flags
138  flags.checkForUnknownFlags({AccessMode::wait_for_new_data});
139 
140  // set address
141  ea.adr(std::string(path).substr(1).c_str()); // strip leading slash
142 
143  // try to read data, to check connectivity and to obtain size of the register
144  doReadTransfer();
145 
146  // obtain number of elements
147  size_t actualLength = dst.array_length();
148  if(actualLength == 0) {
149  actualLength = 1;
150  isArray = false;
151  }
152  else {
153  isArray = true;
154  }
155  if(numberOfWords == 0) {
156  nElements = actualLength;
157  }
158  else {
159  nElements = numberOfWords;
160  }
161  if(nElements + elementOffset > actualLength) {
162  throw DeviceException("Requested number of words exceeds the length of the DOOCS property!",
163  DeviceException::WRONG_PARAMETER);
164  }
165  if(nElements == actualLength && elementOffset == 0) {
166  isPartial = false;
167  }
168  else {
169  isPartial = true;
170  }
171 
172  // allocate buffers
173  if(allocateBuffers) {
174  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
175  NDRegisterAccessor<UserType>::buffer_2D[0].resize(nElements);
176  }
177 
178  // set proper type information in the source EqData
179  src.set_type(dst.type());
180  if(allocateBuffers) {
181  src.length(actualLength);
182  }
183 
184  // use ZeroMQ with AccessMode::wait_for_new_data
185  if(flags.has(AccessMode::wait_for_new_data)) {
186  // set flag
187  useZMQ = true;
188  // subscribe to property
189  int err = dmsg_attach(&ea, &dst, (void*)this, &zmq_callback);
190  if(err) {
191  throw DeviceException(std::string("Cannot subscribe to DOOCS property via ZeroMQ: ")+dst.get_string(),
192  DeviceException::CANNOT_OPEN_DEVICEBACKEND);
193  }
194  // run dmsg_start() once
195  std::unique_lock<std::mutex> lck(DoocsBackend::dmsgStartCalled_mutex);
196  if(!DoocsBackend::dmsgStartCalled) {
197  dmsg_start();
198  DoocsBackend::dmsgStartCalled = true;
199  }
200  }
201  }
202  catch(...) {
203  this->shutdown();
204  throw;
205  }
206  }

266  template<typename UserType>
267  void DoocsBackendRegisterAccessor<UserType>::doReadTransfer() {
268  boost::this_thread::interruption_point();
269  if(!useZMQ) {
270  // read data
271  int rc = eq.get(&ea, &src, &dst);
272  // check error
273  if(rc) {
274  throw DeviceException(std::string("Cannot read from DOOCS property: ")+dst.get_string(),
275  DeviceException::CANNOT_OPEN_DEVICEBACKEND);
276  }
277  }
278  else {
279  // loop to allow retries in case of timeout errors
280  while(true) {
281  // obtain lock as required for the condition variable
282  std::unique_lock<std::mutex> lck(ZMQmtx);
283  // wait until new data has been received
284  while(ZMQbuffer.empty()) {
285  ZMQnotifier.wait_for(lck, std::chrono::milliseconds(200));
286  boost::this_thread::interruption_point();
287  }
288  // obtain new data and remove it from the queue
289  dst = ZMQbuffer.front();
290  ZMQbuffer.pop();
291  // check for an error
292  if(dst.error() != 0) {
293  // try obtaining data through RPC call instead to verify error
294  int rc = eq.get(&ea, &src, &dst);
295  // if again error received, throw exception
296  if(rc) {
297  throw DeviceException(std::string("Cannot read from DOOCS property: ")+dst.get_string(),
298  DeviceException::CANNOT_OPEN_DEVICEBACKEND);
299  }
300  // otherwise try again
301  continue;
302  }
303  // terminate loop, since no retry needed
304  break;
305  }
306  }
307  // check for error code in dst
308  if(dst.error() != 0) {
309  throw DeviceException(std::string("Cannot read from DOOCS property: ")+dst.get_string(),
310  DeviceException::CANNOT_OPEN_DEVICEBACKEND);
311  }
312  }
