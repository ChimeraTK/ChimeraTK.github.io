235  template<typename UserType>
236  VariableNetworkNode VariableNetworkNode::makeConstant(bool makeFeeder, UserType value, size_t length) {
237  VariableNetworkNode node;
238  node.pdata = boost::make_shared<VariableNetworkNode_data>();
239  node.pdata->constNode.reset(new ConstantAccessor<UserType>(value, length));
240  node.pdata->type = NodeType::Constant;
241  node.pdata->valueType = &typeid(UserType);
242  node.pdata->nElements = length;
243  node.pdata->name = "*UNNAMED CONSTANT*";
244  if(makeFeeder) {
245  node.pdata->direction = VariableDirection::feeding;
246  node.pdata->mode = UpdateMode::push;
247  }
248  else {
249  node.pdata->direction = VariableDirection::consuming;
250  node.pdata->mode = UpdateMode::poll;
251  }
252  return node;
253  }
