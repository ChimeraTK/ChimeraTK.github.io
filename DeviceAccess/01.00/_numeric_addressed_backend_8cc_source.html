31  boost::shared_ptr<RegisterInfoMap::RegisterInfo> NumericAddressedBackend::getRegisterInfo(const RegisterPath &registerPathName) {
32  if(!registerPathName.startsWith(numeric_address::BAR)) {
33  boost::shared_ptr<RegisterInfo> info = _catalogue.getRegister(registerPathName);
34  return boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(info);
35  }
36  else {
37  auto components = registerPathName.getComponents();
38  if(components.size() != 3) {
39  throw ChimeraTK::logic_error("Illegal numeric address: '"+(registerPathName)+"'");
40  }
41  auto bar = std::stoi(components[1]);
42  size_t pos = components[2].find_first_of("*");
43  auto address = std::stoi(components[2].substr(0,pos));
44  size_t nBytes;
45  if(pos != std::string::npos) {
46  nBytes = std::stoi(components[2].substr(pos+1));
47  }
48  else {
49  nBytes = sizeof(int32_t);
50  }
51  auto nElements = nBytes/sizeof(int32_t);
52  if(nBytes == 0 || nBytes % sizeof(int32_t) != 0) {
53  throw ChimeraTK::logic_error("Illegal numeric address: '"+(registerPathName)+"'");
54  }
55  return boost::make_shared<RegisterInfoMap::RegisterInfo>(registerPathName, nElements, address, nBytes, bar);
56  }
57  }

100  void NumericAddressedBackend::checkRegister(const std::string &regName,
101  const std::string &regModule, size_t dataSize,
102  uint32_t addRegOffset, uint32_t &retDataSize,
103  uint32_t &retRegOff, uint8_t &retRegBar) const {
104 
105  RegisterInfoMap::RegisterInfo registerInfo;
106  _registerMap->getRegisterInfo(regName, registerInfo, regModule);
107  if (addRegOffset % 4) {
108  throw ChimeraTK::logic_error("Register offset must be divisible by 4");
109  }
110  if (dataSize) {
111  if (dataSize % 4) {
112  throw ChimeraTK::logic_error("Data size must be divisible by 4");
113  }
114  if (dataSize > registerInfo.nBytes - addRegOffset) {
115  throw ChimeraTK::logic_error("Data size exceed register size");
116  }
117  retDataSize = dataSize;
118  } else {
119  retDataSize = registerInfo.nBytes;
120  }
121  retRegBar = registerInfo.bar;
122  retRegOff = registerInfo.address + addRegOffset;
123  }

127  template<typename UserType>
128  boost::shared_ptr< NDRegisterAccessor<UserType> > NumericAddressedBackend::getRegisterAccessor_impl(
129  const RegisterPath &registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
130  boost::shared_ptr< NDRegisterAccessor<UserType> > accessor;
131  // obtain register info
132  boost::shared_ptr<RegisterInfo> info = getRegisterInfo(registerPathName);
133  // 1D or scalar register
134  if(info->getNumberOfDimensions() <= 1) {
135  accessor = boost::shared_ptr< NDRegisterAccessor<UserType> >(
136  new NumericAddressedBackendRegisterAccessor<UserType>(shared_from_this(), registerPathName,
137  numberOfWords, wordOffsetInRegister, flags) );
138  }
139  // 2D multiplexed register
140  else {
141  accessor = boost::shared_ptr< NDRegisterAccessor<UserType> >(
142  new NumericAddressedBackendMuxedRegisterAccessor<UserType>(registerPathName, numberOfWords,
143  wordOffsetInRegister, shared_from_this()) );
144  }
145  return accessor;
146  }
