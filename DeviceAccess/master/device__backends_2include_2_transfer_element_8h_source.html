124  void read() {
125  if(TransferElement::_isInTransferGroup) {
126  throw ChimeraTK::logic_error("Calling read() or write() on the TransferElement '" + _name +

150  bool readNonBlocking() {
151  if(TransferElement::_isInTransferGroup) {
152  throw ChimeraTK::logic_error("Calling read() or write() on the TransferElement '" + _name +
153  "' which is part of a TransferGroup is not allowed.");
154  }
155  this->readTransactionInProgress = false;
156  preReadAndHandleExceptions(TransferType::readNonBlocking);
157  bool updateDataBuffer = false;
158  if(!_activeException) {
159  handleTransferException([&] { updateDataBuffer = readTransferNonBlocking(); });
160  }
161 
162  bool retVal = updateDataBuffer;
163  if(_activeException) {
164  auto previousVersionNumber = _versionNumber;
165  auto previousDataValidity = _dataValidity;
166  // always call postRead with updateDataBuffer = false in case of an exception
167  postRead(TransferType::readNonBlocking, false);
168  // Usually we do not reach this point because postRead() is re-throwing the _activeExeption.
169  // If we reach this point the exception has been suppressed. We have to calculate a
170  // new return value because the dataBuffer has not changed, but the meta data
171  // could have, in which case we have to return true.
172  if((previousVersionNumber != _versionNumber) || (previousDataValidity != _dataValidity)) {
173  retVal = true;
174  }
175  else {
176  retVal = false;
177  }
178  }
179  else {
180  // call postRead with updateDataBuffer as returned by readTransferNonBlocking
181  postRead(TransferType::readNonBlocking, updateDataBuffer);
182  }
183  return retVal;
184  }

190  bool readLatest() {
191  if(_accessModeFlags.has(AccessMode::wait_for_new_data)) {
192  bool updateDataBuffer = false;
193  // Call readNonBlocking until there is no new data to be read any more
194  while(readNonBlocking()) {
195  // remember whether we have new data
196  updateDataBuffer = true;
197  }
198  return updateDataBuffer;
199  }
200  else {
201  // Without wait_for_new_data readNonBlocking always returns true, and the while loop above would never end.
202  // Hence we just call the (synchronous) read and return true;
203  read();
204  return true;
205  }
206  }

470  void postRead(TransferType type, bool updateDataBuffer) {
471  // only delegate to doPostRead() the first time postRead() is called in a row.
472  if(readTransactionInProgress) {
473  readTransactionInProgress = false;

482  // don't clear the active connection. This is done in preRead().
