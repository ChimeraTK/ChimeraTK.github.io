37  void ExceptionHandlingDecorator<UserType>::doPreWrite(TransferType type, VersionNumber versionNumber) {
38  /* For writable accessors, copy data to the recoveryAcessor before perfroming the write.
39  * Otherwise, the decorated accessor may have swapped the data out of the user buffer already.
40  * This obtains a shared lock from the DeviceModule, hence, the regular writing happeniin here
41  * can be performed in shared mode of the mutex and accessors are not blocking each other.
42  * In case of recovery, the DeviceModule thread will take an exclusive lock so that this thread can not
43  * modify the recoveryAcessor's user buffer while data is written to the device.
44  */
45  {
46  hasThrownToInhibitTransfer = false;
47  auto recoverylock{deviceModule.getRecoverySharedLock()};
48 
49  if(_recoveryAccessor != nullptr) {
50  // Access to _recoveryAccessor is only possible channel-wise
51  for(unsigned int ch = 0; ch < _recoveryAccessor->getNumberOfChannels(); ++ch) {
52  _recoveryAccessor->accessChannel(ch) = buffer_2D[ch];
53  }
54  // FIXME: This should be the version number of the owner, which is not set at the moment
55  _recoveryHelper->versionNumber = {};
56  }
57  else {
58  throw ChimeraTK::logic_error(
59  "ChimeraTK::ExceptionhandlingDecorator: Calling write() on a non-writeable accessor is not supported ");
60  }
61 
62  boost::shared_lock<boost::shared_mutex> errorLock(deviceModule.errorMutex);
63  if(deviceModule.deviceHasError) {
64  hasThrownToInhibitTransfer = true;
65  throw ChimeraTK::runtime_error(
66  "ExceptionHandlingDecorator is preventing the write transfer."); // writing will be delayed, done by the recovery accessor
67  }
68 
69  ++deviceModule.synchronousTransferCounter; // must be under the lock
70 
71  } // lock guard goes out of scope
72

74  // preWrite and postWrite are only delegated if the transfer is allowed.
