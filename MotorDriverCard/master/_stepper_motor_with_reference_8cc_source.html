211  int calibNegativeEndSwitchInSteps = _motorControler->getNegativeReferenceSwitchCalibration();

234  bool StepperMotorWithReference::isEndSwitchActive(Sign sign){
235  bool posActive = _motorControler->getReferenceSwitchData().getPositiveSwitchActive() == 1U;
236  bool negActive = _motorControler->getReferenceSwitchData().getNegativeSwitchActive() == 1U;
237 
238  if(posActive && negActive){
239  _errorMode.exchange(StepperMotorError::BOTH_END_SWITCHES_ON);
240  _stateMachine->setAndProcessUserEvent(StepperMotorStateMachine::errorEvent);
241  }
242 
243  if (_toleranceCalculated){
244  if (posActive){
245  if(std::abs(_motorControler->getActualPosition() - _calibPositiveEndSwitchInSteps.load()) > 3 * _tolerancePositiveEndSwitch.load()){
246  _errorMode.exchange(StepperMotorError::CALIBRATION_ERROR);
247  }
248  }
249  else if(negActive){
250  if(std::abs(_motorControler->getActualPosition() - _calibNegativeEndSwitchInSteps.load()) > 3 * _toleranceNegativeEndSwitch.load()){
251  _errorMode.exchange(StepperMotorError::CALIBRATION_ERROR);
252  }
253  }
254  }
255 
256  if (sign == Sign::POSITIVE){
257  if (posActive){
258  return true;
259  }else{
260  return false;
261  }
262  }else{
263  if (negActive){
264  return true;
265  }else{
266  return false;
267  }
268  }
269  }
