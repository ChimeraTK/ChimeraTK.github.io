136 void DoocsServerTestHelper::doocsSet(const std::string& name, TYPE value) {
137  EqAdr ad;
138  EqData ed, res;
139  // obtain location pointer
140  ad.adr(name.c_str());
141  EqFct* p = eq_get(&ad);
142  ASSERT(p != NULL, std::string("Could not get location for property ") + name);
143  // set value
144  ed.set(value);
145  p->lock();
146  p->set(&ad, &ed, &res);
147  p->unlock();
148  // check for error
149  ASSERT(res.error() == 0, std::string("Error writing property ") + name + ": " + res.get_string());
150 }

180  // fill spectrum data structure

182  // obtain location pointer

184  EqFct* p = eq_get(&ad);

189  p->unlock();
190  // check for error

228 template<typename TYPE>
229 std::vector<TYPE> DoocsServerTestHelper::doocsGetArray(const std::string& name) {
230  EqAdr ad;
231  EqData ed, res;
232  // obtain location pointer
233  ad.adr(name.c_str());
234  EqFct* p = eq_get(&ad);
235  ASSERT(p != NULL, std::string("Could not get location for property ") + name);
236  // for D_BSpectrum: set IIII structure to obtain always the latest buffer
237  // (shouldn't hurt for others)
238  IIII iiii;
239  iiii.i1_data = -1;
240  iiii.i2_data = -1;
241  iiii.i3_data = -1;
242  iiii.i4_data = -1;
243  ed.set(&iiii);
244  // obtain values
245  p->lock();
246  p->get(&ad, &ed, &res);
247  p->unlock();
248  // check for errors
249  ASSERT(res.error() == 0, std::string("Error reading property ") + name);
250  // copy to vector and return it
251  std::vector<TYPE> val;
252  if(std::is_integral<TYPE>()) {
253  if(res.type() != DATA_A_LONG) {
254  for(int i = 0; i < res.length(); i++) val.push_back(res.get_int(i));
255  }
256  else {
257  for(int i = 0; i < res.length(); i++) val.push_back(res.get_long(i));
258  }
259  }
260  else if(std::is_floating_point<TYPE>()) {
261  for(int i = 0; i < res.length(); i++) val.push_back(res.get_float(i));
262  }
263  else {
264  ASSERT(false, "Wrong type passed as tempalate argument.");
265  }
266  return val;
267 }
