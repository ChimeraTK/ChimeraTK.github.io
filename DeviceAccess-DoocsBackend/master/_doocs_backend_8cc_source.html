111  bool DoocsBackend::checkZmqAvailability(const std::string& fullLocationPath, const std::string& propertyName) {
112  int rc;
113  float f1;
114  float f2;
115  char* sp;
116  time_t tm;
117  EqAdr ea;
118  EqData dat;
119  EqData dst;
120  EqCall eq;
121  int portp;
122 
123  ea.adr(fullLocationPath + "/SPN");
124 
125  // get channel port number
126  dat.set(1, 0.0f, 0.0f, time_t{0}, propertyName, 0);
127 
128  // call get () to see whether it is supported
129  rc = eq.get(&ea, &dat, &dst);
130  if(rc) {
131  return false;
132  }
133 
134  rc = dst.get_ustr(&portp, &f1, &f2, &tm, &sp, 0);
135  if(rc && !portp && !static_cast<int>(f1 + f2)) rc = 0; // not supported
136 
137  if(!rc) {
138  dst.get_ustr(&portp, &f1, &f2, &tm, &sp, 0);
139  // get () not supported, call set ()
140  rc = eq.set(&ea, &dat, &dst);
141  if(rc) {
142  return false;
143  }
144  }
145 
146  if(dst.type() == DATA_INT) {
147  portp = dst.get_int();
148  }
149  else {
150  dst.get_ustr(&portp, &f1, &f2, &tm, &sp, 0);
151  }
152 
153  return portp != 0;
154  }

163  void DoocsBackend::fillCatalogue(std::string fixedComponents, long level) const {
164  // obtain list of elements within the given partial address
165  EqAdr ea;
166  EqCall eq;
167  EqData src, propList;
168  ea.adr((fixedComponents + "/*").c_str());
169  int rc = eq.names(&ea, &propList);
170  if(rc) {
171  // if the enumeration failes, maybe the server is not available (but exists
172  // in ENS) -> just ignore this address
173  return;
174  }
175 
176  // iterate over list
177  for(int i = 0; i < propList.array_length(); ++i) {
178  // obtain the name of the element
179  char c[255];
180  propList.get_string_arg(i, c, 255);
181  std::string name = c;
182  name = name.substr(0, name.find_first_of(" ")); // ignore comment which is following the space
183 
184  // if we are not yet at the property-level, recursivly call the function
185  // again to resolve the next hierarchy level
186  if(level < 2) {
187  fillCatalogue(fixedComponents + "/" + name, level + 1);
188  }
189  else {
190  // this is a property: create RegisterInfo entry and set its name
191  bool skipRegister = false;
192  for(uint i = 0; i < SIZE_IGNORE_PATTERNS; i++) {
193  std::string pattern = IGNORE_PATTERNS[i];
194  if(ignorePattern(name, pattern)) {
195  skipRegister = true;
196  break;
197  }
198  }
199  if(skipRegister) {
200  continue;
201  }
202  boost::shared_ptr<DoocsBackendRegisterInfo> info(new DoocsBackendRegisterInfo());
203  std::string fqn = fixedComponents + "/" + name;
204  info->name = fqn.substr(std::string(_serverAddress).length());
205 
206  // read property once to determine its length and data type
208  EqData dst;
209  ea.adr(fqn.c_str()); // strip leading slash
210  rc = eq.get(&ea, &src, &dst);
211  if(rc) {
212  // if the property is not accessible, ignore it. This happens frequently
213  // e.g. for archiver-related properties
214  continue;
215  }
216 
217  if(DoocsBackend::checkZmqAvailability(fixedComponents, name))
218  info->accessModeFlags.add(AccessMode::wait_for_new_data);
219 
220  info->length = dst.array_length();
221  if(info->length == 0) info->length = 1; // DOOCS reports 0 if not an array
222  if(dst.type() == DATA_TEXT || dst.type() == DATA_STRING || // in case of strings, DOOCS reports the
223  // length of the string
224  dst.type() == DATA_STRING16 || dst.type() == DATA_USTR) {
225  info->length = 1;
226  info->dataDescriptor =
227  ChimeraTK::RegisterInfo::DataDescriptor(ChimeraTK::RegisterInfo::FundamentalType::string);
228  }
229  else if(dst.type() == DATA_INT || dst.type() == DATA_A_INT || dst.type() == DATA_A_SHORT ||
230  dst.type() == DATA_A_LONG || dst.type() == DATA_A_BYTE || dst.type() == DATA_IIII) { // integral data types
231  size_t digits;
232  if(dst.type() == DATA_A_SHORT) { // 16 bit signed
233  digits = 6;
234  }
235  else if(dst.type() == DATA_A_BYTE) { // 8 bit signed
236  digits = 4;
237  }
238  else if(dst.type() == DATA_A_LONG) { // 64 bit signed
239  digits = 20;
240  }
241  else { // 32 bit signed
242  digits = 11;
243  }
244  if(dst.type() == DATA_IIII) info->length = 4;
245 
246  info->dataDescriptor = ChimeraTK::RegisterInfo::DataDescriptor(
247  ChimeraTK::RegisterInfo::FundamentalType::numeric, true, true, digits);
248  }
249  else if(dst.type() == DATA_IFFF) {
250  info->name = fqn.substr(std::string(_serverAddress).length()) + "/I";
251  info->dataDescriptor = ChimeraTK::RegisterInfo::DataDescriptor(
252  ChimeraTK::RegisterInfo::FundamentalType::numeric, true, true, 11); // 32 bit integer
253 
254  boost::shared_ptr<DoocsBackendRegisterInfo> infoF1(new DoocsBackendRegisterInfo(*info));
255  infoF1->name = fqn.substr(std::string(_serverAddress).length()) + "/F1";
256  infoF1->dataDescriptor = ChimeraTK::RegisterInfo::DataDescriptor(
257  ChimeraTK::RegisterInfo::FundamentalType::numeric, false, true, 320, 300); // float
258 
259  boost::shared_ptr<DoocsBackendRegisterInfo> infoF2(new DoocsBackendRegisterInfo(*infoF1));
260  infoF2->name = fqn.substr(std::string(_serverAddress).length()) + "/F2";
261 
262  boost::shared_ptr<DoocsBackendRegisterInfo> infoF3(new DoocsBackendRegisterInfo(*infoF1));
263  infoF3->name = fqn.substr(std::string(_serverAddress).length()) + "/F3";
264 
265  _catalogue_mutable.addRegister(infoF1);
266  _catalogue_mutable.addRegister(infoF2);
267  _catalogue_mutable.addRegister(infoF3);
268  // the info for the integer is added below
269  }
270  else { // floating point data types: always treat like double
271  info->dataDescriptor = ChimeraTK::RegisterInfo::DataDescriptor(
272  ChimeraTK::RegisterInfo::FundamentalType::numeric, false, true, 320, 300);
273  }
274 
275  // add info to catalogue
276  _catalogue_mutable.addRegister(info);
277  }
278  }

290  void DoocsBackend::fillCatalogue() const {
291  if(!catalogueFilled && _opened) {
292  // Fill the catalogue:
293  // first, count number of elements in address part given in DMAP file to
294  // determine how many components we have to iterate over
295  size_t nSlashes;
296  if(!boost::starts_with(_serverAddress, "doocs://")) {
297  nSlashes = std::count(_serverAddress.begin(), _serverAddress.end(), '/');
298  }
299  else {
300  nSlashes = std::count(_serverAddress.begin(), _serverAddress.end(), '/') - 3;
301  }
302  // next, iteratively call the function to fill the catalogue
303  fillCatalogue(_serverAddress, nSlashes);
304  catalogueFilled = true;
305  }
306  }

322  template<typename UserType>
323  boost::shared_ptr<NDRegisterAccessor<UserType>> DoocsBackend::getRegisterAccessor_impl(
324  const RegisterPath& registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
325  NDRegisterAccessor<UserType>* p;
326  std::string path = _serverAddress + registerPathName;
327 
328  // check for additional hierarchy level, which indicates an access to a field of a complex property data type
329  bool hasExtraLevel = false;
330  if(!boost::starts_with(path, "doocs://") && !boost::starts_with(path, "epics://")) {
331  size_t nSlashes = std::count(path.begin(), path.end(), '/');
332  if(nSlashes == 4) {
333  hasExtraLevel = true;
334  }
335  else if(nSlashes < 3 || nSlashes > 4) {
336  throw ChimeraTK::logic_error(std::string("DOOCS address has an illegal format: ") + path);
337  }
338  }
339  else if(boost::starts_with(path, "doocs://")) {
340  size_t nSlashes = std::count(path.begin(), path.end(), '/');
341  // we have 3 extra slashes compared to the standard syntax without "doocs:://"
342  if(nSlashes == 4 + 3) {
343  hasExtraLevel = true;
344  }
345  else if(nSlashes < 3 + 3 || nSlashes > 4 + 3) {
346  throw ChimeraTK::logic_error(std::string("DOOCS address has an illegal format: ") + path);
347  }
348  }
349 
350  // split the path into property name and field name
351  std::string field;
352  if(hasExtraLevel) {
353  field = path.substr(path.find_last_of('/') + 1);
354  path = path.substr(0, path.find_last_of('/'));
355  }
356 
357  // read property once
358  EqAdr ea;
359  EqCall eq;
360  EqData src, dst;
361  ea.adr(std::string(path).c_str());
362  int rc = eq.get(&ea, &src, &dst);
363  if(rc) {
364  throw ChimeraTK::runtime_error("Cannot open DOOCS property '" + path + "': " + dst.get_string());
365  }
366 
367  // check type and create matching accessor
368  bool extraLevelUsed = false;
369  if(dst.type() == DATA_INT || dst.type() == DATA_A_INT || dst.type() == DATA_BOOL || dst.type() == DATA_A_BOOL ||
370  dst.type() == DATA_A_SHORT) {
371  p = new DoocsBackendIntRegisterAccessor<UserType>(path, numberOfWords, wordOffsetInRegister, flags);
372  }
373  else if(dst.type() == DATA_A_LONG) {
374  p = new DoocsBackendLongRegisterAccessor<UserType>(path, numberOfWords, wordOffsetInRegister, flags);
375  }
376  else if(dst.type() == DATA_FLOAT || dst.type() == DATA_A_FLOAT || dst.type() == DATA_DOUBLE ||
377  dst.type() == DATA_A_DOUBLE || dst.type() == DATA_SPECTRUM) {
378  p = new DoocsBackendFloatRegisterAccessor<UserType>(path, numberOfWords, wordOffsetInRegister, flags);
379  }
380  else if(dst.type() == DATA_TEXT || dst.type() == DATA_STRING || dst.type() == DATA_STRING16) {
381  p = new DoocsBackendStringRegisterAccessor<UserType>(path, numberOfWords, wordOffsetInRegister, flags);
382  }
383  else if(dst.type() == DATA_IIII) {
384  p = new DoocsBackendIIIIRegisterAccessor<UserType>(path, numberOfWords, wordOffsetInRegister, flags);
385  }
386  else if(dst.type() == DATA_IFFF) {
387  extraLevelUsed = true;
388  p = new DoocsBackendIFFFRegisterAccessor<UserType>(path, field, numberOfWords, wordOffsetInRegister, flags);
389  }
390  else {
391  throw ChimeraTK::logic_error("Unsupported DOOCS data type " + std::string(dst.type_string()) + " of property '" +
392  _serverAddress + registerPathName + "'");
393  }
394 
395  // if the field name has been specified but the data type does not use it, throw an exception
396  if(hasExtraLevel && !extraLevelUsed) {
397  throw ChimeraTK::logic_error("Specifiaction of field name is not supported for the DOOCS data type " +
398  std::string(dst.type_string()) + ": " + _serverAddress + registerPathName);
399  }
400 
401  return boost::shared_ptr<NDRegisterAccessor<UserType>>(p);
402  }
