27  LNMBackendChannelAccessor(boost::shared_ptr<DeviceBackend> dev, const RegisterPath &registerPathName,
28  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
29  : SyncNDRegisterAccessor<UserType>(registerPathName),
30  _registerPathName(registerPathName)
31  {
32  try {
33  // check for unknown flags
34  flags.checkForUnknownFlags({AccessMode::raw});
35  // check for illegal parameter combinations
36  if(flags.has(AccessMode::raw)) {
37  throw ChimeraTK::logic_error("LNMBackendChannelAccessor: raw access not yet supported!");
38  }
39  _dev = boost::dynamic_pointer_cast<LogicalNameMappingBackend>(dev);
40  // copy the register info and create the internal accessors, if needed
41  _info = *( boost::static_pointer_cast<LNMBackendRegisterInfo>(
42  _dev->getRegisterCatalogue().getRegister(_registerPathName)) );
43  // check for incorrect usage of this accessor
44  if( _info.targetType != LNMBackendRegisterInfo::TargetType::CHANNEL ) {
45  throw ChimeraTK::logic_error("LNMBackendChannelAccessor used for wrong register type."); // LCOV_EXCL_LINE (impossible to test...)
46  }
47  // get target device and accessor
48  std::string devName = _info.deviceName;
49  boost::shared_ptr<DeviceBackend> targetDevice;
50  if(devName != "this") {
51  targetDevice = _dev->_devices[devName];
52  }
53  else {
54  targetDevice = dev;
55  }
56  _accessor = targetDevice->getRegisterAccessor<UserType>(RegisterPath(_info.registerName), numberOfWords,wordOffsetInRegister, false);
57  // allocate the buffer
58  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
59  NDRegisterAccessor<UserType>::buffer_2D[0].resize(_accessor->getNumberOfSamples());
60  }
61  catch(...) {
62  this->shutdown();
63  throw;
64  }
65  }
