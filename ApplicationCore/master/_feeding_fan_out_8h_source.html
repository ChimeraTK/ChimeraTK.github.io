25  struct RuntimeErrorCollector {
26  std::stringstream ss;
27  bool hasException{false};
28 
29  // Ideally, this would happen in the destructor
30  void unwrap() {
31  if(not hasException) return;
32 
33  auto s = ss.str();
34  s.pop_back();
35 
36  throw runtime_error(s);
37  }

39  bool wrap(std::function<void(void)> f) {
40  try {
41  f();
42  return true;
43  }
44  catch(runtime_error& ex) {
45  ss << ex.what() << "\n";
46  return false;
47  }
48  }

68  void addSlave(boost::shared_ptr<ChimeraTK::NDRegisterAccessor<UserType>> slave, VariableNetworkNode&) override {
69  // check if array shape is compatible, unless the receiver is a trigger
70  // node, so no data is expected
71  if(slave->getNumberOfSamples() != 0 &&
72  (slave->getNumberOfChannels() != 1 || slave->getNumberOfSamples() != this->getNumberOfSamples())) {
73  std::string what = "FeedingFanOut::addSlave(): Trying to add a slave '" + slave->getName();
74  what += "' with incompatible array shape! Name of fan out: '" + this->getName() + "'";
75  throw ChimeraTK::logic_error(what.c_str());
76  }
77 
78  // make sure slave is writeable
79  if(!slave->isWriteable()) {
80  throw ChimeraTK::logic_error("FeedingFanOut::addSlave() has been called "
81  "with a receiving implementation!");
82  }
83 
84  // handle return channels
85  if(_withReturn) {
86  if(slave->isReadable()) {
87  if(_hasReturnSlave) {
88  throw ChimeraTK::logic_error("FeedingFanOut: Cannot add multiple slaves with return channel!");
89  }
90  _hasReturnSlave = true;
91  _returnSlave = slave;
92  }
93  }
94 
95  // add the slave
96  FanOut<UserType>::slaves.push_back(slave);
97  }

126  void doPostRead(TransferType type, bool hasNewData) override {
127  assert(_withReturn);
128  assert(_hasReturnSlave);
129 
130  auto _ = cppext::finally([&] {
131  if(!hasNewData) return;
132  _returnSlave->accessChannel(0).swap(ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0]);
133  // distribute return-channel update to the other slaves
134  for(auto& slave : FanOut<UserType>::slaves) { // send out copies to slaves
135  if(slave == _returnSlave) continue;
136  if(slave->getNumberOfSamples() != 0) { // do not send copy if no data is expected (e.g. trigger)
137  slave->accessChannel(0) = ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0];
138  }
139  slave->writeDestructively(_returnSlave->getVersionNumber());
140  }
141  });
142 
143  _returnSlave->postRead(type, hasNewData);
144  }

151  void doPreWrite(TransferType type, VersionNumber versionNumber) override {
152  for(auto& slave : FanOut<UserType>::slaves) { // send out copies to slaves
153  if(slave->getNumberOfSamples() != 0) { // do not send copy if no data is expected (e.g. trigger)
154  if(slave == FanOut<UserType>::slaves.front()) { // in case of first slave, swap instead of copy
155  slave->accessChannel(0).swap(ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0]);
156  }
157  else { // not the first slave: copy the data from the first slave
158  slave->accessChannel(0) = FanOut<UserType>::slaves.front()->accessChannel(0);
159  }
160  }
161  slave->setDataValidity(this->dataValidity());
162  }
163 
164  // pre write may only be called on the target accessors after we have filled
165  // them all, otherwise the first accessor might take us the data away...
166  RuntimeErrorCollector ec;
167  for(auto& slave : FanOut<UserType>::slaves) {
168  ec.wrap([&] { slave->preWrite(type, versionNumber); });
169  }
170 
171  ec.unwrap();
172  }

174  bool doWriteTransfer(ChimeraTK::VersionNumber versionNumber) override {
175  bool dataLost = false;
176  bool isFirst = true;
177  for(auto& slave : FanOut<UserType>::slaves) {
178  bool ret;
179  if(isFirst) {
180  isFirst = false;
181  ret = slave->writeTransfer(versionNumber);
182  }
183  else {
184  ret = slave->writeTransferDestructively(versionNumber);
185  }
186  if(ret) dataLost = true;
187  }
188  return dataLost;
189  }
