129  StepperMotorError StepperMotorWithReference::getError(){
130  boost::lock_guard<boost::mutex> guard(_mutex);
131  if (_motorControler->getReferenceSwitchData().getPositiveSwitchActive() && _motorControler->getReferenceSwitchData().getNegativeSwitchActive()){
132  return BOTH_END_SWITCHES_ON;
133  }
134  if (stateMachineInIdleAndNoEvent()){
135  if (_toleranceCalcFailed.load() || _calibrationFailed.load()){
136  return ACTION_ERROR;
137  }else if ((_motorControler->getTargetPosition() != _motorControler->getActualPosition()) &&
138  !_motorControler->getReferenceSwitchData().getPositiveSwitchActive() &&
139  !_motorControler->getReferenceSwitchData().getNegativeSwitchActive()){
140  return ACTION_ERROR;
141  }
142  }
143  if (_toleranceCalculated){
144  if (_motorControler->getReferenceSwitchData().getPositiveSwitchActive() &&
145  std::abs(_motorControler->getActualPosition() - _calibPositiveEndSwitchInSteps.load()) > 3 * _tolerancePositiveEndSwitch.load()){
146  return CALIBRATION_LOST;
147  }else if(_motorControler->getReferenceSwitchData().getNegativeSwitchActive() &&
148  std::abs(_motorControler->getActualPosition() - _calibNegativeEndSwitchInSteps.load()) > 3 * _toleranceNegativeEndSwitch.load()){
149  return CALIBRATION_LOST;
150  }
151  }
152  return NO_ERROR;
153  }
