41  mtca4u::TransferElementID Module::readAny() {
42  auto accessorList = getAccessorListRecursive();
43  // put push-type transfer elements into a list suitable for TransferElement::readAny()
44  std::list<std::reference_wrapper<mtca4u::TransferElementAbstractor>> transferElementList;
45  for(auto &accessor : accessorList) {
46  if(accessor.getDirection() != VariableDirection::consuming) continue;
47  if(accessor.getMode() == UpdateMode::push) {
48  transferElementList.emplace_back(accessor.getAppAccessorNoType());
49  }
50  }
51 
52  // wait until one of the push-type accessors receives an update
53  auto ret = Application::getInstance().readAny(transferElementList);
54 
55  // trigger read on the poll-type accessors
56  for(auto accessor : accessorList) {
57  if(accessor.getDirection() != VariableDirection::consuming) continue;
58  if(accessor.getMode() == UpdateMode::poll) {
59  accessor.getAppAccessorNoType().readLatest();
60  }
61  }
62 
63  return ret;
64  }

68  void Module::readAll() {
69  auto accessorList = getAccessorListRecursive();
70  // first blockingly read all push-type variables
71  for(auto accessor : accessorList) {
72  if(accessor.getDirection() != VariableDirection::consuming) continue;
73  if(accessor.getMode() != UpdateMode::push) continue;
74  accessor.getAppAccessorNoType().read();
75  }
76  // next non-blockingly read the latest values of all poll-type variables
77  for(auto accessor : accessorList) {
78  if(accessor.getDirection() != VariableDirection::consuming) continue;
79  if(accessor.getMode() == UpdateMode::push) continue;
80  accessor.getAppAccessorNoType().readLatest();
