129  void LogicalNameMapParser::parseElement(RegisterPath currentPath, const xmlpp::Element *element) {
130  // module tag found: look for registers and sub-modules in module
131  if(element->get_name() == "module") {
132 
133  // obtain name of the module
134  auto nameAttr = element->get_attribute("name");
135  if(!nameAttr) {
136  parsingError("Missing name attribute of 'module' tag.");
137  }
138  std::string moduleName = nameAttr->get_value();
139 
140  // iterate over childs in module
141  for(const auto& child : element->get_children()) {
142  // cast into element, ignore if not an element (e.g. comment)
143  const xmlpp::Element *childElement = dynamic_cast<const xmlpp::Element*>(child);
144  if(!childElement) continue;
145 
146  // parse the element
147  parseElement(currentPath/moduleName, childElement);
148  }
149  }
150 
151  // register tag found: create new entry in map
152  else {
153 
154  // obtain the type
155  std::string type = element->get_name();
156 
157  // create new RegisterInfo object
158  auto info = boost::shared_ptr<LNMBackendRegisterInfo>( new LNMBackendRegisterInfo() );
159 
160  if(type == "redirectedRegister") {
161  info->targetType = LNMBackendRegisterInfo::TargetType::REGISTER;
162  info->deviceName = getValueFromXmlSubnode<std::string>(element, "targetDevice");
163  info->registerName = getValueFromXmlSubnode<std::string>(element, "targetRegister");
164  info->firstIndex = getValueFromXmlSubnode<unsigned int>(element, "targetStartIndex",true,0);
165  info->length = getValueFromXmlSubnode<unsigned int>(element, "numberOfElements",true,0);
166  info->nDimensions = 0;
167  info->nChannels = 0;
168  }
169  else if(type == "redirectedChannel") {
170  info->targetType = LNMBackendRegisterInfo::TargetType::CHANNEL;
171  info->deviceName = getValueFromXmlSubnode<std::string>(element, "targetDevice");
172  info->registerName = getValueFromXmlSubnode<std::string>(element, "targetRegister");
173  info->channel = getValueFromXmlSubnode<unsigned int>(element, "targetChannel");
174  info->firstIndex = 0;
175  info->length = 0;
176  info->nDimensions = 1;
177  info->nChannels = 1;
178  }
179  else if(type == "constant") {
180  std::string constantType = getValueFromXmlSubnode<std::string>(element, "type");
181  if(constantType != "integer") {
182  parsingError("Type '"+constantType+"' is not valid for a constant");
183  }
184  info->targetType = LNMBackendRegisterInfo::TargetType::INT_CONSTANT;
185  info->value = getValueFromXmlSubnode<int>(element, "value");
186  info->firstIndex = 0;
187  info->length = 1;
188  info->nDimensions = 0;
189  info->nChannels = 1;
190  }
191  else if(type == "variable") {
192  std::string constantType = getValueFromXmlSubnode<std::string>(element, "type");
193  if(constantType != "integer") {
194  parsingError("Type '"+constantType+"' is not valid for a variable");
195  }
196  info->targetType = LNMBackendRegisterInfo::TargetType::INT_VARIABLE;
197  info->value = getValueFromXmlSubnode<int>(element, "value");
198  info->firstIndex = 0;
199  info->length = 1;
200  info->nDimensions = 0;
201  info->nChannels = 1;
202  }
203  else {
204  parsingError("Wrong logical register type: "+type);
205  }
206 
207  // obtain name of logical register
208  auto nameAttr = element->get_attribute("name");
209  if(!nameAttr) {
210  parsingError("Missing name attribute of '"+type+"' tag.");
211  }
212  RegisterPath registerName = currentPath/std::string(nameAttr->get_value());
213  info->name = registerName;
214 
215  // add register to catalogue
216  _catalogue.addRegister(info);
217 
218  // iterate over childs of the register to find plugins
219  for(const auto& child : element->get_children()) {
220  // cast into element, ignore if not an element (e.g. comment)
221  const xmlpp::Element *childElement = dynamic_cast<const xmlpp::Element*>(child);
222  if(!childElement) continue;
223  if(childElement->get_name() != "plugin") continue; // look only for plugins
224 
225  // get name of plugin
226  auto pluginNameAttr = childElement->get_attribute("name");
227  if(!pluginNameAttr) {
228  parsingError("Missing name attribute of 'plugin' tag.");
229  }
230  std::string pluginName = pluginNameAttr->get_value();
231 
232  // collect parameters
233  std::map<std::string, DynamicValue<std::string> > parameters;
234  for(const auto& paramchild : childElement->get_children()) {
235  // cast into element, ignore if not an element (e.g. comment)
236  const xmlpp::Element *paramElement = dynamic_cast<const xmlpp::Element*>(paramchild);
237  if(!paramElement) continue;
238 
239  // get name of parameter
240  std::string parameterName = paramElement->get_name();
241 
242  // get value of parameter and store in map
243  parameters[parameterName] = getValueFromXmlSubnode<std::string>(childElement,parameterName);
244 
245  }
246 
247  // create instance of plugin and add to the list in the register info
248  boost::shared_ptr<RegisterInfoPlugin> plugin = RegisterPluginFactory::getInstance().createPlugin(pluginName, parameters);
249  info->pluginList.push_back(plugin);
250 
251  }
