336  int StepperMotor::truncateMotorPosition(int newPosition) {
337  int position = newPosition;
338

352  StepperMotorStatusAndError StepperMotor::determineMotorStatusAndError() {
353 
354  //InternalMutex intMutex(this);
355  boost::lock_guard<boost::mutex> guard(_mutex);
356 
357  _motorStatus = StepperMotorStatusTypes::M_OK;
358  _motorError = StepperMotorErrorTypes::M_NO_ERROR;
359 
360  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
361  // ERROR CONDITION CHECK - it should be done first.
362  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
363 
364 
365  if (_softwareLimitsEnabled && (_maxPositionLimitInSteps <= _minPositionLimitInSteps)) {
366  _motorStatus = StepperMotorStatusTypes::M_ERROR;
367  _motorError = StepperMotorErrorTypes::M_CONFIG_ERROR_MIN_POS_GRATER_EQUAL_TO_MAX;
368 
369  return StepperMotorStatusAndError(_motorStatus, _motorError);
370  }
371 
372  //first check if disabled
373  if (!_motorControler->isEnabled()) {
374  _motorStatus = StepperMotorStatusTypes::M_DISABLED;
375  return StepperMotorStatusAndError(_motorStatus, _motorError);
376  }
377 
378  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
379  // STATUS CHECK
380  // !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
381 
382  if (_motorControler->getStatus().getStandstillIndicator() == 0) {
383  _motorStatus = StepperMotorStatusTypes::M_IN_MOVE;
384  return StepperMotorStatusAndError(_motorStatus, _motorError);
385  }
386 
387 
388  if (_targetPositionInSteps != _motorControler->getActualPosition()) {
389  _motorStatus = StepperMotorStatusTypes::M_NOT_IN_POSITION;
390  return StepperMotorStatusAndError(_motorStatus, _motorError);
391  }
392 
393  if (_softwareLimitsEnabled) {
394  if (_targetPositionInSteps >= _maxPositionLimitInSteps) {
395  _motorStatus = StepperMotorStatusTypes::M_SOFT_POSITIVE_END_SWITCHED_ON;
396  return StepperMotorStatusAndError(_motorStatus, _motorError);
397  }
398 
399  if (_targetPositionInSteps <= _minPositionLimitInSteps) {
400  _motorStatus = StepperMotorStatusTypes::M_SOFT_NEGATIVE_END_SWITCHED_ON;
401  return StepperMotorStatusAndError(_motorStatus, _motorError);
402  }
403  }
404 
405  return StepperMotorStatusAndError(_motorStatus, _motorError);
406  }
