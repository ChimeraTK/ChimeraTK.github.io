396  template<class T>
397  void BidirectionalProcessArray<T>::doPostRead() {
398  _receiver->postRead();
399  // We only update the current value (stored in the sender) when the version
400  // number of the data that we received is greater than the current version
401  // number. This ensures that old updates (that might arrive late due to the
402  // asynchronous nature of the transfer logic) do not overwrite newer values
403  // and also helps to ensure that we do not get a feedback loop where two (or
404  // more) bidirectional process variables "play ping-pong" (see issue #2 for
405  // the full discussion).
406  if (_receiver->getVersionNumber() > _versionNumber) {
407  this->accessChannel(0).swap(_receiver->accessChannel(0));
408  // After receiving, our new time stamp and version number are the ones
409  // that we got from the receiver.
410  _timeStamp = _receiver->getTimeStamp();
411  _versionNumber = _receiver->getVersionNumber();
412  // If we have a persistent data-storage, we have to update it. We have to
413  // do this because a (new) value received from the other side should be
414  // treated like a value sent by this side.
415  if (_persistentDataStorage) {
416  _persistentDataStorage->updateValue(_persistentDataStorageID,
417  _sender->accessChannel(0));
418  }
419  }
420  }

424  template<class T>
425  bool BidirectionalProcessArray<T>::doWriteTransfer(
426  ChimeraTK::VersionNumber versionNumber) {
427  // We have to copy our current value to the sender. We cannot swap it
428  // because this would mean that we would lose the current value.
429  _sender->accessChannel(0) = this->accessChannel(0);
430  // We need a new time stamp.
431  TimeStamp newTimeStamp =
432  _timeStampSource ?
433  _timeStampSource->getCurrentTimeStamp() : TimeStamp::currentTime();
434  // We already copied the value, so the sender does not have to copy the
435  // value again.
436  bool lostData = _sender->writeDestructively(newTimeStamp, versionNumber);
437  // After sending the new value, our current time stamp and version number
438  // are the one from the sender.
439  _timeStamp = newTimeStamp;
440  _versionNumber = versionNumber;
441  // If we have a persistent data-storage, we have to update it.
442  if (_persistentDataStorage) {
443  _persistentDataStorage->updateValue(_persistentDataStorageID,
444  _sender->accessChannel(0));
445  }
446  if (_sendNotificationListener) {
447  typename ProcessArray<T>::SharedPtr partner = _partner.lock();
448  if (partner) {
449  _sendNotificationListener->notify(partner);
450  }
451  }
452  return lostData;
453  }
