212  template<typename UserType>
213  void DoocsBackendRegisterAccessor<UserType>::initialise() {
214  if(isInitialised) return;
215 
216  // try to read data, to check connectivity and to obtain size of the register
217  EqData tmp;
218  int rc = eq.get(&ea, &tmp, &dst);
219  if(rc) {
220  if(rc == eq_errors::ill_property || rc == eq_errors::ill_location ||
221  rc == eq_errors::ill_address) { // no property by that name
222  throw ChimeraTK::logic_error("Property does not exist: " + _path);
223  }
224  // runtime error will later be thrown by the transfer...
225  return;
226  }
227 
228  // obtain number of elements
229  size_t actualLength = dst.array_length();
230  if(actualLength == 0 && dst.length() == 1) {
231  actualLength = 1;
232  isArray = false;
233  }
234  else {
235  if(actualLength == 0) actualLength = dst.length();
236  isArray = true;
237  }
238  if(nElements == 0) {
239  nElements = actualLength;
240  }
241  if(nElements + elementOffset > actualLength) {
242  throw ChimeraTK::logic_error("Requested number of words exceeds the length of the DOOCS property!");
243  }
244  if(nElements == actualLength && elementOffset == 0) {
245  isPartial = false;
246  }
247  else {
248  isPartial = true;
249  }
250 
251  // allocate buffers
252  if(_allocateBuffers) {
253  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
254  NDRegisterAccessor<UserType>::buffer_2D[0].resize(nElements);
255  }
256 
257  // set proper type information in the source EqData
258  src.set_type(dst.type());
259  if(_allocateBuffers && dst.type() != DATA_IIII) {
260  src.length(actualLength);
261  }
262 
263  // use ZeroMQ with AccessMode::wait_for_new_data
264  if(useZMQ) {
265  // subscribe via subscription manager
266  DoocsBackendNamespace::ZMQSubscriptionManager::getInstance().subscribe(_path, this);
267  }
268 
269  initialiseImplementation();
270  isInitialised = true;
271  }

293

296

306  }

311  if(nElements == 0) nElements = actualLength;
