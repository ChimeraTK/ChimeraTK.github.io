27  LNMBackendVariableAccessor(boost::shared_ptr<DeviceBackend> dev, const RegisterPath &registerPathName,
28  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
29  : NDRegisterAccessor<UserType>(registerPathName),
30  _registerPathName(registerPathName),
31  _fixedPointConverter(registerPathName, 32, 0, 1)
32  {
33  try {
34  // check for unknown flags
35  flags.checkForUnknownFlags({AccessMode::raw});
36  // check for illegal parameter combinations
37  if(wordOffsetInRegister != 0 || numberOfWords > 1) {
38  throw DeviceException("LNMBackendBufferingVariableAccessor: offset and number of words not "
39  "supported!", DeviceException::NOT_IMPLEMENTED); // LCOV_EXCL_LINE (impossible to test...)
40  }
41  if(flags.has(AccessMode::raw)) {
42  if(typeid(UserType) != typeid(int32_t)) {
43  throw DeviceException("Given UserType when obtaining the LNMBackendBufferingVariableAccessor in raw mode"
44  " does not match the expected type. Use an int32_t instead!", DeviceException::WRONG_PARAMETER);
45  }
46  }
47  _dev = boost::dynamic_pointer_cast<LogicalNameMappingBackend>(dev);
48  // obtain the register info
49  _info = boost::static_pointer_cast<LNMBackendRegisterInfo>(
50  _dev->getRegisterCatalogue().getRegister(_registerPathName));
51  // check for incorrect usage of this accessor
52  if( _info->targetType != LNMBackendRegisterInfo::TargetType::INT_CONSTANT &&
53  _info->targetType != LNMBackendRegisterInfo::TargetType::INT_VARIABLE ) {
54  throw DeviceException("LNMBackendBufferingVariableAccessor used for wrong register type.",
55  DeviceException::WRONG_PARAMETER); // LCOV_EXCL_LINE (impossible to test...)
56  }
57  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
58  NDRegisterAccessor<UserType>::buffer_2D[0].resize(1);
59  NDRegisterAccessor<UserType>::buffer_2D[0][0] = _fixedPointConverter.toCooked<UserType>(_info->value);
60  }
61  catch(...) {
62  this->shutdown();
63  throw;
64  }
65  }
