152  bool readNonBlocking() {
153  if(TransferElement::_isInTransferGroup) {
154  throw ChimeraTK::logic_error("Calling read() or write() on an accessor "

157  }

159  preReadAndHandleExceptions(TransferType::readNonBlocking);

161  if(!_activeException) {

173  bool readLatest() {
174  if(_accessModeFlags.has(AccessMode::wait_for_new_data)) {
175  bool updateDataBuffer = false;
176  // Call readNonBlocking until there is no new data to be read any more
177  while(readNonBlocking()) {
178  // remember whether we have new data
179  updateDataBuffer = true;
180  }
181  return updateDataBuffer;
182  }
183  else {
184  // Without wait_for_new_data readNonBlocking always returns true, and the while loop above would never end.
185  // Hence we just call the (synchronous) read and return true;
186  read();
187  return true;
188  }
189  }

195  if(TransferElement::_isInTransferGroup) {

198  "is not allowed.");

200  this->writeTransactionInProgress = false;
201  bool previousDataLost =
202  true; // the value here does not matter. If there was an exception, it will be re-thrown in postWrite, so it is never returned

205  if(!_activeException) {

427  void postRead(TransferType type, bool updateDataBuffer) {
428  // only delegate to doPostRead() the first time postRead() is called in a row.
429  if(readTransactionInProgress) {
430  readTransactionInProgress = false;
431  doPostRead(type, updateDataBuffer);
432  }

434  // Throw on each call of postRead(). All high-level elements for a shared low-level transfer element must see the exception.
435  // Note: doPostRead can throw an exception, but in that case _activeException must be false (we can only have one
436  // exception at a time). In case other code is added here later which needs to be executed after doPostRead()
437  // always, a try-catch block may be necessary.
438  if(_activeException) {
439  // don't clear the active connection. This is done in preRead().
440  std::rethrow_exception(_activeException);
