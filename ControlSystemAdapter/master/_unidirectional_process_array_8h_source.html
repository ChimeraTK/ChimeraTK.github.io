633  template<class T>
634  bool UnidirectionalProcessArray<T>::writeInternal(TimeStamp newTimeStamp, VersionNumber newVersionNumber,
635  bool shouldCopy) {
636 
637  // thread safety check, if enabled (only active with debug flags enabled)
638  assert(checkThreadSafety());
639 
640  if(!this->isWriteable()) {
641  throw std::logic_error(
642  "Send operation is only allowed for a sender process variable. Variable name: "+this->getName());
643  }
644 
645  // We have to check that the vector that we currently own still has the
646  // right size. Otherwise, the code using the receiver might get into
647  // trouble when it suddenly experiences a vector of the wrong size.
648  if(ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0].size() != _vectorSize) {
649  throw std::logic_error(
650  "Cannot run receive operation because the size of the vector belonging to the current buffer has been "
651  "modified. Variable name: "+this->getName());
652  }
653 
654  // A version should never be send with a version number that is equal to or
655  // even less than the last version number used. Such an attempt indicates
656  // that there is a problem in the logic attempting the write operation.
657  if (newVersionNumber < _versionNumber) {
658  throw std::logic_error(
659  "The version number passed to write is less than or equal to the last version number used.");
660  }
661 
662  // First update the persistent data storage, if any was associated. This cannot be done after sending, since the
663  // value might no longer be available within this instance.
664  if(_persistentDataStorage) {
665  _persistentDataStorage->updateValue(_persistentDataStorageID, ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0]);
666  }
667 
668  // Set time stamp and version number
669  _localBuffer._timeStamp = newTimeStamp;
670  _localBuffer._versionNumber = newVersionNumber;
671  _timeStamp = newTimeStamp;
672  _versionNumber = newVersionNumber;
673 
674  // set the data by copying or swapping
675  assert(_localBuffer._value.size() == ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0].size() );
676  if(shouldCopy) {
677  _localBuffer._value = ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0];
678  }
679  else {
680  _localBuffer._value.swap( ChimeraTK::NDRegisterAccessor<T>::buffer_2D[0] );
681  }
682 
683  // send the data to the queue
684  bool dataNotLost = _sharedState._queue.push_overwrite(std::move(_localBuffer));
685 
686  // notify send notification listener, if present
687  if(_sendNotificationListener) {
688  _sendNotificationListener->notify(_receiver);
689  }
690 
691  return !dataNotLost;
692  }
