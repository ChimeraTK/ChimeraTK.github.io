168  template<typename UserType>
169  boost::shared_ptr< NDRegisterAccessor<UserType> > SubdeviceBackend::getRegisterAccessor_impl(
170  const RegisterPath &registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
171  assert(type == Type::area);
172 
173  // obtain register info
174  auto info = boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(_catalogue.getRegister(registerPathName));
175 
176  // check that the bar is 0
177  if(info->bar != 0) {
178  throw DeviceException("SubdeviceBackend: BARs other then 0 are not supported. Register '"+registerPathName+
179  "' is in BAR "+std::to_string(info->bar)+".", DeviceException::WRONG_PARAMETER);
180  }
181 
182  // check that the register is not a 2D multiplexed register, which is not yet supported
183  if(info->is2DMultiplexed) {
184  throw DeviceException("SubdeviceBackend: 2D multiplexed registers are not yet supported.",
185  DeviceException::NOT_IMPLEMENTED);
186  }
187 
188  // compute full offset (from map file and function arguments)
189  size_t byteOffset = info->address + sizeof(int32_t)*wordOffsetInRegister;
190  if(byteOffset % 4 != 0) {
191  throw DeviceException("SubdeviceBackend: Only addresses which are a multiple of 4 are supported.",
192  DeviceException::WRONG_PARAMETER);
193  }
194  size_t wordOffset = byteOffset / 4;
195 
196  // compute effective length
197  if(numberOfWords == 0) {
198  numberOfWords = info->nElements;
199  }
200  else if(numberOfWords > info->nElements) {
201  throw DeviceException("SubdeviceBackend: Requested "+std::to_string(numberOfWords)+" elements from register '"+
202  registerPathName+"', which only has a length of "+std::to_string(info->nElements)+
203  " elements.", DeviceException::WRONG_PARAMETER);
204  }
205 
206  // raw transfer: directly return accessor from target device
207  if(flags.has(AccessMode::raw)) {
208  return targetDevice->getRegisterAccessor<UserType>(targetArea, numberOfWords, wordOffset, flags);
209  }
210 
211  // if not a raw transfer: obtain raw accessor from target device and decorate with fixed-point-converting decorator
213  auto rawAcc = targetDevice->getRegisterAccessor<int32_t>(targetArea, numberOfWords, wordOffset, {AccessMode::raw});
214  return boost::make_shared<FixedPointConvertingDecorator<UserType, int32_t>>(rawAcc,
215  FixedPointConverter(registerPathName, info->width, info->nFractionalBits, info->signedFlag) );
216  }
