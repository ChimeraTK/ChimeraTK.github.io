102  while(true) {
103  // receive data
104  boost::this_thread::interruption_point();
105  Profiler::stopMeasurement();
106  auto var = group.readAny();
107  Profiler::startMeasurement();
108  boost::this_thread::interruption_point();
109  // if the update came through the return channel, return it to the feeder
110  if(var == _returnChannelSlave->getId()) {
111  FanOut<UserType>::impl->accessChannel(0).swap(_returnChannelSlave->accessChannel(0));
112  FanOut<UserType>::impl->write(_returnChannelSlave->getVersionNumber());
113  }
114  // send out copies to slaves
115  auto version = FanOut<UserType>::impl->getVersionNumber();
116  for(auto& slave : FanOut<UserType>::slaves) {
117  // do not feed back value returnChannelSlave if it was received from it
118  if(slave->getId() == var) continue;
119  // do not send copy if no data is expected (e.g. trigger)
120  if(slave->getNumberOfSamples() != 0) {
121  slave->accessChannel(0) = FanOut<UserType>::impl->accessChannel(0);
122  }
123  bool dataLoss = slave->write(version);
124  if(dataLoss) Application::incrementDataLossCounter();
125  }
126  }
