402  template<class T>
403  void BidirectionalProcessArray<T>::doPostRead() {
404  _receiver->postRead();
405  // We only update the current value (stored in the sender) when the version
406  // number of the data that we received is greater than the current version
407  // number. This ensures that old updates (that might arrive late due to the
408  // asynchronous nature of the transfer logic) do not overwrite newer values
409  // and also helps to ensure that we do not get a feedback loop where two (or
410  // more) bidirectional process variables "play ping-pong" (see issue #2 for
411  // the full discussion).
412  if (_receiver->getVersionNumber() > _versionNumber) {
413  this->accessChannel(0).swap(_receiver->accessChannel(0));
414  // After receiving, our new time stamp and version number are the ones
415  // that we got from the receiver.
416  _timeStamp = _receiver->getTimeStamp();
417  _versionNumber = _receiver->getVersionNumber();
418  // If we have a persistent data-storage, we have to update it. We have to
419  // do this because a (new) value received from the other side should be
420  // treated like a value sent by this side.
421  if (_persistentDataStorage) {
422  _persistentDataStorage->updateValue(_persistentDataStorageID,
423  _sender->accessChannel(0));
424  }
425  }
426  }

436  template<class T>
437  bool BidirectionalProcessArray<T>::doWriteTransfer(
438  ChimeraTK::VersionNumber versionNumber) {
439  // We have to copy our current value to the sender. We cannot swap it
440  // because this would mean that we would lose the current value.
441  _sender->accessChannel(0) = this->accessChannel(0);
442  // We need a new time stamp.
443  TimeStamp newTimeStamp =
444  _timeStampSource ?
445  _timeStampSource->getCurrentTimeStamp() : TimeStamp::currentTime();
446  // We already copied the value, so the sender does not have to copy the
447  // value again.
448  bool lostData = _sender->writeDestructively(newTimeStamp, versionNumber);
449  // After sending the new value, our current time stamp and version number
450  // are the one from the sender.
451  _timeStamp = newTimeStamp;
452  _versionNumber = versionNumber;
453  // If we have a persistent data-storage, we have to update it.
454  if (_persistentDataStorage) {
455  _persistentDataStorage->updateValue(_persistentDataStorageID,
456  _sender->accessChannel(0));
457  }
458  if (_sendNotificationListener) {
459  typename ProcessArray<T>::SharedPtr partner = _partner.lock();
460  if (partner) {
461  _sendNotificationListener->notify(partner);
462  }
463  }
464  return lostData;
465  }
