26 extern "C" int nanosleep(__const struct timespec *__requested_time, struct timespec *__remaining) {
27 
28  // call original-equivalent version if requested time does not match the magic signature
29  if(__requested_time->tv_sec != doocsServerTestHelper::magic_sleep_time_sec ||
30  __requested_time->tv_nsec != doocsServerTestHelper::magic_sleep_time_usec*1000 ) {
31  return clock_nanosleep(CLOCK_MONOTONIC, 0, __requested_time, __remaining);
32  }
33 
34  // if timing system is not yet initialised, sleep for 1 second
35  if(!doocsServerTestHelper::interceptSystemCalls) {
36  struct timespec tv;
37  tv.tv_sec = 1;
38  tv.tv_nsec = 0;
39  return clock_nanosleep(CLOCK_MONOTONIC, 0, &tv, __remaining);
40  }
41 
42  // if this is the first call, acquire lock and set flag that the server has been started
43  if(!doocsServerTestHelper::serverStarted) {
44  doocsServerTestHelper::serverStarted = true;
45  doocsServerTestHelper::update_mutex.lock();
46  doocsServerTestHelper::sigusr1_mutex.lock();
47  doocsServerTestHelper::allowUpdate = false;
48  doocsServerTestHelper::allowSigusr1 = false;
49  doocsServerTestHelper::sigusr1_mutex.unlock();
