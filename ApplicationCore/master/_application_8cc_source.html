99  for(auto &accessor : module->getAccessorList()) {
100  if(!accessor.hasOwner()) {
101  if(enableUnconnectedVariablesWarning) {

112  assert(callable.done);
113  constantList.emplace_back(callable.theNode);

234 VariableNetwork& Application::connect(VariableNetworkNode a, VariableNetworkNode b) {
235 
236  // if one of the nodes has the value type AnyType, set it to the type of the other
237  // if both are AnyType, nothing changes.
238  if(a.getValueType() == typeid(AnyType)) {
239  a.setValueType(b.getValueType());
240  }
241  else if(b.getValueType() == typeid(AnyType)) {
242  b.setValueType(a.getValueType());
243  }
244 
245  // if one of the nodes has not yet a defined number of elements, set it to the number of elements of the other.
246  // if both are undefined, nothing changes.
247  if(a.getNumberOfElements() == 0) {
248  a.setNumberOfElements(b.getNumberOfElements());
249  }
250  else if(b.getNumberOfElements() == 0) {
251  b.setNumberOfElements(a.getNumberOfElements());
252  }
253  if(a.getNumberOfElements() != b.getNumberOfElements()) {
254  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalParameter>(
255  "Error: Cannot connect array variables with difference number of elements!");
256  }
257 
258  // if both nodes already have an owner, we are done
259  if(a.hasOwner() && b.hasOwner()) {
260  if(&(a.getOwner()) != &(b.getOwner())) {
261  std::stringstream what;
262  what << "*** ERROR: nodes to be connected should have the same owner!" << std::endl;
263  what << "Node A:" << std::endl;
264  a.dump(what);
265  what << "Node B:" << std::endl;
266  b.dump(what);
267  what << "Owner of node A:" << std::endl;
268  a.getOwner().dump("",what);
269  what << "Owner of node B:" << std::endl;
270  b.getOwner().dump("",what);
271  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalVariableNetwork>(what.str());
272  }
273  }
274  // add b to the existing network of a
275  else if(a.hasOwner()) {
276  a.getOwner().addNode(b);
277  }
278 
279  // add a to the existing network of b
280  else if(b.hasOwner()) {
281  b.getOwner().addNode(a);
282  }
283  // create new network
284  else {
285  networkList.emplace_back();
286  networkList.back().addNode(a);
287  networkList.back().addNode(b);
288  }
289  return a.getOwner();
290 }

321 boost::shared_ptr<mtca4u::NDRegisterAccessor<UserType>> Application::createProcessVariable(VariableNetworkNode const &node) {
322 
323  // determine the SynchronizationDirection
324  SynchronizationDirection dir;
325  if(node.getDirection() == VariableDirection::feeding) {
326  dir = SynchronizationDirection::controlSystemToDevice;
327  }
328  else {
329  dir = SynchronizationDirection::deviceToControlSystem;
330  }
331 
332  // create the ProcessArray for the proper UserType
333  auto pvar = _processVariableManager->createProcessArray<UserType>(dir, node.getPublicName(), node.getNumberOfElements(),
334  node.getOwner().getUnit(), node.getOwner().getDescription());
335  assert(pvar->getName() != "");
336 
337  // create variable ID
338  idMap[pvar->getId()] = getNextVariableId();
339  pvIdMap[pvar->getUniqueId()] = idMap[pvar->getId()];
340 
341  // Decorate the process variable if testable mode is enabled and this is the receiving end of the variable.
342  // Also don't decorate, if the mode is polling. Instead flag the variable to be polling, so the TestFacility is aware of this.
343  if(testableMode && node.getDirection() == VariableDirection::feeding) {
344 
345  // The transfer mode of this process variable is considered to be polling, if only one consumer exists and this
346  // consumer is polling. Reason: mulitple consumers will result in the use of a FanOut, so the communication up to
347  // the FanOut will be push-type, even if all consumers are poll-type.
349  auto mode = UpdateMode::push;
350  if(node.getOwner().countConsumingNodes() == 1) {
351  if(node.getOwner().getConsumingNodes().front().getMode() == UpdateMode::poll) mode = UpdateMode::poll;
352  }
353 
354  if(mode != UpdateMode::poll) {
355  auto pvarDec = boost::make_shared<TestDecoratorRegisterAccessor<UserType>>(pvar);
356  testableMode_names[idMap[pvarDec->getId()]] = "ControlSystem:"+node.getPublicName();
357  return pvarDec;
358  }
359  else {
360  testableMode_isPollMode[idMap[pvar->getId()]] = true;
361  }
362  }
363 
364  // return the process variable
365  return pvar;
366 }

444 void Application::optimiseConnections() {
445 
446  // list of iterators of networks to be removed from the networkList after the merge operation
447  std::list<VariableNetwork*> deleteNetworks;
448 
449  // search for networks with the same feeder
450  for(auto it1 = networkList.begin(); it1 != networkList.end(); ++it1) {
451  for(auto it2 = it1; it2 != networkList.end(); ++it2) {
452  if(it1 == it2) continue;
453 
454  auto feeder1 = it1->getFeedingNode();
455  auto feeder2 = it2->getFeedingNode();
456 
457  // this optimisation is only necessary for device-type nodes, since application and control-system nodes will
458  // automatically create merged networks when having the same feeder
460  if(feeder1.getType() != NodeType::Device || feeder2.getType() != NodeType::Device) continue;
461 
462  // check if referrring to same register
463  if(feeder1.getDeviceAlias() != feeder2.getDeviceAlias()) continue;
464  if(feeder1.getRegisterName() != feeder2.getRegisterName()) continue;
465 
466  // check if directions are the same
467  if(feeder1.getDirection() != feeder2.getDirection()) continue;
468 
469  // check if value types and number of elements are compatible
470  if(feeder1.getValueType() != feeder2.getValueType()) continue;
471  if(feeder1.getNumberOfElements() != feeder2.getNumberOfElements()) continue;
472 
473  // check if transfer mode is the same
474  if(feeder1.getMode() != feeder2.getMode()) continue;
475 
476  // check if triggers are compatible, if present
477  if(feeder1.hasExternalTrigger() != feeder2.hasExternalTrigger()) continue;
478  if(feeder1.hasExternalTrigger()) {
479  if(feeder1.getExternalTrigger() != feeder2.getExternalTrigger()) continue;
480  }
481 
482  // everything should be compatible at this point: merge the networks. We will merge the network of the outer
483  // loop into the network of the inner loop, since the network of the outer loop will not be found a second time
484  // in the inner loop.
485  for(auto consumer : it1->getConsumingNodes()) {
486  consumer.clearOwner();
487  it2->addNode(consumer);
488  }
489 
490  // if trigger present, remove corresponding trigger receiver node from the trigger network
491  if(feeder1.hasExternalTrigger()) {
492  for(auto &itTrig : networkList) {
493  if(itTrig.getFeedingNode() != feeder1.getExternalTrigger()) continue;
494  itTrig.removeNodeToTrigger(it1->getFeedingNode());
495  }
496  }
497 
498  // schedule the outer loop network for deletion and stop processing it
499  deleteNetworks.push_back(&(*it1));
500  break;
501  }
502  }
503 
504  // remove networks from the network list
505  for(auto net : deleteNetworks) {
506  networkList.remove(*net);
507  }
508 
509 }

560 template<typename UserType>
561 void Application::typedMakeConnection(VariableNetwork &network) {
562  bool connectionMade = false; // to check the logic...
563 
564  size_t nNodes = network.countConsumingNodes()+1;
565  auto feeder = network.getFeedingNode();
566  auto consumers = network.getConsumingNodes();
567  bool useExternalTrigger = network.getTriggerType() == VariableNetwork::TriggerType::external;
568  bool useFeederTrigger = network.getTriggerType() == VariableNetwork::TriggerType::feeder;
569  bool constantFeeder = feeder.getType() == NodeType::Constant;
570 
571  // 1st case: the feeder requires a fixed implementation
572  if(feeder.hasImplementation() && !constantFeeder) {
573 
574  // Create feeding implementation. Note: though the implementation is derived from the feeder, it will be used as
575  // the implementation of the (or one of the) consumer. Logically, implementations are always pairs of
576  // implementations (sender and receiver), but in this case the feeder already has a fixed implementation pair.
577  // So our feedingImpl will contain the consumer-end of the implementation pair. This is the reason why the
578  // functions createProcessScalar() and createDeviceAccessor() get the VariableDirection::consuming.
579  boost::shared_ptr<mtca4u::NDRegisterAccessor<UserType>> feedingImpl;
580  if(feeder.getType() == NodeType::Device) {
581  feedingImpl = createDeviceVariable<UserType>(feeder.getDeviceAlias(), feeder.getRegisterName(),
582  VariableDirection::consuming, feeder.getMode(), feeder.getNumberOfElements());
583  }
584  else if(feeder.getType() == NodeType::ControlSystem) {
585  feedingImpl = createProcessVariable<UserType>(feeder);
586  }
587  else {
588  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalParameter>("Unexpected node type!"); // LCOV_EXCL_LINE (assert-like)
589  }
590 
591  // if we just have two nodes, directly connect them
592  if(nNodes == 2 && !useExternalTrigger) {
593  auto consumer = consumers.front();
594  if(consumer.getType() == NodeType::Application) {
595  consumer.getAppAccessor<UserType>().replace(feedingImpl);
596  connectionMade = true;
597  }
598  else if(consumer.getType() == NodeType::Device) {
599  auto consumingImpl = createDeviceVariable<UserType>(consumer.getDeviceAlias(), consumer.getRegisterName(),
600  VariableDirection::feeding, consumer.getMode(), consumer.getNumberOfElements());
601  // connect the Device with e.g. a ControlSystem node via a ThreadedFanOut
602  auto fanOut = boost::make_shared<ThreadedFanOut<UserType>>(feedingImpl);
603  fanOut->addSlave(consumingImpl);
604  internalModuleList.push_back(fanOut);
605  connectionMade = true;
606  }
607  else if(consumer.getType() == NodeType::ControlSystem) {
608  auto consumingImpl = createProcessVariable<UserType>(consumer);
609  // connect the ControlSystem with e.g. a Device node via an ThreadedFanOut
610  auto fanOut = boost::make_shared<ThreadedFanOut<UserType>>(feedingImpl);
611  fanOut->addSlave(consumingImpl);
612  internalModuleList.push_back(fanOut);
613  connectionMade = true;
614  }
615  else if(consumer.getType() == NodeType::TriggerReceiver) {
616  consumer.getNodeToTrigger().getOwner().setExternalTriggerImpl(feedingImpl);
617  connectionMade = true;
618  }
619  else {
620  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalParameter>("Unexpected node type!"); // LCOV_EXCL_LINE (assert-like)
621  }
622  }
623  else { /* !(nNodes == 2 && !useExternalTrigger) */
624 
625  // create the right FanOut type
626  boost::shared_ptr<FanOut<UserType>> fanOut;
627  boost::shared_ptr<ConsumingFanOut<UserType>> consumingFanOut;
628  if(useExternalTrigger) {
629  // if external trigger is enabled, use externally triggered threaded FanOut
630  auto triggerNode = feeder.getExternalTrigger();
631  auto triggerFanOut = triggerMap[triggerNode.getUniqueId()];
632  if(!triggerFanOut) {
633  triggerFanOut = boost::make_shared<TriggerFanOut>(network.getExternalTriggerImpl());
634  triggerMap[triggerNode.getUniqueId()] = triggerFanOut;
635  internalModuleList.push_back(triggerFanOut);
636  }
637  fanOut = triggerFanOut->addNetwork(feedingImpl);
638  }
639  else if(useFeederTrigger) {
640  // if the trigger is provided by the pushing feeder, use the treaded version of the FanOut to distribute
641  // new values immediately to all consumers.
642  auto threadedFanOut = boost::make_shared<ThreadedFanOut<UserType>>(feedingImpl);
643  internalModuleList.push_back(threadedFanOut);
644  fanOut = threadedFanOut;
645  }
646  else {
647  assert(network.hasApplicationConsumer()); // checkConnections should catch this
648  consumingFanOut = boost::make_shared<ConsumingFanOut<UserType>>(feedingImpl);
649  fanOut = consumingFanOut;
650  }
651 
652  // In case we have one or more trigger receivers among our consumers, we produce exactly one application variable
653  // for it. We never need more, since the distribution is done with a TriggerFanOut.
654  bool usedTriggerReceiver{false}; // flag if we already have a trigger receiver
655  auto triggerConnection = createApplicationVariable<UserType>(feeder); // will get destroyed if not used
656 
657  // add all consumers to the FanOut
658  for(auto &consumer : consumers) {
659  if(consumer.getType() == NodeType::Application) {
660  if(consumingFanOut && consumer.getMode() == UpdateMode::poll) {
661  consumer.getAppAccessor<UserType>().replace(consumingFanOut);
662  consumingFanOut.reset();
663  }
664  else {
665  auto impls = createApplicationVariable<UserType>(consumer);
666  fanOut->addSlave(impls.first);
667  consumer.getAppAccessor<UserType>().replace(impls.second);
668  }
669  }
670  else if(consumer.getType() == NodeType::ControlSystem) {
671  auto impl = createProcessVariable<UserType>(consumer);
672  fanOut->addSlave(impl);
673  }
674  else if(consumer.getType() == NodeType::Device) {
675  auto impl = createDeviceVariable<UserType>(consumer.getDeviceAlias(), consumer.getRegisterName(),
676  VariableDirection::feeding, consumer.getMode(), consumer.getNumberOfElements());
677  fanOut->addSlave(impl);
678  }
679  else if(consumer.getType() == NodeType::TriggerReceiver) {
680  if(!usedTriggerReceiver) fanOut->addSlave(triggerConnection.first);
681  usedTriggerReceiver = true;
682  consumer.getNodeToTrigger().getOwner().setExternalTriggerImpl(triggerConnection.second);
683  }
684  else {
685  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalParameter>("Unexpected node type!"); // LCOV_EXCL_LINE (assert-like)
686  }
687  }
688  connectionMade = true;
689  }
690  }
691  // 2nd case: the feeder does not require a fixed implementation
692  else if(!constantFeeder) { /* !feeder.hasImplementation() */
693  // we should be left with an application feeder node
694  if(feeder.getType() != NodeType::Application) {
695  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalParameter>("Unexpected node type!"); // LCOV_EXCL_LINE (assert-like)
696  }
697  assert(!useExternalTrigger);
698  // if we just have two nodes, directly connect them
699  if(nNodes == 2) {
700  auto consumer = consumers.front();
701  if(consumer.getType() == NodeType::Application) {
702  auto impls = createApplicationVariable<UserType>(feeder,consumer);
703  feeder.getAppAccessor<UserType>().replace(impls.first);
704  consumer.getAppAccessor<UserType>().replace(impls.second);
705  connectionMade = true;
706  }
707  else if(consumer.getType() == NodeType::ControlSystem) {
708  auto impl = createProcessVariable<UserType>(consumer);
709  feeder.getAppAccessor<UserType>().replace(impl);
710  connectionMade = true;
711  }
712  else if(consumer.getType() == NodeType::Device) {
713  auto impl = createDeviceVariable<UserType>(consumer.getDeviceAlias(), consumer.getRegisterName(),
714  VariableDirection::feeding, consumer.getMode(), consumer.getNumberOfElements());
715  feeder.getAppAccessor<UserType>().replace(impl);
716  connectionMade = true;
717  }
718  else if(consumer.getType() == NodeType::TriggerReceiver) {
719  auto impls = createApplicationVariable<UserType>(feeder,consumer);
720  feeder.getAppAccessor<UserType>().replace(impls.first);
721  consumer.getNodeToTrigger().getOwner().setExternalTriggerImpl(impls.second);
722  connectionMade = true;
723  }
724  else if(consumer.getType() == NodeType::Constant) {
725  auto impl = consumer.getConstAccessor<UserType>();
726  feeder.getAppAccessor<UserType>().replace(impl);
727  connectionMade = true;
728  }
729  else {
730  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalParameter>("Unexpected node type!"); // LCOV_EXCL_LINE (assert-like)
731  }
732  }
733  else {
734  // create FanOut and use it as the feeder implementation
735  auto fanOut = boost::make_shared<FeedingFanOut<UserType>>(feeder.getName(), feeder.getUnit(),
736  feeder.getDescription(), feeder.getNumberOfElements());
737  feeder.getAppAccessor<UserType>().replace(fanOut);
738 
739  // In case we have one or more trigger receivers among our consumers, we produce exactly one application variable
740  // for it. We never need more, since the distribution is done with a TriggerFanOut.
741  bool usedTriggerReceiver{false}; // flag if we already have a trigger receiver
742  auto triggerConnection = createApplicationVariable<UserType>(feeder); // will get destroyed if not used
743 
744  for(auto &consumer : consumers) {
745  if(consumer.getType() == NodeType::Application) {
746  auto impls = createApplicationVariable<UserType>(consumer);
747  fanOut->addSlave(impls.first);
748  consumer.getAppAccessor<UserType>().replace(impls.second);
749  }
750  else if(consumer.getType() == NodeType::ControlSystem) {
751  auto impl = createProcessVariable<UserType>(consumer);
752  fanOut->addSlave(impl);
753  }
754  else if(consumer.getType() == NodeType::Device) {
755  auto impl = createDeviceVariable<UserType>(consumer.getDeviceAlias(), consumer.getRegisterName(),
756  VariableDirection::feeding, consumer.getMode(), consumer.getNumberOfElements());
757  fanOut->addSlave(impl);
758  }
759  else if(consumer.getType() == NodeType::TriggerReceiver) {
760  if(!usedTriggerReceiver) fanOut->addSlave(triggerConnection.first);
761  usedTriggerReceiver = true;
762  consumer.getNodeToTrigger().getOwner().setExternalTriggerImpl(triggerConnection.second);
763  }
764  else {
765  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalParameter>("Unexpected node type!"); // LCOV_EXCL_LINE (assert-like)
766  }
767  }
768  connectionMade = true;
769  }
770  }
771  else { /* constantFeeder */
772  assert(feeder.getType() == NodeType::Constant);
773  auto feedingImpl = feeder.getConstAccessor<UserType>();
774  assert(feedingImpl != nullptr);
775 
776  for(auto &consumer : consumers) {
777  if(consumer.getType() == NodeType::Application) {
778  if(testableMode) {
779  idMap[feedingImpl->getId()] = getNextVariableId();
780  auto pvarDec = boost::make_shared<TestDecoratorRegisterAccessor<UserType>>(feedingImpl);
781  testableMode_names[idMap[pvarDec->getId()]] = "Constant";
782  consumer.getAppAccessor<UserType>().replace(pvarDec);
783  }
784  else {
785  consumer.getAppAccessor<UserType>().replace(feedingImpl);
786  }
787  }
788  else if(consumer.getType() == NodeType::ControlSystem) {
789  auto impl = createProcessVariable<UserType>(consumer);
790  impl->accessChannel(0) = feedingImpl->accessChannel(0);
791  impl->write();
792  }
793  else if(consumer.getType() == NodeType::Device) {
794  auto impl = createDeviceVariable<UserType>(consumer.getDeviceAlias(), consumer.getRegisterName(),
795  VariableDirection::feeding, consumer.getMode(), consumer.getNumberOfElements());
796  impl->accessChannel(0) = feedingImpl->accessChannel(0);
797  impl->write();
798  }
799  else if(consumer.getType() == NodeType::TriggerReceiver) {
800  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalParameter>("Using constants as triggers is not supported!");
801  }
802  else {
803  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalParameter>("Unexpected node type!"); // LCOV_EXCL_LINE (assert-like)
804  }
805  }
806  connectionMade = true;
807 
808  }
809 
810  if(!connectionMade) { // LCOV_EXCL_LINE (assert-like)
811  throw ApplicationExceptionWithID<ApplicationExceptionID::notYetImplemented>( // LCOV_EXCL_LINE (assert-like)
812  "The variable network cannot be handled. Implementation missing!"); // LCOV_EXCL_LINE (assert-like)
813  } // LCOV_EXCL_LINE (assert-like)
814 
815 }

832 void Application::stepApplication() {
833  // testableMode_counter must be non-zero, otherwise there is no input for the application to process
834  if(testableMode_counter == 0) {
835  throw ApplicationExceptionWithID<ApplicationExceptionID::illegalParameter>(
836  "Application::stepApplication() called despite no input was provided to the application to process!");
837  }
838  // let the application run until it has processed all data (i.e. the semaphore counter is 0)
839  size_t oldCounter = 0;
840  while(testableMode_counter > 0) {
841  if(enableDebugTestableMode && ( oldCounter != testableMode_counter) ) { // LCOV_EXCL_LINE (only cout)
842  std::cout << "Application::stepApplication(): testableMode_counter = " << testableMode_counter << std::endl; // LCOV_EXCL_LINE (only cout)
843  oldCounter = testableMode_counter; // LCOV_EXCL_LINE (only cout)
844  }
845  testableModeUnlock("stepApplication");
846  boost::this_thread::yield();
847  testableModeLock("stepApplication");
848  }
849 }

881 void Application::testableModeLock(const std::string& name) {
882  // don't do anything if testable mode is not enabled
883  if(!getInstance().testableMode) return;
884 
885  // debug output if enabled (also prevent spamming the same message)
886  if(getInstance().enableDebugTestableMode && getInstance().testableMode_repeatingMutexOwner == 0) { // LCOV_EXCL_LINE (only cout)
887  std::cout << "Application::testableModeLock(): Thread " << threadName() // LCOV_EXCL_LINE (only cout)
888  << " tries to obtain lock for " << name << std::endl; // LCOV_EXCL_LINE (only cout)
889  } // LCOV_EXCL_LINE (only cout)
890 
891  // if last lock was obtained repeatedly by the same thread, sleep a short time before obtaining the lock to give the
892  // other threads a chance to get the lock first
893  if(getInstance().testableMode_repeatingMutexOwner > 0) usleep(10000);
894 
895  // obtain the lock
896  getTestableModeLockObject().lock();
897 
898  // check if the last owner of the mutex was this thread, which may be a hint that no other thread is waiting for the
899  // lock
900  if(getInstance().testableMode_lastMutexOwner == std::this_thread::get_id()) {
901  // debug output if enabled
902  if(getInstance().enableDebugTestableMode && getInstance().testableMode_repeatingMutexOwner == 0) { // LCOV_EXCL_LINE (only cout)
903  std::cout << "Application::testableModeLock(): Thread " << threadName() // LCOV_EXCL_LINE (only cout)
904  << " repeatedly obtained lock successfully for " << name // LCOV_EXCL_LINE (only cout)
905  << ". Further messages will be suppressed." << std::endl; // LCOV_EXCL_LINE (only cout)
906  } // LCOV_EXCL_LINE (only cout)
907 
908  // increase counter for stall detection
909  getInstance().testableMode_repeatingMutexOwner++;
910 
911  // detect stall: if the same thread got the mutex with no other thread obtaining it in between for one second, we
912  // assume no other thread is able to process data at this time. The test should fail in this case
913  if(getInstance().testableMode_repeatingMutexOwner > 100) {
914  // print an informative message first, which lists also all variables currently containing unread data.
915  std::cout << "*** Tests are stalled due to data which has been sent but not received." << std::endl;
916  std::cout << " The following variables still contain unread values or had data loss due to a queue overflow:" << std::endl;
917  for(auto &pair : Application::getInstance().testableMode_perVarCounter) {
918  if(pair.second > 0) {
919  std::cout << " - " << Application::getInstance().testableMode_names[pair.first];
920  // check if process variable still has data in the queue
921  try {
922  if(getInstance().testableMode_processVars[pair.first]->readNonBlocking()) {
923  std::cout << " (unread data in queue)";
924  }
925  else {
926  std::cout << " (data loss)";
927  }
928  }
929  catch(std::logic_error &e) {
930  // if we receive a logic_error in readNonBlocking() it just means another thread is waiting on a
931  // TransferFuture of this variable, and we actually were not allowed to read...
932  std::cout << " (data loss)";
933  }
934  std::cout << std::endl;
935  }
936  }
937  // throw a specialised exception to make sure whoever catches it really knows what he does...
938  throw TestsStalled();
939  }
940  }
941  else {
942  // last owner of the mutex was different: reset the counter and store the thread id
943  getInstance().testableMode_repeatingMutexOwner = 0;
944  getInstance().testableMode_lastMutexOwner = std::this_thread::get_id();
945 
946  // debug output if enabled
947  if(getInstance().enableDebugTestableMode) { // LCOV_EXCL_LINE (only cout)
948  std::cout << "Application::testableModeLock(): Thread " << threadName() // LCOV_EXCL_LINE (only cout)
949  << " obtained lock successfully for " << name << std::endl; // LCOV_EXCL_LINE (only cout)
950  } // LCOV_EXCL_LINE (only cout)
951  }
952 }
