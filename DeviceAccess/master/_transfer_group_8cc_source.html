58  void TransferGroup::addAccessor(TransferElementAbstractor& accessor) {
59  // check if accessor is already in a transfer group
60  if(accessor.getHighLevelImplElement()->isInTransferGroup) {
61  throw ChimeraTK::logic_error("The given accessor is already in a TransferGroup and cannot be added "
62  "to another.");
63  }
64 
65  // set flag on the accessors that it is now in a transfer group
66  accessor.getHighLevelImplElement()->isInTransferGroup = true;
67 
68  auto highLevelElementsWithNewAccessor = highLevelElements;
69  highLevelElementsWithNewAccessor.insert(accessor.getHighLevelImplElement());
70 
71  // try replacing all internal elements in all high-level elements
72  for(auto& hlElem1 : highLevelElementsWithNewAccessor) {
73  auto list = hlElem1->getInternalElements();
74  list.push_front(hlElem1);
75 
76  for(auto& replacement : list) {
77  // try on the abstractor first, to make sure we replace at the highest
78  // level if possible
79  accessor.replaceTransferElement(replacement);
80  // try on all high-level elements already stored in the list
81  for(auto& hlElem : highLevelElementsWithNewAccessor) {
82  hlElem->replaceTransferElement(replacement); // note: this does nothing, if the replacement cannot
83  // be used by the hlElem!
84  }
85  }
86  }
87 
88  // store the accessor in the list of high-level elements
89  // this must be done only now, since it may have been replaced during the
90  // replacement process
91  highLevelElements.insert(accessor.getHighLevelImplElement());
92 
93  // update the list of hardware-accessing elements, since we might just have
94  // made some of them redundant since we are using a set to store the elements,
95  // duplicates are intrinsically avoided.
96  lowLevelElements.clear();
97  for(auto& hlElem : highLevelElements) {
98  for(auto& hwElem : hlElem->getHardwareAccessingElements()) lowLevelElements.insert(hwElem);
99  }
100 
101  // update the list of CopyRegisterDecorators
102  copyDecorators.clear();
103  for(auto& hlElem : highLevelElements) {
104  if(boost::dynamic_pointer_cast<ChimeraTK::CopyRegisterDecoratorTrait>(hlElem) != nullptr) {
105  copyDecorators.insert(hlElem);
106  }
107  for(auto& hwElem : hlElem->getInternalElements()) {
108  if(boost::dynamic_pointer_cast<ChimeraTK::CopyRegisterDecoratorTrait>(hwElem) != nullptr) {
109  copyDecorators.insert(hwElem);
110  }
111  }
112  }
113 
114  // Update read-only flag
115  if(accessor.isReadOnly()) readOnly = true;
116  }
