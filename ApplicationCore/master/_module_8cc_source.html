63  void Module::readAll() {
64  auto recursiveAccessorList = getAccessorListRecursive();
65  // first blockingly read all push-type variables
66  for(auto accessor : recursiveAccessorList) {
67  if(accessor.getDirection() == VariableDirection{VariableDirection::feeding, false}) continue;
68  if(accessor.getMode() != UpdateMode::push) continue;
69  accessor.getAppAccessorNoType().read();
70  }
71  // next non-blockingly read the latest values of all poll-type variables
72  for(auto accessor : recursiveAccessorList) {
73  if(accessor.getDirection() == VariableDirection{VariableDirection::feeding, false}) continue;
74  if(accessor.getMode() == UpdateMode::push) continue;
75  accessor.getAppAccessorNoType().readLatest();
76  }
77  }

129  Module& Module::submodule(const std::string& moduleName) const {
130  size_t slash = moduleName.find_first_of("/");
131  // no slash found: call subscript operator
132  if(slash == std::string::npos) return (*this)[moduleName];
133  // slash found: split module name at slash
134  std::string upperModuleName = moduleName.substr(0, slash);
135  std::string remainingModuleNames = moduleName.substr(slash + 1);
136  return (*this)[upperModuleName].submodule(remainingModuleNames);
137  }
