102  /* For writable accessors, copy data to the recoveryAcessor before perfroming the write.
103  * Otherwise, the decorated accessor may have swapped the data out of the user buffer already.
104  * This obtains a shared lock from the DeviceModule, hence, the regular writing happeniin here
105  * can be performed in shared mode of the mutex and accessors are not blocking each other.
106  * In case of recovery, the DeviceModule thread will take an exclusive lock so that this thread can not
107  * modify the recoveryAcessor's user buffer while data is written to the device.
108  */
109  {
110  auto lock{deviceModule.getRecoverySharedLock()};
111 
112  if(_recoveryAccessor != nullptr) {
113  // Access to _recoveryAccessor is only possible channel-wise
114  for(unsigned int ch = 0; ch < _recoveryAccessor->getNumberOfChannels(); ++ch) {
115  _recoveryAccessor->accessChannel(ch) = buffer_2D[ch];
116  }
117  // FIXME: This should be the version number of the owner, which is not set at the moment
118  _recoveryHelper->versionNumber = {};
119  }
120  else {
121  throw ChimeraTK::logic_error(
122  "ChimeraTK::ExceptionhandlingDecorator: Calling write() on a non-writeable accessor is not supported ");
123  }
124  } // lock guard goes out of scope

167  void ExceptionHandlingDecorator<UserType>::doPostRead(TransferType type, bool hasNewData) {
168  bool hasException = false;
169  try {
170  // preRead has not been called when the transfer was not allowed. Don't call postRead in this case.
171  if(transferAllowed) {
172  this->_target->postRead(type, hasNewData);
173  }
174  }
175  catch(ChimeraTK::runtime_error& e) {
176  deviceModule.reportException(e.what());
177  hasException = true;
178  // #138 Phase 1: Remove for phase 2
179  // Inform the owner about the failed read (will be informed again after successful recovery
180  setOwnerValidity(hasException);
181  }
182  // #138 Phase 2: change if codition here
183  if(hasException || !transferAllowed) {
184  // Try to recover and read until it succeeds.
185  // We are already behind the delegated postRead, so the transfer in the target is already complemted.
186  // So we have to use a complete blocking preRead, readTransfer, postRead, i.e. _tagret->read()
187  while(true) {
188  deviceModule.waitForRecovery();
189  /* //#138 Phase 2
190  * if (!deviceModule->startTransfer()) continue;
191  * // end of #138 Phase 2
192  */
193  try {
194  this->_target->read();
195  hasException = false;
196  hasNewData = true; // if read() returns there is always new data
197  /* //#138 Phase 2
198  * deviceModule->stopTransfer());
199  * // end of #138 Phase 2
200  */
201  break;
202  }
203  catch(ChimeraTK::runtime_error&) {
204  /* //#138 Phase 2
205  * deviceModule->stopTransfer());
206  * // end of #138 Phase 2
207  */
208  }
209  }
210  }
211  setOwnerValidity(hasException);
212  // only replace the user buffer if there really is new data
213  if(hasNewData) {
214  for(size_t i = 0; i < buffer_2D.size(); ++i)
215  buffer_2D[i].swap(this->_target->accessChannel(static_cast<unsigned int>(i)));
216  }
