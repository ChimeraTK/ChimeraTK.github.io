138  class NumericAddressedBackendRegisterAccessor : public SyncNDRegisterAccessor<UserType> {
139  public:
140  NumericAddressedBackendRegisterAccessor(boost::shared_ptr<DeviceBackend> dev, const RegisterPath& registerPathName,
141  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
142  : SyncNDRegisterAccessor<UserType>(registerPathName), _dataConverter(registerPathName),
143  _registerPathName(registerPathName), _numberOfWords(numberOfWords),
144  _prePostActionsImplementor(NDRegisterAccessor<UserType>::buffer_2D, _rawAccessor, _startAddress, _dataConverter, _isNotWriteable) {
145  try {
146  // check for unknown flags
147  flags.checkForUnknownFlags({AccessMode::raw});
148 
149  // check device backend
150  _dev = boost::dynamic_pointer_cast<NumericAddressedBackend>(dev);
151  if(!_dev) {
152  throw ChimeraTK::logic_error("NumericAddressedBackendRegisterAccessor "
153  "is used with a backend which is not "
154  "a NumericAddressedBackend.");
155  }
156 
157  // obtain register information
158  boost::shared_ptr<RegisterInfo> info = _dev->getRegisterInfo(registerPathName);
159  _registerInfo = boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(info);
160  _bar = _registerInfo->bar;
161  _startAddress = _registerInfo->address + wordOffsetInRegister * sizeof(int32_t);
162 
163  // check number of words
164  if(_numberOfWords == 0) {
165  _numberOfWords = _registerInfo->getNumberOfElements() - wordOffsetInRegister;
166  }
167  if(_numberOfWords + wordOffsetInRegister > _registerInfo->getNumberOfElements()) {
168  throw ChimeraTK::logic_error(
169  "Requested number of words exceeds the size of the register '" + _registerPathName + "'!");
170  }
171  if(wordOffsetInRegister >= _registerInfo->getNumberOfElements()) {
172  throw ChimeraTK::logic_error("Requested offset exceeds the size of the register'" + _registerPathName + "'!");
173  }
174 
175  // Cache writeability
176  _isNotWriteable = (_registerInfo->registerAccess & RegisterInfoMap::RegisterInfo::Access::WRITE) == 0;
177 
178  // create low-level transfer element handling the actual data transfer to
179  // the hardware with raw data
180  _rawAccessor.reset(new NumericAddressedLowLevelTransferElement(_dev, _bar, _startAddress, _numberOfWords));
181 
182  // allocated the buffers
183  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
184  NDRegisterAccessor<UserType>::buffer_2D[0].resize(_numberOfWords);
185 
186  // We don't have to fill it in a special way if the accessor is raw
187  // because we have an overloaded, more efficient implementation
188  // in this case. So we can use it in setAsCooked() and getAsCooked()
189  _dataConverter = detail::createDataConverter<DataConverterType>(_registerInfo);
190 
191  if(flags.has(AccessMode::raw)) {
192  if(typeid(UserType) != typeid(int32_t)) {
193  throw ChimeraTK::logic_error("Given UserType when obtaining the "
194  "NumericAddressedBackendRegisterAccessor in raw mode does not "
195  "match the expected type. Use an int32_t instead! (Register "
196  "name: " +
197  _registerPathName + "')");
198  }
199  // FIXME: this has to move to the creation
200  //isRaw = true;
201  }
202  }
203  catch(...) {
204  this->shutdown();
205  throw;
206  }
207 
208  FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(getAsCooked_impl);
209  FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(setAsCooked_impl);
210  }

328  void replaceTransferElement(boost::shared_ptr<TransferElement> newElement) override {
329  auto casted = boost::dynamic_pointer_cast<NumericAddressedLowLevelTransferElement>(newElement);
330  if(casted && casted->isMergeable(_rawAccessor)) {
331  size_t newStartAddress = std::min(casted->_startAddress, _rawAccessor->_startAddress);
332  size_t newStopAddress = std::max(
333  casted->_startAddress + casted->_numberOfBytes, _rawAccessor->_startAddress + _rawAccessor->_numberOfBytes);
334  size_t newNumberOfWords = (newStopAddress - newStartAddress) / sizeof(int32_t);
335  casted->changeAddress(newStartAddress, newNumberOfWords);
336  _rawAccessor = casted;
337  }
