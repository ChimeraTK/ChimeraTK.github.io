168  std::list< boost::function<void(void)> > DummyBackend::findCallbackFunctionsForAddressRange(
169  AddressRange addressRange){
170  // as callback functions are not sortable, we want to loop the multimap only once.
171  // FIXME: If the same function is registered more than one, it may be executed multiple times
172 
173  // we only want the start address of the range, so we set size to 0
174  AddressRange firstAddressInBar(addressRange.bar, 0, 0);
175  AddressRange endAddress( addressRange.bar, addressRange.offset + addressRange.sizeInBytes, 0 );

178  _writeCallbackFunctions.lower_bound( firstAddressInBar );

194  bool DummyBackend::isWriteRangeOverlap( AddressRange firstRange, AddressRange secondRange){
195  if (firstRange.bar != secondRange.bar){
196  return false;
197  }
198 
199  uint32_t startAddress = std::max( firstRange.offset, secondRange.offset );
200  uint32_t endAddress = std::min( firstRange.offset + firstRange.sizeInBytes,
201  secondRange.offset + secondRange.sizeInBytes );
202 
203  // if at least one register is writable there is an overlap of writable registers
204  for ( uint32_t address = startAddress; address < endAddress; address += sizeof(int32_t) ){
205  if ( isReadOnly(firstRange.bar, address)==false ){
206  return true;
207  }
208  }
209 
210  // we looped all possible registers, none is writable
211  return false;
212  }
