120  void ad16::setSamplingRate(float rate, oversampling oversampling) {
121 
122  // check if opened
123  if(!_isOpen) throw ad16Exception("Device not opened.",ad16Exception::NOT_OPENED);
124 
125  // check if rate exceeds specification
126  if(rate > 100000.) throw ad16Exception("Sampling rate set too high.",ad16Exception::IMPOSSIBLE_TIMING_CONFIGURATION);
127 
128  // compute timing dividers and actual adc frequencies
129  int tdiv = (regClockFrequency[0]/rate)-1.;
130  _samplingFrequency = regClockFrequency[0]/(tdiv+1);
131 
132  // select correct read mode
133  int32_t readMode = 1;
134  double tcycle = 1./_samplingFrequency * 1.e6;
135  double tread = 74./regClockFrequency[1] * 1.e6;
136  double tconv = _conversionTimes[oversampling];
137  double tdelay = 0;
138  if(tcycle <= tdelay+tconv || tconv <= tread) {
139  readMode = 0;
140  if(tcycle <= tdelay+tconv+tread) {
141  throw ad16Exception("Impossible setting of sampling rate and/or oversampling ratio.",ad16Exception::IMPOSSIBLE_TIMING_CONFIGURATION);
142  }
143  }
144 
145  // set ADCA rate
146  regReadModeA->write(readMode);
147  regTimingDivA->write(tdiv);
148  regOversamplingA->write( static_cast<int>(oversampling) );
149 
150  // set ADCB rate
151  regReadModeB->write(readMode);
152  regTimingDivB->write(tdiv);
153  regOversamplingB->write( static_cast<int>(oversampling) );
154 
155 
156  }

175  void ad16::setTriggerMode(trigger trigger, boost::any arg) {
176 
177  // check if opened
178  if(!_isOpen) throw ad16Exception("Device not opened.",ad16Exception::NOT_OPENED);
179 
180  if(trigger == USER) {
181 
182  // disable internal trigger 8 (counting from 0) without changing the others
183  int bitmask = regTimingInternalEnable->read<int>();
184  bitmask = bitmask & 0xFF;
185  regTimingInternalEnable->write(bitmask);
186 
187  // select trigger for starting a new block (trigger 8, which is the user trigger)
188  regTimingTriggerSelect->write(8);
189 
190  }
191  else if(trigger == EXTERNAL) {
192 
193  // get channel number from 2nd argument
194  int channel;
195  try {
196  channel = boost::any_cast<int>(arg);
197  }
198  catch(boost::bad_any_cast &e) {
199  throw ad16Exception("ad16::setTriggerType(): second argument missing or of wrong type for EXTERNAL trigger mode.",ad16Exception::ILLEGAL_PARAMETER);
200  }
201 
202  // check if channel in range
203  if(channel < 0 || channel > 7) {
204  throw ad16Exception("Trigger channel out of range",ad16Exception::INCORRECT_TRIGGER_SETTING);
205  }
206 
207  // disable internal trigger on selected channel number (and on channel 8)
208  int bitmask = regTimingInternalEnable->read<int>();
209  bitmask = bitmask & ( 0xFF ^ (1 << channel) );
210  regTimingInternalEnable->write(bitmask);
211 
212  // select trigger for starting a new block (trigger 8, which is the user trigger)
213  regTimingTriggerSelect->write(channel);
214 
215  }
216  else if(trigger == PERIODIC) {
217 
218  // get frequency from 2nd argument.
219  // We want to have some type tolerance, so we try to cast to double, float and int.
220  float freq;
221  try {
222  freq = boost::any_cast<double>(arg);
223  }
224  catch(boost::bad_any_cast &e) {
225  try {
226  freq = boost::any_cast<float>(arg);
227  }
228  catch(boost::bad_any_cast &e) {
229  try {
230  freq = boost::any_cast<int>(arg);
231  }
232  catch(boost::bad_any_cast &e) {
233  throw ad16Exception("ad16::setTriggerType(): second argument missing or of wrong type for PERIODIC trigger mode.",ad16Exception::ILLEGAL_PARAMETER);
234  }
235  }
236  }
237 
238  // compute frequency divider
239  int div = round(regClockFrequency[0]/freq) - 1;
240 
241  // update frequency divider for trigger 0
242  regTimingFrequency.read();
243  regTimingFrequency[0] = div;
244  regTimingFrequency.write();
245 
246  // enable internal trigger 0
247  int bitmask = regTimingInternalEnable->read<int>();
248  bitmask = bitmask | 1;
249  regTimingInternalEnable->write(bitmask);
250 
251  // select trigger 0

254  }
