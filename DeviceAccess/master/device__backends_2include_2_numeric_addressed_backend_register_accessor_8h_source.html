39  template<typename UserType, typename DataConverterType>
40  class NumericAddressedBackendRegisterAccessor : public SyncNDRegisterAccessor<UserType> {
41  public:
42  NumericAddressedBackendRegisterAccessor(boost::shared_ptr<DeviceBackend> dev,
43  const RegisterPath& registerPathName,
44  size_t numberOfWords,
45  size_t wordOffsetInRegister,
46  AccessModeFlags flags)
47  : SyncNDRegisterAccessor<UserType>(registerPathName), _dataConverter(registerPathName), isRaw(false),
48  _registerPathName(registerPathName), _numberOfWords(numberOfWords) {
49  try {
50  // check for unknown flags
51  flags.checkForUnknownFlags({AccessMode::raw});
52 
53  // check device backend
54  _dev = boost::dynamic_pointer_cast<NumericAddressedBackend>(dev);
55  if(!_dev) {
56  throw ChimeraTK::logic_error("NumericAddressedBackendRegisterAccessor "
57  "is used with a backend which is not "
58  "a NumericAddressedBackend.");
59  }
60 
61  // obtain register information
62  boost::shared_ptr<RegisterInfo> info = _dev->getRegisterInfo(registerPathName);
63  _registerInfo = boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(info);
64  _bar = _registerInfo->bar;
65  _startAddress = _registerInfo->address + wordOffsetInRegister * sizeof(int32_t);
66 
67  // check number of words
68  if(_numberOfWords == 0) {
69  _numberOfWords = _registerInfo->getNumberOfElements() - wordOffsetInRegister;
70  }
71  if(_numberOfWords + wordOffsetInRegister > _registerInfo->getNumberOfElements()) {
72  throw ChimeraTK::logic_error(
73  "Requested number of words exceeds the size of the register '" + _registerPathName + "'!");
74  }
75  if(wordOffsetInRegister >= _registerInfo->getNumberOfElements()) {
76  throw ChimeraTK::logic_error("Requested offset exceeds the size of the register'" + _registerPathName + "'!");
77  }
78 
79  // create low-level transfer element handling the actual data transfer to
80  // the hardware with raw data
81  _rawAccessor.reset(new NumericAddressedLowLevelTransferElement(_dev, _bar, _startAddress, _numberOfWords));
82 
83  // allocated the buffers
84  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
85  NDRegisterAccessor<UserType>::buffer_2D[0].resize(_numberOfWords);
86 
87  // We don't have to fill it in a special way if the accessor is raw
88  // because we have an overloaded, more efficient implementation
89  // in this case. So we can use it in setAsCooked() and getAsCooked()
90  _dataConverter = detail::createDataConverter<DataConverterType>(_registerInfo);
91 
92  if(flags.has(AccessMode::raw)) {
93  if(typeid(UserType) != typeid(int32_t)) {
94  throw ChimeraTK::logic_error("Given UserType when obtaining the "
95  "NumericAddressedBackendRegisterAccessor in raw mode does not "
96  "match the expected type. Use an int32_t instead! (Register "
97  "name: " +
98  _registerPathName + "')");
99  }
100  isRaw = true;
101  }
102  }
103  catch(...) {
104  this->shutdown();
105  throw;
106  }
107 
108  FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(getAsCooked_impl);
109  FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(setAsCooked_impl);
110  }

160  bool mayReplaceOther(const boost::shared_ptr<TransferElement const>& other) const override {
161  auto rhsCasted =
162  boost::dynamic_pointer_cast<const NumericAddressedBackendRegisterAccessor<UserType, DataConverterType>>(
163  other);
164  if(!rhsCasted) return false;

166  if(_bar != rhsCasted->_bar) return false;
167  if(_startAddress != rhsCasted->_startAddress) return false;
168  if(_numberOfWords != rhsCasted->_numberOfWords) return false;
169  if(isRaw != rhsCasted->isRaw) return false;
170  if(_dataConverter != rhsCasted->_dataConverter) return false;
171  return true;

239  void replaceTransferElement(boost::shared_ptr<TransferElement> newElement) override {
240  auto casted = boost::dynamic_pointer_cast<NumericAddressedLowLevelTransferElement>(newElement);
241  if(casted && casted->isMergeable(_rawAccessor)) {
242  size_t newStartAddress = std::min(casted->_startAddress, _rawAccessor->_startAddress);
243  size_t newStopAddress = std::max(
244  casted->_startAddress + casted->_numberOfBytes, _rawAccessor->_startAddress + _rawAccessor->_numberOfBytes);
245  size_t newNumberOfWords = (newStopAddress - newStartAddress) / sizeof(int32_t);
246  casted->changeAddress(newStartAddress, newNumberOfWords);
247  _rawAccessor = casted;
248  }
