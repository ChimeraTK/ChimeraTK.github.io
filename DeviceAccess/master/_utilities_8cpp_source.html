39  Sdm Utilities::parseSdm(std::string sdmString) {
40  Sdm sdmInfo;
41  size_t signatureLen = 6;
42  if(sdmString.empty()) throw ChimeraTK::logic_error("Invalid sdm.");
43  if(sdmString.length() < signatureLen) throw ChimeraTK::logic_error("Invalid sdm.");
44  if(sdmString.substr(0,6) !="sdm://") throw ChimeraTK::logic_error("Invalid sdm.");
45  int pos = 6;
46 
47  std::size_t found = sdmString.find_first_of("/", pos);
48  std::string subUri;
49  if (found != std::string::npos)
50  {
51  sdmInfo._Host = sdmString.substr(pos, found - pos); // Get the Host
52  }
53  else
54  {
55  throw ChimeraTK::logic_error("Invalid sdm.");
56  }
57  if (sdmString.length() < found+1)
58  return sdmInfo;
59  subUri = sdmString.substr(found+1);
60  //let's do a sanity check, only one delimiter occurrence is allowed.
61  if (countOccurence(subUri,':') > 1) /* check ':' */
62  {
63  throw ChimeraTK::logic_error("Invalid sdm.");
64  }
65  if (countOccurence(subUri,';') > 1) /* check ';' */
66  {
67  throw ChimeraTK::logic_error("Invalid sdm.");
68  }
69  if (countOccurence(subUri,'=') > 1) /* check '=' */
70  {
71  throw ChimeraTK::logic_error("Invalid sdm.");
72  }
73  std::vector<std::string> tokens;
74  boost::split(tokens, subUri, boost::is_any_of(":;="));
75  int numOfTokens = tokens.size();
76  if (numOfTokens < 1)
77  return sdmInfo;
78  int counter = 0;
79  sdmInfo._Interface = tokens[counter]; // Get the Interface
80  counter++;
81  if (counter < numOfTokens)
82  {
83  // Get the Instance
84  found = sdmString.find_first_of(":", pos);
85  if (found != std::string::npos) {
86  sdmInfo._Instance = tokens[counter];
87  counter++;
88  }
89  }
90  if (counter < numOfTokens)
91  {
92  // Get the Protocol
93  found = sdmString.find_first_of(";", pos);
94  if (found != std::string::npos) {
95  sdmInfo._Protocol = tokens[counter];
96  counter++;
97  }
98  }
99  if (counter < numOfTokens)
100  {
101  // Get the Parameters
102  found = sdmString.find_first_of("=", pos);
103  if (found != std::string::npos) {
104  std::string parameters = tokens[counter];
105  std::vector<std::string> paramterTokens;
106  boost::split(paramterTokens, parameters, boost::is_any_of(","));
107  for (uint i=0; i < paramterTokens.size();i++ )
108  sdmInfo._Parameters.push_back(paramterTokens[i]);
109  }
110  }
111 
112  return sdmInfo;
113 
114  }
