81  void DoocsSpectrum::write(fstream& fptr) {
82  // DOOCS is normally keeping the location lock until everything is written for that location: all D_spectrum and all
83  // other properties. This can take too long (like seconds), which leads to noticable freezes of the UI. As a
84  // work-around we release the lock here, wait some time and acquire the lock again. Since this happens in a separate
85  // thread (svr_writer), this slow-down should not be of any harm.
86  efp_->unlock();
87  usleep(1000);
88  efp_->lock();
89  if(!modified) return;
90  D_spectrum::write(fptr);
91  }

93  void DoocsSpectrum::updateDoocsBuffer(TransferElementID transferElementId) {
94  // Note: we already own the location lock by specification of the DoocsUpdater
95 
96  // FIXME: A first implementation is checking the data consistency here. Later this should be
97  // before calling this function because calling this function through a function pointer is
98  // comparatively expensive.
99  // Only check the consistency group if there is a macro pulse number associated.
100  // There are only the processArray and the macro pulse number in the consistency
101  // group. The limits are coming asynchronously and not for every macro pulse,
102  // so we just take test latest we have.
103  if(_macroPulseNumberSource && !_consistencyGroup.update(transferElementId)) {
104  return;
105  }
106 
107  // determine time stamp
108  auto sinceEpoch = _processArray->getVersionNumber().getTime().time_since_epoch();
109  auto time = std::chrono::duration_cast<std::chrono::microseconds>(sinceEpoch);
110  auto sec = time.count() / 1000000;
111  auto usec = time.count() % 1000000;
112 
113  // set macro pulse number, buffer number and time stamp
114  size_t ibuf = 0;
115  if(_macroPulseNumberSource != nullptr) {
116  ibuf = _macroPulseNumberSource->accessData(0) % nBuffers;
117  macro_pulse(_macroPulseNumberSource->accessData(0), ibuf);
118  }
119  set_tmstmp(sec, usec, ibuf);
120 
121  if(_processArray->dataValidity() != ChimeraTK::DataValidity::ok) {
122  this->d_error(stale_data);
123  }
124  else {
125  this->d_error(no_error);
126  }
127 
128  // fill the spectrum
129  std::vector<float>& processVector = _processArray->accessChannel(0);
130  if(nBuffers == 1) {
131  // We have to fill the spectrum differently if it is unbuffered, as the internal data structures seem to be
132  // completely different.
133  memcpy(spectrum()->d_spect_array.d_spect_array_val, processVector.data(), processVector.size() * sizeof(float));
134  spectrum()->d_spect_array.d_spect_array_len = processVector.size();
135  }
136  else {
137  fill_spectrum(processVector.data(), processVector.size(), ibuf);
138  }
139  if(publishZMQ) {
140  dmsg_info info;
141  memset(&info, 0, sizeof(info));
142  info.sec = sec;
143  info.usec = usec;
144  if(_macroPulseNumberSource != nullptr) {
145  info.ident = _macroPulseNumberSource->accessData(0);
146  }
147  else {
148  info.ident = 0;
149  }
150  this->send(&info);
151  }
152  }
