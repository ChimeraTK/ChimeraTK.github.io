220  void MotorControlerDummy::moveTowardsTarget(float fraction, bool block, bool bothEndSwitchesAlwaysOn, bool zeroPositions) {
221 
222  if (zeroPositions) {
223  _absolutePosition = 0;
224  _targetPosition = 0;
225  _currentPosition = 0;
226  _positiveEndSwitchEnabled = true;
227  _negativeEndSwitchEnabled = true;
228  }
229 
230  _bothEndSwitchesAlwaysOn = bothEndSwitchesAlwaysOn;
231 
232  _blockMotor = block;
233  if (_blockMotor) {
234  return;
235  }
236 
237  // check if the motor is should to be moving if it was enabled
238  // otherwise return immediately
239  if (!isStepping()) {
240  return;
241  }
242 
243  // range check for the input parameter 'fraction', just force it to 0..1
244  if (fraction <= 0) {
245  return;
246  }
247  if (fraction > 1) {
248  fraction = 1.;
249  }
250 
251  // calculate the new abolute target position for this move operation
252  int relativeSteps = (_targetPosition - _currentPosition) * fraction;
253  int absoluteTargetInThisMove = _absolutePosition + relativeSteps;
254  int targetInThisMove = _currentPosition + relativeSteps;
255 
256  if (!_enabled) {
257  // the motor is stepping to the target position, but the actual positon
258  // does not change. Thus no end switch will be hit.
259  _currentPosition = targetInThisMove;
260  return;
261  }
262 
263  // ok, so the motor is really moving.
264 
265  // check whether an end switch would be hit
266  if (_positiveEndSwitchEnabled &&
267  (absoluteTargetInThisMove > _positiveEndSwitchPosition)) {
268  absoluteTargetInThisMove = _positiveEndSwitchPosition;
269  // also the current positon stops if an end switch is reached
270  int stepsToEndSwitch = _positiveEndSwitchPosition - _absolutePosition;
271  targetInThisMove = _currentPosition + stepsToEndSwitch;
272  }
273  if (_negativeEndSwitchEnabled &&
274  (absoluteTargetInThisMove < _negativeEndSwitchPosition)) {
275  absoluteTargetInThisMove = _negativeEndSwitchPosition;
276  // also the current positon stops if an end switch is reached
277  int stepsToEndSwitch = _negativeEndSwitchPosition - _absolutePosition;
278  targetInThisMove = _currentPosition + stepsToEndSwitch;
279  }
280 
281  // finally 'move' the motor
282  _absolutePosition = absoluteTargetInThisMove;
283  _currentPosition = targetInThisMove;
284  }
