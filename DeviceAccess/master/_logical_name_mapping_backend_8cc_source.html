82  template<typename UserType>
83  boost::shared_ptr< NDRegisterAccessor<UserType> > LogicalNameMappingBackend::getRegisterAccessor_impl(
84  const RegisterPath &registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
85 
86  // obtain register info
87  auto info = boost::static_pointer_cast<LNMBackendRegisterInfo>(_catalogue_mutable.getRegister(registerPathName));
88 
89  // implementation for each type
90  boost::shared_ptr< NDRegisterAccessor<UserType> > ptr;
91  if( info->targetType == LNMBackendRegisterInfo::TargetType::REGISTER) {
92  DeviceBackend *_targetDevice;
93  std::string devName = info->deviceName;
94  if(devName != "this") {
95  _targetDevice = _devices[info->deviceName].get();
96  }
97  else {
98  _targetDevice = this;
99  }
100  // make sure the target device exists
101  if(_targetDevice == nullptr) {
102  throw ChimeraTK::logic_error("Target device for this logical register is not opened. See exception thrown in open()!");
103  }
104  // determine the offset and length
105  size_t actualOffset = size_t(info->firstIndex) + wordOffsetInRegister;
106  size_t actualLength = ( numberOfWords > 0 ? numberOfWords : size_t(info->length) );
107  // obtain underlying register accessor
108  ptr = _targetDevice->getRegisterAccessor<UserType>(RegisterPath(info->registerName),actualLength,actualOffset,flags);
109  }
110  else if( info->targetType == LNMBackendRegisterInfo::TargetType::CHANNEL) {
111  ptr = boost::shared_ptr< NDRegisterAccessor<UserType> >(new LNMBackendChannelAccessor<UserType>(shared_from_this(),
112  registerPathName, numberOfWords, wordOffsetInRegister, flags));
113  }
114  else if( info->targetType == LNMBackendRegisterInfo::TargetType::BIT) {
115  ptr = boost::shared_ptr< NDRegisterAccessor<UserType> >(new LNMBackendBitAccessor<UserType>(shared_from_this(),
116  registerPathName, numberOfWords, wordOffsetInRegister, flags));
117  }
118  else if( info->targetType == LNMBackendRegisterInfo::TargetType::INT_CONSTANT ||
119  info->targetType == LNMBackendRegisterInfo::TargetType::INT_VARIABLE ) {
120  ptr = boost::shared_ptr< NDRegisterAccessor<UserType> >(new LNMBackendVariableAccessor<UserType>(shared_from_this(),
121  registerPathName, numberOfWords, wordOffsetInRegister, flags));
122  }
123  else {
124  throw ChimeraTK::logic_error("For this register type, a RegisterAccessor cannot be obtained (name of logical register: "+
125  registerPathName+").");
126  }
127 
128  return ptr;
129  }

133  const RegisterCatalogue& LogicalNameMappingBackend::getRegisterCatalogue() const {
134 
135  if(catalogueCompleted) return _catalogue_mutable;
136 
137  // fill in information to the catalogue from the target devices
138  for(auto &info : _catalogue_mutable) {
139  LNMBackendRegisterInfo &info_cast = static_cast<LNMBackendRegisterInfo&>(info);
140  auto targetType = info_cast.targetType;
141  if(targetType != LNMBackendRegisterInfo::TargetType::REGISTER &&
142  targetType != LNMBackendRegisterInfo::TargetType::CHANNEL ) continue;
143 
144  std::string devName = info_cast.deviceName;
145  boost::shared_ptr<RegisterInfo> target_info;
146  if(devName != "this") {
147  target_info = _devices.at(devName)->getRegisterCatalogue().getRegister(std::string(info_cast.registerName));
148  }
149  else {
150  target_info = getRegisterCatalogue().getRegister(std::string(info_cast.registerName));
151  }
152 
153  info_cast._dataDescriptor = target_info->getDataDescriptor();
154 
155  if(targetType == LNMBackendRegisterInfo::TargetType::REGISTER) {
156  info_cast.nDimensions = target_info->getNumberOfDimensions();
157  info_cast.nChannels = target_info->getNumberOfChannels();
158  }
159  if((int)info_cast.length == 0) info_cast.length = target_info->getNumberOfElements();
160  }
161 
162  catalogueCompleted = true;
163  return _catalogue_mutable;
164  }
