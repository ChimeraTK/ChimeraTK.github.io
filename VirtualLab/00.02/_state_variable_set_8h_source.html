118  inline const STATE& getState(VirtualTime time) {
119  // check if time is the current time and return the latest element
120  if(time >= currentTime && time < currentTime + validityPeriod) return getLatestState();
121  // search in buffer: find the first element after the requested time
122  auto it = buffer.upper_bound(time);
123  if(it == buffer.begin()) {
124  std::stringstream s;
125  s << "Value request is too far into the past: ";
126  s << "requested time = " << time << ", oldest history = " << buffer.begin()->first;
127  s << ", current time = " << currentTime;
128  throw StateVariableSetException(s.str(),StateVariableSetException::REQUEST_FAR_PAST);
129  }
130  // if this is end(), a new value needs to be computed
131  if(it == buffer.end()) return getValueFromCallback(time);
132  // decrement to get the most recent sample before the requested time
133  --it;
134  // if sample is too old, request one via callback
135  if(time >= it->first + validityPeriod) return getValueFromCallback(time);
136  // return value from buffer
137  return it->second;
138  }

204  inline const STATE& getValueFromCallback(VirtualTime time) {
205 
206  // If request goes into the past (w.r.t. currentTime), erase newer states to force recomputing them.
207  truncateFuture(time);
208 
209  // Compute the number of steps necessary to reach the requested time, obaying hugeGap. The last step at the
210  // requested time itself is not computed here, since maxGap shall be obeyed in the last gap (see below).
211  // Note: this is effectively rounding-up the integer division (time-CurrentTime)/hugeGap
212  unsigned int nHugeSteps = (time-currentTime-1)/hugeGap + 1;
213 
214  // Loop over the time in nHugeSteps steps. The first step is potentially smaller, all consequtive steps are
215  // maxGap big. Omit the last step, so the requested time is not yet reached.
216  for(unsigned int i=0; i<nHugeSteps-1; i++) {
217  VirtualTime t = time-(nHugeSteps-1)*hugeGap + i*hugeGap;
218 
219  // obtain the new state
220  STATE state = compute(t);
221 
222  // save value into buffer and update the current time
223  buffer[t] = state;
224  currentTime = t;
225 
226  }
227 
228  // Compute the number of steps necessary to reach the requested time, obaying maxGap.
229  // Note: this is effectively rounding-up the integer division (time-CurrentTime)/maxGap
230  unsigned int nSteps = (time-currentTime-1)/maxGap + 1;
231 
232  // Loop over the time in nSteps steps. The first step is potentially smaller, all consequtive steps are
233  // maxGap big.
234  for(unsigned int i=0; i<nSteps; i++) {
235  VirtualTime t = time-(nSteps-1)*maxGap + i*maxGap;
236 
237  // obtain the new state
238  STATE state = compute(t);
239 
240  // save value into buffer and update the current time
241  buffer[t] = state;
242  currentTime = t;
243 
244  }
245 
246  // remove old history
247  truncatePast();
248 
249  // return the new state (do not use buffer[], as it will silently insert new objects if something went wrong)
250  return buffer.find(time)->second;
251  }
