476  ProcessArray<T>::ProcessArray(InstanceType instanceType, const mtca4u::RegisterPath& name, const std::string &unit,
477  const std::string &description, const std::vector<T>& initialValue, std::size_t numberOfBuffers)
478  : mtca4u::NDRegisterAccessor<T>(name, unit, description),
479  _instanceType(instanceType),
480  _vectorSize(initialValue.size()),
481  _maySendDestructively(false),
482  _sharedState(boost::make_shared<SharedState>(numberOfBuffers)),
483  _currentIndex(&(_sharedState->_buffers[0])),
484  _tripleBufferIndex(&(_sharedState->_buffers[4]))
485  {
486  try {
487  // allocate and initialise buffer of the base class
488  mtca4u::NDRegisterAccessor<T>::buffer_2D.resize(1);
489  mtca4u::NDRegisterAccessor<T>::buffer_2D[0] = initialValue;
490  // It would be better to do the validation before initializing, but this
491  // would mean that we would have to initialize twice.
492  if (instanceType != RECEIVER) {
493  throw std::invalid_argument("This constructor may only be used for a receiver process variable.");
494  }
495  // We need at least two buffers for the queue (so four buffers in total)
496  // in order to guarantee that we never have to block.
497  if (numberOfBuffers < 2) {
498  throw std::invalid_argument("The number of buffers must be at least two.");
499  }
500  // We have to limit the number of buffers because we cannot allocate
501  // more buffers than can be represented by size_t and the total number
502  // of buffers is the specified number plus two. These extra two buffers
503  // are needed because the sender and the receiver each hold one buffer
504  // at all times and thus only the rest remains for the queue. We use
505  // this definition of the number of buffers to keep it consistent with
506  // what ProcessScalarImpl uses.
507  if (numberOfBuffers > (std::numeric_limits<std::size_t>::max() - (2+3))) {
508  throw std::invalid_argument("The number of buffers is too large.");
509  }
510  // We have to initialize the buffers by copying in the initial vectors.
511  for (auto &i : _sharedState->_buffers) {
512  i.value = initialValue;
513  }
514  // The buffer with the index 0 is assigned to the receiver and the buffer with the index 1 is assigned to the
515  // sender. Indices 2-4 are assigned to the atomic triple buffer (for detailed assignment see definition of
516  // atomic triple buffer in the shared state). All other buffers have to be added to the empty-buffer queue.
517  for (std::size_t i = 2+3; i < _sharedState->_buffers.size(); ++i) {
518  _sharedState->_emptyBufferQueue.push(&(_sharedState->_buffers[i]));
519  }
520  }
521  catch(...) {
522  this->shutdown();
523  throw;
524  }
525  }

617  template<class T>
618  bool ProcessArray<T>::doReadTransferLatest() {
619  // As long as there is more than one valid element on the queue, discard it.
620  // Due to our implementation there is always one unfulfilled future in the queue, so
621  // we must pop until there are two elements left in order not to flush out the newest valid value.
622  auto theFuture = readAsync().getBoostFuture();
623  while( theFuture.wait_for(boost::chrono::duration<int, boost::centi>(0)) != boost::future_status::timeout
624  && _sharedState->_fullBufferQueue.read_available() > 2 ) {
625  // discard data by moving the buffer to the empty buffer queue
626  TransferFuture::Data *discardedBuffer = theFuture.get();
627  _sharedState->_fullBufferQueue.pop();
628  mtca4u::TransferElement::hasActiveFuture = false;
629  _sharedState->_emptyBufferQueue.push(static_cast<Buffer*>(discardedBuffer)); // static cast is ok, we never put something else into the queue
630  theFuture = readAsync().getBoostFuture();
631  }
632  // Check whether data is present in the atomic triple buffer. If it is newer, we also need to discard the last
633  // valid element in the queue. If it is older, we need to discard the data in the triple buffer.
634  // First update the triple buffer if it is not valid on our side
635  if(!(_tripleBufferIndex->_isValid)) {
636  _tripleBufferIndex = _sharedState->_tripleBufferIndex.exchange(_tripleBufferIndex);
637  }
638  // if it now contains a valid element, check if it is newer
639  if(_tripleBufferIndex->_isValid) {
640  uint64_t tripleBufferVersion = _tripleBufferIndex->_internalVersionNumber;
641  uint64_t queueVersion = static_cast<Buffer*>(theFuture.get())->_internalVersionNumber;
642  if(tripleBufferVersion > queueVersion) {
643  TransferFuture::Data *discardedBuffer = theFuture.get();
644  _sharedState->_fullBufferQueue.pop();
645  _sharedState->_emptyBufferQueue.push(static_cast<Buffer*>(discardedBuffer));
646  }
647  else {
648  _tripleBufferIndex->_isValid = false;
649  }
650  }
651  return this->doReadTransferNonBlocking();
652  }

656  template<class T>
657  mtca4u::TransferFuture& ProcessArray<T>::readAsync() {
658 
659  // If we already have a still-active future, we can return it directly. It is not possible that this future is
660  // blocking and there is valid data to be read in the atomic triple buffer.
661  // This is only a preformance optimisation, we would generate the same future again.
662  if(mtca4u::TransferElement::hasActiveFuture) return mtca4u::TransferElement::activeFuture;
663 
664  if (_instanceType != RECEIVER) {
665  throw std::logic_error("Receive operation is only allowed for a receiver process variable.");
666  }
667 
668  // Obtain future from queue but do not pop it yet, since we need to determine first whether to use it nor not.
669  // The future will only be popped from the queue in postRead(). This makes sure that subsequent calls to this
670  // function even if the future has not yet been used still result in the correct behaviour.
671  assert(_sharedState->_fullBufferQueue.read_available() > 0); // guaranteed by our logic
672  TransferFuture::PlainFutureType future = _sharedState->_fullBufferQueue.front();
673 
674  // if the future blocks, check for data in the tripple buffer
675  auto status = future.wait_for(boost::chrono::duration<int, boost::centi>(0));
676  if(status == boost::future_status::timeout) {
677  // if local buffer of triple buffer is invalid, swap with shared buffer first
678  if(!(_tripleBufferIndex->_isValid)) {
679  _tripleBufferIndex = _sharedState->_tripleBufferIndex.exchange(_tripleBufferIndex);
680  }
681  // if now the local buffer contains valid data, use that data first
682  if(_tripleBufferIndex->_isValid) {
683  future = boost::make_ready_future(static_cast<TransferFuture::Data*>(_tripleBufferIndex)).share();
684  }
685  }
686 
687  // return the future
688  mtca4u::TransferElement::activeFuture = mtca4u::TransferFuture(future, static_cast<mtca4u::TransferElement*>(this));
689  mtca4u::TransferElement::hasActiveFuture = true;
690  return mtca4u::TransferElement::activeFuture;
691 
692  }

696  template<class T>
697  void ProcessArray<T>::postRead() {
698  mtca4u::TransferElement::hasActiveFuture = false;
699 
700  // We have to check that the vector that we currently own still has the
701  // right size. Otherwise, the code using the sender might get into
702  // trouble when it suddenly experiences a vector of the wrong size.
703  if(_currentIndex->value.size() != _vectorSize) {
704  throw std::runtime_error("Cannot run read operation because the size of the vector belonging to the current"
705  " buffer has been modified.");
706  }
707 
708  // If local buffer of triple buffer is invalid, swap with shared buffer first
709  if(!(_tripleBufferIndex->_isValid)) {
710  _tripleBufferIndex = _sharedState->_tripleBufferIndex.exchange(_tripleBufferIndex);
711  }
712 
713  // Determine version number of data in the triple buffer
714  bool tripleBufferHasData = _tripleBufferIndex->_isValid;
715 
716  // Determine version number of data in the queue
717  assert(_sharedState->_fullBufferQueue.read_available() > 0); // guaranteed by our logic
718  TransferFuture::PlainFutureType future = _sharedState->_fullBufferQueue.front();
719  auto status = future.wait_for(boost::chrono::duration<int, boost::centi>(0));
720  bool queueHasData = (status != boost::future_status::timeout);
721 
722  // If postRead() has been called, there should be new data somewhere
724  //assert(queueHasData || tripleBufferHasData);
725  if(!(queueHasData || tripleBufferHasData)) {
726  std::cout << "ERROR: assert(queueHasData || tripleBufferHasData) failed! Waiting for new data..." << std::endl;
727  future.wait();
728  }
729 
730  // Determine whether to use the data from the queue or from the triple buffer. If in both places data is present,
731  // use the older data (by the version number).
732  bool useTripleBuffer;
733  if(queueHasData && !tripleBufferHasData) {
734  // only data in queue present
735  useTripleBuffer = false;
736  }
737  else if(!queueHasData && tripleBufferHasData) {
738  // only data in atomic triple buffer present
739  useTripleBuffer = true;
740  }
741  else { // queueHasData && tripleBufferHasData -> see the assert above
742  uint64_t tripleBufferVersion = _tripleBufferIndex->_internalVersionNumber;
743  uint64_t queueVersion = static_cast<Buffer*>(future.get())->_internalVersionNumber;
744  if(tripleBufferVersion < queueVersion) {
745  useTripleBuffer = true;
746  }
747  else {
748  useTripleBuffer = false;
749  }
750  }
751 
752  // perform the determined action
753  if(!useTripleBuffer) {
754  // remove future from queue, since we only obtained it through front() so far
755  _sharedState->_fullBufferQueue.pop();
756  // get buffer from future
757  TransferFuture::Data *nextBuffer = future.get();
758  // put old buffer back on empty buffer queue and update current index
759  _sharedState->_emptyBufferQueue.push(_currentIndex);
760  _currentIndex = static_cast<Buffer*>(nextBuffer); // static_cast should be ok, since we control the queue
761  }
762  else {
763  Buffer *nextBuffer = _tripleBufferIndex;
764  // put old buffer back into triple buffer (after invalidating int) and update current index
765  _currentIndex->_isValid = false;
766  _tripleBufferIndex = _currentIndex;
767  _currentIndex = nextBuffer;
768  }
769 
770  // swap data out of the queue buffer
771  mtca4u::NDRegisterAccessor<T>::buffer_2D[0].swap( _currentIndex->value );
772 
773  }

811  template<class T>
812  bool ProcessArray<T>::writeInternal(VersionNumber newVersionNumber, bool shouldCopy) {
813  if(_instanceType != SENDER) {
814  throw std::logic_error(
815  "Send operation is only allowed for a sender process variable. Variable name: "+this->getName());
816  }
817  // We have to check that the vector that we currently own still has the
818  // right size. Otherwise, the code using the receiver might get into
819  // trouble when it suddenly experiences a vector of the wrong size.
820  if(mtca4u::NDRegisterAccessor<T>::buffer_2D[0].size() != _vectorSize) {
821  throw std::runtime_error(
822  "Cannot run receive operation because the size of the vector belonging to the current buffer has been "
823  "modified. Variable name: "+this->getName());
824  }
825 
826  // First update the persistent data storage, if any was associated. This cannot be done after sending, since the
827  // value might no longer be available within this instance.
828  if(_persistentDataStorage) {
829  _persistentDataStorage->updateValue(_persistentDataStorageID, mtca4u::NDRegisterAccessor<T>::buffer_2D[0]);
830  }
831 
832  // Before sending the value, we have to set the associated time-stamp.
833  TimeStamp newTimeStamp = _timeStampSource ? _timeStampSource->getCurrentTimeStamp() : TimeStamp::currentTime();
834  _currentIndex->timeStamp = newTimeStamp;
835 
836  // set the version numbers
837  _currentIndex->_versionNumber = newVersionNumber;
838  _currentIndex->_internalVersionNumber = _lastInternalVersionNumber;
839  ++_lastInternalVersionNumber;
840 
841  // flag current data as valid
842  _currentIndex->_isValid = true;
843 
844  // set the data by copying or swapping
845  if(shouldCopy) {
846  _currentIndex->value = mtca4u::NDRegisterAccessor<T>::buffer_2D[0];
847  }
848  else {
849  _currentIndex->value.swap( mtca4u::NDRegisterAccessor<T>::buffer_2D[0] );
850  }
851  // send the data to the queue by first obtaining a new empty buffer from the empty buffer queue
852  Buffer *nextIndex;
853  bool lostData = false;
854  if(_sharedState->_emptyBufferQueue.pop(nextIndex)) {
855  // Create a new promise first and push its future to the full buffer queue before fulfilling the current promise.
856  // This makes sure that always at least one unfulfilled future is available for the receier to wait on.
857  TransferFuture::PromiseType nextPromise;
858  _sharedState->_fullBufferQueue.push(nextPromise.get_future());
859  _sharedState->_notificationPromise.set_value(_currentIndex);
860  _sharedState->_notificationPromise = std::move(nextPromise);
861  }
862  else {
863  // No empty buffer available: write to special atomic triple buffer
864  nextIndex = _tripleBufferIndex;
865  _tripleBufferIndex = _currentIndex;
866  _tripleBufferIndex = _sharedState->_tripleBufferIndex.exchange(_tripleBufferIndex);
867  // if the buffer swapped out of the shared state contained valid data, we have now lost that data
868  if(_tripleBufferIndex->_isValid) {
869  lostData = true;
870  // mark as invalid so we can always just swap it with the shared state
871  _tripleBufferIndex->_isValid = false;
872  }
873  }
874 
875  // change current index to the new empty buffer
876  if(shouldCopy) {
877  nextIndex->timeStamp = newTimeStamp;
878  nextIndex->_versionNumber = newVersionNumber;
879  }
880  _currentIndex = nextIndex;
881 
882  // notify send notification listener, if present
883  if(_sendNotificationListener) {
884  _sendNotificationListener->notify(_receiver);
885  }
886 
887  return lostData;
888  }
