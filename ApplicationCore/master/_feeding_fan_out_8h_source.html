99  void doPostRead() override {
100  if(!_withReturn) throw ChimeraTK::logic_error("Read operation called on write-only variable.");
101  assert(_hasReturnSlave);
102  _returnSlave->postRead();
103  _returnSlave->accessChannel(0).swap(ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0]);
104  // distribute return-channel update to the other slaves
105  for(auto &slave : FanOut<UserType>::slaves) { // send out copies to slaves
106  if(slave == _returnSlave) continue;

108  slave->accessChannel(0) = ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0];

120  void doPreWrite() override {

122  if(slave->getNumberOfSamples() != 0) { // do not send copy if no data is expected (e.g. trigger)
123  if(slave == FanOut<UserType>::slaves.front()) { // in case of first slave, swap instead of copy
124  slave->accessChannel(0).swap(ChimeraTK::NDRegisterAccessor<UserType>::buffer_2D[0]);
125  }
126  else { // not the first slave: copy the data from the first slave
127  slave->accessChannel(0) = FanOut<UserType>::slaves.front()->accessChannel(0);
128  }
129  }
130  }

132  // accessor might take us the data away...

134  slave->preWrite();
