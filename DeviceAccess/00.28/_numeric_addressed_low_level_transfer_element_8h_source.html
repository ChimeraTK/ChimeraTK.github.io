93  bool isSameRegister(const boost::shared_ptr<TransferElement const> &other) const override {
94  // accessor type, device and bar must be the same
95  auto rhsCasted = boost::dynamic_pointer_cast< const NumericAddressedLowLevelTransferElement >(other);
