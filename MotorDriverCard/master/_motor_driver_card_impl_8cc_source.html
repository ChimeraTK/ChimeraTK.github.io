162  void MotorDriverCardImpl::checkFirmwareVersion(){
163  ChimeraTK::ScalarRegisterAccessor<int32_t> firmwareVersion
164  = _device->getScalarRegisterAccessor<int32_t>(_moduleName + "/" + PROJECT_VERSION_ADDRESS_STRING);
165  firmwareVersion.read();
166 
167  // only allow firmware versions with the same major number
168  uint32_t maxFirmwareVersion = (MINIMAL_FIRMWARE_VERSION & 0xFF000000) | 0x00FFFFFF;
169 
170  // check the version
171  if ( (static_cast<uint32_t>(firmwareVersion) < static_cast<uint32_t>(MINIMAL_FIRMWARE_VERSION)) ||
172  (static_cast<uint32_t>(firmwareVersion) > maxFirmwareVersion) ){
173  std::stringstream errorMessage;
174  errorMessage << "MotorDriverCardImpl detected wrong firmware version. "
175  << "Current firmware is 0x" << std::hex << firmwareVersion
176  << ", minumum required version is 0x" << MINIMAL_FIRMWARE_VERSION
177  << ", maximum allowed version is 0x" << maxFirmwareVersion << std::dec << ".";
178  std::cerr << errorMessage.str() << std::endl;
179  throw MotorDriverException( errorMessage.str(), MotorDriverException::WRONG_FIRMWARE_VERSION );
180  }
