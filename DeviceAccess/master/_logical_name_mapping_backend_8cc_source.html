23  void LogicalNameMappingBackend::parse() {
24 
25  // don't run, if already parsed
26  if(hasParsed) return;
27 
28  // parse the map fle
29  LogicalNameMapParser parser = LogicalNameMapParser(_lmapFileName);
30  _catalogue = parser.getCatalogue();
31 
32  // create all devices referenced in the map
33  for(auto &devName : parser.getTargetDevices()) {
34  _devices[devName] = BackendFactory::getInstance().createBackend(devName);
35  }
36 
37  // fill in information to the catalogue from the target devices
38  for(auto &info : _catalogue) {
39  LNMBackendRegisterInfo &info_cast = static_cast<LNMBackendRegisterInfo&>(info);
40  auto targetType = info_cast.targetType;
41  if(targetType != LNMBackendRegisterInfo::TargetType::REGISTER &&
42  targetType != LNMBackendRegisterInfo::TargetType::CHANNEL ) continue;
43 
44  std::string devName = info_cast.deviceName;
45  boost::shared_ptr<RegisterInfo> target_info;
46  if(devName != "this") {
47  target_info = _devices[devName]->getRegisterCatalogue().getRegister(std::string(info_cast.registerName));
48  }
49  else {
50  target_info = getRegisterCatalogue().getRegister(std::string(info_cast.registerName));
51  }
52 
53  info_cast._dataDescriptor = target_info->getDataDescriptor();
54 
55  if(targetType == LNMBackendRegisterInfo::TargetType::REGISTER) {
56  info_cast.nDimensions = target_info->getNumberOfDimensions();
57  info_cast.nChannels = target_info->getNumberOfChannels();
58  }
59  if((int)info_cast.length == 0) info_cast.length = target_info->getNumberOfElements();
60  }
61  }

99  template<typename UserType>
100  boost::shared_ptr< NDRegisterAccessor<UserType> > LogicalNameMappingBackend::getRegisterAccessor_impl(
101  const RegisterPath &registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
102 
103  // obtain register info
104  auto info = boost::static_pointer_cast<LNMBackendRegisterInfo>(_catalogue.getRegister(registerPathName));
105 
106  // implementation for each type
107  boost::shared_ptr< NDRegisterAccessor<UserType> > ptr;
108  if( info->targetType == LNMBackendRegisterInfo::TargetType::REGISTER) {
109  DeviceBackend *_targetDevice;
110  std::string devName = info->deviceName;
111  if(devName != "this") {
112  _targetDevice = _devices[info->deviceName].get();
113  }
114  else {
115  _targetDevice = this;
116  }
117  // make sure the target device exists
118  if(_targetDevice == nullptr) {
119  throw DeviceException("Target device for this logical register is not opened. See exception thrown in open()!",
120  DeviceException::NOT_OPENED);
121  }
122  // determine the offset and length
123  size_t actualOffset = size_t(info->firstIndex) + wordOffsetInRegister;
124  size_t actualLength = ( numberOfWords > 0 ? numberOfWords : size_t(info->length) );
125  // obtain underlying register accessor
126  ptr = _targetDevice->getRegisterAccessor<UserType>(RegisterPath(info->registerName),actualLength,actualOffset,flags);
127  }
128  else if( info->targetType == LNMBackendRegisterInfo::TargetType::CHANNEL) {
129  ptr = boost::shared_ptr< NDRegisterAccessor<UserType> >(new LNMBackendChannelAccessor<UserType>(shared_from_this(),
130  registerPathName, numberOfWords, wordOffsetInRegister, flags));
131  }
132  else if( info->targetType == LNMBackendRegisterInfo::TargetType::INT_CONSTANT ||
133  info->targetType == LNMBackendRegisterInfo::TargetType::INT_VARIABLE ) {
134  ptr = boost::shared_ptr< NDRegisterAccessor<UserType> >(new LNMBackendVariableAccessor<UserType>(shared_from_this(),
135  registerPathName, numberOfWords, wordOffsetInRegister, flags));
136  }
137  else {
138  throw DeviceException("For this register type, a RegisterAccessor cannot be obtained (name of logical register: "+
139  registerPathName+").", DeviceException::NOT_IMPLEMENTED);
140  }
141 
142  // allow plugins to decorate the accessor and return it
143  return decorateRegisterAccessor<UserType>(registerPathName,ptr);
144  }
