600  void Device::write(const RegisterPath &registerPathName, const std::vector<UserType> &vector, size_t wordOffsetInRegister,
601  const AccessModeFlags &flags) {
602  auto acc = getOneDRegisterAccessor<UserType>(registerPathName, vector.size(), wordOffsetInRegister, flags);
603  // the vector stays constant, but we have to work around so we can swap it and later swap it back...
604  std::vector<UserType> &mutable_vector = const_cast<std::vector<UserType>&>(vector);
605  acc.swap(mutable_vector);
606  acc.write();
607  acc.swap(mutable_vector);
