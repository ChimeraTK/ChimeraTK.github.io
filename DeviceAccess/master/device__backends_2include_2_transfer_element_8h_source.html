110  bool readNonBlocking() {
111  if(hasActiveFuture) {

131  bool readLatest() {
132  bool ret = false;
133  if(hasActiveFuture) {

135  activeFuture.wait();
136  ret = true;
137  }
138  else {
139  return false;
140  }
141  }

143  preRead();
144  bool ret2 = doReadTransferLatest();
145  if(ret2) postRead();
146  return ret || ret2;

174  TransferFuture& readAsync() {
175  if(!hasActiveFuture) {
176  // call preRead
177  this->readTransactionInProgress = false;
178  this->preRead();
179 
180  // initiate asynchronous transfer and return future
181  activeFuture = doReadTransferAsync();
182  hasActiveFuture = true;
183  }
184  return activeFuture;

191  // Note: this override is final to prevent implementations from implementing
