24  template<typename UserType>
25  class LNMBackendVariableAccessor : public NDRegisterAccessor<UserType> {
26  public:
27  LNMBackendVariableAccessor(boost::shared_ptr<DeviceBackend> dev, const RegisterPath& registerPathName,
28  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
29  : NDRegisterAccessor<UserType>(registerPathName, flags), _registerPathName(registerPathName),
30  _wordOffsetInRegister(wordOffsetInRegister), _fixedPointConverter(registerPathName, 32, 0, 1) {
31  // cast device
32  _dev = boost::dynamic_pointer_cast<LogicalNameMappingBackend>(dev);
33  // check for unknown flags
34  flags.checkForUnknownFlags({AccessMode::raw});
35  // obtain the register info
36  _info = boost::static_pointer_cast<LNMBackendRegisterInfo>(
37  _dev->getRegisterCatalogue().getRegister(_registerPathName));
38  if(numberOfWords == 0) numberOfWords = _info->length;
39  // check for illegal parameter combinations
40  if(wordOffsetInRegister + numberOfWords > _info->length) {
41  throw ChimeraTK::logic_error("Requested number of words and/or offset "
42  "exceeds length of register '" +
43  registerPathName + "'.");
44  }
45  if(flags.has(AccessMode::raw)) {
46  throw ChimeraTK::logic_error("AccessMode::raw is not supported on a "
47  "variable/constant-type register in a logical "
48  "name mapping device.");
49  }
50  // check for incorrect usage of this accessor
51  if(_info->targetType != LNMBackendRegisterInfo::TargetType::CONSTANT &&
52  _info->targetType != LNMBackendRegisterInfo::TargetType::VARIABLE) {
53  throw ChimeraTK::logic_error("LNMBackendVariableAccessor used for "
54  "wrong register type."); // LCOV_EXCL_LINE
55  // (impossible to
56  // test...)
57  }
58  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
59  NDRegisterAccessor<UserType>::buffer_2D[0].resize(numberOfWords);
60  }
61 
62  void doReadTransferSynchronously() override {
63  if(_dev->_hasException) {
64  throw ChimeraTK::runtime_error("previous, unrecovered fault");
65  }
66  }
67 
68  void doPreWrite(TransferType type, VersionNumber) override {
69  std::ignore = type;
70  if(_info->targetType ==
71  LNMBackendRegisterInfo::TargetType::
72  CONSTANT) { // repeat the condition in isReadOnly() here to avoid virtual function call.
73  throw ChimeraTK::logic_error("Writing to constant-type registers of logical name mapping devices "
74  "is not possible.");
75  }
76  if(!_dev->_opened) { // directly use member variables as friend to avoid virtual function calls
77  throw ChimeraTK::logic_error("Cannot write to a closed device.");
78  }
79 
80  _valueTableDataValidity = this->_dataValidity;
81  }
82 
83  bool doWriteTransfer(ChimeraTK::VersionNumber) override {
84  if(_dev->_hasException) {
85  throw ChimeraTK::runtime_error("previous, unrecovered fault");
86  }
87  std::lock_guard<std::mutex> lock(_info->valueTable_mutex);
88  for(size_t i = 0; i < NDRegisterAccessor<UserType>::buffer_2D[0].size(); ++i) {
89  callForType(_info->valueType, [&, this](auto arg) {
90  boost::fusion::at_key<decltype(arg)>(_info->valueTable.table)[i + _wordOffsetInRegister] =
91  userTypeToUserType<decltype(arg)>(this->buffer_2D[0][i]);
92  });
93  }
94  return false;
95  }
96 
97  bool mayReplaceOther(const boost::shared_ptr<TransferElement const>&) const override {
98  return false; // never replace, since it does not optimise anything
99  }
100 
101  bool isReadOnly() const override { return _info->targetType == LNMBackendRegisterInfo::TargetType::CONSTANT; }
102 
103  bool isReadable() const override { return true; }
104 
105  bool isWriteable() const override { return _info->targetType != LNMBackendRegisterInfo::TargetType::CONSTANT; }
106 
107  void doPreRead(TransferType) override {
108  if(!_dev->_opened) {
109  throw ChimeraTK::logic_error("Cannot read from a closed device.");
110  }
111  }
112 
113  void doPostRead(TransferType, bool hasNewData) override {
114  if(!hasNewData) return;
115 
116  std::lock_guard<std::mutex> lock(_info->valueTable_mutex);
117  for(size_t i = 0; i < NDRegisterAccessor<UserType>::buffer_2D[0].size(); ++i) {
118  callForType(_info->valueType, [&, this](auto arg) {
119  this->buffer_2D[0][i] = userTypeToUserType<UserType>(
120  boost::fusion::at_key<decltype(arg)>(_info->valueTable.table)[i + _wordOffsetInRegister]);
121  });
122  }
123 
124  this->_versionNumber = {};
125  this->_dataValidity = _valueTableDataValidity;
126  }
127 
128  protected:
130  RegisterPath _registerPathName;
131 
133  boost::shared_ptr<LogicalNameMappingBackend> _dev;
134 
138  boost::shared_ptr<LNMBackendRegisterInfo> _info;
139 
141  DataValidity _valueTableDataValidity{DataValidity::ok};

158 
159  // Version number of the last transfer
160  VersionNumber currentVersion{nullptr};
