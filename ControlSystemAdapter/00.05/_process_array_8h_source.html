362  void doReadTransfer() override {
363  // If previously a TransferFuture has been requested by readAsync(), first make sure that that future is
364  // already fulfilled. Otherwise we might run out of futures on the notification queue.
365  if(mtca4u::TransferElement::hasActiveFuture) {
366  mtca4u::TransferElement::activeFuture.getBoostFuture().wait();
367  mtca4u::TransferElement::hasActiveFuture = false;
368  if(!_asyncReadTransferNeeded) return;
369  _asyncReadTransferNeeded = false;
370  }
371  // Obtain futures from the notification queue and wait on them until we receive data. We start with checking
372  // for data before obtaining a future from the notification queue, since this is faster and the notification
373  // queue is shorter than the data queue.
374  while(!doReadTransferNonBlocking()) {
375  boost::shared_future<void> future;
376  bool atLeastOneFuturePresent = _sharedState->_notificationQueue.pop(future);
377  assert(atLeastOneFuturePresent); // if this is not true, there is something wrong with the algorithm
378  (void)atLeastOneFuturePresent; // prevent warning in case asserts are disabled
379  future.wait();
380  boost::this_thread::interruption_point();
381  }

384  bool doReadTransferNonBlocking() override {
385  if (_instanceType != RECEIVER) {
386  throw std::logic_error("Receive operation is only allowed for a receiver process variable.");
387  }
388  // If previously a TransferFuture has been requested by readAsync(), first make sure that that future is
389  // already fulfilled. Otherwise we might run out of futures on the notification queue.
390  if(mtca4u::TransferElement::hasActiveFuture) {
391  auto theFuture = mtca4u::TransferElement::activeFuture.getBoostFuture();
392  auto status = theFuture.wait_for(boost::chrono::duration<int, boost::centi>(0));
393  if(status == boost::future_status::timeout) return false;
394  mtca4u::TransferElement::hasActiveFuture = false;
395  if(!_asyncReadTransferNeeded) return true;
396  _asyncReadTransferNeeded = false;
397  }
398  // We have to check that the vector that we currently own still has the
399  // right size. Otherwise, the code using the sender might get into
400  // trouble when it suddenly experiences a vector of the wrong size.
401  if (_sharedState->_buffers[_currentIndex].value.size() != _vectorSize) {
402  throw std::runtime_error("Cannot run receive operation because the size of the vector belonging to the current"
403  " buffer has been modified.");
404  }
405  std::size_t nextIndex;
406  if (_sharedState->_fullBufferQueue.pop(nextIndex)) {
407  // We only use the incoming update if it has a higher version number
408  // than the current value. This check is disabled when there is no
409  // version number source.
410  if (!_versionNumberSource || (_sharedState->_buffers[nextIndex]).versionNumber > getVersionNumber()) {
411  _sharedState->_emptyBufferQueue.push(_currentIndex);
412  _currentIndex = nextIndex;
413  return true;
414  } else {
415  _sharedState->_emptyBufferQueue.push(nextIndex);
416  return false;
417  }
418  } else {
419  return false;
420  }

429  mtca4u::TransferFuture& readAsync() override {
430  if (_instanceType != RECEIVER) {
431  throw std::logic_error("Receive operation is only allowed for a receiver process variable.");
432  }
433  ChimeraTK::ExperimentalFeatures::check("asynchronous read");
434  if(mtca4u::TransferElement::hasActiveFuture) return mtca4u::TransferElement::activeFuture; // the last future given out by this fuction is still active
435 
436  // check if we already have data
437  boost::promise<void> promise;
438  boost::shared_future<void> readyFuture = promise.get_future().share();
439  promise.set_value();
440  if(doReadTransferNonBlocking()) {
441  mtca4u::TransferElement::activeFuture = mtca4u::TransferFuture(readyFuture, static_cast<mtca4u::TransferElement*>(this));
442  mtca4u::TransferElement::hasActiveFuture = true;
443  _asyncReadTransferNeeded = false;
444  return mtca4u::TransferElement::activeFuture;
445  }
446  else {
447  boost::shared_future<void> future;
448  // obtain future from queue
449  bool atLeastOneFuturePresent = _sharedState->_notificationQueue.pop(future);
450  assert(atLeastOneFuturePresent); // if this is not true, there is something wrong with the algorithm
451  (void)atLeastOneFuturePresent; // prevent warning in case asserts are disabled
452  // if this future is already fulfilled, obtain the next one
453  if(future.is_ready()) {
454  // before we can obtain the next future, we must check for new data once more
455  if(doReadTransferNonBlocking()) {
456  mtca4u::TransferElement::activeFuture = mtca4u::TransferFuture(readyFuture, static_cast<mtca4u::TransferElement*>(this));
457  mtca4u::TransferElement::hasActiveFuture = true;
458  _asyncReadTransferNeeded = false;
459  return mtca4u::TransferElement::activeFuture;
460  }
461  atLeastOneFuturePresent = _sharedState->_notificationQueue.pop(future);
462  assert(atLeastOneFuturePresent);
463  }
464  // return the future
465  mtca4u::TransferElement::activeFuture = mtca4u::TransferFuture(future, static_cast<mtca4u::TransferElement*>(this));
466  mtca4u::TransferElement::hasActiveFuture = true;
467  _asyncReadTransferNeeded = true;
468  return mtca4u::TransferElement::activeFuture;

785  bool writeInternal(VersionNumber newVersionNumber, bool shouldCopy) {
786  if (_instanceType != SENDER) {
787  throw std::logic_error(
788  "Send operation is only allowed for a sender process variable. Variable name: "+this->getName());
789  }
790  // We have to check that the vector that we currently own still has the
791  // right size. Otherwise, the code using the receiver might get into
792  // trouble when it suddenly experiences a vector of the wrong size.
793  if (mtca4u::NDRegisterAccessor<T>::buffer_2D[0].size() != _vectorSize) {
794  throw std::runtime_error(
795  "Cannot run receive operation because the size of the vector belonging to the current buffer has been "
796  "modified. Variable name: "+this->getName());
797  }
798  // First update the persistent data storage, if any was associated. This cannot be done after sending, since the
799  // value might no longer be available within this instance.
800  if (_persistentDataStorage) {
801  _persistentDataStorage->updateValue(_persistentDataStorageID, mtca4u::NDRegisterAccessor<T>::buffer_2D[0]);
802  }
803 
804  // Before sending the value, we have to update the associated
805  // time-stamp.
806  TimeStamp newTimeStamp =
807  _timeStampSource ?
808  _timeStampSource->getCurrentTimeStamp() :
809  TimeStamp::currentTime();
810  (_sharedState->_buffers[_currentIndex]).timeStamp = newTimeStamp;
811  (_sharedState->_buffers[_currentIndex]).versionNumber = newVersionNumber;
812  if (shouldCopy) {
813  _sharedState->_buffers[_currentIndex].value = mtca4u::NDRegisterAccessor<T>::buffer_2D[0];
814  }
815  else {
816  _sharedState->_buffers[_currentIndex].value.swap( mtca4u::NDRegisterAccessor<T>::buffer_2D[0] );
817  }
818  std::size_t nextIndex;
819  bool lostData = false;
820  if (_sharedState->_emptyBufferQueue.pop(nextIndex)) {
821  _sharedState->_fullBufferQueue.push(_currentIndex);
822  } else {
823  // We can still send, but we will lose older data.
824  if (_sharedState->_fullBufferQueue.pop(nextIndex)) {
825  _sharedState->_fullBufferQueue.push(_currentIndex);
826  lostData = true;
827  } else {
828  // It is possible, that we did not find an empty buffer but before
829  // we could get a full buffer, the receiver processed all full
830  // buffers. In this case, the queue of empty buffers cannot be empty
831  // any longer because we have at least four buffers.
832  if (!_sharedState->_emptyBufferQueue.pop(nextIndex)) {
833  // This should never happen and is just an assertion for extra
834  // safety.
835  throw std::runtime_error(
836  "Assertion that empty-buffer queue has at least one element failed.");
837  }
838  _sharedState->_fullBufferQueue.push(_currentIndex);
839  }
840  }
841  if (shouldCopy) {
842  _sharedState->_buffers[nextIndex].timeStamp = newTimeStamp;
843  _sharedState->_buffers[nextIndex].versionNumber = newVersionNumber;
844  }
845  _currentIndex = nextIndex;
846  if (_sendNotificationListener) {
847  _sendNotificationListener->notify(_receiver);
848  }
849  // Notify the receiver through the notification queue. It must be guaranteed that there is at least one
850  // unfulfilled future in the notification queue at all times, so we try pushing a new one first. If this
851  // failes, there is already one unfulfilled and one fulfilled future in the queue, so we are done.
852  boost::promise<void> nextPromise;
853  if(_sharedState->_notificationQueue.push(nextPromise.get_future())) {
854  try {
855  _sharedState->_notificationPromise.set_value();
856  }
857  catch(boost::promise_already_satisfied) {
858  // ignore -> this may happen only during shutdown
859  }
860  _sharedState->_notificationPromise = std::move(nextPromise);
861  }
862 
863  return lostData;
864  }
