131  bool advance(VirtualTime tval) {
132  if(tval < 0) return false;
133  current += tval;
134  bool hasFired = false;
135  do {
136  VirtualTime tstep = fmin(tval, getRemaining());
137  boost::fusion::for_each(*timers, advanceTimer(tstep,hasFired));
138  tval -= tstep;
