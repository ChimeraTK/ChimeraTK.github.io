223 template <typename T>
224 class DummyMultiplexedRegisterAccessor : public DummyRegisterAddressChecker {
225 public:
232  DummyMultiplexedRegisterAccessor(DummyBackend *dev, std::string module,
233  std::string name)
234  : _dev(dev), pitch(0) {
235  _dev->_registerMapping->getRegisterInfo(MULTIPLEXED_SEQUENCE_PREFIX + name,
236  registerInfo, module);
237 
238  int i = 0;
239  while (true) {
240  // obtain register information for sequence
241  RegisterInfoMap::RegisterInfo elem;
242  std::stringstream sequenceNameStream;
243  sequenceNameStream << SEQUENCE_PREFIX << name << "_" << i++;
244  try {
245  _dev->_registerMapping->getRegisterInfo(sequenceNameStream.str(), elem,
246  module);
247  } catch (ChimeraTK::logic_error &) {
248  break;
249  }
250  // create fixed point converter for sequence
251  fpc.push_back(FixedPointConverter(module + "/" + name, elem.width,
252  elem.nFractionalBits, elem.signedFlag));
253  // store offsets and number of bytes per word
254  offsets.push_back(elem.address);
255  nbytes.push_back(elem.nBytes);
256  // determine pitch
257  pitch += elem.nBytes;
258  }
259 
260  if (fpc.empty()) {
261  throw ChimeraTK::logic_error("No sequenes found for name \"" + name +
262  "\".");
263  }
264 
265  // compute number of elements per sequence
266  nElements = registerInfo.nBytes / pitch;
267  }
268 
270  unsigned int getNumberOfElements() { return nElements; }
271 
273  unsigned int getNumberOfSequences() { return fpc.size(); }
274 
280  inline proxies::DummyRegisterSequence<T> operator[](unsigned int sequence) {
281  int8_t *basePtr =
282  reinterpret_cast<int8_t *>(_dev->_barContents[registerInfo.bar].data());
283  uint32_t *seq = reinterpret_cast<uint32_t *>(basePtr + offsets[sequence]);
284  return proxies::DummyRegisterSequence<T>(&(fpc[sequence]), nbytes[sequence],
285  pitch, seq);
286  }
287 
288 protected:
290  DummyBackend *_dev;
291 
293  std::vector<FixedPointConverter> fpc;
294 
296  std::vector<int> offsets;
297 
299  std::vector<int> nbytes;
300 
302  int pitch;
303 
305  unsigned int nElements;
306 
307 private:
309  void operator=(const DummyMultiplexedRegisterAccessor &rightHandSide) const;
310 };
