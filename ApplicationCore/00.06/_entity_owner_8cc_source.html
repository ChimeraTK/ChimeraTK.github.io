92  void EntityOwner::findTagAndAppendToModule(VirtualModule &module, const std::string &tag, bool eliminateAllHierarchies,
93  bool eliminateFirstHierarchy) const {
94 
95  VirtualModule nextmodule{_name, _description};
96  VirtualModule *moduleToAddTo;
97 
98  bool needToAddSubModule = false;
99  if(!getEliminateHierarchy() && !eliminateAllHierarchies && !eliminateFirstHierarchy) {
100  moduleToAddTo = &nextmodule;
101  needToAddSubModule = true;
102  }
103  else {
104  moduleToAddTo = &module;
105  }
106 
107  // add nodes to the module if matching the tag
108  for(auto node : getAccessorList()) {
109  if(node.getTags().count(tag) > 0) {
110  moduleToAddTo->registerAccessor(node);
111  }
112  }
113 
114  // iterate through submodules
115  for(auto submodule : getSubmoduleList()) {
116  submodule->findTagAndAppendToModule(*moduleToAddTo, tag, eliminateAllHierarchies);
117  }
118 
119  if(needToAddSubModule) {
120  if( nextmodule.getAccessorList().size() > 0 || nextmodule.getSubmoduleList().size() > 0 ) {
121  module.addSubModule(nextmodule);
122  }
123  }

125  }
