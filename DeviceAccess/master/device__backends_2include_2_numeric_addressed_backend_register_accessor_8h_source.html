25  NumericAddressedBackendRegisterAccessor(boost::shared_ptr<DeviceBackend> dev,
26  const RegisterPath &registerPathName, size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
27  : SyncNDRegisterAccessor<UserType>(registerPathName),
28  _fixedPointConverter(registerPathName),
29  isRaw(false),
30  _registerPathName(registerPathName),
31  _numberOfWords(numberOfWords)
32  {
33  try {
34  // check for unknown flags
35  flags.checkForUnknownFlags({AccessMode::raw});
36 
37  // check device backend
38  _dev = boost::dynamic_pointer_cast<NumericAddressedBackend>(dev);
39  if(!_dev) {
40  throw ChimeraTK::logic_error("NumericAddressedBackendRegisterAccessor is used with a backend which is not "
41  "a NumericAddressedBackend.");
42  }
43 
44  // obtain register information
45  boost::shared_ptr<RegisterInfo> info = _dev->getRegisterInfo(registerPathName);
46  _registerInfo = boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(info);
47  _bar = _registerInfo->bar;
48  _startAddress = _registerInfo->address + wordOffsetInRegister*sizeof(int32_t);
49 
50  // check number of words
51  if(_numberOfWords == 0) {
52  _numberOfWords = _registerInfo->getNumberOfElements() - wordOffsetInRegister;
53  }
54  if(_numberOfWords + wordOffsetInRegister > _registerInfo->getNumberOfElements()) {
55  throw ChimeraTK::logic_error("Requested number of words exceeds the size of the register '"+_registerPathName+"'!");
56  }
57  if(wordOffsetInRegister >= _registerInfo->getNumberOfElements()) {
58  throw ChimeraTK::logic_error("Requested offset exceeds the size of the register'"+_registerPathName+"'!");
59  }
60 
61  // create low-level transfer element handling the actual data transfer to the hardware with raw data
62  _rawAccessor.reset(new NumericAddressedLowLevelTransferElement(_dev,_bar,_startAddress,_numberOfWords));
63 
64  // allocated the buffers
65  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
66  NDRegisterAccessor<UserType>::buffer_2D[0].resize(_numberOfWords);
67 
68  // configure fixed point converter
69  // We don't have to fill it in a special way if the accessor is raw
70  // because we have an overloaded, more efficient implementation
71  // in this case. So we can use it in setAsCoocked() and getAsCoocked()
72  _fixedPointConverter = FixedPointConverter(_registerPathName,
73  _registerInfo->width,
74  _registerInfo->nFractionalBits,
75  _registerInfo->signedFlag);
76  if(flags.has(AccessMode::raw)) {
77  if(typeid(UserType) != typeid(int32_t)) {
78  throw ChimeraTK::logic_error("Given UserType when obtaining the NumericAddressedBackendRegisterAccessor in raw mode does not "
79  "match the expected type. Use an int32_t instead! (Register name: "+_registerPathName+"')");
80  }
81  isRaw = true;
82  }
83  }
84  catch(...) {
85  this->shutdown();
86  throw;
87  }
88 
89  FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(getAsCoocked_impl);
90  FILL_VIRTUAL_FUNCTION_TEMPLATE_VTABLE(setAsCoocked_impl);
91  }

224  void replaceTransferElement(boost::shared_ptr<TransferElement> newElement) override {
225  auto casted = boost::dynamic_pointer_cast< NumericAddressedLowLevelTransferElement >(newElement);
226  if(casted && casted->isMergeable(_rawAccessor)) {
227  size_t newStartAddress = std::min(casted->_startAddress, _rawAccessor->_startAddress);
228  size_t newStopAddress = std::max(casted->_startAddress+casted->_numberOfBytes,
229  _rawAccessor->_startAddress+_rawAccessor->_numberOfBytes);
230  size_t newNumberOfWords = (newStopAddress-newStartAddress)/sizeof(int32_t);
231  casted->changeAddress(newStartAddress,newNumberOfWords);
232  _rawAccessor = casted;
233  }
