318  _receiver->postRead(type, hasNewData);
319  if(hasNewData) {
320  this->accessChannel(0).swap(_receiver->accessChannel(0));
321  // After receiving, our new time stamp and version number are the ones

324

346  template<class T>
347  bool BidirectionalProcessArray<T>::doWriteTransfer(ChimeraTK::VersionNumber versionNumber) {
348  // We have to copy our current value to the sender. We cannot swap it
349  // because this would mean that we would lose the current value.
350  _sender->accessChannel(0) = this->accessChannel(0);
351  // We already copied the value, so the sender does not have to copy the
352  // value again.
353 
354  // Propagate validity flag
355  _sender->setDataValidity(TransferElement::dataValidity());
356 
357  bool lostData = _sender->writeDestructively(versionNumber);
358 
359  // If we have a persistent data-storage, we have to update it.
360  if(_persistentDataStorage) {
361  _persistentDataStorage->updateValue(_persistentDataStorageID, this->accessChannel(0));
362  }
363  if(_sendNotificationListener) {
364  typename ProcessArray<T>::SharedPtr partner = _partner.lock();
365  if(partner) {
366  _sendNotificationListener->notify(partner);
367  }
368  }
369  return lostData;
370  }
