71  boost::shared_ptr<NDRegisterAccessor<UserType>> getRegisterAccessor_impl(const RegisterPath& registerPathName,
72  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags) {
73  // Suffix to mark writeable references to read-only registers
74  static const std::string DUMMY_WRITEABLE_SUFFIX{".DUMMY_WRITEABLE"};
75 
76  bool isDummyWriteableAccessor = false;
77  RegisterPath actualRegisterPath{registerPathName};
78 
79  // Check if register name ends on DUMMY_WRITEABLE_SUFFIX,
80  // in that case, set actual path to the "real" register
81  // which exists in the catalogue.
82  const std::string regPathNameStr{registerPathName};
83  std::smatch match;
84  const std::regex re{DUMMY_WRITEABLE_SUFFIX + "$"};
85  std::regex_search(regPathNameStr, match, re);
86 
87  if(!match.empty()) {
88  isDummyWriteableAccessor = true;
89  actualRegisterPath = RegisterPath{match.prefix()};
90  }
91 
92  auto accessor = NumericAddressedBackend::getRegisterAccessor_impl<UserType>(
93  actualRegisterPath, numberOfWords, wordOffsetInRegister, flags);
94 
95  // Modify writeability of the NumericAddressedBackendRegisterAccessor
96  if(isDummyWriteableAccessor) {
97  const auto info{getRegisterInfo(actualRegisterPath)};
98 
99  if(info->dataType == RegisterInfoMap::RegisterInfo::Type::FIXED_POINT) {
100  if(flags.has(AccessMode::raw)) {
101  boost::dynamic_pointer_cast<NumericAddressedBackendRegisterAccessor<UserType, FixedPointConverter, true>>(
102  accessor)
103  ->makeWriteable();
104  }
105  else {
106  if(info->getNumberOfDimensions() < 2) {
107  boost::dynamic_pointer_cast<
108  NumericAddressedBackendRegisterAccessor<UserType, FixedPointConverter, false>>(accessor)
109  ->makeWriteable();
110  }
111  else {
112  boost::dynamic_pointer_cast<NumericAddressedBackendMuxedRegisterAccessor<UserType>>(accessor)
113  ->makeWriteable();
114  }
115  }
116  }
117  else if(info->dataType == RegisterInfoMap::RegisterInfo::Type::IEEE754) {
118  if(flags.has(AccessMode::raw)) {
119  boost::dynamic_pointer_cast<
120  NumericAddressedBackendRegisterAccessor<UserType, IEEE754_SingleConverter, true>>(accessor)
121  ->makeWriteable();
122  }
123  else {
124  boost::dynamic_pointer_cast<
125  NumericAddressedBackendRegisterAccessor<UserType, IEEE754_SingleConverter, false>>(accessor)
126  ->makeWriteable();
127  }
128  }
129  }
130 
131  return accessor;
132  }

163  boost::shared_ptr<DeviceBackend> ptr(new T(std::forward<Args>(arguments)...));
