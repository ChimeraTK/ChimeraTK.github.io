99  inline bool DataConsistencyGroup::update(TransferElementID transferElementID) {
100  // ignore transferElementID does not belong to DataConsistencyGroup
101  if(push_elements.find(transferElementID) == push_elements.end()) {
102  return false;
103  }
104  auto getVNFromElemenst = push_elements[transferElementID].getVersionNumber();
105  if(versionNumberToBeConsistentTo != getVNFromElemenst) {
106  versionNumberToBeConsistentTo = getVNFromElemenst;
107  consistentElements.clear();
108  }
109  consistentElements.insert(transferElementID);
110  if(consistentElements.size() == push_elements.size()) {
111  lasteSateOfConsistentElements = consistentElements;
112  return true;
113  }
114  return false;
115  }
