81 std::vector<int> RegisterItem::fetchElementsFromCard(
82  const TableWidgetData& tabledata) {
83  RegisterInfo_t regInfo = this->getRegisterMapElement();
84 
85  int numberOfElements = regInfo.nElements;
86  std::vector<int> buffer(numberOfElements);
87 
88  unsigned int registerBar = regInfo.bar;
89  Device_t const& mtcadevice = tabledata.device;
90  int maxrows = tabledata.tableMaxRowCount;
91  size_t nBytesToRead =
92  std::min(numberOfElements, maxrows) * qthardmon::WORD_SIZE_IN_BYTES;
93  unsigned int registerAddress = regInfo.address;
94 
95  if (registerBar == 0xD) {
96  mtcadevice->readDMA(registerBar, registerAddress, &(buffer[0]),
97  nBytesToRead);
98  } else {
99  mtcadevice->read(registerBar, registerAddress, &(buffer[0]), nBytesToRead);
100  }
101  return buffer;
102 }
