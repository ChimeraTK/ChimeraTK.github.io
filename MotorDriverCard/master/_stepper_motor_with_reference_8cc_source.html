213  void StepperMotorWithReference::loadEndSwitchCalibration(){
214  LockGuard guard(_mutex);
215 
216  if(_motorControler->getCalibrationTime() != 0){
217 
218  int calibPositiveEndSwitchInSteps = _motorControler->getPositiveReferenceSwitchCalibration();
219  int calibNegativeEndSwitchInSteps = _motorControler->getNegativeReferenceSwitchCalibration();
220  _calibPositiveEndSwitchInSteps.exchange(calibPositiveEndSwitchInSteps);
221  _calibNegativeEndSwitchInSteps.exchange(calibNegativeEndSwitchInSteps);
222 
223  if(calibPositiveEndSwitchInSteps == std::numeric_limits<int>::max()
224  && calibNegativeEndSwitchInSteps == std::numeric_limits<int>::min()){
225  _calibrationMode.exchange(StepperMotorCalibrationMode::SIMPLE);
226  }

228  _calibrationMode.exchange(StepperMotorCalibrationMode::FULL);

242  bool StepperMotorWithReference::isEndSwitchActive(Sign sign){
243  bool posActive = _motorControler->getReferenceSwitchData().getPositiveSwitchActive() == 1U;
244  bool negActive = _motorControler->getReferenceSwitchData().getNegativeSwitchActive() == 1U;
245 
246  if(posActive && negActive){
247  _errorMode.exchange(StepperMotorError::BOTH_END_SWITCHES_ON);
248  _stateMachine->setAndProcessUserEvent(StateMachine::errorEvent);
249  }
250 
251  if (_toleranceCalculated){
252  if (posActive){
253  if(std::abs(_motorControler->getActualPosition() - _calibPositiveEndSwitchInSteps.load()) > 3 * _tolerancePositiveEndSwitch.load()){
254  _errorMode.exchange(StepperMotorError::CALIBRATION_ERROR);
255  }
256  }
257  else if(negActive){
258  if(std::abs(_motorControler->getActualPosition() - _calibNegativeEndSwitchInSteps.load()) > 3 * _toleranceNegativeEndSwitch.load()){
259  _errorMode.exchange(StepperMotorError::CALIBRATION_ERROR);
260  }
261  }
262  }
263 
264  if (sign == Sign::POSITIVE){
265  if (posActive){
266  return true;
267  }else{
268  return false;
269  }
270  }else{
271  if (negActive){
272  return true;
273  }else{
274  return false;
275  }
276  }
