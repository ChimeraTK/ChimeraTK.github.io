23  template<typename UserType>
24  class LNMBackendBitAccessor : public SyncNDRegisterAccessor<UserType> {
25  public:
26  LNMBackendBitAccessor(boost::shared_ptr<DeviceBackend> dev, const RegisterPath& registerPathName,
27  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
28  : SyncNDRegisterAccessor<UserType>(registerPathName), _registerPathName(registerPathName),
29  _fixedPointConverter(registerPathName, 32, 0, 1) {
30  try {
31  // check for unknown flags
32  flags.checkForUnknownFlags({AccessMode::raw});
33  // check for illegal parameter combinations
34  if(flags.has(AccessMode::raw)) {
35  throw ChimeraTK::logic_error("LNMBackendBitAccessor: raw access not supported!");
36  }
37  _dev = boost::dynamic_pointer_cast<LogicalNameMappingBackend>(dev);
38  // copy the register info and create the internal accessors, if needed
39  auto info = boost::static_pointer_cast<LNMBackendRegisterInfo>(
40  _dev->getRegisterCatalogue().getRegister(_registerPathName));
41  // check for incorrect usage of this accessor
42  if(info->targetType != LNMBackendRegisterInfo::TargetType::BIT) {
43  throw ChimeraTK::logic_error(
44  "LNMBackendBitAccessor used for wrong register type."); // LCOV_EXCL_LINE (impossible to test...)
45  }
46  // get target device and accessor
47  std::string devName = info->deviceName;
48  boost::shared_ptr<DeviceBackend> targetDevice;
49  if(devName != "this") {
50  targetDevice = _dev->_devices[devName];
51  }
52  else {
53  targetDevice = dev;
54  }
55  auto& map = boost::fusion::at_key<uint64_t>(_dev->sharedAccessorMap.table);
56  auto it = map.find(RegisterPath(info->registerName));
57  if(it == map.end()) {
58  _accessor = targetDevice->getRegisterAccessor<uint64_t>(
59  RegisterPath(info->registerName), numberOfWords, wordOffsetInRegister, false);
60  map[RegisterPath(info->registerName)].accessor = _accessor;
61  }
62  else {
63  _accessor = map[RegisterPath(info->registerName)].accessor;
64  }
65  _mutex = &map[RegisterPath(info->registerName)].mutex;
66  // allocate and initialise the buffer
67  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
68  NDRegisterAccessor<UserType>::buffer_2D[0].resize(1);
69  NDRegisterAccessor<UserType>::buffer_2D[0][0] = numericToUserType<UserType>(false);
70  // set the bit mask
71  _bitMask = 1 << info->bit;
72  }
73  catch(...) {
74  this->shutdown();
75  throw;
76  }
77  }

110  void doPostRead(TransferType type, bool hasNewData) override {
111  std::lock_guard<std::mutex> lock(*_mutex);
112  _accessor->postRead(type, hasNewData);
113  if(!hasNewData) return;
114  if(_accessor->accessData(0) & _bitMask) {
115  NDRegisterAccessor<UserType>::buffer_2D[0][0] = numericToUserType<UserType>(true);
116  }
117  else {
118  NDRegisterAccessor<UserType>::buffer_2D[0][0] = numericToUserType<UserType>(false);
119  }
120  _versionNumber = {}; // VersionNumber needs to be decoupled from target accessor
121  }
122 
123  void doPreWrite(TransferType type, VersionNumber) override {
124  std::lock_guard<std::mutex> lock(*_mutex);
125  if(!_fixedPointConverter.toRaw<UserType>(NDRegisterAccessor<UserType>::buffer_2D[0][0])) {
126  _accessor->accessData(0) &= ~(_bitMask);
127  }
128  else {
129  _accessor->accessData(0) |= _bitMask;
130  }
131 
132  _versionNumberTemp = {};
133  _accessor->preWrite(type, _versionNumberTemp);
134  }
135 
136  void doPostWrite(TransferType type, bool dataLost) override {
137  std::lock_guard<std::mutex> lock(*_mutex);
138  _accessor->postWrite(type, dataLost);
139  }

141  bool mayReplaceOther(const boost::shared_ptr<TransferElement const>& other) const override {
142  auto rhsCasted = boost::dynamic_pointer_cast<const LNMBackendBitAccessor<UserType>>(other);
143  if(!rhsCasted) return false;
144  if(_registerPathName != rhsCasted->_registerPathName) return false;
145  if(_dev != rhsCasted->_dev) return false;
146  return true;
147  }
