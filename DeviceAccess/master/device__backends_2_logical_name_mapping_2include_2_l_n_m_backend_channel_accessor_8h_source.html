22  template<typename UserType>
23  class LNMBackendChannelAccessor : public SyncNDRegisterAccessor<UserType> {
24  public:
25  LNMBackendChannelAccessor(boost::shared_ptr<DeviceBackend> dev, const RegisterPath& registerPathName,
26  size_t numberOfWords, size_t wordOffsetInRegister, AccessModeFlags flags)
27  : SyncNDRegisterAccessor<UserType>(registerPathName), _registerPathName(registerPathName) {
28  try {
29  // check for unknown flags
30  flags.checkForUnknownFlags({AccessMode::raw});
31  // check for illegal parameter combinations
32  if(flags.has(AccessMode::raw)) {
33  throw ChimeraTK::logic_error("LNMBackendChannelAccessor: raw access not yet supported!");
34  }
35  _dev = boost::dynamic_pointer_cast<LogicalNameMappingBackend>(dev);
36  // copy the register info and create the internal accessors, if needed
37  _info = boost::static_pointer_cast<LNMBackendRegisterInfo>(
38  _dev->getRegisterCatalogue().getRegister(_registerPathName));
39  // check for incorrect usage of this accessor
40  if(_info->targetType != LNMBackendRegisterInfo::TargetType::CHANNEL) {
41  throw ChimeraTK::logic_error("LNMBackendChannelAccessor used for wrong "
42  "register type."); // LCOV_EXCL_LINE
43  // (impossible to
44  // test...)
45  }
46  // get target device and accessor
47  std::string devName = _info->deviceName;
48  boost::shared_ptr<DeviceBackend> targetDevice;
49  if(devName != "this") {
50  targetDevice = _dev->_devices[devName];
51  }
52  else {
53  targetDevice = dev;
54  }
55  _accessor = targetDevice->getRegisterAccessor<UserType>(
56  RegisterPath(_info->registerName), numberOfWords, wordOffsetInRegister, false);
57  // verify channel number
58  if(_info->channel >= _accessor->getNumberOfChannels()) {
59  throw ChimeraTK::logic_error("LNMBackendChannelAccessor: Requested channel number " +
60  std::to_string(_info->channel) +
61  " exceeds number of channels of target register,"
62  " in accesor for register '" +
63  registerPathName + "'.");
64  }
65  // allocate the buffer
66  NDRegisterAccessor<UserType>::buffer_2D.resize(1);
67  NDRegisterAccessor<UserType>::buffer_2D[0].resize(_accessor->getNumberOfSamples());
68  }
69  catch(...) {
70  this->shutdown();
71  throw;
72  }
73  }

101  bool mayReplaceOther(const boost::shared_ptr<TransferElement const>& other) const override {
102  auto rhsCasted = boost::dynamic_pointer_cast<const LNMBackendChannelAccessor<UserType>>(other);
103  if(!rhsCasted) return false;
104  if(_registerPathName != rhsCasted->_registerPathName) return false;
105  if(_dev != rhsCasted->_dev) return false;
106  return true;
107  }

142  void replaceTransferElement(boost::shared_ptr<TransferElement> newElement) override {
143  auto casted = boost::dynamic_pointer_cast<NDRegisterAccessor<UserType>>(newElement);
144  if(casted && _accessor->mayReplaceOther(newElement)) {
145  _accessor = casted;
146  }
147  else {
148  _accessor->replaceTransferElement(newElement);
149  }
