49  template <typename ConvertedDataType>
50  void read(ConvertedDataType *convertedData, size_t nWords = 1, uint32_t wordOffsetInRegister = 0) const {
51  if(nWords == 0) return;
52  // obtain the accessor through the handler
53  auto & accessorHandler = boost::fusion::at_key<ConvertedDataType>(_convertingAccessorHandlers.table);
54  accessorHandler.checkAndResize(nWords, wordOffsetInRegister, false /*not raw*/,
55  _backend, _registerPathName);
56  // now we can be sure that the accessor is valid and has enough memory to perform the memcpy
57  accessorHandler.accessor->read();
58  // copy data to target buffer. The accessor might not start at index 0, so we have to calculate the correct
59  // offset inside the accessor
60  memcpy(convertedData, accessorHandler.accessor->accessChannel(0).data() + wordOffsetInRegister-accessorHandler.beginIndex,
61  nWords*sizeof(ConvertedDataType));
62  }

218  template<class UserType>
219  class AccessorHandler{
220  public:
221  AccessorHandler() : beginIndex(0),endIndex(0) {}
222 
223  boost::shared_ptr<NDRegisterAccessor<UserType> > accessor;
224  size_t beginIndex; // the first index of the accessor;
225  size_t endIndex; // the end index (first index after the valid range, liken end iterator)
226  void checkAndResize(size_t nWords, size_t wordOffsetInRegister, bool isRaw, boost::shared_ptr<DeviceBackend> const & backend, RegisterPath const & registerPathName){
227  // no need for out of register size checking. getRegisterAccessor does the job for us and throws in case of error.
228  if ( (wordOffsetInRegister < beginIndex) || (wordOffsetInRegister+nWords > (endIndex) || !accessor)){
229  // we have to (re)allocate
230  size_t newBeginIndex;
231  if (!accessor){
232  newBeginIndex = wordOffsetInRegister;
233  }else{
234  newBeginIndex = std::min(wordOffsetInRegister, beginIndex);
235  }
236  size_t newEndIndex = std::max(wordOffsetInRegister+nWords, endIndex);
237  accessor = backend->getRegisterAccessor<UserType>(registerPathName, newEndIndex-newBeginIndex, newBeginIndex, isRaw); // 0 offset, raw
238  // we have to update the accessor with the current hardware information. It might be that a partial write is going to happen.
239  accessor->read();
240  // only if creating the new accessor succeeds we change the index bookkeeping variables
241  beginIndex = newBeginIndex;
242  endIndex = newEndIndex;
243  }
244  }
245  };
