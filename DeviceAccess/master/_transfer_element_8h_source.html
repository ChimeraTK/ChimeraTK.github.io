523  inline TransferElement::ID TransferElement::readAny(std::list<std::reference_wrapper<TransferElement>> elementsToRead) {
524 
525  // build list of TransferFutures for all elements. Since readAsync() is a virtual call and we need to visit all
526  // elements at least twice (once in wait_for_any and a second time for sorting by version number), this is assumed
527  // to be less expensive than calling readAsync() on the fly in the TransferFutureIterator instead.
528  std::list<TransferFuture> futureList;
529  for(auto &elem : elementsToRead) {
530  futureList.push_back(elem.get().readAsync());
531  }
532  // wait until any future is ready
533  auto iter = boost::wait_for_any(TransferFutureIterator(futureList.begin()), TransferFutureIterator(futureList.end()));
534 
535  // Find the variable which has the oldest version number (to guarantee the order of updates).
536  // Start with assuming that the future returned by boost::wait_for_any() has the oldes version number.
537  TransferFuture theUpdate = iter.getTransferFuture();
538  for(auto future : futureList) {
539  // skip if this is the future returned by boost::wait_for_any()
540  if(future == theUpdate) continue;
541  // also skip if the future is not yet ready
542  if(!future.hasNewData()) continue;
543  // compare version number with the version number of the stored future
544  if(future.getBoostFuture().get()->_versionNumber < theUpdate.getBoostFuture().get()->_versionNumber) {
545  theUpdate = future;
546  }
547  }
548 
549  // complete the transfer (i.e. run postRead())
550  theUpdate.wait();
551 
552  // return the transfer element as a shared pointer
553  return theUpdate.getTransferElement().getId();
554  }
