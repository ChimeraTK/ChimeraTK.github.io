81  bool RegisterInfoMap::check(ErrorList &err, RegisterInfoMap::ErrorList::ErrorElem::TYPE level) {
82  std::vector<addresses> v_addresses;
83  std::vector<addresses>::iterator v_iter;
84  addresses address;
85 
86  std::vector<RegisterInfo> map_file;
87  for(auto it = _catalogue.begin(); it != _catalogue.end(); ++it) {
88  auto infoCast = boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(it.get());
89  map_file.push_back(*infoCast);
90  }
91 
92  std::vector<RegisterInfo>::iterator iter_p, iter_n;
93  bool ret = true;
94 
95  err.clear();
96  if (map_file.size() < 2)
97  return true;
98  std::sort(map_file.begin(), map_file.end(), compareRegisterInfosByName_functor());
99  iter_p = map_file.begin();
100  iter_n = iter_p + 1;
101  while (1) {
102  if ( (iter_p->name == iter_n->name) && (iter_p->module == iter_n->module) ) {
103  err.insert(ErrorList::ErrorElem(ErrorList::ErrorElem::ERROR,
104  ErrorList::ErrorElem::NONUNIQUE_REGISTER_NAME,
105  *iter_p, *iter_n, _mapFileName));
106  ret = false;
107  }
108  iter_p = iter_n;
109  iter_n = ++iter_n;
110  if (iter_n == map_file.end()) {
111  break;
112  }
113  }
114 
115  for (iter_p = map_file.begin(); iter_p != map_file.end(); ++iter_p) {
116  address.start = iter_p->address;
117  address.end = iter_p->address + iter_p->nBytes;
118  address.iter = iter_p;
119  for (v_iter = v_addresses.begin(); v_iter != v_addresses.end(); ++v_iter) {
120  // only addresses in the same module are considered to overlap
121  if (iter_p->module != v_iter->iter->module){
122  continue;
123  }
124 
125  if (iter_p->bar == (v_iter->iter)->bar) {
126  if ( (v_iter->start >= iter_p->address) &&
127  (v_iter->start < (iter_p->address + iter_p->nBytes)) ) {
128  if (level >= ErrorList::ErrorElem::WARNING){
129  err.insert(ErrorList::ErrorElem(ErrorList::ErrorElem::WARNING,
130  ErrorList::ErrorElem::WRONG_REGISTER_ADDRESSES,
131  *iter_p, *(v_iter->iter), _mapFileName));
132  ret = false;
133  }
134  } else if ( (v_iter->start <= iter_p->address) && (iter_p->address < v_iter->end) ) {
135  if (level >= ErrorList::ErrorElem::WARNING){
136  err.insert(ErrorList::ErrorElem(ErrorList::ErrorElem::WARNING,
137  ErrorList::ErrorElem::WRONG_REGISTER_ADDRESSES,
138  *iter_p, *(v_iter->iter), _mapFileName));
139  ret = false;
140  }
141  }
142  }
143  }
144  v_addresses.push_back(address);
145  }
146  return ret;
147  }

255  std::list< RegisterInfoMap::RegisterInfo > RegisterInfoMap::getRegistersInModule( std::string const & moduleName){
256  // first sort all elements accordind the names (module first, then register in module)
257  // make a copy to keep the original order from the map file
258  std::vector<RegisterInfo> sortedRegisterInfos;
259  for(auto it = _catalogue.begin(); it != _catalogue.end(); ++it) {
260  auto infoCast = boost::static_pointer_cast<RegisterInfoMap::RegisterInfo>(it.get());
261  sortedRegisterInfos.push_back(*infoCast);
262  }
263  std::sort(sortedRegisterInfos.begin(), sortedRegisterInfos.end(), compareRegisterInfosByName_functor());
264 
265  // The vector is sorted, first module, than register name.
266  // Find the first iterator with the module name, starting at the beginning,
267  // then search for the first iterator with another name, starting from the previousy found first match.
268  std::vector<RegisterInfo>::iterator firstMatchingIterator =
269  std::find_if( sortedRegisterInfos.begin(), sortedRegisterInfos.end(), compareModuleName_pred( moduleName ) );
270  std::vector<RegisterInfo>::iterator firstNotMatchingIterator =
271  std::find_if( firstMatchingIterator, sortedRegisterInfos.end(),
272  std::not1( compareModuleName_pred(moduleName) ) );
273 
274  // fill the list
275  std::list<RegisterInfoMap::RegisterInfo> registerInfoList;
276  for (std::vector<RegisterInfo>::iterator it = firstMatchingIterator;
277  it != firstNotMatchingIterator; ++it){
278  registerInfoList.push_back( *it );
279  }
280 
281  return registerInfoList;
282  }
