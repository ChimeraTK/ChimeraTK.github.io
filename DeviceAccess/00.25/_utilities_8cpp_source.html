37  Sdm Utilities::parseSdm(std::string sdmString) {
38  Sdm sdmInfo;
39  size_t signatureLen = 6;
40  if (sdmString.empty())
41  throw SdmUriParseException("Invalid sdm.", SdmUriParseException::INVALID_SDM);
42  if (sdmString.length() < signatureLen)
43  throw SdmUriParseException("Invalid sdm.", SdmUriParseException::INVALID_SDM);
44  if (sdmString.substr(0,6) !="sdm://")
45  throw SdmUriParseException("Invalid sdm.", SdmUriParseException::INVALID_SDM);
46  int pos = 6;
47 
48  std::size_t found = sdmString.find_first_of("/", pos);
49  std::string subUri;
50  if (found != std::string::npos)
51  {
52  sdmInfo._Host = sdmString.substr(pos, found - pos); // Get the Host
53  }
54  else
55  {
56  throw SdmUriParseException("Invalid sdm.", SdmUriParseException::INVALID_SDM);
57  }
58  if (sdmString.length() < found+1)
59  return sdmInfo;
60  subUri = sdmString.substr(found+1);
61  //let's do a sanity check, only one delimiter occurrence is allowed.
62  if (countOccurence(subUri,':') > 1) /* check ':' */
63  {
64  throw SdmUriParseException("Invalid sdm.", SdmUriParseException::INVALID_SDM);
65  }
66  if (countOccurence(subUri,';') > 1) /* check ';' */
67  {
68  throw SdmUriParseException("Invalid sdm.", SdmUriParseException::INVALID_SDM);
69  }
70  if (countOccurence(subUri,'=') > 1) /* check '=' */
71  {
72  throw SdmUriParseException("Invalid sdm.", SdmUriParseException::INVALID_SDM);
73  }
74  std::vector<std::string> tokens;
75  boost::split(tokens, subUri, boost::is_any_of(":;="));
76  int numOfTokens = tokens.size();
77  if (numOfTokens < 1)
78  return sdmInfo;
79  int counter = 0;
80  sdmInfo._Interface = tokens[counter]; // Get the Interface
81  counter++;
82  if (counter < numOfTokens)
83  {
84  // Get the Instance
85  found = sdmString.find_first_of(":", pos);
86  if (found != std::string::npos) {
87  sdmInfo._Instance = tokens[counter];
88  counter++;
89  }
90  }
91  if (counter < numOfTokens)
92  {
93  // Get the Protocol
94  found = sdmString.find_first_of(";", pos);
95  if (found != std::string::npos) {
96  sdmInfo._Protocol = tokens[counter];
97  counter++;
98  }
99  }
100  if (counter < numOfTokens)
101  {
102  // Get the Parameters
103  found = sdmString.find_first_of("=", pos);
104  if (found != std::string::npos) {
105  std::string parameters = tokens[counter];
106  std::vector<std::string> paramterTokens;
107  boost::split(paramterTokens, parameters, boost::is_any_of(","));
108  for (uint i=0; i < paramterTokens.size();i++ )
109  sdmInfo._Parameters.push_back(paramterTokens[i]);
110  }
111  }
112 
113  return sdmInfo;
114 
115  }
