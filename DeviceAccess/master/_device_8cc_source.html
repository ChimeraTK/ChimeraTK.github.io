82  void Device::readReg(const std::string &regName,
83  const std::string &regModule, int32_t *data,
84  size_t dataSize, uint32_t addRegOffset) const {
85  if(dataSize % sizeof(int32_t) != 0) {
86  throw DeviceException("Wrong data size - must be dividable by 4", DeviceException::WRONG_PARAMETER);
87  }
88  if(addRegOffset % sizeof(int32_t) != 0) {
89  throw DeviceException("Wrong additional register offset - must be dividable by 4", DeviceException::WRONG_PARAMETER);
90  }
91  try {
92  auto vec = read<int32_t>(RegisterPath(regModule)/regName, dataSize/sizeof(int32_t), addRegOffset/sizeof(int32_t), true);
93  memcpy(data,vec.data(),vec.size()*sizeof(int32_t));
94  }
95  catch(DeviceException &e) { // translate exception for compatibility
96  if(e.getID() == DeviceException::REGISTER_DOES_NOT_EXIST) {
97  throw ChimeraTK::MapFileException(e.what(), ChimeraTK::LibMapException::EX_NO_REGISTER_IN_MAP_FILE);
98  }
99  throw;
100  }
101  }
