163  ConfigReader::ConfigReader(EntityOwner* owner, const std::string& name, const std::string& fileName,
164  const std::unordered_set<std::string>& tags)
165  : ApplicationModule(owner, name, "Configuration read from file '" + fileName + "'", false, tags),
166  _fileName(fileName) {
167  // parse the file into a DOM structure
168  xmlpp::DomParser parser;
169  try {
170  parser.parse_file(fileName);
171  }
172  catch(xmlpp::exception& e) {
173  throw ChimeraTK::logic_error("ConfigReader: Error opening the config file '" + fileName + "': " + e.what());
174  }
175 
176  // get root element
177  const auto root = parser.get_document()->get_root_node();
178  if(root->get_name() != "configuration") {
179  parsingError("Expected 'configuration' tag instead of: " + root->get_name());
180  }
181 
182  // parsing loop
183  for(const auto& child : root->get_children()) {
184  // cast into element, ignore if not an element (e.g. comment)
185  const xmlpp::Element* element = dynamic_cast<const xmlpp::Element*>(child);
186  if(!element) continue;
187  if(element->get_name() != "variable") {
188  parsingError("Expected 'variable' tag instead of: " + root->get_name());
189  }
190 
191  // obtain attributes from the element
192  auto name = element->get_attribute("name");
193  if(!name) parsingError("Missing attribute 'name' for the 'variable' tag.");
194  auto type = element->get_attribute("type");
195  if(!type) parsingError("Missing attribute 'type' for the 'variable' tag.");
196 
197  // scalar value: obtain value from attribute
198  auto value = element->get_attribute("value");
199  if(value) {
200  // create accessor and store value in map using the functor
201  bool processed{false};
202  boost::fusion::for_each(
203  variableMap.table, FunctorFill(this, type->get_value(), name->get_value(), value->get_value(), processed));
204  if(!processed)
205  parsingError("Incorrect value '" + type->get_value() + "' for attribute 'type' of the 'variable' tag.");
206  }
207  // array value: obtain values from child elements
208  else {
209  bool valueFound = false;
210  std::map<size_t, std::string> values;
211  for(const auto& valueChild : child->get_children()) {
212  // obtain value child element and extract index and value from
213  // attributes
214  const xmlpp::Element* valueElement = dynamic_cast<const xmlpp::Element*>(valueChild);
215  if(!valueElement) continue; // ignore comments etc.
216  if(valueElement->get_name() != "value") {
217  parsingError("Expected 'value' tag instead of: " + valueElement->get_name());
218  }
219  valueFound = true;
220  auto index = valueElement->get_attribute("i");
221  if(!index) parsingError("Missing attribute 'index' for the 'value' tag.");
222  auto value = valueElement->get_attribute("v");
223  if(!value) parsingError("Missing attribute 'value' for the 'value' tag.");
224 
225  // get index as number and store value as a string
226  size_t intIndex;
227  try {
228  intIndex = std::stoi(index->get_value());
229  }
230  catch(std::exception& e) {
231  parsingError(
232  "Cannot parse string '" + std::string(index->get_value()) + "' as an index number: " + e.what());
233  }
234  values[intIndex] = value->get_value();
235  }
236 
237  // make sure there is at least one value
238  if(!valueFound) {
239  parsingError("Each variable must have a value, either specified as an "
240  "attribute or as child tags.");
241  }
242 
243  // create accessor and store array value in map using functor
244  bool processed{false};
245  boost::fusion::for_each(
246  variableMap.table, ArrayFunctorFill(this, type->get_value(), name->get_value(), values, processed));
247  if(!processed)
248  parsingError("Incorrect value '" + type->get_value() + "' for attribute 'type' of the 'variable' tag.");
249  }
250  }
