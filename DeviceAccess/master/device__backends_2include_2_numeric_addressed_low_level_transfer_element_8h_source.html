77  bool isMergeable(const boost::shared_ptr<TransferElement const>& other) const {
78  if(!_dev->canMergeRequests()) return false;
79 
80  // accessor type, device and bar must be the same
81  auto rhsCasted = boost::dynamic_pointer_cast<const NumericAddressedLowLevelTransferElement>(other);
82  if(!rhsCasted) return false;
83  if(_dev != rhsCasted->_dev) return false;
84  if(_bar != rhsCasted->_bar) return false;
85 
86  // only allow adjacent and overlapping address areas to be merged
87  if(_startAddress + _numberOfBytes < rhsCasted->_startAddress) return false;
88  if(_startAddress > rhsCasted->_startAddress + rhsCasted->_numberOfBytes) return false;
89  return true;
90  }
