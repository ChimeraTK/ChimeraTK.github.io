22  // Generate a static data member of type Tag::type in which to store
23  // the address of a private member. It is crucial that Tag does not
24  // depend on the /value/ of the the stored address in any way so that
25  // we can access it from ordinary code without directly touching
26  // private data.
27  template <class Tag>
28  struct stowed
29  {
30  static typename Tag::type value;
31  };
32  template <class Tag>
33  typename Tag::type stowed<Tag>::value;
34 
35  // Generate a static data member whose constructor initializes
36  // stowed<Tag>::value. This type will only be named in an explicit
37  // instantiation, where it is legal to pass the address of a private
38  // member.
39  template <class Tag, typename Tag::type x>
40  struct stow_private
41  {
42  stow_private() { stowed<Tag>::value = x; }
43  static stow_private instance;
44  };
45  template <class Tag, typename Tag::type x>
46  stow_private<Tag,x> stow_private<Tag,x>::instance;
