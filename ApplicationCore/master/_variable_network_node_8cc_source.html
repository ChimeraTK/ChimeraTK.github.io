153  VariableNetworkNode VariableNetworkNode::operator>>(VariableNetworkNode other) {
154  if(pdata->direction == VariableDirection::invalid) {
155  if(!other.hasOwner()) {
156  pdata->direction = VariableDirection::feeding;
157  }
158  else {
159  if(other.getOwner().hasFeedingNode()) {
160  pdata->direction = VariableDirection::consuming;
161  if(getType() == NodeType::Device) { // special treatment for Device-type variables: consumers are push-type
162  pdata->mode = UpdateMode::push;
163  }
164  }
165  else {
166  pdata->direction = VariableDirection::feeding;
167  }
168  }
169  }
170  if(other.pdata->direction == VariableDirection::invalid) {
171  if(!hasOwner()) {
172  other.pdata->direction = VariableDirection::consuming;
173  if(other.getType() == NodeType::Device) { // special treatment for Device-type variables: consumers are push-type
174  other.pdata->mode = UpdateMode::push;
175  }
176  }
177  else {
178  if(getOwner().hasFeedingNode()) {
179  other.pdata->direction = VariableDirection::consuming;
180  if(other.getType() == NodeType::Device) { // special treatment for Device-type variables: consumers are push-type
181  other.pdata->mode = UpdateMode::push;
182  }
183  }
184  else {
185  other.pdata->direction = VariableDirection::feeding;
186  }
187  }
188  }
189  Application::getInstance().connect(*this, other);
190  return *this;
191  }

195  VariableNetworkNode VariableNetworkNode::operator[](VariableNetworkNode trigger) {
196 
197  // check if node already has a trigger
198  if(pdata->externalTrigger.getType() != NodeType::invalid) {
199  throw ChimeraTK::logic_error("Only one external trigger per variable network is allowed.");
200  }
201 
202  // force direction of the node we are operating on to be feeding
203  if(pdata->direction == VariableDirection::invalid) pdata->direction = VariableDirection::feeding;
204  assert(pdata->direction == VariableDirection::feeding);
205 
206  // force direction of the triggering node to be feeding
207  if(trigger.pdata->direction == VariableDirection::invalid) trigger.pdata->direction = VariableDirection::feeding;
208  assert(trigger.pdata->direction == VariableDirection::feeding);
209 
210  // check if already existing in map
211  if(pdata->nodeWithTrigger.count(trigger) > 0) {
212  return pdata->nodeWithTrigger[trigger];
213  }
214 
215  // create copy of the node
216  pdata->nodeWithTrigger[trigger].pdata = boost::make_shared<VariableNetworkNode_data>(*pdata);
217 
218  // add ourselves as a trigger receiver to the other network
219  if(!trigger.hasOwner()) {
220  Application::getInstance().createNetwork().addNode(trigger);
221  }
222  trigger.getOwner().addNodeToTrigger(pdata->nodeWithTrigger[trigger]);
223 
224  // set flag and store pointer to other network
225  pdata->nodeWithTrigger[trigger].pdata->externalTrigger = trigger;
226 
227  // return the new node
228  return pdata->nodeWithTrigger[trigger];
229  }
