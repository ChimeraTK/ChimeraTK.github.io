51  void TransferGroup::addAccessor<TransferElement>(TransferElement &accessor) {
52 
53  // check if accessor is already in a transfer group
54  if(accessor.getHighLevelImplElement()->isInTransferGroup) {
55  throw DeviceException("The given accessor is already in a TransferGroup and cannot be added to another.",
56  DeviceException::WRONG_PARAMETER);
57  }
58 
59  // Store the accessor itself in the high-level list
60  highLevelElements.push_back(accessor.getHighLevelImplElement());
61  accessor.getHighLevelImplElement()->isInTransferGroup = true;
62 
63  // Iterate over all new hardware-accessing elements and add them to the list
64  auto newElements = accessor.getHardwareAccessingElements();
65  for(unsigned int i=0; i<newElements.size(); i++) {
66  elements.push_back(newElements[i]);
67  }
68 
69  // Iterate over all hardware-accessing elements and merge any potential duplicates
70  // Note: This cannot be done only for the newly added elements, since e.g. in NumericAddressedBackend the new
71  // accessor might fill the gap (in the address space) between to already added accessors, in which case also the
72  // already added accessors must be merged.
73  // We go backwards through the list, since the latest element is the recently added accessor, for which we still
74  // have the original pimpl frontend (variable "accessor"). In cases were we need to replace the entire accessor
75  // implementation we need to do this on the pimpl frontent instead of the implementation stored in the
76  // highLevelElements list.
77  std::vector< std::vector< boost::shared_ptr<TransferElement> >::iterator > eraseList;
78  for(int i=elements.size()-1; i>0; --i) {
79  for(int k=i-1; k>=0; --k) {
80  if(elements[i]->isSameRegister(elements[k])) {
81  // replace the TransferElement inside any high-level accessor with the merged version
82  accessor.replaceTransferElement(elements[k]); // might need to replace the implementation of the pimpl accessor frontend
83  for(auto &m : highLevelElements) {
84  m->replaceTransferElement(elements[k]);
85  }
86  eraseList.push_back(elements.begin() + i); // store iterator to element to be deleted
87  break;
88  }
89  }
90  }
91 
92  // Erase the duplicates (cannot do this inside the loop since it would invalidate the iterators)
93  for(auto &e : eraseList) elements.erase(e);
94 
95  // remove elements which are on both the highLevelElements and the elements lists from the highLevelElements, since
96  // we should not run postRead() resp. preWrite()/postWrite() on them
97  for(auto &elem : elements) {
98  bool inHLE = false;
99  std::vector< boost::shared_ptr<TransferElement> >::iterator deleteIt;
100  for(auto hle = highLevelElements.begin(); hle != highLevelElements.end(); ++hle) {
101  if(*hle == elem) { // this is comparing the shared pointers
102  inHLE = true;
103  deleteIt = hle;
104  }
105  }
106  if(inHLE) {
107  highLevelElements.erase(deleteIt);
108  elem->isInTransferGroup = false; // only high level elements should count as being in the group
109  }
110  }
111 
112  // Update read-only flag
113  if(accessor.isReadOnly()) readOnly = true;
114  }
