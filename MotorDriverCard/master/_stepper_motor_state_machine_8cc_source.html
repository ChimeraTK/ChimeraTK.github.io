53  void BasicStepperMotor::StateMachine::waitForStandstill(){
54 
55  if(!_motorControler->isMotorMoving()){
56 
57  _asyncActionActive.exchange(false);
58 
59  // Move may be stopped by
60  // user requesting state transition
61  if(hasRequestedState()){
62  moveToRequestedState();
63  return;
64  }
65 
66  Event stateExitEvnt = stopEvent;
67 
68  // Motor stopped by itself
69  if(!_stepperMotor.verifyMoveAction()){
70  _stepperMotor._errorMode.exchange(StepperMotorError::MOVE_INTERRUPTED);
71 
72  _motorControler->setTargetPosition(_motorControler->getActualPosition());
73  stateExitEvnt = errorEvent;
74  }
75  performTransition(stateExitEvnt);
76  return;
77  }
78  }
