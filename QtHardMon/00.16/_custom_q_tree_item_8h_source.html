259 template <typename T>
260 inline void CustomQTreeItem::putValuesIntoTable(
261  const TableWidgetData& tabledata, std::vector<T> const& buffer) {
262  int column = 0;
263  if (typeid(T) == typeid(int)) {
264  column = qthardmon::FIXED_POINT_DISPLAY_COLUMN;
265  } else if (typeid(T) == typeid(double)) {
266  column = qthardmon::FLOATING_POINT_DISPLAY_COLUMN;
267  }
268 
269  clearTable(tabledata);
270 
271  QTableWidget* table = tabledata.table;
272  unsigned int maxRow = tabledata.tableMaxRowCount;
273 
274  size_t rowCount = (buffer.size() > maxRow) ? maxRow : buffer.size();
275  // Prepare the appropriate number of rows in the table
276  table->setRowCount(rowCount);
277 
278  for (size_t row = 0; row < rowCount; row++) {
279  // Prepare a data item with a QVariant. The QVariant takes care that the
280  // type is recognised as
281  // int and a proper editor (spin box) is used when editing in the GUI.
282  QTableWidgetItem* dataItem = new QTableWidgetItem();
283  QTableWidgetItem* rowItem = new QTableWidgetItem();
284  std::stringstream rowAsText;
285  rowAsText << row;
286 
287  rowItem->setText(rowAsText.str().c_str());
288  table->setVerticalHeaderItem(row, rowItem);
289 
290  if (row == maxRow) { // The register is too large to display. Show that it
291  // is truncated and stop reading
292  dataItem->setText("truncated");
293  dataItem->setFlags(dataItem->flags() & ~Qt::ItemIsSelectable &
294  ~Qt::ItemIsEditable);
295  dataItem->setToolTip("List is truncated. You can change the number of "
296  "words displayed in the preferences.");
297  table->setItem(row, 0, dataItem);
298  break;
299  }
300 
301  T registerContent = buffer[row];
302  dataItem->setData(0, QVariant(registerContent)); // 0 is the default role
303  table->setItem(row, column, dataItem);
304  } // for row
305 }

307 template <typename T>
308 inline std::vector<T> CustomQTreeItem::copyValuesFromTable(
309  const TableWidgetData& tabledata, unsigned int count) {
310 
311  unsigned int maxRowCount = tabledata.tableMaxRowCount;
312  QTableWidget const* table = tabledata.table;
313  std::vector<T> buffer(count);
314 
315  for (unsigned int row = 0; (row < count) && (row < maxRowCount); row++) {
316  int registerContent =
317  table->item(row, qthardmon::FIXED_POINT_DISPLAY_COLUMN)
318  ->data(0 /*default role*/)
319  .toInt();
320  buffer[row] = registerContent;
321  }
322  return buffer;
323 }
